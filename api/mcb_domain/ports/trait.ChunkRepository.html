Expand description

Repository for managing code chunks

Persists extracted code chunks and provides retrieval by collection or ID. Chunks are the granular unit of semantic indexing - each chunk can be independently searched, updated, or deleted.

§Example

use mcb_domain::ports::ChunkRepository;
use mcb_domain::entities::CodeChunk;
use std::sync::Arc;

async fn save_chunk(repo: Arc<dyn ChunkRepository>) -> mcb_domain::Result<()> {
    // Create a code chunk
    let chunk = CodeChunk {
        id: "chunk_001".to_string(),
        content: "fn hello() { println!(\"Hello!\"); }".to_string(),
        file_path: "src/lib.rs".to_string(),
        start_line: 1,
        end_line: 3,
        language: "rust".to_string(),
        metadata: serde_json::json!({"type": "function"}),
    };

    // Save to collection
    let collection = "my-codebase";
    let id = repo.save(collection, &chunk).await?;
    println!("Saved chunk: {}", id);
    Ok(())
}
