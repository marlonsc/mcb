3use std::future::Future;
4use std::path::Path;
5use std::pin::Pin;
6use std::sync::Arc;
7
8use async_trait::async_trait;
9use mcb_domain::entities::project::ProjectType;
10use mcb_domain::ports::ProjectDetectorService;
11
12use super::detect_all_projects;
13
14/// An async function that detects project types at a given path.
15type DetectFn = dyn for<'a> Fn(&'a Path) -> Pin<Box<dyn Future<Output = Vec<ProjectType>> + Send + 'a>>
16    + Send
17    + Sync;
18
19/// Thin adapter implementing the domain port via the detect-all facade.
20struct UniversalProjectDetector {
21    /// The detection function used to discover project types.
22    detect_fn: Arc<DetectFn>,
23}
24
25#[async_trait]
26impl ProjectDetectorService for UniversalProjectDetector {
27    async fn detect_all(&self, path: &Path) -> Vec<ProjectType> {
28        (self.detect_fn)(path).await
29    }
30}
31
32mcb_domain::register_project_detection_service!(
33    mcb_utils::constants::DEFAULT_LANGUAGE_PROVIDER,
34    "Universal project detector using all language-specific detectors",
35    |_config| {
36        let detect_fn: Arc<DetectFn> = Arc::new(|path| Box::pin(detect_all_projects(path)));
37        Ok(Arc::new(UniversalProjectDetector { detect_fn }))
38    }
39);
