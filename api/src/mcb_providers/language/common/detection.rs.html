4//! Language Detection
5//!
6//! Provides functions to detect programming languages from file extensions,
7//! check language support for AST chunking, and retrieve language-specific
8//! configuration values.
9
10use super::constants::{CHUNK_SIZE_GENERIC, EXTENSION_LANG_MAP, LANG_CHUNK_SIZE_MAP};
11use mcb_domain::constants::lang::*;
12
13/// Detect language from file extension
14///
15/// Returns a string identifier for the programming language based on the file extension.
16/// Returns "unknown" for unsupported or unrecognized extensions.
17#[must_use]
18pub fn language_from_extension(ext: &str) -> String {
19    let ext_lower = ext.to_lowercase();
20    EXTENSION_LANG_MAP
21        .iter()
22        .find(|(exts, _)| exts.iter().any(|e| *e == ext_lower))
23        .map_or_else(|| LANG_UNKNOWN.to_owned(), |(_, lang)| (*lang).to_owned())
24}
25
26/// Check if a language is supported for AST-based chunking
27#[must_use]
28pub fn is_language_supported(language: &str) -> bool {
29    matches!(
30        language,
31        LANG_RUST
32            | LANG_PYTHON
33            | LANG_JAVASCRIPT
34            | LANG_TYPESCRIPT
35            | LANG_GO
36            | LANG_JAVA
37            | LANG_C
38            | LANG_CPP
39            | LANG_CSHARP
40            | LANG_RUBY
41            | LANG_PHP
42            | LANG_SWIFT
43            | LANG_KOTLIN
44    )
45}
46
47/// Get the chunk size for a specific language
48#[must_use]
49pub fn get_chunk_size(language: &str) -> usize {
50    LANG_CHUNK_SIZE_MAP
51        .iter()
52        .find(|(langs, _)| langs.contains(&language))
53        .map_or(CHUNK_SIZE_GENERIC, |(_, size)| *size)
54}
55
56/// Get a list of all supported languages
57#[must_use]
58pub fn supported_languages() -> Vec<String> {
59    vec![
60        LANG_RUST.to_owned(),
61        LANG_PYTHON.to_owned(),
62        LANG_JAVASCRIPT.to_owned(),
63        LANG_TYPESCRIPT.to_owned(),
64        LANG_GO.to_owned(),
65        LANG_JAVA.to_owned(),
66        LANG_C.to_owned(),
67        LANG_CPP.to_owned(),
68        LANG_CSHARP.to_owned(),
69        LANG_RUBY.to_owned(),
70        LANG_PHP.to_owned(),
71        LANG_SWIFT.to_owned(),
72        LANG_KOTLIN.to_owned(),
73    ]
74}
