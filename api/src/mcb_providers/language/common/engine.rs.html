89/// Intelligent chunking engine using tree-sitter
90#[derive(Default)]
91pub struct IntelligentChunker;
92
93impl IntelligentChunker {
94    /// Create a new intelligent chunker
95    #[must_use]
96    pub fn new() -> Self {
97        Self
98    }
99
100    /// Chunk code based on language-specific structural analysis
101    pub fn chunk_code(
102        &self,
103        content: &str,
104        file_name: &str,
105        language: &Language,
106    ) -> Vec<CodeChunk> {
107        if let Some(processor) = LANGUAGE_PROCESSORS.get(language) {
108            match Self::parse_with_tree_sitter(content, &processor.get_language()) {
109                Ok(tree) => {
110                    let chunks = processor
111                        .extract_chunks_with_tree_sitter(&tree, content, file_name, language);
112                    if !chunks.is_empty() {
113                        return chunks;
114                    }
115                }
116                Err(e) => {
117                    mcb_domain::warn!(
118                        "engine",
119                        "tree-sitter parse failed, using generic chunking",
120                        &format!("file = {file_name}, error = {e}")
121                    );
122                }
123            }
124        }
125
126        Self::chunk_generic(content, file_name, language)
127    }
128
129    /// Chunk code asynchronously (offloads to blocking thread)
130    pub async fn chunk_code_async(
131        &self,
132        content: String,
133        file_name: String,
134        language: Language,
135    ) -> Vec<CodeChunk> {
136        tokio::task::spawn_blocking(move || {
137            let chunker = Self::new();
138            chunker.chunk_code(&content, &file_name, &language)
139        })
140        .await
141        .unwrap_or_default()
142    }
143
144    /// Generic chunking for unsupported languages
145    fn chunk_generic(content: &str, file_name: &str, language: &Language) -> Vec<CodeChunk> {
146        let lines: Vec<&str> = content.lines().collect();
147        let mut chunks = Vec::new();
148        let chunk_size = CHUNK_SIZE_GENERIC;
149        // Clone language once before loop to avoid repeated allocations
150        let lang = language.clone();
151        let file = file_name.to_owned();
152
153        for (chunk_idx, chunk_lines) in lines.chunks(chunk_size).enumerate() {
154            let start_line = chunk_idx * chunk_size;
155            let end_line = start_line + chunk_lines.len() - 1;
156
157            let content = chunk_lines.join("\n").trim().to_owned();
158            if content.is_empty() || content.len() < 20 {
159                continue;
160            }
161
162            chunks.push(CodeChunk {
163                id: format!("{file_name}_{chunk_idx}"),
164                content,
165                file_path: file.clone(),
166                start_line: start_line as u32,
167                end_line: end_line as u32,
168                language: lang.clone(),
169                metadata: serde_json::json!({
170                    "file": file_name,
171                    "chunk_index": chunk_idx,
172                    "chunk_type": "generic"
173                }),
174            });
175        }
176
177        chunks
178    }
179
180    /// Parse code with tree-sitter
181    fn parse_with_tree_sitter(
182        content: &str,
183        language: &tree_sitter::Language,
184    ) -> Result<tree_sitter::Tree> {
185        let mut parser = tree_sitter::Parser::new();
186        parser
187            .set_language(language)
188            .map_err(|e| Error::internal(format!("Failed to set tree-sitter language: {e:?}")))?;
189
190        let tree = parser
191            .parse(content, None)
192            .ok_or_else(|| Error::internal("Tree-sitter parsing failed".to_owned()))?;
193
194        Ok(tree)
195    }
196}

198#[async_trait]
199impl CodeChunker for IntelligentChunker {
200    async fn chunk_file(
201        &self,
202        file_path: &Path,
203        _options: ChunkingOptions,
204    ) -> Result<ChunkingResult> {
205        let content = tokio::fs::read_to_string(file_path)
206            .await
207            .map_err(|e| Error::io(e.to_string()))?;
208
209        let file_name = mcb_domain::utils::path::path_to_utf8_string(file_path)
210            .map_err(|e| Error::io(e.to_string()))?;
211        let ext = file_path.extension().and_then(|e| e.to_str()).unwrap_or("");
212        let language = language_from_extension(ext);
213
214        self.chunk_content(&content, &file_name, language, _options)
215            .await
216    }
217
218    async fn chunk_content(
219        &self,
220        content: &str,
221        file_name: &str,
222        language: Language,
223        _options: ChunkingOptions,
224    ) -> Result<ChunkingResult> {
225        let chunks = self.chunk_code(content, file_name, &language);
226        let used_ast = is_language_supported(&language);
227
228        Ok(ChunkingResult {
229            file_path: file_name.to_owned(),
230            language,
231            chunks,
232            used_ast,
233        })
234    }
235
236    async fn chunk_batch(
237        &self,
238        file_paths: &[&Path],
239        options: ChunkingOptions,
240    ) -> Result<Vec<ChunkingResult>> {
241        let mut results = Vec::with_capacity(file_paths.len());
242        for path in file_paths {
243            results.push(self.chunk_file(path, options).await?);
244        }
245        Ok(results)
246    }
247
248    fn supported_languages(&self) -> Vec<Language> {
249        LANGUAGE_PROCESSORS.keys().cloned().collect()
250    }
251}

253/// Universal Language Chunking Provider
254///
255/// A provider that supports all languages by delegating to the `IntelligentChunker`.
256/// This is used for dependency injection where we need a single provider that
257/// can handle any supported language.
258pub struct UniversalLanguageChunkingProvider {
259    chunker: IntelligentChunker,
260}
261
262impl UniversalLanguageChunkingProvider {
263    /// Create a new universal language chunking provider
264    #[must_use]
265    pub fn new() -> Self {
266        Self {
267            chunker: IntelligentChunker::new(),
268        }
269    }
270}
271
272impl Default for UniversalLanguageChunkingProvider {
273    fn default() -> Self {
274        Self::new()
275    }
276}
277
278impl mcb_domain::ports::LanguageChunkingProvider for UniversalLanguageChunkingProvider {
279    fn language(&self) -> mcb_domain::value_objects::Language {
280        "universal".to_owned()
281    }
282
283    fn extensions(&self) -> &[&'static str] {
284        &[
285            "rs", "py", "js", "ts", "java", "go", "c", "cpp", "cs", "rb", "php", "swift", "kt",
286        ]
287    }
288
289    fn chunk(&self, content: &str, file_path: &str) -> Vec<mcb_domain::entities::CodeChunk> {
290        let path = std::path::Path::new(file_path);
291        let ext = path.extension().and_then(|e| e.to_str()).unwrap_or("");
292        let language = super::detection::language_from_extension(ext);
293        self.chunker.chunk_code(content, file_path, &language)
294    }
295
296    fn provider_name(&self) -> &str {
297        "universal"
298    }
299}
