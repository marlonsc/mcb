6/// Implements `LanguageProcessor` by delegating to an inner `BaseProcessor` field.
7#[macro_export]
8macro_rules! impl_processor_delegate {
9    ($processor_ty:ty, $inner_field:ident) => {
10        impl $crate::language::common::LanguageProcessor for $processor_ty {
11            fn config(&self) -> &$crate::language::common::LanguageConfig {
12                self.$inner_field.config()
13            }
14
15            fn extract_chunks_with_tree_sitter(
16                &self,
17                tree: &tree_sitter::Tree,
18                content: &str,
19                file_name: &str,
20                language: &mcb_domain::value_objects::Language,
21            ) -> Vec<mcb_domain::entities::CodeChunk> {
22                self.$inner_field
23                    .extract_chunks_with_tree_sitter(tree, content, file_name, language)
24            }
25        }
26    };
27}
28
29/// Implements `Default` via `new()` and `LanguageProcessor` via delegation.
30#[macro_export]
31macro_rules! impl_delegating_language_processor {
32    ($processor_ty:ty, $inner_field:ident) => {
33        impl Default for $processor_ty {
34            fn default() -> Self {
35                Self::new()
36            }
37        }
38
39        $crate::impl_processor_delegate!($processor_ty, $inner_field);
40    };
41}

43/// Generates a complete language processor struct with `new()`, `Default`, and `LanguageProcessor` impl.
44#[macro_export]
45macro_rules! impl_simple_language_processor {
46    (
47        $processor_ty:ident,
48        language = $language:expr,
49        chunk_size = $chunk_size:expr,
50        max_depth = $max_depth:expr,
51        nodes = [$($node_type:expr),+ $(,)?]
52    ) => {
53
54
55        /// Language processor generated by `impl_simple_language_processor!`.
56        pub struct $processor_ty {
57            processor: $crate::language::common::BaseProcessor,
58        }
59
60
61        impl $processor_ty {
62            /// Create a new processor instance with default configuration.
63            pub fn new() -> Self {
64                let config = $crate::language::common::LanguageConfig::new($language)
65                    .with_rules(vec![$crate::language::common::NodeExtractionRule {
66                        node_types: vec![$($node_type.to_string()),+],
67                        min_length: 30,
68                        min_lines: 2,
69                        max_depth: $max_depth,
70                        priority: 5,
71                        include_context: true,
72                    }])
73                    .with_chunk_size($chunk_size);
74
75                Self {
76                    processor: $crate::language::common::BaseProcessor::new(config),
77                }
78            }
79        }
80
81        $crate::impl_delegating_language_processor!($processor_ty, processor);
82    };
83}
