14/// Intelligent chunking engine using tree-sitter
15#[derive(Default)]
16pub struct IntelligentChunker;
17
18impl IntelligentChunker {
19    /// Create a new intelligent chunker
20    #[must_use]
21    pub fn new() -> Self {
22        Self
23    }
24    /// Chunk code based on language-specific structural analysis
25    pub fn chunk_code(
26        &self,
27        content: &str,
28        file_name: &str,
29        language: &Language,
30    ) -> Vec<CodeChunk> {
31        if let Some(processor) = LANGUAGE_PROCESSORS.get(language) {
32            match Self::parse_with_tree_sitter(content, &processor.get_language()) {
33                Ok(tree) => {
34                    let chunks = processor
35                        .extract_chunks_with_tree_sitter(&tree, content, file_name, language);
36                    if !chunks.is_empty() {
37                        return chunks;
38                    }
39                }
40                Err(e) => {
41                    mcb_domain::warn!(
42                        "engine",
43                        "tree-sitter parse failed, using generic chunking",
44                        &format!("file = {file_name}, error = {e}")
45                    );
46                }
47            }
48        }
49        Self::chunk_generic(content, file_name, language)
50    }
51    /// Chunk code asynchronously (offloads to blocking thread)
52    pub async fn chunk_code_async(
53        &self,
54        content: String,
55        file_name: String,
56        language: Language,
57    ) -> Vec<CodeChunk> {
58        tokio::task::spawn_blocking(move || {
59            let chunker = Self::new();
60            chunker.chunk_code(&content, &file_name, &language)
61        })
62        .await
63        .unwrap_or_else(|e| {
64            mcb_domain::error!(
65                "chunking_engine",
66                "spawn_blocking panic in chunking engine",
67                &e
68            );
69            Default::default()
70        })
71    }
72    /// Generic chunking for unsupported languages
73    fn chunk_generic(content: &str, file_name: &str, language: &Language) -> Vec<CodeChunk> {
74        let lines: Vec<&str> = content.lines().collect();
75        let mut chunks = Vec::new();
76        let chunk_size = CHUNK_SIZE_GENERIC;
77        // Clone language once before loop to avoid repeated allocations
78        let lang = language.clone();
79        let file = file_name.to_owned();
80        for (chunk_idx, chunk_lines) in lines.chunks(chunk_size).enumerate() {
81            let start_line = chunk_idx * chunk_size;
82            let end_line = start_line + chunk_lines.len() - 1;
83
84            let content = chunk_lines.join("\n").trim().to_owned();
85            if content.is_empty() || content.len() < 20 {
86                continue;
87            }
88
89            chunks.push(CodeChunk {
90                id: format!("{file_name}_{chunk_idx}"),
91                content,
92                file_path: file.clone(),
93                start_line: start_line as u32,
94                end_line: end_line as u32,
95                language: lang.clone(),
96                metadata: serde_json::json!({
97                    "file": file_name,
98                    "chunk_index": chunk_idx,
99                    "chunk_type": "generic"
100                }),
101            });
102        }
103        chunks
104    }
105    /// Parse code with tree-sitter
106    fn parse_with_tree_sitter(
107        content: &str,
108        language: &tree_sitter::Language,
109    ) -> Result<tree_sitter::Tree> {
110        let mut parser = tree_sitter::Parser::new();
111        parser
112            .set_language(language)
113            .map_err(|e| Error::internal(format!("Failed to set tree-sitter language: {e:?}")))?;
114
115        let tree = parser
116            .parse(content, None)
117            .ok_or_else(|| Error::internal("Tree-sitter parsing failed".to_owned()))?;
118        Ok(tree)
119    }
120}

122#[async_trait]
123impl CodeChunker for IntelligentChunker {
124    async fn chunk_file(
125        &self,
126        file_path: &std::path::Path,
127        _options: ChunkingOptions,
128    ) -> Result<ChunkingResult> {
129        let content = tokio::fs::read_to_string(file_path)
130            .await
131            .map_err(|e| Error::io(e.to_string()))?;
132
133        let file_name = mcb_utils::utils::path::path_to_utf8_string(file_path)
134            .map_err(|e| Error::io(e.to_string()))?;
135        let ext = file_path.extension().and_then(|e| e.to_str()).unwrap_or("");
136        let language = language_from_extension(ext);
137
138        self.chunk_content(&content, &file_name, language, _options)
139            .await
140    }
141    async fn chunk_content(
142        &self,
143        content: &str,
144        file_name: &str,
145        language: Language,
146        _options: ChunkingOptions,
147    ) -> Result<ChunkingResult> {
148        let chunks = self.chunk_code(content, file_name, &language);
149        let used_ast = is_language_supported(&language);
150        Ok(ChunkingResult {
151            file_path: file_name.to_owned(),
152            language,
153            chunks,
154            used_ast,
155        })
156    }
157    async fn chunk_batch(
158        &self,
159        file_paths: &[&std::path::Path],
160        options: ChunkingOptions,
161    ) -> Result<Vec<ChunkingResult>> {
162        let mut results = Vec::with_capacity(file_paths.len());
163        for path in file_paths {
164            results.push(self.chunk_file(path, options).await?);
165        }
166        Ok(results)
167    }
168    fn supported_languages(&self) -> Vec<Language> {
169        LANGUAGE_PROCESSORS.keys().cloned().collect()
170    }
171}
