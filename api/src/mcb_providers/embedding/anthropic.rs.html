44impl AnthropicEmbeddingProvider {
45    /// Get the maximum tokens for this model
46    #[must_use]
47    pub fn max_tokens(&self) -> usize {
48        // All Voyage AI models support the same max tokens
49        ANTHROPIC_MAX_INPUT_TOKENS
50    }
51
52    /// Send embedding request and get response data
53    async fn fetch_embeddings(&self, texts: &[String]) -> Result<serde_json::Value> {
54        let payload = serde_json::json!({
55            (EMBEDDING_PARAM_INPUT): texts,
56            (EMBEDDING_PARAM_MODEL): self.client.model,
57            "input_type": "document",
58            "encoding_format": "float"
59        });
60
61        let headers = vec![
62            (
63                HTTP_HEADER_AUTHORIZATION,
64                format!("Bearer {}", self.client.api_key),
65            ),
66            (HTTP_HEADER_CONTENT_TYPE, CONTENT_TYPE_JSON.to_owned()),
67        ];
68
69        send_json_request(JsonRequestParams {
70            client: &self.client.client,
71            method: reqwest::Method::POST,
72            url: format!("{}{}", self.base_url(), EMBEDDING_API_ENDPOINT),
73            timeout: self.client.timeout,
74            provider: "Anthropic",
75            operation: EMBEDDING_OPERATION_NAME,
76            kind: RequestErrorKind::Embedding,
77            headers: &headers,
78            body: Some(&payload),
79            retry: Some(RetryConfig::new(
80                PROVIDER_RETRY_COUNT,
81                std::time::Duration::from_millis(PROVIDER_RETRY_BACKOFF_MS),
82            )),
83        })
84        .await
85    }
86
87    /// Parse embedding vector from response data
88    fn parse_embedding(&self, index: usize, item: &serde_json::Value) -> Result<Embedding> {
89        parse_standard_embedding(&self.client.model, self.dimensions(), index, item)
90    }
91}
