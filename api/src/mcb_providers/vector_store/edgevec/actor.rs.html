9/// Single-threaded owner of the `EdgeVec` index, storage, and metadata.
10///
11/// Runs in its own task and serializes all vector-store operations by
12/// processing [`EdgeVecMessage`] values received on `receiver`.
13pub struct EdgeVecActor {
14    receiver: mpsc::Receiver<EdgeVecMessage>,
15    index: edgevec::HnswIndex,
16    storage: edgevec::VectorStorage,
17    metadata_store: DashMap<String, CollectionMetadata>,
18    id_map: DashMap<String, VectorId>,
19    config: EdgeVecConfig,
20}
21
22impl EdgeVecActor {
23    pub fn new(receiver: mpsc::Receiver<EdgeVecMessage>, config: EdgeVecConfig) -> Result<Self> {
24        let hnsw_config = edgevec::HnswConfig {
25            m: config.hnsw_config.m,
26            m0: config.hnsw_config.m0,
27            ef_construction: config.hnsw_config.ef_construction,
28            ef_search: config.hnsw_config.ef_search,
29            dimensions: config.dimensions as u32,
30            metric: match config.metric {
31                MetricType::L2Squared => edgevec::HnswConfig::METRIC_L2_SQUARED,
32                MetricType::Cosine => edgevec::HnswConfig::METRIC_COSINE,
33                MetricType::DotProduct => edgevec::HnswConfig::METRIC_DOT_PRODUCT,
34            },
35            _reserved: [0; 2],
36        };
37
38        let storage = edgevec::VectorStorage::new(&hnsw_config, None);
39        let index = edgevec::HnswIndex::new(hnsw_config, &storage)
40            .map_err(|e| Error::vector_db(format!("Failed to create EdgeVec HNSW index: {e}")))?;
41
42        Ok(Self {
43            receiver,
44            index,
45            storage,
46            metadata_store: DashMap::new(),
47            id_map: DashMap::new(),
48            config,
49        })
50    }
51}

53impl EdgeVecActor {
54    fn handle_create_collection(&self, name: String) -> Result<()> {
55        self.metadata_store.insert(name, HashMap::new());
56        Ok(())
57    }
58
59    fn handle_delete_collection(&mut self, name: &str) -> Result<()> {
60        if let Some((_, collection_metadata)) = self.metadata_store.remove(name) {
61            for external_id in collection_metadata.keys() {
62                if let Some(vector_id) = self.id_map.remove(external_id) {
63                    let _ = self.index.soft_delete(vector_id.1);
64                }
65            }
66        }
67        Ok(())
68    }
69
70    fn handle_collection_exists(&self, name: &str) -> Result<bool> {
71        Ok(self.metadata_store.contains_key(name))
72    }
73}

75impl EdgeVecActor {
76    fn get_collection_metadata(
77        &self,
78        name: &str,
79    ) -> Option<dashmap::mapref::one::Ref<'_, String, CollectionMetadata>> {
80        self.metadata_store.get(name)
81    }
82
83    fn collection_metadata_len(&self, name: &str) -> usize {
84        self.get_collection_metadata(name).map_or(0, |m| m.len())
85    }
86
87    fn handle_insert_vectors(
88        &mut self,
89        collection: &str,
90        vectors: Vec<Embedding>,
91        metadata: Vec<HashMap<String, serde_json::Value>>,
92    ) -> Result<Vec<String>> {
93        let mut ids = Vec::with_capacity(vectors.len());
94        let mut collection_metadata = self
95            .metadata_store
96            .entry(collection.to_owned())
97            .or_default();
98
99        for (vector, meta) in vectors.into_iter().zip(metadata) {
100            let external_id = format!("{}_{}", collection, id::generate());
101
102            match self.index.insert(&vector.vector, &mut self.storage) {
103                Ok(vector_id) => {
104                    self.id_map.insert(external_id.clone(), vector_id);
105                    let mut enriched_metadata = meta.clone();
106                    enriched_metadata.insert("id".to_owned(), serde_json::json!(external_id));
107                    collection_metadata
108                        .insert(external_id.clone(), serde_json::json!(enriched_metadata));
109                    ids.push(external_id);
110                }
111                Err(e) => {
112                    return Err(Error::vector_db(format!("Failed to insert vector: {e}")));
113                }
114            }
115        }
116        Ok(ids)
117    }
118
119    fn handle_delete_vectors(&mut self, collection: &str, ids: Vec<String>) -> Result<()> {
120        if let Some(mut collection_metadata) = self.metadata_store.get_mut(collection) {
121            for id in ids {
122                if let Some((_, vector_id)) = self.id_map.remove(&id) {
123                    let _ = self.index.soft_delete(vector_id);
124                }
125                collection_metadata.remove(&id);
126            }
127        }
128        Ok(())
129    }
130
131    fn handle_get_vectors_by_ids(&self, collection: &str, ids: Vec<String>) -> Vec<SearchResult> {
132        let mut final_results = Vec::new();
133        if let Some(collection_metadata) = self.get_collection_metadata(collection) {
134            for id in ids {
135                if let Some(meta_val) = collection_metadata.get(&id) {
136                    final_results.push(search_result_from_json_metadata(id, meta_val, 1.0));
137                }
138            }
139        }
140        final_results
141    }
142
143    fn handle_list_vectors(&self, collection: &str, limit: usize) -> Vec<SearchResult> {
144        let mut final_results = Vec::new();
145        if let Some(collection_metadata) = self.get_collection_metadata(collection) {
146            for (ext_id, meta_val) in collection_metadata.iter().take(limit) {
147                final_results.push(search_result_from_json_metadata(
148                    ext_id.to_owned(),
149                    meta_val,
150                    1.0,
151                ));
152            }
153        }
154        final_results
155    }
156}

158impl EdgeVecActor {
159    fn handle_search_similar(
160        &self,
161        collection: &str,
162        query_vector: &[f32],
163        limit: usize,
164    ) -> Result<Vec<SearchResult>> {
165        // The HNSW index is global (shared across all collections), so when
166        // multiple collections exist we must over-fetch to ensure enough
167        // results survive the per-collection metadata filter.
168        let total_vectors = self.index.len();
169        let collection_size = self.collection_metadata_len(collection);
170        let fetch_limit = if collection_size > 0 && total_vectors > collection_size {
171            let ratio = (total_vectors as f64 / collection_size as f64).ceil() as usize;
172            (limit * ratio).min(total_vectors)
173        } else {
174            limit
175        };
176
177        let results = self
178            .index
179            .search(query_vector, fetch_limit, &self.storage)
180            .map_err(|e| Error::vector_db(format!("Search failed: {e}")))?;
181
182        let Some(collection_metadata) = self.get_collection_metadata(collection) else {
183            return Ok(Vec::new());
184        };
185
186        let mut final_results = Vec::with_capacity(limit);
187        for res in results {
188            if final_results.len() >= limit {
189                break;
190            }
191            let Some(ext_id) = self.external_id_for(res.vector_id) else {
192                continue;
193            };
194            if let Some(meta_val) = collection_metadata.get(&ext_id) {
195                final_results.push(search_result_from_json_metadata(
196                    ext_id,
197                    meta_val,
198                    res.distance as f64,
199                ));
200            }
201        }
202        Ok(final_results)
203    }
204
205    /// Resolve the external string id for an internal HNSW vector id.
206    fn external_id_for(&self, vector_id: VectorId) -> Option<String> {
207        self.id_map
208            .iter()
209            .find(|entry| *entry.value() == vector_id)
210            .map(|entry| entry.key().to_owned())
211    }
212}

214impl EdgeVecActor {
215    fn handle_get_stats(&self, collection: &str) -> HashMap<String, serde_json::Value> {
216        let vector_count = self.collection_metadata_len(collection);
217        let mut stats = HashMap::new();
218        stats.insert(
219            STATS_FIELD_COLLECTION.to_owned(),
220            serde_json::json!(collection),
221        );
222        stats.insert(
223            STATS_FIELD_VECTORS_COUNT.to_owned(),
224            serde_json::json!(vector_count),
225        );
226        stats.insert(
227            "total_indexed_vectors".to_owned(),
228            serde_json::json!(self.index.len()),
229        );
230        stats.insert(
231            "dimensions".to_owned(),
232            serde_json::json!(self.config.dimensions),
233        );
234        stats
235    }

238impl EdgeVecActor {
239    fn handle_list_collections(&self) -> Vec<CollectionInfo> {
240        self.metadata_store
241            .iter()
242            .map(|entry| {
243                let name = entry.key().clone();
244                let vector_count = entry.value().len() as u64;
245
246                // Count unique file paths
247                let file_paths: std::collections::HashSet<&str> = entry
248                    .value()
249                    .values()
250                    .filter_map(|v| {
251                        v.as_object()
252                            .and_then(|o| o.get(VECTOR_FIELD_FILE_PATH))
253                            .and_then(|v| v.as_str())
254                    })
255                    .collect();
256                let file_count = file_paths.len() as u64;
257
258                CollectionInfo::new(name, vector_count, file_count, None, "edgevec")
259            })
260            .collect()
261    }
262
263    fn handle_list_file_paths(&self, collection: &str, limit: usize) -> Result<Vec<FileInfo>> {
264        let collection_metadata = self
265            .get_collection_metadata(collection)
266            .ok_or_else(|| Error::vector_db(format!("Collection '{collection}' not found")))?;
267
268        let mut file_map: HashMap<String, (u32, String)> = HashMap::new();
269
270        for meta_val in collection_metadata.values() {
271            if let Some(meta) = meta_val.as_object()
272                && let Some(file_path) = meta.get(VECTOR_FIELD_FILE_PATH).and_then(|v| v.as_str())
273            {
274                let language = meta
275                    .get(VECTOR_FIELD_LANGUAGE)
276                    .and_then(|v| v.as_str())
277                    .unwrap_or("unknown")
278                    .to_owned();
279
280                let entry = file_map
281                    .entry(file_path.to_owned())
282                    .or_insert((0, language));
283                entry.0 += 1;
284            }
285        }
286
287        let files = file_map
288            .into_iter()
289            .take(limit)
290            .map(|(path, (chunk_count, language))| FileInfo::new(path, chunk_count, language, None))
291            .collect();
292        Ok(files)
293    }
294
295    fn handle_get_chunks_by_file(
296        &self,
297        collection: &str,
298        file_path: &str,
299    ) -> Result<Vec<SearchResult>> {
300        let mut results = Vec::new();
301        // Normalize to forward slashes for cross-platform path matching
302        let normalized_query = file_path.replace('\\', "/");
303        if let Some(collection_metadata) = self.get_collection_metadata(collection) {
304            for (ext_id, meta_val) in collection_metadata.iter() {
305                if let Some(meta) = meta_val.as_object()
306                    && meta
307                        .get(VECTOR_FIELD_FILE_PATH)
308                        .and_then(|v| v.as_str())
309                        .is_some_and(|p| p.replace('\\', "/") == normalized_query)
310                {
311                    let mut result =
312                        search_result_from_json_metadata(ext_id.to_owned(), meta_val, 1.0);
313                    result.file_path = file_path.to_owned();
314                    results.push(result);
315                }
316            }
317        }
318        // Sort by start_line
319        results.sort_by_key(|r| r.start_line);
320        Ok(results)
321    }
322}

324impl EdgeVecActor {
325    pub async fn run(mut self) {
326        while let Some(msg) = self.receiver.recv().await {
327            match msg {
328                EdgeVecMessage::Core(core) => self.handle_core_message(core),
329                EdgeVecMessage::Query(query) => self.handle_query_message(query),
330                EdgeVecMessage::Browse(browse) => self.handle_browse_message(browse),
331            }
332        }
333    }
334
335    fn handle_core_message(&mut self, msg: CoreMessage) {
336        match msg {
337            CoreMessage::CreateCollection { name, tx } => {
338                let _ = tx.send(self.handle_create_collection(name));
339            }
340            CoreMessage::DeleteCollection { name, tx } => {
341                let _ = tx.send(self.handle_delete_collection(&name));
342            }
343            CoreMessage::InsertVectors {
344                collection,
345                vectors,
346                metadata,
347                tx,
348            } => {
349                let _ = tx.send(self.handle_insert_vectors(&collection, vectors, metadata));
350            }
351            CoreMessage::SearchSimilar {
352                collection,
353                query_vector,
354                limit,
355                tx,
356            } => {
357                let _ = tx.send(self.handle_search_similar(&collection, &query_vector, limit));
358            }
359            CoreMessage::DeleteVectors {
360                collection,
361                ids,
362                tx,
363            } => {
364                let _ = tx.send(self.handle_delete_vectors(&collection, ids));
365            }
366        }
367    }
368
369    fn handle_query_message(&mut self, msg: QueryMessage) {
370        match msg {
371            QueryMessage::GetStats { collection, tx } => {
372                let _ = tx.send(Ok(self.handle_get_stats(&collection)));
373            }
374            QueryMessage::ListVectors {
375                collection,
376                limit,
377                tx,
378            } => {
379                let _ = tx.send(Ok(self.handle_list_vectors(&collection, limit)));
380            }
381            QueryMessage::GetVectorsByIds {
382                collection,
383                ids,
384                tx,
385            } => {
386                let _ = tx.send(Ok(self.handle_get_vectors_by_ids(&collection, ids)));
387            }
388            QueryMessage::CollectionExists { name, tx } => {
389                let _ = tx.send(self.handle_collection_exists(&name));
390            }
391        }
392    }
393
394    fn handle_browse_message(&mut self, msg: BrowseMessage) {
395        match msg {
396            BrowseMessage::ListCollections { tx } => {
397                let _ = tx.send(Ok(self.handle_list_collections()));
398            }
399            BrowseMessage::ListFilePaths {
400                collection,
401                limit,
402                tx,
403            } => {
404                let _ = tx.send(self.handle_list_file_paths(&collection, limit));
405            }
406            BrowseMessage::GetChunksByFile {
407                collection,
408                file_path,
409                tx,
410            } => {
411                let _ = tx.send(self.handle_get_chunks_by_file(&collection, &file_path));
412            }
413        }
414    }
