14pub(super) fn query_row_count(query_results: &[FieldColumn]) -> usize {
15    query_results.first().map_or(0, FieldColumn::len)
16}

18/// Convert Milvus query results to domain `SearchResult` objects.
19///
20/// Extracts file path, start line, and content from Milvus field columns.
21///
22/// # Errors
23///
24/// Returns an error if required fields are missing from the query results.
25pub fn convert_query_results(
26    query_results: &[FieldColumn],
27    file_path_override: Option<&str>,
28) -> Result<Vec<SearchResult>> {
29    let mut results = Vec::new();
30    for index in 0..query_row_count(query_results) {
31        let file_path = match file_path_override {
32            Some(path) => path.to_owned(),
33            None => extract_string_field(query_results, VECTOR_FIELD_FILE_PATH, index)?,
34        };
35        let start_line = extract_long_field(query_results, VECTOR_FIELD_START_LINE, index)? as u32;
36
37        results.push(SearchResult {
38            id: value_to_id_string(
39                query_results
40                    .iter()
41                    .find(|column| column.name == VECTOR_FIELD_ID)
42                    .and_then(|column| column.get(index)),
43            ),
44            file_path,
45            start_line,
46            content: extract_string_field(query_results, VECTOR_FIELD_CONTENT, index)?,
47            score: 1.0,
48            language: "unknown".to_owned(),
49        });
50    }
51    Ok(results)
52}

54fn value_to_id_string(value: Option<milvus::value::Value<'_>>) -> String {
55    use milvus::value::Value;
56    match value {
57        Some(Value::Long(id)) => id.to_string(),
58        Some(Value::String(id)) => id.to_string(),
59        _ => "unknown".to_owned(),
60    }
61}

63fn convert_to_file_infos(query_results: &[FieldColumn], limit: usize) -> Result<Vec<FileInfo>> {
64    let mut file_counts: HashMap<String, u32> = HashMap::new();
65
66    for index in 0..query_row_count(query_results) {
67        let path = extract_string_field(query_results, VECTOR_FIELD_FILE_PATH, index)?;
68        *file_counts.entry(path).or_insert(0) += 1;
69    }
70
71    Ok(file_counts
72        .into_iter()
73        .take(limit)
74        .map(|(path, chunk_count)| FileInfo::new(path, chunk_count, "unknown", None))
75        .collect())
76}

82    async fn list_collections(&self) -> Result<Vec<CollectionInfo>> {
83        let collection_names =
84            Self::map_milvus_error(self.client.list_collections().await, "list collections")?;
85
86        let mut collections = Vec::new();
87
88        for name in collection_names {
89            let _collection_id = CollectionId::from_name(&name);
90            // Get stats for each collection
91            let stats = self.client.get_collection_stats(&name).await.map_err(|e| {
92                Error::vector_db(format!("Failed to get stats for collection '{name}': {e}"))
93            })?;
94            let vector_count = stats
95                .get("row_count")
96                .and_then(|value: &String| value.parse::<u64>().ok())
97                .ok_or_else(|| {
98                    Error::vector_db(format!(
99                        "Milvus collection '{name}' stats missing 'row_count'"
100                    ))
101                })?;
102
103            // For now, we don't have a quick way to count unique files without querying all data
104            // In a future optimization, we could cache this or use Milvus aggregation
105            collections.push(CollectionInfo::new(
106                name,
107                vector_count,
108                0, // file_count will be populated when listing files
109                None,
110                self.provider_name(),
111            ));
112        }
113
114        Ok(collections)
115    }

117    async fn list_file_paths(
118        &self,
119        collection: &CollectionId,
120        limit: usize,
121    ) -> Result<Vec<FileInfo>> {
122        if limit == 0 {
123            return Ok(Vec::new());
124        }
125        let name_str = to_milvus_name(collection);
126
127        // Ensure collection is loaded
128        if let Err(e) = self.client.load_collection(&name_str, None).await {
129            let err_str = e.to_string();
130            if is_collection_not_found(&err_str) {
131                return Err(Error::vector_db(format!(
132                    "Collection '{collection}' not found when listing file paths"
133                )));
134            }
135            return Err(Error::vector_db(format!(
136                "Failed to load collection '{collection}': {e}"
137            )));
138        }
139
140        use milvus::query::QueryOptions;
141
142        let expr = "id >= 0".to_owned();
143        let query_options = QueryOptions::new()
144            .limit(mcb_utils::constants::vector_store::MILVUS_DEFAULT_QUERY_LIMIT)
145            .output_fields(vec![VECTOR_FIELD_FILE_PATH.to_owned()]);
146
147        let query_results = match self.client.query(&name_str, &expr, &query_options).await {
148            Ok(results) => results,
149            Err(e) => {
150                let msg = format!("Failed to query file paths in collection '{collection}': {e}");
151                mcb_domain::warn!("milvus", "query file paths failed", &msg);
152                return Err(Error::vector_db(msg));
153            }
154        };
155
156        convert_to_file_infos(&query_results, limit)
157    }

159    async fn get_chunks_by_file(
160        &self,
161        collection: &CollectionId,
162        file_path: &str,
163    ) -> Result<Vec<SearchResult>> {
164        let name_str = to_milvus_name(collection);
165        // Ensure collection is loaded
166        if let Err(e) = self.client.load_collection(&name_str, None).await {
167            let err_str = e.to_string();
168            if is_collection_not_found(&err_str) {
169                return Err(Error::vector_db(format!(
170                    "Collection '{collection}' not found when querying chunks by file"
171                )));
172            }
173            return Err(Error::vector_db(format!(
174                "Failed to load collection '{collection}': {e}"
175            )));
176        }
177
178        use milvus::query::QueryOptions;
179
180        // Query with filter on file_path
181        let expr = format!("file_path == \"{}\"", file_path.replace('"', "\\\""));
182        let query_options = QueryOptions::new()
183            .limit(
184                mcb_utils::constants::BROWSE_MAX_CHUNKS_PER_FILE
185                    .try_into()
186                    .unwrap_or(1000),
187            ) // Reasonable limit for chunks per file
188            .output_fields(Self::default_output_fields());
189
190        let query_results = match self.client.query(&name_str, &expr, &query_options).await {
191            Ok(results) => results,
192            Err(e) => {
193                let msg =
194                    format!("Failed to query chunks by file in collection '{collection}': {e}");
195                mcb_domain::warn!("milvus", "query chunks by file failed", &msg);
196                return Err(Error::vector_db(msg));
197            }
198        };
199
200        let mut results = convert_query_results(&query_results, Some(file_path))?;
201        results.sort_by_key(|r| r.start_line);
202
203        Ok(results)
204    }
