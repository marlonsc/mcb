1//! Qdrant provider factory and auto-registration.
2
3use super::QdrantVectorStoreProvider;
4use mcb_utils::constants::vector_store::QDRANT_DEFAULT_PORT;
5
6mcb_domain::register_vector_store_provider!(
7    mcb_utils::constants::PROVIDER_SLUG_QDRANT,
8    "Qdrant vector search engine (open-source, cloud and self-hosted)",
9    qdrant_factory
10);
11
12fn qdrant_factory(
13    config: &mcb_domain::registry::vector_store::VectorStoreProviderConfig,
14) -> mcb_domain::error::Result<std::sync::Arc<dyn mcb_domain::ports::VectorStoreProvider>> {
15    use crate::utils::http::{DEFAULT_HTTP_TIMEOUT, create_default_client};
16
17    let base_url = config
18        .uri
19        .clone()
20        .unwrap_or_else(|| format!("http://localhost:{QDRANT_DEFAULT_PORT}"));
21    let api_key = config.api_key.clone();
22    let http_client = create_default_client()?;
23
24    Ok(std::sync::Arc::new(QdrantVectorStoreProvider::new(
25        &base_url,
26        api_key,
27        DEFAULT_HTTP_TIMEOUT,
28        http_client,
29    )))
30}
