21/// Pinecone vector store provider
22///
23/// Implements the vector store domain ports using Pinecone's cloud REST API.
24/// Supports index management, vector upsert, search, and metadata filtering.
25pub struct PineconeVectorStoreProvider {
26    pub(super) api_key: String,
27    pub(super) host: String,
28    pub(super) timeout: Duration,
29    pub(super) http_client: Client,
30    /// Track collections (namespaces) locally with their dimensions
31    pub(super) collections: Arc<DashMap<String, usize>>,
32    /// Default dimensions sourced from provider config (embedding model).
33    /// Used when a collection's dimensions are unknown locally.
34    pub(super) default_dimensions: Option<usize>,
35}
36
37impl PineconeVectorStoreProvider {
38    /// Create a new Pinecone vector store provider
39    ///
40    /// # Arguments
41    /// * `api_key` - Pinecone API key
42    /// * `host` - Pinecone index host URL
43    /// * `timeout` - Request timeout duration
44    /// * `http_client` - Reqwest HTTP client for making API requests
45    /// * `default_dimensions` - Default vector dimensions from config (embedding model)
46    #[must_use]
47    pub fn new(
48        api_key: &str,
49        host: &str,
50        timeout: Duration,
51        http_client: Client,
52        default_dimensions: Option<usize>,
53    ) -> Self {
54        Self {
55            api_key: api_key.trim().to_owned(),
56            host: host.trim_end_matches('/').to_owned(),
57            timeout,
58            http_client,
59            collections: Arc::new(DashMap::new()),
60            default_dimensions,
61        }
62    }
63
64    /// Build a URL for the Pinecone API
65    pub(crate) fn api_url(&self, path: &str) -> String {
66        format!("{}{}", self.host, path)
67    }
68
69    /// Make an authenticated request to Pinecone
70    pub(crate) async fn request(
71        &self,
72        method: reqwest::Method,
73        path: &str,
74        body: Option<Value>,
75    ) -> Result<Value> {
76        let headers = vec![
77            (PINECONE_API_KEY_HEADER, self.api_key.clone()),
78            (HTTP_HEADER_CONTENT_TYPE, CONTENT_TYPE_JSON.to_owned()),
79        ];
80
81        send_vector_db_request(VectorDbRequestParams {
82            client: &self.http_client,
83            method,
84            url: self.api_url(path),
85            timeout: self.timeout,
86            provider: "Pinecone",
87            operation: path,
88            headers: &headers,
89            body: body.as_ref(),
90            retry_attempts: PROVIDER_RETRY_COUNT,
91            retry_backoff_ms: PROVIDER_RETRY_BACKOFF_MS,
92        })
93        .await
94    }
95
96    pub(crate) fn extract_json_field<'a, T, F>(
97        obj: &'a Value,
98        field: &str,
99        parser: F,
100        context: &str,
101        expected: &str,
102    ) -> Result<T>
103    where
104        F: FnOnce(&'a Value) -> Option<T>,
105    {
106        obj.get(field).and_then(parser).ok_or_else(|| {
107            Error::vector_db(format!(
108                "Invalid Pinecone {context}: missing or non-{expected} '{field}' field"
109            ))
110        })
111    }
112
113    pub(crate) async fn query_match_results(&self, payload: Value) -> Result<Vec<SearchResult>> {
114        let response = self
115            .request(reqwest::Method::POST, "/query", Some(payload))
116            .await?;
117        let matches =
118            Self::extract_json_field(&response, "matches", Value::as_array, "response", "array")?;
119        matches
120            .iter()
121            .map(|m| {
122                let score =
123                    Self::extract_json_field(m, "score", Value::as_f64, "match", "numeric")?;
124                Self::match_to_search_result(m, score)
125            })
126            .collect()
127    }
128
129    pub(crate) fn collection_dimensions(&self, collection: &str) -> Result<usize> {
130        if let Some(d) = self.collections.get(collection) {
131            return Ok(*d.value());
132        }
133        self.default_dimensions.ok_or_else(|| {
134            Error::vector_db(format!(
135                "Unknown dimensions for collection '{collection}': \
136                 set `dimensions` in vector store config or call create_collection() first"
137            ))
138        })
139    }
140
141    /// Convert Pinecone match result to domain `SearchResult`.
142    ///
143    /// # Errors
144    ///
145    /// Returns an error if required fields are missing from the match result.
146    pub fn match_to_search_result(item: &Value, score: f64) -> Result<SearchResult> {
147        let id = Self::extract_json_field(item, "id", Value::as_str, "match", "string")?.to_owned();
148        let metadata = Self::extract_json_field(item, "metadata", Some, "match", "value")?;
149        Ok(search_result_from_json_metadata(id, metadata, score))
150    }
151}
