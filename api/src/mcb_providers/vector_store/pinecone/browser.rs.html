1//! Browser operations for Pinecone vector store provider.
2
3use async_trait::async_trait;
4use mcb_domain::error::Result;
5use mcb_domain::ports::{VectorStoreAdmin, VectorStoreBrowser, VectorStoreProvider};
6use mcb_domain::value_objects::{CollectionId, CollectionInfo, FileInfo, SearchResult};
7
8use mcb_utils::constants::vector_store::VECTOR_FIELD_FILE_PATH;
9
10use super::PineconeVectorStoreProvider;
11
12#[async_trait]
13impl VectorStoreBrowser for PineconeVectorStoreProvider {
14    // --- Browser Methods ---
15
16    async fn list_collections(&self) -> Result<Vec<CollectionInfo>> {
17        // Query Pinecone to get the authoritative list of namespaces.
18        // `/describe_index_stats` returns all namespaces in the index.
19        let response = self
20            .request(
21                reqwest::Method::POST,
22                "/describe_index_stats",
23                Some(serde_json::json!({ "filter": {} })),
24            )
25            .await?;
26
27        let collections = response
28            .get("namespaces")
29            .and_then(serde_json::Value::as_object)
30            .map(|namespaces| {
31                namespaces
32                    .keys()
33                    .map(|name| {
34                        let vector_count = namespaces
35                            .get(name)
36                            .and_then(|ns| ns.get("vectorCount"))
37                            .and_then(serde_json::Value::as_u64)
38                            .unwrap_or(0);
39                        CollectionInfo::new(name, vector_count, 0, None, self.provider_name())
40                    })
41                    .collect()
42            })
43            .unwrap_or_default();
44
45        Ok(collections)
46    }
47
48    async fn list_file_paths(
49        &self,
50        collection: &CollectionId,
51        limit: usize,
52    ) -> Result<Vec<FileInfo>> {
53        let results = self.list_vectors(collection, limit).await?;
54        Ok(crate::utils::vector_store::build_file_info_from_results(
55            results,
56        ))
57    }
58
59    async fn get_chunks_by_file(
60        &self,
61        collection: &CollectionId,
62        file_path: &str,
63    ) -> Result<Vec<SearchResult>> {
64        let filter = serde_json::json!({
65            (VECTOR_FIELD_FILE_PATH): { "$eq": file_path }
66        });
67
68        let collection_str = collection.to_string();
69        let dimensions = self.collection_dimensions(&collection_str)?;
70
71        let zero_vector = vec![0.0f32; dimensions];
72
73        let mut results = self
74            .query_match_results(serde_json::json!({
75                "vector": zero_vector,
76                "topK": mcb_utils::constants::BROWSE_MAX_CHUNKS_PER_FILE,
77                "namespace": collection_str,
78                "includeMetadata": true,
79                "filter": filter
80            }))
81            .await?;
82
83        results.sort_by_key(|r| r.start_line);
84        Ok(results)
85    }
86}
