27/// Classification used to map HTTP request failures to domain errors.
28pub(crate) enum RequestErrorKind {
29    /// Embedding provider request.
30    Embedding,
31    /// Vector database provider request.
32    VectorDb,
33}

35/// Create an HTTP client with the specified timeout
36///
37/// # Arguments
38/// * `timeout_secs` - Request timeout in seconds
39///
40/// # Returns
41/// Configured reqwest Client or error message string
42///
43/// # Example
44/// ```ignore
45/// use mcb_providers::utils::http::create_client;
46///
47/// let client = create_client(30)?;
48/// ```
49pub(crate) fn create_client(timeout_secs: u64) -> std::result::Result<Client, String> {
50    Client::builder()
51        .timeout(Duration::from_secs(timeout_secs))
52        .build()
53        .map_err(|e| format!("Failed to create HTTP client: {e}"))
54}
55
56/// Create an HTTP client with the default 30-second timeout
57pub(crate) fn create_default_client() -> std::result::Result<Client, String> {
58    create_client(30)
59}

61pub(crate) fn handle_request_error_with_kind(
62    error: &reqwest::Error,
63    timeout: Duration,
64    provider: &str,
65    operation: &str,
66    kind: RequestErrorKind,
67) -> Error {
68    match kind {
69        RequestErrorKind::Embedding => {
70            if error.is_timeout() {
71                Error::embedding(format!("{ERROR_MSG_REQUEST_TIMEOUT} {timeout:?}"))
72            } else {
73                Error::embedding(format!("HTTP request to {provider} failed: {error}"))
74            }
75        }
76        RequestErrorKind::VectorDb => {
77            if error.is_timeout() {
78                Error::vector_db(format!(
79                    "{provider} {operation} request timed out after {timeout:?}"
80                ))
81            } else {
82                Error::vector_db(format!(
83                    "{provider} HTTP request for {operation} failed: {error}"
84                ))
85            }
86        }
87    }
88}

90/// Parse embedding vector from JSON response item
91///
92/// Extracts embedding vector from a JSON value, handling errors appropriately.
93///
94/// # Arguments
95/// * `item` - JSON value containing the embedding
96/// * `field` - Field name to extract (e.g., "embedding", "values")
97/// * `index` - Text index for error messages
98///
99/// # Returns
100/// Parsed embedding vector or domain error
101pub(crate) fn parse_embedding_vector(
102    item: &serde_json::Value,
103    field: &str,
104    index: usize,
105) -> mcb_domain::error::Result<Vec<f32>> {
106    item[field]
107        .as_array()
108        .ok_or_else(|| Error::embedding(format!("Invalid embedding format for text {index}")))?
109        .iter()
110        .map(|v| {
111            v.as_f64().map(|n| n as f32).ok_or_else(|| {
112                Error::embedding(format!("Invalid number in embedding for text {index}"))
113            })
114        })
115        .collect()
116}

128    /// Request timeout
129    pub timeout: Duration,
130    /// Configured HTTP client
131    pub client: Client,

134/// Create configuration for an HTTP-based embedding provider
135///
136/// This helper extracts common configuration from `EmbeddingProviderConfig`,
137/// applies provider-specific defaults, and creates the HTTP client.
138///
139/// # Arguments
140/// * `config` - The provider configuration from registry
141/// * `provider_name` - Name of the provider (for error messages)
142/// * `default_model` - Default model to use if none specified
143///
144/// # Returns
145/// Configured `HttpProviderConfig` or error message
146///
147/// # Example
148/// ```ignore
149/// let cfg = create_http_provider_config(config, "openai", "text-embedding-3-small")?;
150/// Ok(Arc::new(OpenAIEmbeddingProvider::new(
151///     cfg.api_key,
152///     cfg.base_url,
153///     cfg.model,
154///     cfg.timeout,
155///     cfg.client,
156/// )))
157/// ```
158pub(crate) fn create_http_provider_config(
159    config: &mcb_domain::registry::embedding::EmbeddingProviderConfig,
160    provider_name: &str,
161    default_model: &str,
162) -> std::result::Result<HttpProviderConfig, String> {
163    let api_key = config
164        .api_key
165        .clone()
166        .ok_or_else(|| format!("{provider_name} requires api_key"))?;
167
168    let base_url = config.base_url.clone();
169
170    let model = config
171        .model
172        .clone()
173        .unwrap_or_else(|| default_model.to_owned());
174
175    let client = create_default_client()?;
176
177    Ok(HttpProviderConfig {
178        api_key,
179        base_url,
180        model,
181        timeout: DEFAULT_HTTP_TIMEOUT,
182        client,
183    })
184}

186/// Parameters for [`send_json_request`].
187pub(crate) struct JsonRequestParams<'a> {
188    /// HTTP client to use.
189    pub client: &'a Client,
190    /// HTTP method (GET, POST, etc.).
191    pub method: reqwest::Method,
192    /// Target URL.
193    pub url: String,
194    /// Request timeout.
195    pub timeout: Duration,
196    /// Provider name for error messages.
197    pub provider: &'a str,
198    /// Operation name for error messages.
199    pub operation: &'a str,
200    /// Error classification kind.
201    pub kind: RequestErrorKind,
202    /// Additional headers.
203    pub headers: &'a [(&'a str, String)],
204    /// Optional JSON body.
205    pub body: Option<&'a Value>,
206    /// Optional retry configuration for transient errors (rate limits, 5xx, timeouts).
207    pub retry: Option<RetryConfig>,
208}

210/// Check whether a domain error represents a transient HTTP failure worth retrying.
211fn is_retryable_error(error: &Error) -> bool {
212    let msg = error.to_string();
213    msg.contains("rate limit exceeded")
214        || msg.contains("server error (5")
215        || msg.contains("timed out")
216        || msg.contains("timeout")
217}

219/// Send a JSON request with configurable parameters and optional retry.
220pub(crate) async fn send_json_request(
221    params: JsonRequestParams<'_>,
222) -> mcb_domain::error::Result<Value> {
223    let JsonRequestParams {
224        client,
225        method,
226        url,
227        timeout,
228        provider,
229        operation,
230        kind,
231        headers,
232        body,
233        retry,
234    } = params;
235
236    let execute = || async {
237        let mut builder = client.request(method.clone(), &url).timeout(timeout);
238
239        for (key, value) in headers {
240            builder = builder.header(*key, value);
241        }
242
243        if let Some(payload) = body {
244            builder = builder.json(payload);
245        }
246
247        let response = builder
248            .send()
249            .await
250            .map_err(|e| handle_request_error_with_kind(&e, timeout, provider, operation, kind))?;
251
252        HttpResponseUtils::check_and_parse(response, provider).await
253    };
254
255    match retry {
256        None => execute().await,
257        Some(config) => retry_with_backoff(config, |_| execute(), is_retryable_error).await,
258    }
259}

274pub(crate) async fn send_vector_db_request(

278        client,
279        method,
280        url,
281        timeout,
282        provider,
283        operation,
284        headers,
285        body,
286        retry_attempts,
287        retry_backoff_secs,

289
290    send_json_request(JsonRequestParams {
291        client,
292        method,
293        url,
294        timeout,
295        provider,
296        operation,
297        kind: RequestErrorKind::VectorDb,
298        headers,
299        body,
300        retry: Some(RetryConfig::new(
301            retry_attempts,
302            Duration::from_secs(retry_backoff_secs),
303        )),
304    })
305    .await
