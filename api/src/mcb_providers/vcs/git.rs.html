30    fn open_repo(path: &Path) -> Result<Repository> {
31        Repository::open(path).map_err(|e| {
32            if e.code() == git2::ErrorCode::NotFound {
33                Error::repository_not_found(path.display().to_string())
34            } else {
35                Error::vcs_with_source(format!("Failed to open repository: {}", path.display()), e)
36            }
37        })
38    }
39
40    fn get_root_commit_hash(repo: &Repository) -> Result<String> {
41        let mut revwalk = repo
42            .revwalk()
43            .map_err(|e| Error::vcs_with_source("Failed to create revwalk", e))?;
44
45        revwalk
46            .push_head()
47            .map_err(|e| Error::vcs_with_source("Failed to push HEAD to revwalk", e))?;
48
49        // INTENTIONAL: Sorting preference is non-critical; default ordering is acceptable
50        revwalk.set_sorting(Sort::TIME | Sort::REVERSE).ok();
51
52        let first_oid = revwalk
53            .next()
54            .ok_or_else(|| Error::vcs("Repository has no commits"))?
55            .map_err(|e| Error::vcs_with_source("Failed to get first commit", e))?;
56
57        Ok(first_oid.to_string())
58    }
59
60    fn get_default_branch(repo: &Repository) -> Result<String> {
61        repo.head()
62            // INTENTIONAL: Best-effort default branch detection; falls back to None
63            .ok()
64            .and_then(|head| head.shorthand().map(String::from))
65            .ok_or_else(|| {
66                Error::vcs(
67                    "Cannot determine default branch: repository has no HEAD (possibly empty/uninitialized)",
68                )
69            })
70    }
71
72    fn get_remote_url(repo: &Repository) -> Option<String> {
73        repo.find_remote("origin")
74            // INTENTIONAL: Best-effort remote URL detection; falls back to None
75            .ok()
76            .and_then(|remote| remote.url().map(String::from))
77    }
78
79    fn list_branch_names(repo: &Repository) -> Result<Vec<String>> {
80        let branches = repo
81            .branches(Some(BranchType::Local))
82            .map_err(|e| Error::vcs_with_source("Failed to list branches", e))?;
83
84        let mut names = Vec::new();
85        for branch_result in branches {
86            let (branch, _) =
87                branch_result.map_err(|e| Error::vcs_with_source("Failed to read branch", e))?;
88            // INTENTIONAL: Optional branch name extraction; None means detached HEAD
89            if let Some(name) = branch.name().ok().flatten() {
90                names.push(name.to_owned());
91            }
92        }
93
94        Ok(names)
95    }
96
97    /// Resolve a local branch reference, returning a useful error if not found.
98    fn find_branch_ref<'r>(repo: &'r Repository, branch: &str) -> Result<git2::Reference<'r>> {
99        let branch_ref = repo.find_branch(branch, BranchType::Local).map_err(|e| {
100            if e.code() == git2::ErrorCode::NotFound {
101                Error::branch_not_found(branch)
102            } else {
103                Error::vcs_with_source(format!("Failed to find branch: {branch}"), e)
104            }
105        })?;
106        Ok(branch_ref.into_reference())
107    }
108
109    /// Convert a git2 delta status to our domain `DiffStatus`.
110    fn delta_to_status(delta: git2::Delta) -> DiffStatus {
111        #[allow(clippy::wildcard_enum_match_arm)]
112        match delta {
113            git2::Delta::Added => DiffStatus::Added,
114            git2::Delta::Deleted => DiffStatus::Deleted,
115            git2::Delta::Renamed => DiffStatus::Renamed,
116            _ => DiffStatus::Modified,
117        }
118    }
119
120    /// Convert a git2 commit to a `VcsCommit` domain entity.
121    fn commit_to_vcs(repo_id: &RepositoryId, commit: &git2::Commit<'_>) -> VcsCommit {
122        let oid = commit.id();
123        let author = commit.author();
124        let parent_hashes: Vec<String> = commit.parent_ids().map(|id| id.to_string()).collect();
125
126        VcsCommit::new(VcsCommitInput {
127            id: format!("{repo_id}:{oid}"),
128            hash: oid.to_string(),
129            message: commit.message().unwrap_or("").to_owned(),
130            author: author.name().unwrap_or("Unknown").to_owned(),
131            author_email: author.email().unwrap_or("").to_owned(),
132            timestamp: commit.time().seconds(),
133            parent_hashes,
134        })
135    }

137    /// Resolve two git refs into their tree objects for diffing.
138    fn resolve_trees<'r>(
139        repo: &'r Repository,
140        base_ref: &str,
141        head_ref: &str,
142    ) -> Result<(git2::Tree<'r>, git2::Tree<'r>)> {
143        let base_obj = repo
144            .revparse_single(base_ref)
145            .map_err(|e| Error::vcs_with_source(format!("Failed to resolve ref: {base_ref}"), e))?;
146        let head_obj = repo
147            .revparse_single(head_ref)
148            .map_err(|e| Error::vcs_with_source(format!("Failed to resolve ref: {head_ref}"), e))?;
149        let base_tree = base_obj
150            .peel_to_tree()
151            .map_err(|e| Error::vcs_with_source("Failed to get base tree", e))?;
152        let head_tree = head_obj
153            .peel_to_tree()
154            .map_err(|e| Error::vcs_with_source("Failed to get head tree", e))?;
155        Ok((base_tree, head_tree))
156    }
157
158    /// Collect file diffs and line counts from a git2 diff.
159    fn collect_diff_files(diff: &git2::Diff<'_>) -> Result<(Vec<FileDiff>, usize, usize)> {
160        let mut files = Vec::new();
161        let mut total_additions = 0;
162        let mut total_deletions = 0;
163
164        diff.foreach(
165            &mut |delta, _| {
166                let path = delta
167                    .new_file()
168                    .path()
169                    .or_else(|| delta.old_file().path())
170                    .map(PathBuf::from)
171                    .unwrap_or_default();
172                files.push(FileDiff {
173                    id: id::generate().to_string(),
174                    path,
175                    status: Self::delta_to_status(delta.status()),
176                    additions: 0,
177                    deletions: 0,
178                });
179                true
180            },
181            None,
182            None,
183            Some(&mut |_delta, _hunk, line| {
184                if line.origin() == '+' {
185                    total_additions += 1;
186                } else if line.origin() == '-' {
187                    total_deletions += 1;
188                }
189                true
190            }),
191        )
192        .map_err(|e| Error::vcs_with_source("Failed to iterate diff", e))?;
193
194        Ok((files, total_additions, total_deletions))
195    }
196}

204#[async_trait]
205impl VcsProvider for GitProvider {
206    async fn open_repository(&self, path: &Path) -> Result<VcsRepository> {
207        let repo = Self::open_repo(path)?;
208
209        let root_hash = Self::get_root_commit_hash(&repo)?;
210        let id = RepositoryId::from_name(&root_hash);
211        let default_branch = Self::get_default_branch(&repo)?;
212        let branches = Self::list_branch_names(&repo)?;
213        let remote_url = Self::get_remote_url(&repo);
214
215        Ok(VcsRepository::new(
216            id,
217            path.to_path_buf(),
218            default_branch,
219            branches,
220            remote_url,
221        ))
222    }
223
224    fn repository_id(&self, repo: &VcsRepository) -> RepositoryId {
225        *repo.id()
226    }
227
228    async fn list_branches(&self, repo: &VcsRepository) -> Result<Vec<VcsBranch>> {
229        let git_repo = Self::open_repo(repo.path())?;
230
231        let branches = git_repo
232            .branches(Some(BranchType::Local))
233            .map_err(|e| Error::vcs_with_source("Failed to list branches", e))?;
234
235        let mut result = Vec::new();
236        for branch_result in branches {
237            let (branch, _) =
238                branch_result.map_err(|e| Error::vcs_with_source("Failed to read branch", e))?;
239
240            let name = match branch.name() {
241                Ok(Some(n)) => n.to_owned(),
242                Ok(None) => continue,
243                Err(_non_utf8) => continue,
244            };
245
246            let head_commit = match branch.get().peel_to_commit() {
247                Ok(c) => c.id().to_string(),
248                Err(_unsupported) => continue,
249            };
250
251            let upstream = branch
252                .upstream()
253                .ok()
254                .and_then(|u| u.name().ok().flatten().map(String::from));
255
256            let is_default = name == repo.default_branch();
257            result.push(VcsBranch::new(
258                format!("{}::{}", repo.id(), name),
259                name,
260                head_commit,
261                is_default,
262                upstream,
263            ));
264        }
265
266        Ok(result)
267    }

269    async fn commit_history(
270        &self,
271        repo: &VcsRepository,
272        branch: &str,
273        limit: Option<usize>,
274    ) -> Result<Vec<VcsCommit>> {
275        let git_repo = Self::open_repo(repo.path())?;
276
277        let branch_ref = Self::find_branch_ref(&git_repo, branch)?;
278        let branch_commit = branch_ref
279            .peel_to_commit()
280            .map_err(|e| Error::vcs_with_source("Failed to get branch commit", e))?;
281
282        let mut revwalk = git_repo
283            .revwalk()
284            .map_err(|e| Error::vcs_with_source("Failed to create revwalk", e))?;
285        revwalk
286            .push(branch_commit.id())
287            .map_err(|e| Error::vcs_with_source("Failed to push commit to revwalk", e))?;
288        // INTENTIONAL: Sorting preference is non-critical; default ordering is acceptable
289        revwalk.set_sorting(Sort::TIME).ok();
290
291        let max_commits = limit.unwrap_or(usize::MAX);
292        let mut commits = Vec::new();
293
294        for oid_result in revwalk {
295            if commits.len() >= max_commits {
296                break;
297            }
298            let oid =
299                oid_result.map_err(|e| Error::vcs_with_source("Failed to iterate commits", e))?;
300            let commit = git_repo
301                .find_commit(oid)
302                .map_err(|e| Error::vcs_with_source("Failed to find commit", e))?;
303
304            commits.push(Self::commit_to_vcs(repo.id(), &commit));
305        }
306
307        Ok(commits)
308    }
309
310    async fn list_files(&self, repo: &VcsRepository, branch: &str) -> Result<Vec<PathBuf>> {
311        let git_repo = Self::open_repo(repo.path())?;
312
313        let branch_ref = Self::find_branch_ref(&git_repo, branch)?;
314
315        let tree = branch_ref
316            .peel_to_tree()
317            .map_err(|e| Error::vcs_with_source("Failed to get branch tree", e))?;
318
319        let mut files = Vec::new();
320        tree.walk(git2::TreeWalkMode::PreOrder, |dir, entry| {
321            if entry.kind() == Some(git2::ObjectType::Blob)
322                && let Some(name) = entry.name()
323            {
324                let path = if dir.is_empty() {
325                    PathBuf::from(name)
326                } else {
327                    PathBuf::from(dir).join(name)
328                };
329                files.push(path);
330            }
331            git2::TreeWalkResult::Ok
332        })
333        .map_err(|e| Error::vcs_with_source("Failed to walk tree", e))?;
334
335        Ok(files)
336    }

338    async fn read_file(&self, repo: &VcsRepository, branch: &str, path: &Path) -> Result<String> {
339        let git_repo = Self::open_repo(repo.path())?;
340
341        let branch_ref = Self::find_branch_ref(&git_repo, branch)?;
342
343        let tree = branch_ref
344            .peel_to_tree()
345            .map_err(|e| Error::vcs_with_source("Failed to get branch tree", e))?;
346
347        let path_str = mcb_utils::utils::path::path_to_utf8_string(path)
348            .map_err(|e| Error::vcs_with_source("non-UTF-8 path", e))?;
349        let entry = tree.get_path(path).map_err(|e| {
350            Error::vcs_with_source(format!("File not found in branch: {path_str}"), e)
351        })?;
352
353        let blob = entry
354            .to_object(&git_repo)
355            .and_then(|obj| obj.peel_to_blob());
356
357        match blob {
358            Ok(blob) => {
359                if blob.is_binary() {
360                    return Err(Error::vcs(format!("Binary file: {path_str}")));
361                }
362                String::from_utf8(blob.content().to_vec())
363                    .map_err(|e| Error::vcs_with_source(format!("Invalid UTF-8: {path_str}"), e))
364            }
365            Err(e) => Err(Error::vcs_with_source(
366                format!("Failed to read file: {path_str}"),
367                e,
368            )),
369        }
370    }

372    fn vcs_name(&self) -> &str {
373        mcb_utils::constants::DEFAULT_VCS_PROVIDER
374    }

376    async fn diff_refs(
377        &self,
378        repo: &VcsRepository,
379        base_ref: &str,
380        head_ref: &str,
381    ) -> Result<RefDiff> {
382        let git_repo = Self::open_repo(repo.path())?;
383        let (base_tree, head_tree) = Self::resolve_trees(&git_repo, base_ref, head_ref)?;
384
385        let diff = git_repo
386            .diff_tree_to_tree(Some(&base_tree), Some(&head_tree), None)
387            .map_err(|e| Error::vcs_with_source("Failed to create diff", e))?;
388
389        let (files, total_additions, total_deletions) = Self::collect_diff_files(&diff)?;
390
391        Ok(RefDiff {
392            id: id::generate().to_string(),
393            base_ref: base_ref.to_owned(),
394            head_ref: head_ref.to_owned(),
395            files,
396            total_additions,
397            total_deletions,
398        })
399    }

401    async fn list_repositories(&self, root: &Path) -> Result<Vec<VcsRepository>> {
402        use walkdir::WalkDir;
403
404        let mut repositories = Vec::new();
405
406        // Check if root itself is a repository
407        let root_is_repo = root.join(".git").exists();
408        if root_is_repo && let Ok(repo) = self.open_repository(root).await {
409            repositories.push(repo);
410        }
411
412        // Recursively scan subdirectories for repositories (limit depth to 3)
413        // Use walkdir instead of fs::read_dir for recursion
414        let walker = WalkDir::new(root)
415            .min_depth(1)
416            .max_depth(3)
417            .follow_links(false)
418            .into_iter()
419            .filter_map(std::result::Result::ok)
420            .filter(|e| e.file_type().is_dir());
421
422        for entry in walker {
423            let path = entry.path();
424            // Skip .git directories themselves
425            if path.file_name().and_then(|n| n.to_str()) == Some(".git") {
426                continue;
427            }
428
429            // Check for .git child directory
430            if path.join(".git").exists()
431                && let Ok(repo) = self.open_repository(path).await
432            {
433                repositories.push(repo);
434            }
435        }
436
437        Ok(repositories)
438    }
