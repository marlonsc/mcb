16/// Creates the complete SeaORM-backed repository bundle for the database registry.
17///
18/// # Errors
19///
20/// Returns an error when the boxed connection is not a `sea_orm::DatabaseConnection`.
21fn create_seaorm_repositories(
22    connection: Arc<dyn Any + Send + Sync>,
23    project_id: String,
24) -> mcb_domain::error::Result<DatabaseRepositories> {
25    let db = connection.downcast::<DatabaseConnection>().map_err(|_| {
26        mcb_domain::error::Error::configuration(
27            "Expected sea_orm::DatabaseConnection but received different type",
28        )
29    })?;
30    let db = db;
31
32    let observation_repo = SeaOrmObservationRepository::new((*db).clone());
33    let agent_repo = SeaOrmAgentRepository::new(Arc::clone(&db));
34    let project_repo = SeaOrmProjectRepository::new((*db).clone());
35    let entity_repo = Arc::new(SeaOrmEntityRepository::new(Arc::clone(&db)));
36    let index_repo = SeaOrmIndexRepository::new(Arc::clone(&db), project_id);
37
38    Ok(DatabaseRepositories {
39        memory: Arc::new(observation_repo),
40        auth: Arc::new(SeaOrmAuthRepositoryAdapter::new((*db).clone())),
41        dashboard: Arc::new(SeaOrmDashboardAdapter::new((*db).clone())),
42        agent: Arc::new(agent_repo),
43        project: Arc::new(project_repo),
44        vcs_entity: Arc::clone(&entity_repo) as _,
45        plan_entity: Arc::clone(&entity_repo) as _,
46        issue_entity: Arc::clone(&entity_repo) as _,
47        org_entity: Arc::clone(&entity_repo) as _,
48        file_hash: Arc::new(index_repo),
49    })
50}
51
52mcb_domain::register_database_repository!(
53    mcb_utils::constants::DEFAULT_DATABASE_PROVIDER,
54    "SeaORM database repositories (SQLite, PostgreSQL, MySQL)",
55    create_seaorm_repositories
56        as fn(
57            Arc<dyn Any + Send + Sync>,
58            String,
59        ) -> mcb_domain::error::Result<DatabaseRepositories>,
