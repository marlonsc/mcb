109impl SeaOrmProjectRepository {
110    /// Verifies a project exists in the given org.
111    async fn verify_project_exists(&self, org_id: &str, project_id: &str) -> Result<()> {
112        let exists = project::Entity::find()
113            .filter(project::Column::Id.eq(project_id))
114            .filter(project::Column::OrgId.eq(org_id))
115            .count(&self.db)
116            .await
117            .map_err(db_error("verify project ownership"))?
118            > 0;
119        if !exists {
120            return Err(Error::not_found(format!(
121                "Project with id {project_id} not found in org"
122            )));
123        }
124        Ok(())
125    }
126
127    /// Applies `IssueFilter` fields to a `SeaORM` `Select` query.
128    fn apply_issue_filter(
129        mut q: sea_orm::Select<project_issue::Entity>,
130        filter: &IssueFilter,
131    ) -> sea_orm::Select<project_issue::Entity> {
132        if let Some(project_id) = &filter.project_id {
133            q = q.filter(project_issue::Column::ProjectId.eq(project_id.to_owned()));
134        }
135        if let Some(phase_id) = &filter.phase_id {
136            q = q.filter(project_issue::Column::PhaseId.eq(phase_id.to_owned()));
137        }
138        if let Some(issue_type) = &filter.issue_type {
139            q = q.filter(project_issue::Column::IssueType.eq(issue_type.to_string()));
140        }
141        if let Some(status) = &filter.status {
142            q = q.filter(project_issue::Column::Status.eq(status.to_string()));
143        }
144        if let Some(priority) = filter.priority {
145            q = q.filter(project_issue::Column::Priority.eq(i64::from(priority)));
146        }
147        if let Some(assignee) = &filter.assignee {
148            q = q.filter(project_issue::Column::Assignee.eq(assignee.to_owned()));
149        }
150        if let Some(label) = &filter.label {
151            let like_pattern = format!("%\"{label}\"%");
152            q = q.filter(project_issue::Column::Labels.like(like_pattern));
153        }
154        if let Some(limit) = filter.limit {
155            q = q.limit(limit as u64);
156        }
157        q
158    }
159
160    /// Lists issues using a rich filter.
161    ///
162    /// # Errors
163    /// Returns an error if the database query fails.
164    pub async fn list_issues_filtered(
165        &self,
166        org_id: &str,
167        filter: &IssueFilter,
168    ) -> Result<Vec<ProjectIssue>> {
169        if let Some(project_id) = &filter.project_id {
170            self.verify_project_exists(org_id, project_id).await?;
171        }
172
173        let q = project_issue::Entity::find()
174            .filter(project_issue::Column::OrgId.eq(org_id.to_owned()))
175            .order_by_desc(project_issue::Column::UpdatedAt);
176
177        let models = Self::apply_issue_filter(q, filter)
178            .all(&self.db)
179            .await
180            .map_err(db_error("list filtered project issues"))?;
181
182        Ok(models.into_iter().map(Into::into).collect())
183    }
184}

202    /// Lists all dependencies attached to an issue.
203    ///
204    /// # Errors
205    /// Returns an error if the database query fails.
206    pub async fn list_dependencies(&self, issue_id: &str) -> Result<Vec<ProjectDependency>> {
207        let models = project_dependency::Entity::find()
208            .filter(
209                sea_orm::Condition::any()
210                    .add(project_dependency::Column::FromIssueId.eq(issue_id.to_owned()))
211                    .add(project_dependency::Column::ToIssueId.eq(issue_id.to_owned())),
212            )
213            .order_by_asc(project_dependency::Column::CreatedAt)
214            .all(&self.db)
215            .await
216            .map_err(db_error("list project dependencies"))?;
217
218        Ok(models.into_iter().map(Into::into).collect())
219    }
220
221    /// Traverses issue dependencies breadth-first up to `max_depth`.
222    ///
223    /// # Errors
224    /// Returns an error if the database query fails.
225    pub async fn traverse_dependencies(
226        &self,
227        issue_id: &str,
228        max_depth: usize,
229    ) -> Result<Vec<ProjectDependency>> {
230        let mut queue = VecDeque::new();
231        queue.push_back((issue_id.to_owned(), 0usize));
232
233        let mut visited_nodes = HashSet::new();
234        visited_nodes.insert(issue_id.to_owned());
235
236        let mut visited_edges = HashSet::new();
237        let mut traversed = Vec::new();
238
239        while let Some((current, depth)) = queue.pop_front() {
240            if depth >= max_depth {
241                continue;
242            }
243
244            let edges = self.outgoing_dependencies(&current).await?;
245            for dependency in edges {
246                if !visited_edges.insert(dependency.id.clone()) {
247                    continue;
248                }
249                let next = dependency.to_issue_id.clone();
250                traversed.push(dependency);
251                if visited_nodes.insert(next.clone()) {
252                    queue.push_back((next, depth + 1));
253                }
254            }
255        }
256
257        Ok(traversed)
258    }
259
260    /// Loads the dependency edges originating from a single issue.
261    async fn outgoing_dependencies(&self, from_issue_id: &str) -> Result<Vec<ProjectDependency>> {
262        let models = project_dependency::Entity::find()
263            .filter(project_dependency::Column::FromIssueId.eq(from_issue_id.to_owned()))
264            .order_by_asc(project_dependency::Column::CreatedAt)
265            .all(&self.db)
266            .await
267            .map_err(db_error("traverse project dependencies"))?;
268        Ok(models.into_iter().map(ProjectDependency::from).collect())
269    }
