1//! VCS entity repository implementation.
2//!
3//! Implements `VcsRepositoryRegistry`, `VcsBranchRegistry`, `VcsWorktreeRegistry`,
4//! and `AgentAssignmentManager` for managing repositories, branches, worktrees,
5//! and agent worktree assignments.
6
7use super::*;
8
9sea_impl_crud_scoped!(VcsRepositoryRegistry for SeaOrmEntityRepository { db: db,
10    entity: repository, domain: Repository, label: "Repository",
11    scope_col: repository::Column::OrgId,
12    create: create_repository(repo),
13    get: get_repository,
14    list: list_repositories(repository::Column::ProjectId => project_id),
15    update: update_repository(repo),
16    delete: delete_repository
17});
18
19sea_impl_crud_mixed!(VcsBranchRegistry for SeaOrmEntityRepository { db: db,
20    entity: branch, domain: Branch, label: "Branch",
21    scope_col: branch::Column::OrgId,
22    create: create_branch(b),
23    get: get_branch,
24    list: list_branches(branch::Column::RepositoryId => repository_id),
25    update: update_branch(b),
26    delete: delete_branch(id),
27});
28
29sea_impl_crud!(VcsWorktreeRegistry for SeaOrmEntityRepository { db: db,
30    entity: worktree, domain: Worktree, label: "Worktree",
31    create: create_worktree(wt),
32    get: get_worktree(id),
33    list: list_worktrees(worktree::Column::RepositoryId => repository_id),
34    update: update_worktree(wt),
35    delete: delete_worktree(id)
36});
37
38#[async_trait]
39impl AgentAssignmentManager for SeaOrmEntityRepository {
40    async fn create_assignment(&self, asgn: &AgentWorktreeAssignment) -> Result<()> {
41        sea_repo_insert!(
42            self.db(),
43            agent_worktree_assignment,
44            asgn,
45            "create assignment"
46        )
47    }
48
49    async fn get_assignment(&self, id: &str) -> Result<AgentWorktreeAssignment> {
50        sea_repo_get!(
51            self.db(),
52            agent_worktree_assignment,
53            AgentWorktreeAssignment,
54            "Assignment",
55            id,
56            "get assignment"
57        )
58    }
59
60    async fn list_assignments_by_worktree(
61        &self,
62        worktree_id: &str,
63    ) -> Result<Vec<AgentWorktreeAssignment>> {
64        sea_repo_list!(self.db(), agent_worktree_assignment, AgentWorktreeAssignment, "list assignments",
65            agent_worktree_assignment::Column::WorktreeId => worktree_id)
66    }
67
68    async fn release_assignment(&self, id: &str, released_at: i64) -> Result<()> {
69        sea_repo_set_field!(
70            self.db(),
71            agent_worktree_assignment,
72            id,
73            "Assignment",
74            "release assignment",
75            released_at = Some(released_at)
76        )
77    }
78}
