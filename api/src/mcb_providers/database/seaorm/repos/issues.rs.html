1//! Issue registry and label management implementations.
2//!
3//! Implements `IssueRegistry`, `IssueCommentRegistry`, `IssueLabelRegistry`,
4//! and `IssueLabelAssignmentManager` for managing issues, comments, labels,
5//! and label assignments.
6
7use super::*;
8
9sea_impl_crud_scoped!(IssueRegistry for SeaOrmEntityRepository { db: db,
10    entity: project_issue, domain: ProjectIssue, label: "Issue",
11    scope_col: project_issue::Column::OrgId,
12    create: create_issue(issue),
13    get: get_issue,
14    list: list_issues(project_issue::Column::ProjectId => project_id),

18
19sea_impl_crud!(IssueCommentRegistry for SeaOrmEntityRepository { db: db,
20    entity: issue_comment, domain: IssueComment, label: "IssueComment",

27sea_impl_crud!(IssueLabelRegistry for SeaOrmEntityRepository { db: db,

35#[async_trait]
36impl IssueLabelAssignmentManager for SeaOrmEntityRepository {
37    async fn assign_label(&self, assignment: &IssueLabelAssignment) -> Result<()> {
38        sea_repo_insert!(
39            self.db(),
40            issue_label_assignment,
41            assignment,
42            "assign issue label"
43        )
44    }
45
46    async fn unassign_label(&self, issue_id: &str, label_id: &str) -> Result<()> {
47        sea_repo_delete!(
48            self.db(),
49            issue_label_assignment,
50            (issue_id.to_owned(), label_id.to_owned()),
51            "unassign issue label"
52        )
53    }
54
55    async fn list_labels_for_issue(&self, issue_id: &str) -> Result<Vec<IssueLabel>> {
56        let assignments = issue_label_assignment::Entity::find()
57            .filter(issue_label_assignment::Column::IssueId.eq(issue_id))
58            .all(self.db())
59            .await
60            .map_err(db_err)?;
61
62        if assignments.is_empty() {
63            return Ok(vec![]);
64        }
65
66        let label_ids: Vec<String> = assignments.into_iter().map(|a| a.label_id).collect();
67        let labels = issue_label::Entity::find()
68            .filter(issue_label::Column::Id.is_in(label_ids))
69            .all(self.db())
70            .await
71            .map_err(db_err)?;
72
73        Ok(labels.into_iter().map(IssueLabel::from).collect())
74    }
75}
