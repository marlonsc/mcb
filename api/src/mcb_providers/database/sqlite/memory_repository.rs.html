8//! # Overview
9//! The `SqliteMemoryRepository` provides persistent storage for observations and session summaries
10//! using a `SQLite` database. It serves as the primary source of truth for structured
11//! memory data and enables full-text search (FTS) capabilities.
12//!
13//! # Responsibilities
14//! - **Observation Persistence**: Storing and retrieving immutable observation records.
15//! - **FTS Implementation**: Leveraging `SQLite`'s FTS5 extension to perform efficient text searches.
16//! - **Session Summaries**: Managing the persistence of high-level session insights.
17//! - **Timeline Construction**: Querying observations efficiently by creation time.
18//!
19//! # Architecture
20//! This repository implements `MemoryRepository` and sits in the Providers layer.
21//! It depends on `DatabaseExecutor` (Port) to execute SQL, ensuring it remains
22//! decoupled from the specific SQL client library (e.g., sqlx, rusqlite).

37/// Serializes the complex JSON fields of a `SessionSummary` for database storage.
38fn serialize_summary_fields(
39    summary: &SessionSummary,
40) -> Result<(String, String, String, String, String)> {
41    let topics = serde_json::to_string(&summary.topics)
42        .map_err(|e| Error::memory_with_source("serialize topics", e))?;
43    let decisions = serde_json::to_string(&summary.decisions)
44        .map_err(|e| Error::memory_with_source("serialize decisions", e))?;
45    let next_steps = serde_json::to_string(&summary.next_steps)
46        .map_err(|e| Error::memory_with_source("serialize next_steps", e))?;
47    let key_files = serde_json::to_string(&summary.key_files)
48        .map_err(|e| Error::memory_with_source("serialize key_files", e))?;
49    let origin_context = serde_json::to_string(&summary.origin_context)
50        .map_err(|e| Error::memory_with_source("serialize origin_context", e))?;
51    Ok((topics, decisions, next_steps, key_files, origin_context))
52}

54/// Builds the base SQL `WHERE` clause and parameters from an optional `MemoryFilter`.
55///
56/// Returns a `(sql_fragment, params)` tuple where `sql_fragment` starts with
57/// `"SELECT * FROM observations WHERE 1=1"` followed by any filter conditions.
58fn build_timeline_filter_sql(filter: Option<&MemoryFilter>) -> (String, Vec<SqlParam>) {
59    let mut sql = String::from("SELECT * FROM observations WHERE 1=1");
60    let mut params: Vec<SqlParam> = Vec::new();
61
62    let Some(f) = filter else {
63        return (sql, params);
64    };
65
66    if let Some(session_id) = &f.session_id {
67        sql.push_str(" AND json_extract(metadata, '$.session_id') = ?");
68        params.push(SqlParam::String(session_id.clone()));
69    }
70    if let Some(parent_session_id) = &f.parent_session_id {
71        sql.push_str(" AND json_extract(metadata, '$.origin_context.parent_session_id') = ?");
72        params.push(SqlParam::String(parent_session_id.clone()));
73    }
74    if let Some(repo_id) = &f.repo_id {
75        sql.push_str(" AND json_extract(metadata, '$.repo_id') = ?");
76        params.push(SqlParam::String(repo_id.clone()));
77    }
78    if let Some(branch) = &f.branch {
79        sql.push_str(" AND json_extract(metadata, '$.branch') = ?");
80        params.push(SqlParam::String(branch.clone()));
81    }
82    if let Some(commit) = &f.commit {
83        sql.push_str(" AND json_extract(metadata, '$.commit') = ?");
84        params.push(SqlParam::String(commit.clone()));
85    }
86    if let Some(obs_type) = &f.r#type {
87        sql.push_str(" AND observation_type = ?");
88        params.push(SqlParam::String(obs_type.as_str().to_owned()));
89    }
90
91    (sql, params)
92}

94/// Assembles a timeline from before/after row sets plus the anchor observation.
95async fn assemble_timeline(
96    before_rows: &[Arc<dyn mcb_domain::ports::SqlRow>],
97    after_rows: &[Arc<dyn mcb_domain::ports::SqlRow>],
98    repo: &SqliteMemoryRepository,
99    anchor_id: &ObservationId,
100) -> Result<Vec<Observation>> {
101    let mut timeline = Vec::new();
102    for row in before_rows.iter().rev() {
103        timeline.push(
104            row_convert::row_to_observation(row.as_ref())
105                .map_err(|e| Error::memory_with_source("decode observation", e))?,
106        );
107    }
108    if let Some(anchor_obs) = repo.get_observation(anchor_id).await? {
109        timeline.push(anchor_obs);
110    }
111    for row in after_rows {
112        timeline.push(
113            row_convert::row_to_observation(row.as_ref())
114                .map_err(|e| Error::memory_with_source("decode observation", e))?,
115        );
116    }
117    Ok(timeline)
118}

120/// SQLite-based implementation of the `MemoryRepository`.
121///
122/// Uses standard SQL queries to manage `observations` and `session_summaries` tables.
123/// Handles JSON serialization of complex fields (metadata, tags) and enforces
124/// referential integrity via `ensure_parent` checks.
125pub struct SqliteMemoryRepository {
126    executor: Arc<dyn DatabaseExecutor>,
127}
128
129impl SqliteMemoryRepository {
130    /// Create a repository that uses the given executor (from provider factory).
131    pub fn new(executor: Arc<dyn DatabaseExecutor>) -> Self {
132        Self { executor }
133    }
134
135    /// Queries a time window of observations relative to an anchor timestamp.
136    async fn query_timeline_window(
137        &self,
138        base_sql: &str,
139        base_params: &[SqlParam],
140        anchor_time: i64,
141        limit: usize,
142        order: &str,
143    ) -> Result<Vec<Arc<dyn mcb_domain::ports::SqlRow>>> {
144        let op = if order == "DESC" { "<" } else { ">" };
145        let sql = format!("{base_sql} AND created_at {op} ? ORDER BY created_at {order} LIMIT ?");
146        let mut params = base_params.to_vec();
147        params.push(SqlParam::I64(anchor_time));
148        params.push(SqlParam::I64(limit as i64));
149        self.executor.query_all(&sql, &params).await
150    }
151}

153#[async_trait]
154/// Persistent memory repository using `SQLite`.
155impl MemoryRepository for SqliteMemoryRepository {
156    /// Stores an observation record.
157    async fn store_observation(&self, observation: &Observation) -> Result<()> {
158        // Ensure default org and project exist
159        super::ensure_parent::ensure_org_and_project(
160            self.executor.as_ref(),
161            &observation.project_id,
162            observation.created_at,
163        )
164        .await?;
165
166        let tags_json = serde_json::to_string(&observation.tags)
167            .map_err(|e| Error::memory_with_source("serialize tags", e))?;
168        let metadata_json = serde_json::to_string(&observation.metadata)
169            .map_err(|e| Error::memory_with_source("serialize metadata", e))?;
170
171        let params = [
172            SqlParam::String(observation.id.clone()),
173            SqlParam::String(observation.project_id.clone()),
174            SqlParam::String(observation.content.clone()),
175            SqlParam::String(observation.content_hash.clone()),
176            SqlParam::String(tags_json),
177            SqlParam::String(observation.r#type.as_str().to_owned()),
178            SqlParam::String(metadata_json),
179            SqlParam::I64(observation.created_at),
180            observation
181                .embedding_id
182                .as_ref()
183                .map_or(SqlParam::Null, |s| SqlParam::String(s.clone())),
184        ];
185
186        self.executor
187            .execute(
188                "
189                INSERT INTO observations (id, project_id, content, content_hash, tags, observation_type, metadata, created_at, embedding_id)
190                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
191                ON CONFLICT(content_hash) DO UPDATE SET
192                    tags = excluded.tags,
193                    metadata = excluded.metadata
194                ",
195                &params,
196            )
197            .await?;
198
199        mcb_domain::debug!("sqlite", "Stored observation", &observation.id);
200        Ok(())
201    }

225    /// Searches observations using FTS.
226    async fn search(&self, query: &str, limit: usize) -> Result<Vec<FtsSearchResult>> {
227        let rows = self
228            .executor
229            .query_all(
230                "SELECT id, rank FROM observations_fts WHERE observations_fts MATCH ? ORDER BY rank LIMIT ?",
231                &[SqlParam::String(query.to_owned()), SqlParam::I64(limit as i64)],
232            )
233            .await?;
234
235        let mut results = Vec::with_capacity(rows.len());
236        for row in rows {
237            let id = row
238                .try_get_string("id")?
239                .ok_or_else(|| Error::memory("FTS result missing id"))?;
240            let rank = row.try_get_f64("rank")?.unwrap_or(0.0);
241            results.push(FtsSearchResult { id, rank });
242        }
243        Ok(results)
244    }

256    /// Retrieves multiple observations by ID.
257    async fn get_observations_by_ids(&self, ids: &[ObservationId]) -> Result<Vec<Observation>> {
258        if ids.is_empty() {
259            return Ok(Vec::new());
260        }
261
262        let placeholders: Vec<String> = ids.iter().map(|_| "?".to_owned()).collect();
263        let sql = format!(
264            "SELECT * FROM observations WHERE id IN ({})",
265            placeholders.join(",")
266        );
267        let params: Vec<SqlParam> = ids
268            .iter()
269            .map(|id| SqlParam::String(id.to_string()))
270            .collect();
271
272        query_helpers::query_all(
273            &self.executor,
274            &sql,
275            &params,
276            row_convert::row_to_observation,
277            "observation",
278        )
279        .await
280    }

282    /// Retrieves a timeline of observations centered around an anchor observation.
283    ///
284    /// The timeline includes a specified number of observations before and after the anchor,
285    /// optionally filtered by session, repository, or observation type.
286    async fn get_timeline(
287        &self,
288        anchor_id: &ObservationId,
289        before: usize,
290        after: usize,
291        filter: Option<MemoryFilter>,
292    ) -> Result<Vec<Observation>> {
293        let anchor = self.get_observation(anchor_id).await?;
294        let anchor_time = match anchor {
295            Some(obs) => obs.created_at,
296            None => return Ok(Vec::new()),
297        };
298
299        let (base_sql, base_params) = build_timeline_filter_sql(filter.as_ref());
300
301        let before_rows = self
302            .query_timeline_window(&base_sql, &base_params, anchor_time, before, "DESC")
303            .await?;
304        let after_rows = self
305            .query_timeline_window(&base_sql, &base_params, anchor_time, after, "ASC")
306            .await?;
307
308        assemble_timeline(&before_rows, &after_rows, self, anchor_id).await
309    }

311    /// Persists a session summary to the database, updating it if it already exists.
312    ///
313    /// Handles serialization of topics, decisions, and other complex fields into JSON.
314    async fn store_session_summary(&self, summary: &SessionSummary) -> Result<()> {
315        super::ensure_parent::ensure_org_and_project(
316            self.executor.as_ref(),
317            &summary.project_id,
318            summary.created_at,
319        )
320        .await?;
321
322        let (topics, decisions, next_steps, key_files, origin_ctx) =
323            serialize_summary_fields(summary)?;
324
325        let params = [
326            SqlParam::String(summary.id.clone()),
327            SqlParam::String(summary.project_id.clone()),
328            SqlParam::String(summary.org_id.clone()),
329            SqlParam::String(summary.session_id.clone()),
330            SqlParam::String(topics),
331            SqlParam::String(decisions),
332            SqlParam::String(next_steps),
333            SqlParam::String(key_files),
334            SqlParam::String(origin_ctx),
335            SqlParam::I64(summary.created_at),
336        ];
337
338        self.executor
339            .execute(
340                "
341                INSERT INTO session_summaries (id, project_id, org_id, session_id, topics, decisions, next_steps, key_files, origin_context, created_at)
342                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
343                ON CONFLICT(id) DO UPDATE SET
344                    topics = excluded.topics,
345                    decisions = excluded.decisions,
346                    next_steps = excluded.next_steps,
347                    key_files = excluded.key_files,
348                    origin_context = excluded.origin_context
349                ",
350                &params,
351            )
352            .await?;
353
354        mcb_domain::debug!(
355            "sqlite",
356            "Stored session summary for session",
357            &mask_id(&summary.session_id.clone())
358        );
359        Ok(())
360    }

370        .await
