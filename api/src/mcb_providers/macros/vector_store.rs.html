6//! - `send_actor_msg!`: Actor message send/await pattern (edgevec only)
7
8/// Send a message to the `EdgeVec` actor and await the oneshot response.
9///
10/// Constructs the message with a oneshot channel, sends it via the actor's
11/// sender, and awaits the response.
12///
13/// ## Example
14///
15/// ```ignore
16/// send_actor_msg!(self, Core(CoreMessage::CreateCollection {
17///     name: collection.to_string()
18/// }))
19/// ```
20macro_rules! send_actor_msg {
21    (
22        $self:expr,
23        $outer:ident ( $inner:ident :: $variant:ident {
24            $($field:ident : $val:expr),* $(,)?
25        })
26    ) => {{
27        let (tx, rx) = tokio::sync::oneshot::channel();
28        let _ = $self
29            .sender
30            .send(EdgeVecMessage::$outer($inner::$variant {
31                $($field: $val,)*
32                tx,
33            }))
34            .await;
35        rx.await
36            .unwrap_or_else(|_| Err(mcb_domain::error::Error::internal("Actor closed")))
37    }};
38}
