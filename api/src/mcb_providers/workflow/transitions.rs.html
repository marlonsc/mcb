4//! FSM transition logic and state validation rules for ADR-034 Workflow.
5//!
6//! Implements the state machine rules that govern valid transitions
7//! and side effects for the 8-state workflow model.
8
9use mcb_domain::entities::{TransitionTrigger, WorkflowSession, WorkflowState};
10use mcb_utils::constants::FALLBACK_UNKNOWN;
11
12type Result<T> = std::result::Result<T, String>;
13
14/// Apply a transition trigger to a workflow session, validating FSM rules.
15///
16/// Returns the target state if transition is valid, error otherwise.
17///
18/// # Errors
19///
20/// Returns an error string if the transition is not valid from the current state.
21pub fn apply_transition(
22    session: &mut WorkflowSession,
23    trigger: &TransitionTrigger,
24) -> Result<WorkflowState> {
25    // Terminal state — no transitions allowed
26    if matches!(session.current_state, WorkflowState::Completed) {
27        return Err("Cannot transition from terminal state Completed".to_owned());
28    }
29
30    // Global error transition: any state → Failed
31    if let TransitionTrigger::Error { message } = trigger {
32        return Ok(WorkflowState::Failed {
33            error: message.clone(),
34            recoverable: true,
35        });
36    }
37
38    resolve_transition(&session.current_state, trigger)
39}
40
41/// Resolve a non-error, non-terminal transition.
42fn resolve_transition(state: &WorkflowState, trigger: &TransitionTrigger) -> Result<WorkflowState> {
43    match (state, trigger) {
44        // Initializing → Ready
45        (WorkflowState::Initializing, TransitionTrigger::ContextDiscovered { context_id }) => {
46            Ok(WorkflowState::Ready {
47                context_id: context_id.clone(),
48            })
49        }
50
51        // Ready | PhaseComplete → Planning
52        (
53            WorkflowState::Ready { .. } | WorkflowState::PhaseComplete { .. },
54            TransitionTrigger::StartPlanning { phase_id },
55        ) => Ok(WorkflowState::Planning {
56            phase_id: phase_id.clone(),
57        }),
58
59        // → Executing (from Ready, Planning, Executing, Verifying)
60        (state, trigger) if is_executing_transition(state, trigger) => {
61            Ok(resolve_executing_state(state, trigger))
62        }
63
64        // Executing → Verifying
65        (WorkflowState::Executing { phase_id, .. }, TransitionTrigger::StartVerification) => {
66            Ok(WorkflowState::Verifying {
67                phase_id: phase_id.clone(),
68            })
69        }
70
71        _ => resolve_verification_transition(state, trigger),
72    }
73}

75/// Resolve verification, completion, and recovery transitions (latter half of the FSM).
76fn resolve_verification_transition(
77    state: &WorkflowState,
78    trigger: &TransitionTrigger,
79) -> Result<WorkflowState> {
80    match (state, trigger) {
81        // Verifying → PhaseComplete
82        (WorkflowState::Verifying { phase_id }, TransitionTrigger::VerificationPassed) => {
83            Ok(WorkflowState::PhaseComplete {
84                phase_id: phase_id.clone(),
85            })
86        }
87
88        // PhaseComplete | Failed → Completed
89        (
90            WorkflowState::PhaseComplete { .. } | WorkflowState::Failed { .. },
91            TransitionTrigger::EndSession,
92        ) => Ok(WorkflowState::Completed),
93
94        // Failed → Executing (recovery)
95        (WorkflowState::Failed { .. }, TransitionTrigger::Recover) => {
96            Ok(WorkflowState::Executing {
97                phase_id: FALLBACK_UNKNOWN.to_owned(),
98                task_id: None,
99            })
100        }
101
102        // Invalid transition
103        (from, trigger) => Err(format!(
104            "Invalid FSM transition: {from} + {trigger} not allowed"
105        )),
106    }
107}

109/// Check if this is a transition that results in `Executing` state.
110fn is_executing_transition(state: &WorkflowState, trigger: &TransitionTrigger) -> bool {
111    matches!(
112        (state, trigger),
113        (
114            WorkflowState::Ready { .. },
115            TransitionTrigger::StartExecution { .. }
116        ) | (
117            WorkflowState::Planning { .. },
118            TransitionTrigger::StartExecution { .. }
119        ) | (
120            WorkflowState::Executing { .. },
121            TransitionTrigger::CompleteTask { .. }
122        ) | (
123            WorkflowState::Executing { .. },
124            TransitionTrigger::ClaimTask { .. }
125        ) | (
126            WorkflowState::Verifying { .. },
127            TransitionTrigger::VerificationFailed { .. }
128        )
129    )
130}

132/// Resolve the specific `Executing` variant based on the trigger.
133fn resolve_executing_state(state: &WorkflowState, trigger: &TransitionTrigger) -> WorkflowState {
134    let phase_id = match (state, trigger) {
135        (_, TransitionTrigger::StartExecution { phase_id })
136        | (
137            WorkflowState::Executing { phase_id, .. }
138            | WorkflowState::Planning { phase_id }
139            | WorkflowState::Verifying { phase_id },
140            _,
141        ) => phase_id.clone(),
142        _ => FALLBACK_UNKNOWN.to_owned(),
143    };
144
145    let task_id = if let TransitionTrigger::ClaimTask { task_id } = trigger {
146        Some(task_id.clone())
147    } else {
148        None
149    };
150
151    WorkflowState::Executing { phase_id, task_id }
152}
