4//! FSM transition logic and state validation rules for ADR-034 Workflow.
5//!
6//! Implements the state machine rules that govern valid transitions
7//! and side effects for the 8-state workflow model.
8
9use mcb_domain::entities::{TransitionTrigger, WorkflowSession, WorkflowState};
10
11type Result<T> = std::result::Result<T, String>;
12
13/// Apply a transition trigger to a workflow session, validating FSM rules.
14///
15/// Returns the target state if transition is valid, error otherwise.
16///
17/// # Errors
18///
19/// Returns an error string if the transition is not valid from the current state.
20pub fn apply_transition(
21    session: &mut WorkflowSession,
22    trigger: &TransitionTrigger,
23) -> Result<WorkflowState> {
24    let new_state = match (&session.current_state, trigger) {
25        // Initializing → Ready (must have context)
26        (WorkflowState::Initializing, TransitionTrigger::ContextDiscovered { context_id }) => {
27            WorkflowState::Ready {
28                context_id: context_id.clone(),
29            }
30        }
31
32        // Ready | PhaseComplete → Planning
33        (WorkflowState::Ready { .. }, TransitionTrigger::StartPlanning { phase_id })
34        | (WorkflowState::PhaseComplete { .. }, TransitionTrigger::StartPlanning { phase_id }) => {
35            WorkflowState::Planning {
36                phase_id: phase_id.clone(),
37            }
38        }
39
40        // Ready → Executing (skip planning) | Planning → Executing |
41        // Executing → Executing (complete task) | Verifying → Executing (verification failed)
42        (WorkflowState::Ready { .. }, TransitionTrigger::StartExecution { phase_id })
43        | (WorkflowState::Planning { phase_id }, TransitionTrigger::StartExecution { .. })
44        | (WorkflowState::Executing { phase_id, .. }, TransitionTrigger::CompleteTask { .. })
45        | (WorkflowState::Verifying { phase_id }, TransitionTrigger::VerificationFailed { .. }) => {
46            WorkflowState::Executing {
47                phase_id: phase_id.clone(),
48                task_id: None,
49            }
50        }
51
52        // Executing → Executing (claim task)
53        (WorkflowState::Executing { phase_id, .. }, TransitionTrigger::ClaimTask { task_id }) => {
54            WorkflowState::Executing {
55                phase_id: phase_id.clone(),
56                task_id: Some(task_id.clone()),
57            }
58        }
59
60        // Executing → Verifying
61        (WorkflowState::Executing { phase_id, .. }, TransitionTrigger::StartVerification) => {
62            WorkflowState::Verifying {
63                phase_id: phase_id.clone(),
64            }
65        }
66
67        // Verifying → PhaseComplete (verification passed)
68        (WorkflowState::Verifying { phase_id }, TransitionTrigger::VerificationPassed) => {
69            WorkflowState::PhaseComplete {
70                phase_id: phase_id.clone(),
71            }
72        }
73
74        // PhaseComplete → Completed
75        (WorkflowState::PhaseComplete { .. }, TransitionTrigger::EndSession) => {
76            WorkflowState::Completed
77        }
78
79        // Any state → Failed (error trigger)
80        (_, TransitionTrigger::Error { message }) => WorkflowState::Failed {
81            error: message.clone(),
82            recoverable: true,
83        },
84
85        // Failed → Executing (recovery)
86        (WorkflowState::Failed { .. }, TransitionTrigger::Recover) => {
87            // Return to last known good executing state
88            WorkflowState::Executing {
89                phase_id: "unknown".to_owned(),
90                task_id: None,
91            }
92        }
93
94        // Failed → Completed (give up)
95        (WorkflowState::Failed { .. }, TransitionTrigger::EndSession) => WorkflowState::Completed,
96
97        // Completed → (no transitions allowed)
98        (WorkflowState::Completed, _) => {
99            return Err("Cannot transition from terminal state Completed".to_owned());
100        }
101
102        // Invalid transition
103        (from, trigger) => {
104            return Err(format!(
105                "Invalid FSM transition: {from} + {trigger} not allowed"
106            ));
107        }
108    };
109
110    Ok(new_state)
111}
