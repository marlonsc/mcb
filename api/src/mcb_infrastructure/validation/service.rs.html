4//! Validation Service Implementation
5//!
6//! # Overview
7//! The `InfraValidationService` adapts the `mcb-validate` toolkit into the domain's
8//! `ValidationServiceInterface` port. It serves as the bridge between the core domain's
9//! need for quality assurance and the infrastructure-level tools that perform analysis.
10//!
11//! # Responsibilities
12//! - **Workspace Validation**: Running suite of validators against the entire project.
13//! - **File Analysis**: Targeted validation and complexity analysis for individual files.
14//! - **Rule Discovery**: Exposing available validation rules and their metadata.
15//! - **Complexity Metrics**: Calculating cyclomatic and cognitive complexity scores.
16

26/// Infrastructure validation service using mcb-validate.

28/// A stateless adapter that orchestrates the `mcb-validate` library to perform
29/// architectural compliance checks, code quality analysis, and rule enforcement.

46#[async_trait]
47impl ValidationServiceInterface for InfraValidationService {
48    async fn validate(
49        &self,
50        workspace_root: &Path,
51        validators: Option<&[String]>,
52        severity_filter: Option<&str>,
53    ) -> Result<ValidationReport> {
54        run_validation(workspace_root, validators, severity_filter)
55    }
56
57    async fn list_validators(&self) -> Result<Vec<String>> {
58        let mut validators = mcb_domain::registry::validation::list_validator_names();
59        validators.sort_unstable();
60        Ok(validators)
61    }
62
63    async fn validate_file(
64        &self,
65        file_path: &Path,
66        validators: Option<&[String]>,
67    ) -> Result<ValidationReport> {
68        run_file_validation(file_path, validators)
69    }
70
71    async fn get_rules(&self, category: Option<&str>) -> Result<Vec<RuleInfo>> {
72        let entries = mcb_domain::registry::validation::list_validator_entries();
73        let mut rules: Vec<RuleInfo> = entries
74            .iter()
75            .filter(|(name, _)| category.is_none_or(|c| *name == c))
76            .map(|(name, description)| RuleInfo {
77                id: (*name).to_owned(),
78                category: (*name).to_owned(),
79                severity: "warning".to_owned(),
80                description: (*description).to_owned(),
81                engine: "linkme".to_owned(),
82            })
83            .collect();
84        rules.sort_unstable_by(|a, b| a.id.cmp(&b.id));
85        Ok(rules)
86    }
87
88    async fn analyze_complexity(
89        &self,
90        file_path: &Path,
91        include_functions: bool,
92    ) -> Result<ComplexityReport> {
93        analyze_file_complexity(file_path, include_functions)
94    }
95}

97fn run_validation(
98    workspace_root: &Path,
99    validators: Option<&[String]>,
100    severity_filter: Option<&str>,
101) -> Result<ValidationReport> {
102    let root = workspace_root.to_path_buf();
103    let validators_list = if let Some(names) = validators {
104        mcb_domain::registry::validation::build_named_validators(&root, names)?
105    } else {
106        mcb_domain::registry::validation::build_all_validators(&root)?
107    };
108
109    let config = mcb_domain::ports::validation::ValidationConfig::new(&root);
110
111    let mut all_violations: Vec<Box<dyn mcb_domain::ports::validation::Violation>> = Vec::new();
112    for v in &validators_list {
113        match v.validate(&config) {
114            Ok(violations) => all_violations.extend(violations),
115            Err(e) => {
116                return Err(mcb_domain::error::Error::internal(format!(
117                    "validator '{}' failed: {}",
118                    v.name(),
119                    e
120                )));
121            }
122        }
123    }
124
125    Ok(mcb_domain::registry::validation::build_report(
126        &all_violations,
127        severity_filter,
128    ))
129}

131/// Traverse parent directories to find the workspace root (directory containing Cargo.toml).
132fn find_workspace_root_from(start: &Path) -> Option<std::path::PathBuf> {
133    let mut current = if start.is_file() {
134        start.parent()?
135    } else {
136        start
137    };
138    loop {
139        if current.join("Cargo.toml").exists() {
140            return Some(current.to_path_buf());
141        }
142        current = current.parent()?;
143    }
144}

146fn run_file_validation(
147    file_path: &Path,
148    validators: Option<&[String]>,
149) -> Result<ValidationReport> {
150    // For single file validation, we need to find the workspace root
151    // and run validation scoped to that file
152    let workspace_root = find_workspace_root_from(file_path)
153        .unwrap_or_else(|| file_path.parent().unwrap_or(file_path).to_path_buf());
154
155    // Run standard validation - mcb-validate doesn't have single-file mode yet
156    // So we run full validation and filter to the specific file
157    let full_report = run_validation(&workspace_root, validators, None)?;
158
159    let file_str = file_path.to_str().unwrap_or_default().to_owned();
160    let file_violations: Vec<ViolationEntry> = full_report
161        .violations
162        .into_iter()
163        .filter(|v| v.file.as_ref().is_some_and(|f| f.contains(&file_str)))
164        .collect();
165
166    let total = file_violations.len();
167    let errors = file_violations
168        .iter()
169        .filter(|v| v.severity == "ERROR")
170        .count();
171    let warnings = file_violations
172        .iter()
173        .filter(|v| v.severity == "WARNING")
174        .count();
175    let infos = total.saturating_sub(errors).saturating_sub(warnings);
176    Ok(ValidationReport {
177        total_violations: total,
178        errors,
179        warnings,
180        infos,
181        passed: errors == 0,
182        violations: file_violations,
183    })
184}

186fn count_sloc(content: &str) -> usize {
187    content
188        .lines()
189        .filter(|line| {
190            let trimmed = line.trim();
191            !trimmed.is_empty() && !trimmed.starts_with("//")
192        })
193        .count()
194}

196fn calculate_maintainability_index(sloc: usize, avg_cyclomatic: f64) -> f64 {
197    if sloc == 0 {
198        return 100.0;
199    }
200    let raw = 171.0
201        - 5.2 * avg_cyclomatic.max(1.0).ln()
202        - 0.23 * avg_cyclomatic
203        - 16.2 * (sloc as f64).ln();
204    (raw * 100.0 / 171.0).clamp(0.0, 100.0)
205}

207fn build_function_metrics(
208    functions: &[mcb_domain::utils::analysis::FunctionRecord],
209    include_functions: bool,
210) -> Vec<FunctionComplexity> {
211    if !include_functions {
212        return vec![];
213    }
214    functions
215        .iter()
216        .map(|f| FunctionComplexity {
217            name: f.name.clone(),
218            line: f.line,
219            cyclomatic: f64::from(f.complexity),
220            cognitive: f64::from(f.complexity) * 1.2,
221            sloc: 0,
222        })
223        .collect()
224}
225
226fn analyze_file_complexity(file_path: &Path, include_functions: bool) -> Result<ComplexityReport> {
227    let content = std::fs::read_to_string(file_path).map_err(|e| {
228        mcb_domain::error::Error::io_with_source(
229            format!("failed to read {}", file_path.display()),
230            e,
231        )
232    })?;
233
234    let sloc = count_sloc(&content);
235    let files = vec![(file_path.to_path_buf(), content)];
236    let functions = mcb_domain::utils::analysis::collect_functions(&files)?;
237
238    let total_cyclomatic: f64 = functions.iter().map(|f| f64::from(f.complexity)).sum();
239    let fn_count = functions.len().max(1) as f64;
240    let avg_cyclomatic = total_cyclomatic / fn_count;
241    let cognitive = avg_cyclomatic * 1.2;
242    let mi = calculate_maintainability_index(sloc, avg_cyclomatic);
243    let function_metrics = build_function_metrics(&functions, include_functions);
244
245    Ok(ComplexityReport {
246        file: file_path.display().to_string(),
247        cyclomatic: avg_cyclomatic,
248        cognitive,
249        maintainability_index: mi,
250        sloc,
251        functions: function_metrics,
252    })
253}
