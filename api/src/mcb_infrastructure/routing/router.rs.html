17/// Default provider router with health-aware selection
18///
19/// Selects providers based on health status and context preferences.
20/// Must be constructed via DI - never instantiate directly.
21pub struct DefaultProviderRouter {
22    /// Health monitor for tracking provider status
23    health_monitor: Arc<dyn HealthMonitor>,
24    /// Available embedding providers
25    embedding_providers: Vec<String>,
26    /// Available vector store providers
27    vector_store_providers: Vec<String>,
28}
29
30impl DefaultProviderRouter {
31    /// Create a new router with injected dependencies
32    ///
33    /// This should only be called by the DI container.
34    pub fn new(
35        health_monitor: Arc<dyn HealthMonitor>,
36        embedding_providers: Vec<String>,
37        vector_store_providers: Vec<String>,
38    ) -> Self {
39        Self {
40            health_monitor,
41            embedding_providers,
42            vector_store_providers,
43        }
44    }
45
46    fn filter_available_providers<'a>(
47        providers: &'a [String],
48        context: &ProviderContext,
49    ) -> Vec<&'a String> {
50        providers
51            .iter()
52            .filter(|p| !context.excluded_providers.contains(*p))
53            .collect()
54    }
55
56    fn find_preferred_provider<'a>(
57        &self,
58        available: &[&'a String],
59        context: &ProviderContext,
60    ) -> Option<&'a String> {
61        for preferred in &context.preferred_providers {
62            if let Some(&p) = available.iter().find(|&&x| x == preferred) {
63                let health = self.health_monitor.get_health(preferred);
64                if health != ProviderHealthStatus::Unhealthy {
65                    return Some(p);
66                }
67            }
68        }
69        None
70    }
71
72    fn find_healthiest_provider<'a>(&self, available: &[&'a String]) -> Option<&'a String> {
73        let mut best_provider: Option<&&String> = None;
74        let mut best_health = ProviderHealthStatus::Unhealthy;
75
76        for provider in available {
77            let health = self.health_monitor.get_health(provider);
78
79            let is_better = match (health, best_health) {
80                (ProviderHealthStatus::Healthy, _) => best_health != ProviderHealthStatus::Healthy,
81                (ProviderHealthStatus::Degraded, ProviderHealthStatus::Unhealthy) => true,
82                (ProviderHealthStatus::Degraded | ProviderHealthStatus::Unhealthy, _) => {
83                    best_provider.is_none()
84                }
85            };
86
87            if is_better {
88                best_provider = Some(provider);
89                best_health = health;
90            }
91        }
92
93        best_provider.copied()
94    }
95
96    /// Select the best provider from a list based on health and preferences
97    fn select_best_provider(
98        &self,
99        providers: &[String],
100        context: &ProviderContext,
101    ) -> Result<String> {
102        let available = Self::filter_available_providers(providers, context);
103
104        if available.is_empty() {
105            return Err(Error::infrastructure(
106                "No providers available after exclusions",
107            ));
108        }
109
110        if let Some(preferred) = self.find_preferred_provider(&available, context) {
111            return Ok(preferred.to_owned());
112        }
113
114        let best = self
115            .find_healthiest_provider(&available)
116            .cloned()
117            .ok_or_else(|| Error::infrastructure("No healthy providers available"))?;
118
119        Ok(best.clone())
120    }
121}
122
123#[async_trait]
124impl ProviderRouter for DefaultProviderRouter {
125    async fn select_embedding_provider(&self, context: &ProviderContext) -> Result<String> {
126        self.select_best_provider(&self.embedding_providers, context)
127    }
128
129    async fn select_vector_store_provider(&self, context: &ProviderContext) -> Result<String> {
130        self.select_best_provider(&self.vector_store_providers, context)
131    }
132
133    async fn get_provider_health(&self, provider_id: &str) -> Result<ProviderHealthStatus> {
134        Ok(self.health_monitor.get_health(provider_id))
135    }
136
137    async fn report_failure(&self, provider_id: &str, _error: &str) -> Result<()> {
138        self.health_monitor.record_failure(provider_id);
139        Ok(())
140    }
141
142    async fn report_success(&self, provider_id: &str) -> Result<()> {
143        self.health_monitor.record_success(provider_id);
144        Ok(())
145    }
146
147    async fn get_all_health(&self) -> Result<HashMap<String, ProviderHealthStatus>> {
148        Ok(self.health_monitor.get_all_health())
149    }
150
151    async fn get_stats(&self) -> HashMap<String, serde_json::Value> {
152        let mut stats = HashMap::new();
153        stats.insert(
154            "provider".to_owned(),
155            serde_json::json!(mcb_utils::constants::DEFAULT_NAMESPACE),
156        );
157        stats.insert(
158            "embedding_providers".to_owned(),
159            serde_json::json!(self.embedding_providers),
160        );
161        stats.insert(
162            "vector_store_providers".to_owned(),
163            serde_json::json!(self.vector_store_providers),
164        );
165        stats.insert(
166            "health_summary".to_owned(),
167            serde_json::json!(self.health_monitor.get_all_health()),
168        );
169        stats
170    }
171}

173impl std::fmt::Debug for DefaultProviderRouter {
174    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
175        f.debug_struct("DefaultProviderRouter")
176            .field("embedding_providers", &self.embedding_providers)
177            .field("vector_store_providers", &self.vector_store_providers)
178            .finish()
179    }
