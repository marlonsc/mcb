4//! Highlight Service Use Case
5//!
6//! # Overview
7//! The `HighlightService` provides backend-agnostic syntax highlighting capabilities using
8//! Tree-sitter. It parses source code into an abstract syntax tree (AST) to identify
9//! tokens and apply semantic highlighting rules, independent of the final output format.
10//!
11//! # Responsibilities
12//! - **Multi-Language Support**: Parsing and highlighting code for supported languages (Rust, Python, JS, etc.).
13//! - **Tree-Sitter Integration**: Leveraging widely-used grammars for accurate syntax analysis.
14//! - **Abstract Representation**: Producing a generic `HighlightedCode` structure (spans + categories)
15//!   that can be rendered to HTML, ANSI, or other formats.
16

26/// Language-specific highlighting configuration
27struct HighlightLanguageConfig {
28    name: &'static str,
29    language: Language,
30    highlights_query: &'static str,
31}
32
33impl HighlightLanguageConfig {
34    fn new(name: &'static str, language: Language, highlights_query: &'static str) -> Self {
35        Self {
36            name,
37            language,
38            highlights_query,
39        }
40    }
41}

43/// Maps tree-sitter highlight names to our category enum.
44#[must_use]
45pub fn map_highlight_to_category(name: &str) -> HighlightCategory {
46    match name {
47        "keyword" => HighlightCategory::Keyword,
48        "string" => HighlightCategory::String,
49        "comment" => HighlightCategory::Comment,
50        "function" => HighlightCategory::Function,
51        "type" => HighlightCategory::Type,
52        "variable" => HighlightCategory::Variable,
53        "number" => HighlightCategory::Number,
54        "operator" => HighlightCategory::Operator,
55        "punctuation" => HighlightCategory::Punctuation,
56        _ => HighlightCategory::Other,
57    }
58}

60/// Concrete highlight service implementation using tree-sitter.
61///
62/// Manages a thread-safe `Highlighter` instance and specific language configurations
63/// to perform efficient, on-demand syntax highlighting.
64pub struct HighlightServiceImpl {
65    highlighter: Arc<tokio::sync::Mutex<Highlighter>>,
66}
67
68impl HighlightServiceImpl {
69    /// Creates a syntax highlight service with an internal tree-sitter highlighter.
70    #[must_use]
71    pub fn new() -> Self {
72        Self {
73            highlighter: Arc::new(tokio::sync::Mutex::new(Highlighter::new())),
74        }
75    }

77    /// Get language configuration for supported languages
78    fn get_language_config(language: &str) -> Result<HighlightLanguageConfig, HighlightError> {
79        let normalized = language.trim().to_lowercase();
80
81        match normalized.as_str() {
82            "rust" => Ok(HighlightLanguageConfig::new(
83                "rust",
84                tree_sitter_rust::LANGUAGE.into(),
85                tree_sitter_rust::HIGHLIGHTS_QUERY,
86            )),
87            "python" => Ok(HighlightLanguageConfig::new(
88                "python",
89                tree_sitter_python::LANGUAGE.into(),
90                tree_sitter_python::HIGHLIGHTS_QUERY,
91            )),
92            "javascript" | "js" => Ok(HighlightLanguageConfig::new(
93                "javascript",
94                tree_sitter_javascript::LANGUAGE.into(),
95                tree_sitter_javascript::HIGHLIGHT_QUERY,
96            )),
97            "typescript" | "ts" => Ok(HighlightLanguageConfig::new(
98                "typescript",
99                tree_sitter_typescript::LANGUAGE_TYPESCRIPT.into(),
100                tree_sitter_typescript::HIGHLIGHTS_QUERY,
101            )),
102            "tsx" => Ok(HighlightLanguageConfig::new(
103                "tsx",
104                tree_sitter_typescript::LANGUAGE_TSX.into(),
105                tree_sitter_typescript::HIGHLIGHTS_QUERY,
106            )),
107            "go" => Ok(HighlightLanguageConfig::new(
108                "go",
109                tree_sitter_go::LANGUAGE.into(),
110                tree_sitter_go::HIGHLIGHTS_QUERY,
111            )),
112            "java" => Ok(HighlightLanguageConfig::new(
113                "java",
114                tree_sitter_java::LANGUAGE.into(),
115                tree_sitter_java::HIGHLIGHTS_QUERY,
116            )),
117            "c" => Ok(HighlightLanguageConfig::new(
118                "c",
119                tree_sitter_c::LANGUAGE.into(),
120                tree_sitter_c::HIGHLIGHT_QUERY,
121            )),
122            "cpp" | "c++" => Ok(HighlightLanguageConfig::new(
123                "cpp",
124                tree_sitter_cpp::LANGUAGE.into(),
125                tree_sitter_cpp::HIGHLIGHT_QUERY,
126            )),
127            "ruby" => Ok(HighlightLanguageConfig::new(
128                "ruby",
129                tree_sitter_ruby::LANGUAGE.into(),
130                tree_sitter_ruby::HIGHLIGHTS_QUERY,
131            )),
132            "php" => Ok(HighlightLanguageConfig::new(
133                "php",
134                tree_sitter_php::LANGUAGE_PHP.into(),
135                tree_sitter_php::HIGHLIGHTS_QUERY,

146    /// Create highlight configuration from language config
147    fn create_highlight_config(
148        lang_config: HighlightLanguageConfig,
149    ) -> Result<HighlightConfiguration, HighlightError> {
150        let mut config = HighlightConfiguration::new(
151            lang_config.language,
152            lang_config.name,
153            lang_config.highlights_query,
154            "",
155            "",
156        )
157        .map_err(|e| HighlightError::ConfigurationError(e.to_string()))?;
158
159        config.configure(&HIGHLIGHT_NAMES);
160        Ok(config)
161    }
162
163    /// Highlight code using tree-sitter
164    fn highlight_code_internal(
165        &self,
166        code: &str,
167        language: &str,
168    ) -> Result<HighlightedCode, HighlightError> {
169        if code.is_empty() {
170            return Ok(HighlightedCode {
171                original: code.to_owned(),
172                spans: vec![],
173                language: language.to_owned(),
174            });
175        }
176
177        let config_err = |op: &str, e: HighlightError| {
178            HighlightError::ConfigurationError(format!("failed to {op} for '{language}': {e}"))
179        };
180
181        let lang_config = Self::get_language_config(language)?;
182
183        let config = Self::create_highlight_config(lang_config)
184            .map_err(|e| config_err("create highlight config", e))?;
185
186        let mut highlighter = self.highlighter.blocking_lock();
187
188        let highlights = highlighter
189            .highlight(&config, code.as_bytes(), None, |_: &str| None)
190            .map_err(|e| HighlightError::HighlightingFailed(e.to_string()))?;
191
192        let mut spans = Vec::new();
193        let mut position = 0;
194        let mut open_spans: Vec<(usize, &str)> = Vec::new();
195
196        for event in highlights {
197            match event {
198                Ok(HighlightEvent::Source { end, .. }) => {
199                    position = end;
200                }
201                Ok(HighlightEvent::HighlightStart(Highlight(highlight))) => {
202                    if let Some(class_name) = HIGHLIGHT_NAMES.get(highlight) {
203                        open_spans.push((position, class_name));
204                    }
205                }
206                Ok(HighlightEvent::HighlightEnd) => {
207                    if let Some((start, class_name)) = open_spans.pop() {
208                        let category = map_highlight_to_category(class_name);
209                        spans.push(HighlightSpan {
210                            start,
211                            end: position,
212                            category,
213                        });
214                    }
215                }
216                Err(e) => {
217                    return Err(HighlightError::HighlightingFailed(e.to_string()));
218                }
219            }
220        }
221
222        Ok(HighlightedCode {
223            original: code.to_owned(),
224            spans,
225            language: language.to_owned(),
226        })
227    }
228}

230impl Default for HighlightServiceImpl {
231    fn default() -> Self {
232        Self::new()
233    }
234}

236#[async_trait::async_trait]
237impl HighlightServiceInterface for HighlightServiceImpl {
238    async fn highlight(&self, code: &str, language: &str) -> mcb_domain::Result<HighlightedCode> {
239        let code = code.to_owned();
240        let language = language.to_owned();
241        let highlighter = Arc::clone(&self.highlighter);
242
243        let result = tokio::task::spawn_blocking(move || {
244            let service = HighlightServiceImpl { highlighter };
245            service.highlight_code_internal(&code, &language)
246        })
247        .await
248        .map_err(|e| HighlightError::HighlightingFailed(format!("Blocking task failed: {e}")))?;
249
250        result.map_err(mcb_domain::Error::from)
251    }
252}
