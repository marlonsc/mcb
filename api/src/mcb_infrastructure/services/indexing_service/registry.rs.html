24/// Build the `IndexingService` from the application registry.
25///
26/// # Errors
27///
28/// Returns an error if any required dependency cannot be resolved from the registry.
29fn build_indexing_service_from_registry(
30    context: &dyn std::any::Any,
31) -> Result<Arc<dyn IndexingServiceInterface>> {
32    let ctx = context
33        .downcast_ref::<mcb_domain::registry::ServiceResolutionContext>()
34        .ok_or_else(|| {
35            mcb_domain::error::Error::internal(
36                "Indexing registry builder requires ServiceResolutionContext",
37            )
38        })?;
39
40    let app_config = ctx
41        .config
42        .downcast_ref::<crate::config::app::AppConfig>()
43        .ok_or_else(|| {
44            mcb_domain::error::Error::internal(
45                "Indexing service requires AppConfig in resolution context",
46            )
47        })?;
48    let db = Arc::clone(&ctx.db);
49
50    let context_service = resolve_context_service(context)?;
51    let language_chunker =
52        resolve_language_provider(&LanguageProviderConfig::new(DEFAULT_LANGUAGE_PROVIDER))?;
53
54    // Use "seaorm" — the actual registry provider — not the user-facing config name.
55    let repositories =
56        resolve_database_repositories(DEFAULT_DATABASE_PROVIDER, db, DEFAULT_NAMESPACE.to_owned())?;
57
58    let indexing_ops: Arc<dyn mcb_domain::ports::IndexingOperationsInterface> =
59        resolve_indexing_operations_provider(&IndexingOperationsProviderConfig::new(
60            DEFAULT_INDEXING_OP_PROVIDER,
61        ))?;
62    let event_bus = Arc::clone(&ctx.event_bus);
63
64    Ok(Arc::new(
65        IndexingServiceImpl::new_with_file_hash_repository(IndexingServiceWithHashDeps {
66            service: IndexingServiceDeps {
67                context_service,
68                language_chunker,
69                indexing_ops,
70                event_bus,
71                supported_extensions: app_config.mcp.indexing.supported_extensions.clone(),
72            },
73            file_hash_repository: repositories.file_hash,
74        }),
75    ))
76}
77
78mcb_domain::register_service!(
79    mcb_utils::constants::SERVICE_NAME_INDEXING,
80    ServiceBuilder::Indexing(build_indexing_service_from_registry),
81);
