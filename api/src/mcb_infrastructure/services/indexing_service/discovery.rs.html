1//! File discovery and filtering for indexing.
2//!
3//! This module handles recursive directory traversal, file filtering,
4//! and extension validation during the indexing process.
5
6use std::path::Path;
7
8use ignore::WalkBuilder;
9
10use mcb_utils::constants::use_cases::SKIP_DIRS;
11
12use super::{IndexingProgress, IndexingServiceImpl};
13
14impl IndexingServiceImpl {
15    /// Discover files recursively from a path
16    pub(crate) async fn discover_files(
17        &self,
18        path: &Path,
19        progress: &mut IndexingProgress,
20    ) -> Vec<std::path::PathBuf> {
21        let mut files = Vec::new();
22        let walker = WalkBuilder::new(path)
23            .hidden(false)
24            .filter_entry(|entry| {
25                if !entry.file_type().is_some_and(|ft| ft.is_dir()) {
26                    return true;
27                }
28
29                entry
30                    .file_name()
31                    .to_str()
32                    .is_none_or(|name| !SKIP_DIRS.contains(&name))
33            })
34            .build();
35
36        for entry_result in walker {
37            match entry_result {
38                Ok(entry) => {
39                    if entry.file_type().is_some_and(|ft| ft.is_file())
40                        && self.is_supported_file(entry.path())
41                    {
42                        files.push(entry.path().to_path_buf());
43                    }
44                }
45                Err(e) => {
46                    progress.record_error("Failed to read directory entry", path, e);
47                }
48            }
49        }
50
51        files
52    }
53
54    /// Check if file has a supported extension
55    fn is_supported_file(&self, path: &Path) -> bool {
56        path.extension()
57            .and_then(|ext| ext.to_str())
58            .is_some_and(|ext| {
59                let ext_lower = ext.to_ascii_lowercase();
60                self.supported_extensions.contains(&ext_lower)
61            })
62    }
63}
