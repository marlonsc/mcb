1//! Observation storage and management operations.
2//!
3//! Handles storing observations in both relational and vector stores,
4//! with deduplication via content hashing.
5

12    METADATA_KEY_CONTENT, METADATA_KEY_FILE_PATH, METADATA_KEY_SESSION_ID, METADATA_KEY_START_LINE,

14};
15use mcb_utils::utils::id;
16use mcb_utils::utils::id::compute_content_hash;

23/// Owned inputs describing an observation to be stored.
24pub(crate) struct ObservationInput {
25    pub project_id: String,
26    pub content: String,
27    pub r#type: ObservationType,
28    pub tags: Vec<String>,
29    pub metadata: ObservationMetadata,
30}
31
32/// Result of persisting the observation embedding into the vector store.
33///
34/// Carries the data needed to finalize the relational record and to roll the
35/// vectors back if the relational write fails.
36struct VectorWriteOutcome<'a> {
37    content_hash: String,
38    embedding_id: Option<String>,
39    collection_id: &'a CollectionId,
40    ids: Vec<String>,
41}
42
43impl MemoryServiceImpl {
44    fn build_vector_metadata(input: &ObservationInput) -> HashMap<String, serde_json::Value> {
45        let mut vector_metadata = HashMap::new();
46        vector_metadata.insert(
47            METADATA_KEY_CONTENT.to_owned(),
48            serde_json::Value::String(input.content.clone()),
49        );
50        vector_metadata.insert(
51            METADATA_KEY_TYPE.to_owned(),
52            serde_json::Value::String(input.r#type.as_str().to_owned()),
53        );
54        vector_metadata.insert(METADATA_KEY_TAGS.to_owned(), serde_json::json!(input.tags));
55        vector_metadata.insert(
56            "project_id".to_owned(),
57            serde_json::Value::String(input.project_id.clone()),
58        );
59
60        if let Some(session_id) = &input.metadata.session_id {
61            vector_metadata.insert(
62                METADATA_KEY_SESSION_ID.to_owned(),
63                serde_json::Value::String(session_id.clone()),
64            );
65        }
66
67        vector_metadata.insert(
68            METADATA_KEY_FILE_PATH.to_owned(),
69            serde_json::Value::String(
70                input
71                    .metadata
72                    .file_path
73                    .clone()
74                    .unwrap_or_else(|| "memory".to_owned()),
75            ),
76        );
77        vector_metadata.insert(
78            METADATA_KEY_START_LINE.to_owned(),
79            serde_json::Value::Number(serde_json::Number::from(0)),
80        );
81
82        vector_metadata
83    }
84
85    async fn insert_into_vector_store(
86        &self,
87        input: &ObservationInput,
88        collection_id: &CollectionId,
89    ) -> Result<Vec<String>> {
90        let embedding = self.embedding_provider.embed(&input.content).await?;
91        let vector_metadata = Self::build_vector_metadata(input);
92
93        self.vector_store
94            .insert_vectors(collection_id, &[embedding], vec![vector_metadata])
95            .await
96    }
97
98    async fn create_and_store_observation(
99        &self,
100        input: ObservationInput,
101        vector: VectorWriteOutcome<'_>,
102    ) -> Result<String> {
103        let observation = Observation {
104            id: id::generate().to_string(),
105            project_id: input.project_id,
106            content: input.content,
107            content_hash: vector.content_hash,
108            tags: input.tags,
109            r#type: input.r#type,
110            metadata: input.metadata,
111            created_at: domain_time::epoch_secs_i64()?,
112            embedding_id: vector.embedding_id,
113        };
114
115        if let Err(err) = self.repository.store_observation(&observation).await {
116            let _ = self
117                .vector_store
118                .delete_vectors(vector.collection_id, &vector.ids)
119                .await;
120            return Err(err);
121        }
122
123        Ok(observation.id)
124    }
125
126    pub(crate) async fn store_observation_impl(
127        &self,
128        input: ObservationInput,
129    ) -> Result<(String, bool)> {
130        if input.project_id.trim().is_empty() {
131            return Err(mcb_domain::error::Error::invalid_argument(
132                "Project ID cannot be empty for memory storage",
133            ));
134        }
135
136        let content_hash = compute_content_hash(&input.content);
137
138        if let Some(existing) = self.repository.find_by_hash(&content_hash).await? {
139            return Ok((existing.id, true));
140        }
141
142        let collection_id =
143            CollectionId::from_uuid(id::deterministic("collection", MEMORY_COLLECTION_NAME));
144
145        let ids = self
146            .insert_into_vector_store(&input, &collection_id)
147            .await?;
148        let vector = VectorWriteOutcome {
149            content_hash,
150            embedding_id: ids.first().cloned(),
151            collection_id: &collection_id,
152            ids,
153        };
154
155        let id = self.create_and_store_observation(input, vector).await?;
156
157        Ok((id, false))
158    }
