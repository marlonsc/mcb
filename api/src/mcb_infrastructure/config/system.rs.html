89/// Event bus backend.
90#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Default)]
91#[serde(rename_all = "lowercase")]
92pub enum EventBusBackend {
93    /// In-process broadcast channel.
94    #[default]
95    #[serde(alias = "tokio", alias = "in_process")]
96    InProcess,
97}
98
99impl EventBusBackend {
100    /// Registry provider name for this backend.
101    #[must_use]
102    pub fn provider_name(&self) -> &'static str {
103        match self {
104            Self::InProcess => "inprocess",
105        }
106    }
107}
108
109/// Event bus configuration.
110#[derive(Debug, Clone, Serialize, Deserialize)]
111#[serde(deny_unknown_fields)]
112pub struct EventBusConfig {
113    /// Event bus backend to use.
114    pub provider: EventBusBackend,
115    /// Buffer capacity for the event bus.
116    pub capacity: usize,
117    /// Connection timeout in milliseconds.
118    pub connection_timeout_ms: u64,
119    /// Maximum reconnection attempts.
120    pub max_reconnect_attempts: u32,
121}
122
123impl EventBusConfig {
124    /// Creates default in-process event bus configuration.
125    #[must_use]
126    pub fn in_process() -> Self {
127        Self {
128            provider: EventBusBackend::InProcess,
129            capacity: EVENT_BUS_DEFAULT_CAPACITY,
130            connection_timeout_ms: EVENT_BUS_CONNECTION_TIMEOUT_MS,
131            max_reconnect_attempts: EVENT_BUS_MAX_RECONNECT_ATTEMPTS,
132        }
133    }
134
135    /// Creates in-process event bus configuration with custom buffer capacity.
136    #[must_use]
137    pub fn with_capacity(capacity: usize) -> Self {
138        Self {
139            provider: EventBusBackend::InProcess,
140            capacity,
141            connection_timeout_ms: EVENT_BUS_CONNECTION_TIMEOUT_MS,
142            max_reconnect_attempts: EVENT_BUS_MAX_RECONNECT_ATTEMPTS,
143        }
144    }
145}
