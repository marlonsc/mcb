13/// Loco YAML configuration provider.
14///
15/// Registered via `#[linkme::distributed_slice]` in `CONFIG_PROVIDERS`.
16pub(crate) struct LocoYamlConfigProvider;
17
18impl ConfigProvider for LocoYamlConfigProvider {
19    fn load_config(&self) -> Result<Box<dyn Any + Send + Sync>> {
20        let config = loader::load_app_config()?;
21        Ok(Box::new(config))
22    }
23
24    fn deserialize_from_value(&self, settings: &dyn Any) -> Result<Box<dyn Any + Send + Sync>> {
25        let json_value = settings
26            .downcast_ref::<serde_json::Value>()
27            .ok_or_else(|| {
28                Error::internal(
29                    "ConfigProvider::deserialize_from_value: expected serde_json::Value",
30                )
31            })?;
32
33        let config: app::AppConfig = serde_json::from_value(json_value.clone()).map_err(|e| {
34            Error::config_with_source("Failed to deserialize AppConfig from JSON", e)
35        })?;
36
37        validation::validate_app_config(&config)?;
38
39        Ok(Box::new(config))
40    }
41
42    fn validate_config(&self, config: &dyn Any) -> Result<()> {
43        let app_config = config.downcast_ref::<app::AppConfig>().ok_or_else(|| {
44            Error::internal("ConfigProvider::validate_config: expected AppConfig")
45        })?;
46        validation::validate_app_config(app_config)
47    }
48
49    fn provider_name(&self) -> &str {
50        mcb_utils::constants::DEFAULT_CONFIG_PROVIDER
51    }
52}
53
54mcb_domain::register_config_provider!(
55    mcb_utils::constants::DEFAULT_CONFIG_PROVIDER,
56    "YAML configuration loader following Loco conventions",
57    |_config| Ok(Arc::new(LocoYamlConfigProvider))
58);
