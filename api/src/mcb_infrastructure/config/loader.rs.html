1//! Production-mirroring configuration loader.
2//!
3//! Loads `AppConfig` using the **same deserialization path as production**:
4//! YAML file → parse → extract `settings` → JSON conversion →
5//! `serde_json::from_value::<AppConfig>()` → `validate_app_config()`.
6//!
7//! This ensures tests load configuration identically to the Loco runtime
8//! (see `crates/mcb/src/initializers/mcp_server.rs`).
9
10use std::path::{Path, PathBuf};
11
12use mcb_domain::error::{Error, Result};
13
14use super::app::AppConfig;
15
16/// Load `AppConfig` using the production Loco deserialization path.
17///
18/// 1. Reads `LOCO_ENV` (defaults to `"test"`)
19/// 2. Walks ancestors from `CARGO_MANIFEST_DIR` looking for
20///    `config/{env}.local.yaml` then `config/{env}.yaml`
21/// 3. Parses YAML, extracts `settings:` key
22/// 4. Converts YAML → JSON via `serde_json::to_value`
23/// 5. Deserializes via `serde_json::from_value::<AppConfig>()` — the production path
24/// 6. Validates via `validate_app_config()`
25///
26/// # Errors
27///
28/// Returns an error if the config file is missing, unreadable, or invalid.
29pub fn load_app_config() -> Result<AppConfig> {
30    let env_name = std::env::var("LOCO_ENV").unwrap_or_else(|_| "test".to_owned());
31
32    let filenames = [format!("{env_name}.local.yaml"), format!("{env_name}.yaml")];
33
34    let manifest_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR"));
35    for dir in manifest_dir.ancestors() {
36        for filename in &filenames {
37            let candidate = dir.join("config").join(filename);
38            if candidate.exists() {
39                return load_from_path(&candidate);
40            }
41        }
42    }
43
44    Err(Error::ConfigMissing(format!(
45        "No config file found for env '{env_name}'"
46    )))
47}
48
49/// Load and deserialize `AppConfig` from a specific YAML file path.
50fn load_from_path(path: &Path) -> Result<AppConfig> {
51    let content = std::fs::read_to_string(path)
52        .map_err(|e| Error::config_with_source(format!("Failed to read {}", path.display()), e))?;
53
54    let yaml: serde_yaml::Value = serde_yaml::from_str(&content)
55        .map_err(|e| Error::config_with_source("Failed to parse YAML", e))?;
56
57    let settings = yaml
58        .get("settings")
59        .ok_or_else(|| Error::ConfigMissing("No 'settings' key in config".into()))?;
60
61    // Production path: YAML value → JSON value → serde_json::from_value
62    // Mirrors crates/mcb/src/initializers/mcp_server.rs:44-45
63    let json_settings = serde_json::to_value(settings)
64        .map_err(|e| Error::config_with_source("Failed to convert YAML settings to JSON", e))?;
65
66    let config: AppConfig = serde_json::from_value(json_settings)
67        .map_err(|e| Error::config_with_source("Failed to deserialize AppConfig", e))?;
68
69    super::validation::validate_app_config(&config)?;
70
71    Ok(config)
72}
