15/// In-process domain event bus backed by a broadcast channel.
16#[derive(Clone)]
17pub struct BroadcastEventBus {
18    sender: Arc<broadcast::Sender<DomainEvent>>,
19}
20
21impl BroadcastEventBus {
22    /// Create a new event bus with default buffer size.
23    #[must_use]
24    pub fn new() -> Self {
25        let (sender, _) = broadcast::channel(EVENT_BUS_BUFFER_SIZE);
26        Self {
27            sender: Arc::new(sender),
28        }
29    }
30}
31
32impl Default for BroadcastEventBus {
33    fn default() -> Self {
34        Self::new()
35    }
36}
37
38impl std::fmt::Debug for BroadcastEventBus {
39    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
40        f.debug_struct("BroadcastEventBus")
41            .field("subscribers", &self.sender.receiver_count())
42            .finish()
43    }
44}
45
46#[async_trait]
47impl EventBusProvider for BroadcastEventBus {
48    async fn publish_event(&self, event: DomainEvent) -> Result<()> {
49        match self.sender.send(event) {
50            Ok(count) => mcb_domain::debug!(
51                "event_bus",
52                &format!("Published event to {count} subscribers")
53            ),
54            Err(_) => mcb_domain::debug!("event_bus", "Published event but no subscribers"),
55        }
56        Ok(())
57    }
58
59    async fn subscribe_events(&self) -> Result<DomainEventStream> {
60        let receiver = self.sender.subscribe();
61        let stream = stream::unfold(receiver, |mut rx| async move {
62            loop {
63                match rx.recv().await {
64                    Ok(event) => return Some((event, rx)),
65                    Err(broadcast::error::RecvError::Lagged(n)) => {
66                        mcb_domain::warn!(
67                            "event_bus",
68                            "Event stream lagged",
69                            &format!("{n} events")
70                        );
71                        continue;
72                    }
73                    Err(broadcast::error::RecvError::Closed) => return None,
74                }
75            }
76        });
77        Ok(Box::pin(stream))
78    }
79
80    fn has_subscribers(&self) -> bool {
81        self.sender.receiver_count() > 0
82    }
83
84    async fn publish(&self, topic: &str, payload: &[u8]) -> Result<()> {
85        let event: DomainEvent = match serde_json::from_slice(payload) {
86            Ok(e) => e,
87            Err(e) => {
88                mcb_domain::warn!(
89                    "event_bus",
90                    "Failed to deserialize event payload for topic",
91                    &format!("topic={topic} error={e}")
92                );
93                return Ok(());
94            }
95        };
96        mcb_domain::debug!(
97            "event_bus",
98            &format!("Publishing event to topic '{topic}': {event:?}")
99        );
100        self.publish_event(event).await
101    }
102
103    async fn subscribe(&self, topic: &str) -> Result<String> {
104        Ok(format!("broadcast-{topic}-{}", id::generate()))
105    }
106}
