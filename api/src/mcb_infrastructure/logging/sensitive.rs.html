1//! Shim that delegates to the global `OperationLogger` (set by catalog).
2//! Also provides the domain log-facade shim so `mcb_domain::error!` etc. route here.
3
4use std::sync::Arc;
5use std::sync::OnceLock;
6
7use mcb_domain::ports::{LogLevel, OperationLogger};
8
9static GLOBAL_LOGGER: OnceLock<Arc<dyn OperationLogger>> = OnceLock::new();
10
11/// Sets the global operation logger. Called from catalog after building the provider.
12pub fn set_global_operation_logger(logger: Arc<dyn OperationLogger>) {
13    let _ = GLOBAL_LOGGER.set(logger);
14}
15
16/// Forwards domain macro calls to the global logger (registered with `set_log_fn`).
17pub fn forward(
18    level: LogLevel,
19    context: &str,
20    message: &str,
21    detail: Option<&dyn std::fmt::Display>,
22) {
23    dispatch(level, context, message, detail);
24}
25
26/// Logs via the global logger at the given level, or fallback (message only).
27fn dispatch(level: LogLevel, context: &str, message: &str, detail: Option<&dyn std::fmt::Display>) {
28    if let Some(logger) = GLOBAL_LOGGER.get() {
29        logger.log(level, context, message, detail);
30    } else {
31        match level {
32            LogLevel::Error => tracing::error!(context = %context, "{}", message),
33            LogLevel::Warn => tracing::warn!(context = %context, "{}", message),
34            LogLevel::Info => tracing::info!(context = %context, "{}", message),
35            LogLevel::Debug => tracing::debug!(context = %context, "{}", message),
36            LogLevel::Trace => tracing::trace!(context = %context, "{}", message),
37        }
38    }
39}
