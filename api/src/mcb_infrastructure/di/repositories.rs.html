4//! Repository factories for standalone/server composition.
5//!
6//! Provides default repository implementations so that the server layer does not
7//! import concrete providers directly (CA006).
8//!
9//! Consumer crates (mcb-server tests, golden tests, etc.) MUST use these
10//! wrappers instead of importing from `mcb-providers` directly.
11
12use std::path::PathBuf;
13use std::sync::Arc;
14
15use mcb_domain::error::Result;
16use mcb_domain::ports::{DatabaseExecutor, MemoryRepository, VcsEntityRepository};
17use mcb_providers::database;
18
19/// Creates a VCS entity repository backed by the provided database executor.
20///
21/// Centralizes repository instantiation in the infrastructure layer.
22pub fn create_vcs_entity_repository(
23    executor: Arc<dyn DatabaseExecutor>,
24) -> Arc<dyn VcsEntityRepository> {
25    database::create_vcs_entity_repository_from_executor(executor)
26}
27
28/// Create a file-backed memory repository.
29///
30/// Wraps the provider factory so consumer crates never import `mcb_providers`.
31///
32/// # Errors
33///
34/// Returns an error if the database connection or schema initialization fails.
35pub async fn create_memory_repository(path: PathBuf) -> Result<Arc<dyn MemoryRepository>> {
36    database::create_memory_repository(path).await
37}
38
39/// Create a file-backed memory repository and its database executor.
40///
41/// Wraps the provider factory so consumer crates never import `mcb_providers`.
42///
43/// # Errors
44///
45/// Returns an error if the database connection or schema initialization fails.
46pub async fn create_memory_repository_with_executor(
47    path: PathBuf,
48) -> Result<(Arc<dyn MemoryRepository>, Arc<dyn DatabaseExecutor>)> {
49    database::create_memory_repository_with_executor(path).await
50}
