4//! Generic Provider Handle - Runtime-swappable provider wrapper
5//!
6//! A single generic implementation replacing four specific handle types.
7//! Wraps providers in `RwLock` to allow runtime reconfiguration via admin API.
8//!
9//! ## Pattern
10//!
11//! ```text
12//! linkme registry → Resolver → Handle<T> (RwLock) → Domain Services
13//!                      ↑
14//!              AdminService.switch_provider()
15//! ```
16
17use std::sync::{Arc, RwLock};
18
19/// Generic handle for runtime-swappable providers
20///
21/// Wraps any provider type in a `RwLock`, allowing admin API to switch
22/// providers without restarting the application.
23///
24/// # Type Parameters
25///
26/// * `T` - Provider trait type (e.g., `dyn EmbeddingProvider`)
27///
28/// # Example
29///
30/// ```
31/// use std::sync::Arc;
32/// use mcb_infrastructure::di::Handle;
33///
34/// // Handle works with any Send + Sync trait object
35/// trait MyTrait: Send + Sync {
36///     fn name(&self) -> &str;
37/// }
38///
39/// struct MyImpl;
40/// impl MyTrait for MyImpl {
41///     fn name(&self) -> &str { "impl" }
42/// }
43///
44/// // Create handle with initial provider
45/// let handle: Handle<dyn MyTrait> = Handle::new(Arc::new(MyImpl));
46///
47/// // Get current provider
48/// let provider = handle.get();
49/// assert_eq!(provider.name(), "impl");
50/// ```
51pub struct Handle<T: ?Sized + Send + Sync> {
52    inner: RwLock<Arc<T>>,
53}
54
55impl<T: ?Sized + Send + Sync> Handle<T> {
56    /// Create a new handle with an initial provider
57    pub fn new(provider: Arc<T>) -> Self {
58        Self {
59            inner: RwLock::new(provider),
60        }
61    }
62
63    /// Get the current provider
64    ///
65    /// Returns a cloned Arc to the current provider, allowing shared access.
66    pub fn get(&self) -> Arc<T> {
67        self.inner
68            .read()
69            .unwrap_or_else(|poisoned| {
70                mcb_domain::error!(
71                    "handle",
72                    "lock poisoned, recovering with poisoned value",
73                    &"action = get"
74                );
75                poisoned.into_inner()
76            })
77            .clone()
78    }
79
80    /// Set a new provider (used by admin service)
81    ///
82    /// Replaces the current provider with a new one. Existing references
83    /// to the old provider remain valid until dropped.
84    pub fn set(&self, new_provider: Arc<T>) {
85        *self.inner.write().unwrap_or_else(|poisoned| {
86            mcb_domain::error!(
87                "handle",
88                "lock poisoned during set, recovering",
89                &"action = set"
90            );
91            poisoned.into_inner()
92        }) = new_provider;
93    }
94}
95
96impl<T: ?Sized + Send + Sync> std::fmt::Debug for Handle<T> {
97    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
98        f.debug_struct("Handle").finish()
99    }
100}
