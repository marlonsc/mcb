12/// Implement `ProviderResolver<P, C>` for a concrete resolver type
13///
14/// Delegates all trait methods to the resolver's inherent methods.
15macro_rules! impl_provider_resolver {
16    ($resolver:ty, $provider:ty, $config:ty) => {
17        impl ProviderResolver<$provider, $config> for $resolver {
18            fn resolve_from_config(&self) -> mcb_domain::error::Result<Arc<$provider>> {
19                <$resolver>::resolve_from_config(self)
20            }
21
22            fn resolve_from_override(
23                &self,
24                config: &$config,
25            ) -> mcb_domain::error::Result<Arc<$provider>> {
26                <$resolver>::resolve_from_override(self, config)
27            }
28
29            fn list_available(&self) -> Vec<(&'static str, &'static str)> {
30                <$resolver>::list_available(self)
31            }
32        }
33    };
34}

40/// Implement an admin interface trait for a concrete admin service type
41///
42/// Two variants:
43/// - Basic: `list_providers`, `switch_provider`, `reload_from_config`
44/// - `with_current_provider`: adds `current_provider()` via handle
45macro_rules! impl_admin_interface {
46    ($service:ty, $trait:ty, $config:ty) => {
47        impl $trait for $service {
48            fn list_providers(&self) -> Vec<ProviderInfo> {
49                AdminService::list_providers(self)
50            }
51
52            fn switch_provider(&self, config: $config) -> Result<(), String> {
53                AdminService::switch_provider(self, &config)
54            }
55
56            fn reload_from_config(&self) -> Result<(), String> {
57                AdminService::reload_from_config(self)
58            }
59        }
60    };
61    ($service:ty, $trait:ty, $config:ty, with_current_provider) => {
62        impl $trait for $service {
63            fn list_providers(&self) -> Vec<ProviderInfo> {
64                AdminService::list_providers(self)
65            }
66
67            fn current_provider(&self) -> String {
68                self.handle.get().provider_name().to_owned()
69            }
70
71            fn switch_provider(&self, config: $config) -> Result<(), String> {
72                AdminService::switch_provider(self, &config)
73            }
74
75            fn reload_from_config(&self) -> Result<(), String> {
76                AdminService::reload_from_config(self)
77            }
78        }
79    };
80}

86/// Generate shared `AppContext` test infrastructure for a test binary.
87///
88/// Expands to `try_shared_app_context()`, `shared_app_context()`, and
89/// `shared_fastembed_test_cache_dir()` functions backed by process-wide
90/// `OnceLock` statics so the ONNX embedding model is loaded only once.
91///
92/// Config is loaded from `config/default.toml` + env overrides via
93/// [`TestConfigBuilder`](crate::config::TestConfigBuilder), then the
94/// database path and embedding cache are overridden for test isolation.
95///
96/// ## Usage
97///
98/// ```ignore
99/// extern crate mcb_providers;
100/// mcb_infrastructure::define_shared_test_context!("my-test.db");
101/// ```
102#[macro_export]
103macro_rules! define_shared_test_context {
104    ($db_name:literal) => {
105        /// Process-wide shared `AppContext`, or `None` when the ONNX model is
106        /// unavailable (offline / CI without model cache) or the init thread
107        /// panics (e.g. `ort` runtime initialization failure).
108        ///
109        /// Config is loaded from `default.toml` via `TestConfigBuilder` with
110        /// test-specific overrides (temp database, shared embedding cache).
111        pub fn try_shared_app_context() -> Option<&'static $crate::di::bootstrap::AppContext> {
112            static CTX: std::sync::OnceLock<Option<$crate::di::bootstrap::AppContext>> =
113                std::sync::OnceLock::new();
114
115            CTX.get_or_init(|| {
116                std::thread::spawn(|| {
117                    let rt = tokio::runtime::Runtime::new()
118                        .unwrap_or_else(|e| panic!("create init runtime: {e}"));
119                    let result = rt.block_on(async {
120                        let (config, _temp_dir) = $crate::config::TestConfigBuilder::new()?
121                            .with_temp_db($db_name)?
122                            .with_fastembed_shared_cache()?
123                            .build()?;
124
125                        // Keep the temp dir alive for the process lifetime
126                        if let Some(td) = _temp_dir {
127                            let _ = td.keep();
128                        }
129
130                        $crate::di::bootstrap::init_app(config).await
131                    });
132                    let _rt = std::mem::ManuallyDrop::new(rt);
133
134                    match result {
135                        Ok(ctx) => Some(ctx),
136                        Err(e) => {
137                            let msg = e.to_string();
138                            if msg.contains("model.onnx")
139                                || msg.contains("Failed to initialize FastEmbed")
140                            {
141                                None
142                            } else {
143                                panic!("shared init_app failed: {e}");
144                            }
145                        }
146                    }
147                })
148                .join()
149                .unwrap_or_else(|_| {
150                    eprintln!(
151                        "WARN: shared context init thread panicked \
152                         (ONNX/ort model likely unavailable — skipping dependent tests)"
153                    );
154                    None
155                })
156            })
157            .as_ref()
158        }
159
160        /// Convenience wrapper that panics when the ONNX model is unavailable.
161        ///
162        /// # Panics
163        ///
164        /// Panics if the shared `AppContext` could not be initialized.
165        pub fn shared_app_context() -> &'static $crate::di::bootstrap::AppContext {
166            try_shared_app_context()
167                .expect("shared AppContext init failed — ONNX model may be unavailable")
168        }
169
170        /// Persistent shared cache dir for the `FastEmbed` ONNX model.
171        ///
172        /// Ensures the model is downloaded once and reused across test invocations.
173        ///
174        /// # Panics
175        ///
176        /// Panics if the cache directory cannot be created.
177        pub fn shared_fastembed_test_cache_dir() -> std::path::PathBuf {
178            static DIR: std::sync::OnceLock<std::path::PathBuf> = std::sync::OnceLock::new();
179            DIR.get_or_init(|| {
180                let cache_dir = std::env::var_os("MCB_FASTEMBED_TEST_CACHE_DIR").map_or_else(
181                    || std::env::temp_dir().join("mcb-fastembed-test-cache"),
182                    std::path::PathBuf::from,
183                );
184                std::fs::create_dir_all(&cache_dir)
185                    .expect("create shared fastembed test cache dir");
186                cache_dir
187            })
188            .clone()
189        }
190    };
191}
