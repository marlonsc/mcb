1//!
2//! Logging facade: single registration point for the operation logger.
3//!
4//! Infra (or tests) call `set_log_fn` at startup so that macros `trace!` .. `error!`
5//! forward to the real implementation. Domain has no dependency on tracing or infra.
6
7use std::sync::OnceLock;
8
9use crate::ports::LogLevel;
10
11/// Function type that the infrastructure layer registers to handle log events.
12pub type LogFn = fn(LogLevel, &str, &str, Option<&dyn std::fmt::Display>);
13
14static LOG_FN: OnceLock<LogFn> = OnceLock::new();
15
16/// Registers the log implementation (e.g. infra shim). Call once at startup.
17pub fn set_log_fn(f: LogFn) {
18    let _ = LOG_FN.set(f);
19}
20
21/// Dispatches to the registered log function, or no-op if none set.
22#[inline]
23pub fn dispatch(
24    level: LogLevel,
25    context: &str,
26    message: &str,
27    detail: Option<&dyn std::fmt::Display>,
28) {
29    if let Some(f) = LOG_FN.get() {
30        f(level, context, message, detail);
31    }
32}
