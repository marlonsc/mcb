4/// Logical column type used by the canonical schema model.
5#[derive(Debug, Clone, PartialEq, Eq)]
6pub enum ColumnType {
7    /// UTF-8 text value.
8    Text,
9    /// Signed integer value.
10    Integer,
11    /// Floating-point value.
12    Real,
13    /// Boolean represented by backend-specific primitive.
14    Boolean,
15    /// Arbitrary binary payload.
16    Blob,
17    /// JSON-encoded text value.
18    Json,
19    /// UUID stored in backend-supported format.
20    Uuid,
21    /// Unix timestamp value.
22    Timestamp,
23}
24
25/// Canonical column definition.
26#[derive(Debug, Clone)]
27pub struct ColumnDef {
28    /// Column name.
29    pub name: String,
30    /// Logical value type.
31    pub type_: ColumnType,
32    /// Whether this column is part of the primary key.
33    pub primary_key: bool,
34    /// Whether this column has a unique constraint.
35    pub unique: bool,
36    /// Whether this column is non-nullable.
37    pub not_null: bool,
38    /// Whether this column auto-increments.
39    pub auto_increment: bool,
40}
41
42/// Canonical table definition.
43#[derive(Debug, Clone)]
44pub struct TableDef {
45    /// Table name.
46    pub name: String,
47    /// Ordered list of columns.
48    pub columns: Vec<ColumnDef>,
49}
50
51/// Canonical secondary index definition.
52#[derive(Debug, Clone)]
53pub struct IndexDef {
54    /// Index name.
55    pub name: String,
56    /// Table that owns the index.
57    pub table: String,
58    /// Ordered indexed columns.
59    pub columns: Vec<String>,
60}
61
62/// Full-text search virtual table definition.
63#[derive(Debug, Clone)]
64pub struct FtsDef {
65    /// Virtual table name.
66    pub virtual_table_name: String,
67    /// Content table used by the virtual table.
68    pub content_table: String,
69    /// Content columns indexed by FTS.
70    pub content_columns: Vec<String>,
71    /// Primary key column mirrored into the virtual table.
72    pub id_column: String,
73}
74
75/// Foreign key relationship definition.
76#[derive(Debug, Clone)]
77pub struct ForeignKeyDef {
78    /// Source table.
79    pub from_table: String,
80    /// Source column.
81    pub from_column: String,
82    /// Referenced table.
83    pub to_table: String,
84    /// Referenced column.
85    pub to_column: String,
86}
87
88/// Multi-column uniqueness constraint definition.
89#[derive(Debug, Clone)]
90pub struct UniqueConstraintDef {
91    /// Table that owns the constraint.
92    pub table: String,
93    /// Ordered columns in the unique set.
94    pub columns: Vec<String>,
95}

100/// Canonical database schema: all tables, FTS, indexes, FKs, unique constraints.
101///
102/// Single source of truth for the full persistence model.
103/// Each backend (`SQLite`, `PostgreSQL`, `MySQL`) implements [`SchemaDdlGenerator`]
104/// to produce dialect-specific DDL from this schema.
105#[derive(Debug, Clone)]
106pub struct Schema {
107    /// All physical tables in canonical creation order.
108    pub tables: Vec<TableDef>,
109    /// Optional full-text search definition.
110    pub fts: Option<FtsDef>,
111    /// Secondary indexes across all tables.
112    pub indexes: Vec<IndexDef>,
113    /// Foreign key relationships.
114    pub foreign_keys: Vec<ForeignKeyDef>,
115    /// Uniqueness constraints.
116    pub unique_constraints: Vec<UniqueConstraintDef>,
117}
118
119/// Port for generating DDL from the canonical schema.
120pub trait SchemaDdlGenerator: Send + Sync {
121    /// Generates backend-specific DDL statements from the canonical schema.
122    fn generate_ddl(&self, schema: &Schema) -> Vec<String>;
123}
