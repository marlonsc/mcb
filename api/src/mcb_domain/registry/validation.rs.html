4//! Validation Provider Registry
5//!
6//! Auto-registration system for validation providers using linkme distributed slices.
7//! Providers register themselves via `#[linkme::distributed_slice]` and are
8//! discovered at runtime.
9
10use std::collections::HashMap;
11use std::path::PathBuf;
12
13/// Configuration for validation provider creation
14///
15/// Contains all configuration options that a validation provider might need.
16/// Providers should use what they need and ignore the rest.
17#[derive(Debug, Clone, Default)]
18pub struct ValidationProviderConfig {
19    /// Provider name (e.g., "mcb-validate")
20    pub provider: String,
21    /// Workspace root for validation scans
22    pub workspace_root: Option<PathBuf>,
23    /// Validator names to run (None = all)
24    pub validators: Option<Vec<String>>,
25    /// Minimum severity to report (error, warning, info)
26    pub severity_filter: Option<String>,
27    /// Path patterns to exclude from validation
28    pub exclude_patterns: Option<Vec<String>>,
29    /// Maximum files to validate
30    pub max_files: Option<usize>,
31    /// Additional provider-specific configuration
32    pub extra: HashMap<String, String>,
33}
34
35crate::impl_config_builder!(ValidationProviderConfig {
36    /// Set workspace root
37    workspace_root: with_workspace_root(into PathBuf),
38    /// Set validators to run
39    validators: with_validators(Vec<String>),
40    /// Set severity filter
41    severity_filter: with_severity_filter(into String),
42    /// Set exclude patterns
43    exclude_patterns: with_exclude_patterns(Vec<String>),
44    /// Set maximum files to validate
45    max_files: with_max_files(usize),
46});
47
48crate::impl_registry!(
49    provider_trait: crate::ports::ValidationProvider,
50    config_type: ValidationProviderConfig,
51    entry_type: ValidationProviderEntry,
52    slice_name: VALIDATION_PROVIDERS,
53    resolve_fn: resolve_validation_provider,
54    list_fn: list_validation_providers
55);
