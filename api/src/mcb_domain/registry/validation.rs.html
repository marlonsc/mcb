4//! Validator Registry
5//!
6//! Auto-registration system for validators using linkme distributed slices.
7//! Validators register themselves via `#[linkme::distributed_slice(VALIDATOR_ENTRIES)]`
8//! in implementing crates (e.g. mcb-validate) and are discovered at runtime.
9
10use std::path::{Path, PathBuf};
11
12use crate::error::Result;
13use crate::ports::services::validation_service::{ValidationReport, ViolationEntry};
14use crate::ports::validation::Validator;

16// ============================================================================
17// Validator entries (linkme-discovered validators)
18// ============================================================================
19
20/// Registry entry for a single validator (e.g. `clean_architecture`, `quality`).
21///
22/// Validators register via `#[linkme::distributed_slice(VALIDATOR_ENTRIES)]`
23/// in implementing crates (e.g. mcb-validate).
24pub struct ValidatorEntry {
25    /// Unique validator name (e.g. `"clean_architecture"`).
26    pub name: &'static str,
27    /// Human-readable description
28    pub description: &'static str,
29    /// Build a validator instance for the given workspace root
30    pub build: fn(PathBuf) -> Result<Box<dyn Validator>>,
31}
32
33#[linkme::distributed_slice]
34/// Distributed slice of all registered validators.
35pub static VALIDATOR_ENTRIES: [ValidatorEntry] = [..];
36
37/// List all registered validator (name, description) pairs.
38#[must_use]
39pub fn list_validator_entries() -> Vec<(&'static str, &'static str)> {
40    VALIDATOR_ENTRIES
41        .iter()
42        .map(|e| (e.name, e.description))
43        .collect()
44}
45
46/// List all registered validator names (single source of truth for CLI and handlers).
47#[must_use]
48pub fn list_validator_names() -> Vec<String> {
49    // Sort for a deterministic order: `linkme` registration order is not stable
50    // across builds/link units, which otherwise makes consumers (list_rules,
51    // snapshots, error messages) non-deterministic.
52    let mut names: Vec<String> = VALIDATOR_ENTRIES
53        .iter()
54        .map(|e| e.name.to_owned())
55        .collect();
56    names.sort_unstable();
57    names
58}
59
60/// Return the number of registered validators.
61#[must_use]
62pub fn validator_count() -> usize {
63    VALIDATOR_ENTRIES.len()
64}
65
66/// Build all registered validators for the given workspace root.
67///
68/// # Errors
69///
70/// Returns an error if any validator's build function fails.
71pub fn build_all_validators(workspace_root: &Path) -> Result<Vec<Box<dyn Validator>>> {
72    let mut out = Vec::with_capacity(VALIDATOR_ENTRIES.len());
73    for entry in VALIDATOR_ENTRIES {
74        let v = (entry.build)(workspace_root.to_path_buf()).map_err(|e| {
75            crate::error::Error::configuration(format!("validator '{}': {}", entry.name, e))
76        })?;
77        out.push(v);
78    }
79    Ok(out)
80}
81
82/// Build only the requested validators for the given workspace root.
83///
84/// Unknown validator names are ignored to preserve existing behavior.
85///
86/// # Errors
87///
88/// Returns an error if any selected validator's build function fails.
89pub fn build_named_validators(
90    workspace_root: &Path,
91    validator_names: &[String],
92) -> Result<Vec<Box<dyn Validator>>> {
93    let requested: std::collections::HashSet<&str> =
94        validator_names.iter().map(String::as_str).collect();
95    let mut out = Vec::new();
96
97    for entry in VALIDATOR_ENTRIES {
98        if !requested.contains(entry.name) {
99            continue;
100        }
101
102        let validator = (entry.build)(workspace_root.to_path_buf()).map_err(|e| {
103            crate::error::Error::configuration(format!("validator '{}': {}", entry.name, e))
104        })?;
105        out.push(validator);
106    }
107
108    Ok(out)
109}

111/// Convert a `dyn Violation` to a serializable `ViolationEntry`.
112///
113/// This is the canonical conversion used by both mcb-validate and mcb-infrastructure
114/// to turn trait objects into string-based entries for reports.
115pub fn violation_to_entry(v: &dyn crate::ports::validation::Violation) -> ViolationEntry {
116    ViolationEntry {
117        id: v.id().to_owned(),
118        category: v.category().to_string(),
119        severity: v.severity().to_string(),
120        file: v.file().map(|p| p.display().to_string()),
121        line: v.line(),
122        message: v.message(),
123        suggestion: v.suggestion(),
124    }
125}

131/// Build a `ValidationReport` from a list of trait-object violations.
132///
133/// Handles counting by severity and applying an optional severity filter.
134#[must_use]
135pub fn build_report(
136    violations: &[Box<dyn crate::ports::validation::Violation>],
137    severity_filter: Option<&str>,
138) -> ValidationReport {
139    let entries: Vec<ViolationEntry> = violations
140        .iter()
141        .map(|v| violation_to_entry(v.as_ref()))
142        .collect();
143
144    let filtered = filter_violations_by_severity(&entries, severity_filter);
145
146    let errors = filtered.iter().filter(|e| e.severity == "ERROR").count();
147    let warnings = filtered.iter().filter(|e| e.severity == "WARNING").count();
148    let infos = filtered.iter().filter(|e| e.severity == "INFO").count();
149
150    // `passed` is based on unfiltered error count
151    let total_errors = entries.iter().filter(|e| e.severity == "ERROR").count();
152
153    ValidationReport {
154        total_violations: filtered.len(),
155        errors,
156        warnings,
157        infos,
158        violations: filtered,
159        passed: total_errors == 0,
160    }
161}
162
163fn filter_violations_by_severity(
164    violations: &[ViolationEntry],
165    severity_filter: Option<&str>,
166) -> Vec<ViolationEntry> {
167    let level = match severity_filter {
168        Some("error") => 1,
169        Some("warning") => 2,
170        Some("info") => 3,
171        _ => return violations.to_vec(),
172    };
173    violations
174        .iter()
175        .filter(|v| {
176            let l = match v.severity.as_str() {
177                "ERROR" => 1,
178                "WARNING" => 2,
179                _ => 3,
180            };
181            l <= level
182        })
183        .cloned()
184        .collect()
185}
