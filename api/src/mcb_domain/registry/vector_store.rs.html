4//! Vector Store Provider Registry
5//!
6//! Auto-registration system for vector store providers using linkme distributed slices.
7//! Providers register themselves via `#[linkme::distributed_slice]` and are
8//! discovered at runtime.
9
10use std::collections::HashMap;
11
12/// Configuration for vector store provider creation
13///
14/// Contains all configuration options that a vector store provider might need.
15/// Providers should use what they need and ignore the rest.
16#[derive(Debug, Clone, Default)]
17pub struct VectorStoreProviderConfig {
18    /// Provider name (e.g., "milvus", "memory", "null")
19    pub provider: String,
20    /// Connection URI or path
21    pub uri: Option<String>,
22    /// Collection/index name
23    pub collection: Option<String>,
24    /// Embedding dimensions
25    pub dimensions: Option<usize>,
26    /// API key or token for authentication
27    pub api_key: Option<String>,
28    /// Enable encryption
29    pub encrypted: Option<bool>,
30    /// Encryption key (if encrypted)
31    pub encryption_key: Option<String>,
32    /// Additional provider-specific configuration
33    pub extra: HashMap<String, String>,
34}
35
36crate::impl_config_builder!(VectorStoreProviderConfig {
37    /// Set the URI
38    uri: with_uri(into String),
39    /// Set the collection name
40    collection: with_collection(into String),
41    /// Set the embedding dimensions
42    dimensions: with_dimensions(usize),
43    /// Set the API key
44    api_key: with_api_key(into String),
45});
46
47impl VectorStoreProviderConfig {
48    /// Enable encryption
49    #[must_use]
50    pub fn with_encryption(mut self, key: impl Into<String>) -> Self {
51        self.encrypted = Some(true);
52        self.encryption_key = Some(key.into());
53        self
54    }
55}

63    list_fn: list_vector_store_providers
