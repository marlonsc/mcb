1//! MCP text extraction utilities.
2//!
3//! Provides functions for extracting text segments from MCP content values.
4//! These are domain-level utilities used across the application for processing
5//! MCP protocol responses.
6//!
7//! The functions accept `serde_json::Value` slices, keeping the domain layer
8//! free from protocol-specific types like `rmcp::model::Content`.
9
10use serde::Serialize;
11
12/// Concatenate all text segments from a slice of JSON content values
13/// using a custom separator.
14///
15/// Each value is expected to be an object containing a `"text"` field.
16/// Non-text values (images, resources, etc.) are silently skipped.
17#[must_use]
18pub fn extract_text_with_sep(content: &[serde_json::Value], sep: &str) -> String {
19    let mut result = String::new();
20    let mut first = true;
21
22    for value in content {
23        if let Some(text) = value.get("text").and_then(|t| t.as_str()) {
24            if !first {
25                result.push_str(sep);
26            }
27            first = false;
28            result.push_str(text);
29        }
30    }
31
32    result
33}
34
35/// Concatenate all text segments from a slice of JSON content values,
36/// separated by newlines.
37#[must_use]
38pub fn extract_text(content: &[serde_json::Value]) -> String {
39    extract_text_with_sep(content, "\n")
40}
41
42/// Extract text from any serializable MCP content sequence, returning a
43/// `Result` to surface serialization failures to callers.
44///
45/// Serializes the content to JSON, then extracts `"text"` fields.
46/// This allows callers to pass protocol-specific types (e.g. `rmcp::model::Content`)
47/// without the domain layer depending on those types directly.
48///
49/// # Errors
50///
51/// Returns [`serde_json::Error`] if any element in `content` fails to serialize to a JSON value.
52pub fn try_extract_text_from<T: Serialize>(content: &[T]) -> Result<String, serde_json::Error> {
53    let values: Result<Vec<serde_json::Value>, serde_json::Error> =
54        content.iter().map(serde_json::to_value).collect();
55
56    values.map(|vals| extract_text(&vals))
57}
58
59/// Extract text from any serializable MCP content sequence.
60///
61/// This is a convenience wrapper around [`try_extract_text_from`] that
62/// silently returns an empty string on serialization failure.
63#[must_use]
64pub fn extract_text_from<T: Serialize>(content: &[T]) -> String {
65    try_extract_text_from(content).unwrap_or_default()
66}

68#[cfg(test)]
69mod tests {
70    use super::{extract_text, extract_text_with_sep};
71    use serde_json::json;
72
73    #[test]
74    fn joins_text_segments_with_separator() {
75        let content = [json!({"text": "a"}), json!({"text": "b"})];
76        assert_eq!(extract_text_with_sep(&content, ", "), "a, b");
77    }
78
79    #[test]
80    fn skips_non_text_values() {
81        let content = [
82            json!({"text": "a"}),
83            json!({"image": "x"}),
84            json!({"text": "b"}),
85        ];
86        assert_eq!(extract_text(&content), "a\nb");
87    }
88
89    #[test]
90    fn empty_input_yields_empty_string() {
91        let content: [serde_json::Value; 0] = [];
92        assert_eq!(extract_text_with_sep(&content, "-"), "");
93    }
94
95    #[test]
96    fn single_segment_has_no_separator() {
97        let content = [json!({"text": "only"})];
98        assert_eq!(extract_text_with_sep(&content, ", "), "only");
99    }
