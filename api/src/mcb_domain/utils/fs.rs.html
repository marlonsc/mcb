6use crate::error::{Error, Result};
7use std::path::{Path, PathBuf};
8
9fn read_entries(dir: &Path) -> Result<std::fs::ReadDir> {
10    std::fs::read_dir(dir)
11        .map_err(|e| Error::internal(format!("Failed to read directory {}: {}", dir.display(), e)))
12}
13
14fn read_file_type(entry: &std::fs::DirEntry, path: &Path) -> Result<std::fs::FileType> {
15    entry.file_type().map_err(|e| {
16        Error::internal(format!(
17            "Failed to get file type for {}: {}",
18            path.display(),
19            e
20        ))
21    })
22}
23
24fn has_matching_extension(path: &Path, extensions: &[&str]) -> bool {
25    path.extension()
26        .and_then(|e| e.to_str())
27        .is_some_and(|ext| extensions.iter().any(|&e| e.eq_ignore_ascii_case(ext)))
28}
29
30/// Collect all files recursively from a directory that match specific extensions.
31///
32/// # Errors
33///
34/// Returns an error if directory traversal fails.
35pub fn find_files_by_extensions(root: &Path, extensions: &[&str]) -> Result<Vec<PathBuf>> {
36    if !root.exists() {
37        return Ok(Vec::new());
38    }
39
40    let mut files = Vec::new();
41    let mut stack = vec![root.to_path_buf()];
42
43    while let Some(dir) = stack.pop() {
44        let entries = read_entries(&dir)?;
45
46        for entry in entries {
47            let entry = entry.map_err(|e| {
48                Error::internal(format!(
49                    "Failed to read directory entry in {}: {}",
50                    dir.display(),
51                    e
52                ))
53            })?;
54            let path = entry.path();
55            let file_type = read_file_type(&entry, &path)?;
56
57            if file_type.is_dir() {
58                stack.push(path);
59                continue;
60            }
61
62            if file_type.is_file() && has_matching_extension(&path, extensions) {
63                files.push(path);
64            }
65        }
66    }
67
68    Ok(files)
69}
