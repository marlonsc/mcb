12/// Generates a new random UUID v4.
13#[must_use]
14pub fn generate() -> Uuid {
15    Uuid::new_v4()
16}

18/// Generates a deterministic UUID v5 from a namespace string and key.
19#[must_use]
20pub fn deterministic(namespace: &str, key: &str) -> Uuid {
21    let ns = Uuid::new_v5(&Uuid::NAMESPACE_OID, namespace.as_bytes());
22    Uuid::new_v5(&ns, key.as_bytes())
23}

25/// Deterministic UUID v5 correlation string for a `kind+raw_id` pair.
26///
27/// Replaces the old HMAC-SHA256 `compute_stable_id_hash` / `hash_id` functions.
28/// Same (kind, `raw_id`) always produces the same UUID string.
29#[must_use]
30pub fn correlate_id(kind: &str, raw_id: &str) -> String {
31    deterministic(kind, raw_id).to_string()
32}

34/// SHA-256 hex digest of content for deduplication.
35#[must_use]
36pub fn compute_content_hash(content: &str) -> String {
37    let mut hasher = Sha256::new();
38    hasher.update(content.as_bytes());
39    hex::encode(hasher.finalize())
40}

42/// Mask sensitive ID for logging â€” shows first 8 chars + "...".
43#[must_use]
44pub fn mask_id(id: &str) -> String {
45    if id.len() <= MASKED_ID_PREFIX_LENGTH {
46        id.to_owned()
47    } else {
48        format!("{}...", &id[..MASKED_ID_PREFIX_LENGTH])
49    }
50}

52/// Compute SHA-256 hash of a file's content.
53///
54/// # Errors
55///
56/// Returns an error if the file cannot be opened or read.
57pub fn compute_file_hash(path: &Path) -> Result<String> {
58    let file =
59        File::open(path).map_err(|e| Error::io(format!("Failed to open file {path:?}: {e}")))?;
60    let mut reader = BufReader::new(file);
61    let mut hasher = Sha256::new();
62    let mut buffer = [0u8; FILE_READ_BUFFER_SIZE];
63    loop {
64        let count = reader
65            .read(&mut buffer)
66            .map_err(|e| Error::io(format!("Failed to read file {path:?}: {e}")))?;
67        if count == 0 {
68            break;
69        }
70        hasher.update(&buffer[..count]);
71    }
72    Ok(hex::encode(hasher.finalize()))
73}
