12/// Record of a discovered function.
13#[derive(Debug, Clone)]
14pub struct FunctionRecord {
15    /// Path to the file containing the function.
16    pub file: PathBuf,
17    /// Name of the function.
18    pub name: String,
19    /// Line number where the function starts.
20    pub line: usize,
21    /// Cyclomatic complexity score of the function.
22    pub complexity: u32,
23}

25/// Computes cyclomatic complexity using control flow keyword counting.
26///
27/// Returns 1 + count of branching constructs (if, for, while, loop, match, &&, ||).
28///
29/// # Errors
30///
31/// Returns an error if the internal complexity regex fails to compile.
32pub fn compute_complexity_score(content: &str, start_pos: usize) -> Result<u32> {
33    let body = extract_function_body(content, start_pos).unwrap_or_default();
34    let re = Regex::new(r"\b(if|for|while|loop|match)\b|&&|\|\|")
35        .map_err(|e| Error::invalid_argument(format!("invalid complexity regex: {e}")))?;
36    let count = re.find_iter(&body).count() as u32;
37    Ok(1 + count)
38}

40/// Extracts the function body by balancing braces.
41///
42/// Returns the content within the function body, including the braces.
43#[must_use]
44pub fn extract_function_body(content: &str, start_pos: usize) -> Option<String> {
45    let after_start = &content[start_pos..];
46    let brace_index = after_start.find('{')?;
47    let body_start = start_pos + brace_index;
48
49    let bytes = content.as_bytes();
50    let mut depth = 0_i32;
51    let mut i = body_start;
52    while i < bytes.len() {
53        match bytes[i] {
54            b'{' => depth += 1,
55            b'}' => {
56                depth -= 1;
57                if depth == 0 {
58                    return Some(content[body_start..=i].to_string());
59                }
60            }
61            _ => {}
62        }
63        i += 1;
64    }
65    None
66}

68/// Extracts a code block defined by balanced braces `{}` from an iterator of lines.
69/// Returns the number of lines from the start of the iterator that contain the balanced block.
70#[must_use]
71pub fn count_balanced_block_lines<I, S>(lines: I, max_search_offset: usize) -> Option<usize>
72where
73    I: IntoIterator<Item = S>,
74    S: AsRef<str>,
75{
76    let mut brace_balance = 0;
77    let mut found_start = false;
78
79    for (offset, line) in lines.into_iter().enumerate() {
80        let line_ref = line.as_ref();
81
82        let open_count = line_ref.chars().filter(|c| *c == '{').count() as i32;
83        let close_count = line_ref.chars().filter(|c| *c == '}').count() as i32;
84
85        if open_count > 0 && !found_start {
86            found_start = true;
87        }
88
89        if found_start {
90            brace_balance += open_count;
91            brace_balance -= close_count;
92
93            if brace_balance <= 0 {
94                return Some(offset + 1); // Length in lines
95            }
96        } else if offset > max_search_offset {
97            return None;
98        }
99    }
100
101    None
102}

104/// Checks if a symbol should be exempt from dead code detection (e.g. main, tests).
105#[must_use]
106pub fn is_exempt_symbol(name: &str) -> bool {
107    matches!(name, "main") || name.starts_with("test_")
108}

110/// Counts occurrences of symbols across a set of file contents.
111///
112/// Returns a map of symbol -> count.
113///
114/// # Errors
115///
116/// Returns an error if a symbol produces an invalid regex pattern.
117pub fn count_symbol_occurrences(
118    file_contents: &[String],
119    symbols: &[String],
120) -> Result<HashMap<String, usize>> {
121    let mut map = HashMap::new();
122    for symbol in symbols {
123        let escaped = regex::escape(symbol);
124        let re = Regex::new(&format!(r"\b{escaped}\b"))
125            .map_err(|e| Error::invalid_argument(format!("invalid symbol regex: {e}")))?;
126        let count = file_contents
127            .iter()
128            .map(|content| re.find_iter(content).count())
129            .sum();
130        map.insert(symbol.clone(), count);
131    }
132    Ok(map)
133}

135/// Collects function definitions from a set of source files.
136///
137/// Discovers all `fn` declarations (including `pub` and `async` variants),
138/// computes their line number and cyclomatic complexity.
139///
140/// # Errors
141///
142/// Returns an error if the function-matching regex or complexity computation fails.
143pub fn collect_functions(files: &[(PathBuf, String)]) -> Result<Vec<FunctionRecord>> {
144    let fn_re = Regex::new(r"(?m)^\s*(?:pub\s+)?(?:async\s+)?fn\s+([A-Za-z_][A-Za-z0-9_]*)")
145        .map_err(|e| Error::invalid_argument(format!("invalid function regex: {e}")))?;
146
147    let mut records = Vec::new();
148    for (file, content) in files {
149        for captures in fn_re.captures_iter(content) {
150            let Some(name_match) = captures.get(1) else {
151                continue;
152            };
153            let name = name_match.as_str().to_owned();
154            let fn_start = name_match.start();
155            let line = content[..fn_start].bytes().filter(|b| *b == b'\n').count() + 1;
156            let complexity = compute_complexity_score(content, fn_start)?;
157            records.push(FunctionRecord {
158                file: file.clone(),
159                name,
160                line,
161                complexity,
162            });
163        }
164    }
165    Ok(records)
166}

168/// Filters functions exceeding a complexity threshold.
169///
170/// Returns a [`ComplexityFinding`] for each function whose score is above the threshold.
171#[must_use]
172pub fn filter_complex_functions(
173    functions: Vec<FunctionRecord>,
174    threshold: u32,
175) -> Vec<ComplexityFinding> {
176    functions
177        .into_iter()
178        .filter(|f| f.complexity > threshold)
179        .map(|f| ComplexityFinding {
180            file: f.file,
181            function: f.name,
182            complexity: f.complexity,
183        })
184        .collect()
185}

187/// Detects potentially dead functions by counting symbol occurrences.
188///
189/// A function is considered dead if it appears at most once across all file contents
190/// and is not an exempt symbol (e.g. `main`, `test_*`).
191///
192/// # Errors
193///
194/// Returns an error if symbol occurrence counting fails.
195pub fn detect_dead_functions(
196    functions: Vec<FunctionRecord>,
197    file_contents: &[String],
198) -> Result<Vec<DeadCodeFinding>> {
199    let names: Vec<String> = functions.iter().map(|f| f.name.clone()).collect();
200    let occurrences = count_symbol_occurrences(file_contents, &names)?;
201
202    Ok(functions
203        .into_iter()
204        .filter(|f| !is_exempt_symbol(&f.name))
205        .filter(|f| occurrences.get(&f.name).copied().unwrap_or(0) <= 1)
206        .map(|f| DeadCodeFinding {
207            file: f.file,
208            line: f.line,
209            item_type: "function".to_owned(),
210            name: f.name,
211        })
212        .collect())
213}

215/// Computes TDG (Technical Debt Grade) scores per file.
216///
217/// Aggregates max complexity, dead code count, and SLOC into a composite score.
218/// Only files exceeding the threshold are returned.
219#[must_use]
220pub fn compute_tdg_scores(
221    files: &[(PathBuf, String)],
222    functions: Vec<FunctionRecord>,
223    dead_code: &[DeadCodeFinding],
224    threshold: u32,
225) -> Vec<TdgFinding> {
226    let mut complexity_by_file: HashMap<PathBuf, u32> = HashMap::new();
227    for function in functions {
228        let entry = complexity_by_file.entry(function.file).or_insert(0);
229        *entry = (*entry).max(function.complexity);
230    }
231
232    let mut dead_by_file: HashMap<PathBuf, u32> = HashMap::new();
233    for finding in dead_code {
234        *dead_by_file.entry(finding.file.clone()).or_insert(0) += 1;
235    }
236
237    let mut findings = Vec::new();
238    for (path, content) in files {
239        let sloc = content.lines().filter(|l| !l.trim().is_empty()).count() as u32;
240        let complexity = complexity_by_file.get(path).copied().unwrap_or(1);
241        let dead = dead_by_file.get(path).copied().unwrap_or(0);
242
243        let tdg_score = complexity.saturating_mul(2)
244            + dead.saturating_mul(10)
245            + ((sloc / 200).saturating_mul(10));
246
247        if tdg_score > threshold {
248            findings.push(TdgFinding {
249                file: path.clone(),
250                score: tdg_score,
251            });
252        }
253    }
254
255    findings
256}
