20/// Value Object: Embedding Provider Configuration
21///
22/// Configuration for connecting to and using embedding providers.
23/// Defines which provider to use and how to authenticate with it.
24#[derive(Clone, Serialize, Deserialize, PartialEq)]
25pub struct EmbeddingConfig {
26    /// Provider name (openai, ollama, fastembed, etc.)
27    pub provider: EmbeddingProviderKind,
28    /// Model identifier specific to the provider
29    pub model: String,
30    /// API key for cloud providers
31    pub api_key: Option<String>,
32    /// Custom API endpoint URL
33    pub base_url: Option<String>,
34    /// Output embedding dimensions
35    pub dimensions: Option<usize>,
36    /// Maximum input token limit
37    pub max_tokens: Option<usize>,
38}
39
40impl fmt::Debug for EmbeddingConfig {
41    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
42        f.debug_struct("EmbeddingConfig")
43            .field("provider", &self.provider)
44            .field("model", &self.model)
45            .field("api_key", &self.api_key.as_ref().map(|_| REDACTED))
46            .field("base_url", &self.base_url.as_ref().map(|_| REDACTED))
47            .field("dimensions", &self.dimensions)
48            .field("max_tokens", &self.max_tokens)
49            .finish()
50    }
51}

119/// Value Object: Synchronization Batch
120///
121/// Represents a batch of files queued for synchronization/re-indexing.
122/// Used by the file watcher daemon to batch file changes.
123#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
124pub struct SyncBatch {
125    /// Unique batch identifier
126    pub id: String,
127    /// Repository or collection identifier
128    pub collection: String,
129    /// List of file paths to process
130    pub files: Vec<String>,
131    /// Priority level (higher numbers = higher priority)
132    pub priority: u8,
133    /// Timestamp when batch was created
134    pub created_at: i64,
135}
