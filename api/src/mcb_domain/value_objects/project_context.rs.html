6/// Auto-resolved project identity derived from the git repository.
7///
8/// Without authentication the project is identified by `owner/repo`
9/// parsed from `git remote.origin.url`.  This is stable across
10/// worktrees, directory renames, and multiple checkouts of the same
11/// repository.
12#[derive(Debug, Clone, TypedBuilder)]
13pub struct ProjectContext {
14    /// Stable project identifier (e.g. `"marlonsc/mcb"`).
15    #[builder(setter(into))]
16    pub project_id: String,
17    /// Human-readable short name (e.g. `"mcb"`).
18    #[builder(setter(into))]
19    pub project_name: String,
20    /// Whether this repository is a git submodule.
21    #[builder(default)]
22    pub is_submodule: bool,
23    /// The superproject's owner/repo if this is a submodule.
24    #[builder(default, setter(strip_option, into))]
25    pub superproject_id: Option<String>,
26}
27
28impl ProjectContext {
29    /// Explicit constructor for tests and overrides.
30    pub fn new(project_id: impl Into<String>, project_name: impl Into<String>) -> Self {
31        Self::builder()
32            .project_id(project_id.into())
33            .project_name(project_name.into())
34            .build()
35    }
36}

38/// Parse a git remote URL into `owner/repo` format.
39///
40/// Supports SSH shorthand (`git@host:owner/repo.git`),
41/// SSH URL (`ssh://git@host/owner/repo.git`),
42/// and HTTPS (`https://host/owner/repo[.git]`).
43#[must_use]
44pub fn parse_owner_repo(url: &str) -> Option<String> {
45    // SSH shorthand: git@host:owner/repo.git
46    if let Some((_host, path)) = url.strip_prefix("git@").and_then(|s| s.split_once(':')) {
47        let path = path.trim_end_matches(".git");
48        return normalize_owner_repo(path);
49    }
50
51    // URL form: scheme://host/owner/repo.git
52    let path = url
53        .split("://")
54        .nth(1)
55        .and_then(|s| s.split_once('/'))
56        .map(|(_, path)| path)?;
57    let path = path.trim_end_matches(".git");
58    normalize_owner_repo(path)
59}
60
61/// Normalize a repository path into `owner/repo` or `org/subgroup/repo`.
62///
63/// Returns `None` when the input path does not contain at least one
64/// non-empty segment.
65#[must_use]
66pub fn normalize_owner_repo(path: &str) -> Option<String> {
67    let parts: Vec<&str> = path.split('/').filter(|s| !s.is_empty()).collect();
68    if parts.len() >= 2 {
69        Some(parts.join("/"))
70    } else if parts.len() == 1 {
71        Some(parts[0].to_owned())
72    } else {
73        None
74    }
