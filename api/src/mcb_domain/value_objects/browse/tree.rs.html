4use derive_more::Display;
5use serde::{Deserialize, Serialize};
6
7/// Tree node for hierarchical file navigation
8#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, Display)]
9#[display("{}", self.to_ansi())]
10pub struct FileTreeNode {
11    /// Display name of the node (file or directory name)
12    pub name: String,
13    /// Full path from repository root
14    pub path: String,
15    /// Whether this node is a directory
16    pub is_dir: bool,
17    /// Child nodes (empty for files)
18    pub children: Vec<FileTreeNode>,
19    /// Number of chunks (only for files)
20    pub chunk_count: Option<u32>,
21    /// Detected language (only for files)
22    pub language: Option<String>,
23}
24
25impl FileTreeNode {
26    /// Create a new file node
27    pub fn file(
28        name: impl Into<String>,
29        path: impl Into<String>,
30        chunk_count: u32,
31        language: impl Into<String>,
32    ) -> Self {
33        Self {
34            name: name.into(),
35            path: path.into(),
36            is_dir: false,
37            children: Vec::new(),
38            chunk_count: Some(chunk_count),
39            language: Some(language.into()),
40        }
41    }
42
43    /// Create a new directory node
44    pub fn directory(name: impl Into<String>, path: impl Into<String>) -> Self {
45        Self {
46            name: name.into(),
47            path: path.into(),
48            is_dir: true,
49            children: Vec::new(),
50            chunk_count: None,
51            language: None,
52        }
53    }
54
55    /// Add a child node to this directory (builder pattern)
56    #[must_use]
57    pub fn with_child(mut self, child: FileTreeNode) -> Self {
58        self.children.push(child);
59        self
60    }
61
62    /// Sort children: directories first, then files, alphabetically
63    #[must_use]
64    pub fn sorted(mut self) -> Self {
65        self.sort_children();
66        self
67    }
68
69    /// Sort children: directories first, then files, alphabetically (in-place)
70    pub fn sort_children(&mut self) {
71        self.children.sort_by(|a, b| match (a.is_dir, b.is_dir) {
72            (true, false) => std::cmp::Ordering::Less,
73            (false, true) => std::cmp::Ordering::Greater,
74            _ => a.name.cmp(&b.name),
75        });
76        for child in &mut self.children {
77            child.sort_children();
78        }
79    }
80
81    /// Traverse the tree and call a callback for each node
82    pub fn traverse(&self, callback: &mut dyn FnMut(&FileTreeNode)) {
83        callback(self);
84        for child in &self.children {
85            child.traverse(callback);
86        }
87    }
88
89    /// Convert tree to ANSI-formatted string with colors and tree structure
90    #[must_use]
91    pub fn to_ansi(&self) -> String {
92        let mut output = String::new();
93        self.format_ansi(&mut output, "", true);
94        output
95    }
96
97    fn format_ansi(&self, output: &mut String, prefix: &str, is_last: bool) {
98        let connector = if is_last { "‚îî‚îÄ‚îÄ " } else { "‚îú‚îÄ‚îÄ " };
99        let color = if self.is_dir { "\x1b[34m" } else { "\x1b[0m" };
100        let reset = "\x1b[0m";
101
102        output.push_str(prefix);
103        output.push_str(connector);
104        output.push_str(color);
105        output.push_str(&self.name);
106        output.push_str(reset);
107
108        if let Some(count) = self.chunk_count {
109            use std::fmt::Write;
110            let _ = write!(output, " ({count})");
111        }
112        output.push('\n');
113
114        let new_prefix = format!("{prefix}{}", if is_last { "    " } else { "‚îÇ   " });
115
116        for (i, child) in self.children.iter().enumerate() {
117            let is_last_child = i == self.children.len() - 1;
118            child.format_ansi(output, &new_prefix, is_last_child);
119        }
120    }
121
122    /// Convert tree to HTML-formatted string with nesting
123    #[must_use]
124    pub fn to_html(&self) -> String {
125        let mut output = String::new();
126        self.format_html(&mut output);
127        output
128    }
129
130    fn format_html(&self, output: &mut String) {
131        let icon = if self.is_dir { "üìÅ" } else { "üìÑ" };
132        let name_html = html_escape(&self.name);
133
134        output.push_str("<ul>\n");
135        output.push_str("<li>");
136        output.push_str(icon);
137        output.push(' ');
138        output.push_str(&name_html);
139
140        if let Some(count) = self.chunk_count {
141            use std::fmt::Write;
142            let _ = write!(output, " <span style=\"color: #888;\">({count})</span>");
143        }
144
145        if !self.children.is_empty() {
146            output.push('\n');
147            for child in &self.children {
148                child.format_html(output);
149            }
150            output.push_str("</li>\n");
151            output.push_str("</ul>\n");
152        } else {
153            output.push_str("</li>\n");
154            output.push_str("</ul>\n");
155        }
156    }
157}

159/// HTML escape a string to prevent XSS
160fn html_escape(text: &str) -> String {
161    text.replace('&', "&amp;")
162        .replace('<', "&lt;")
163        .replace('>', "&gt;")
164        .replace('"', "&quot;")
165        .replace('\'', "&#39;")
166}
