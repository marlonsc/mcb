6use serde::{Deserialize, Serialize};

8/// Highlight category for code tokens
9#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
10pub enum HighlightCategory {
11    /// Language keyword (e.g., if, while, return)
12    Keyword,
13    /// String literal
14    String,
15    /// Code comment
16    Comment,
17    /// Function or method name
18    Function,
19    /// Variable or identifier
20    Variable,
21    /// Type name or type annotation
22    Type,
23    /// Numeric literal
24    Number,
25    /// Operator symbol (e.g., +, -, *)
26    Operator,
27    /// Punctuation (e.g., brackets, semicolons)
28    Punctuation,
29    /// Other token type
30    Other,
31}
32
33/// Represents a span of code with a specific highlight category.
34///
35/// Defines a range within source code and the category of syntax highlighting
36/// to apply to that range.
37#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
38pub struct HighlightSpan {
39    /// Starting byte offset of the span in the source code
40    pub start: usize,
41    /// Ending byte offset of the span in the source code
42    pub end: usize,
43    /// The highlight category for this span
44    pub category: HighlightCategory,
45}
46
47/// Complete result of syntax highlighting for source code.
48///
49/// Contains the original source code, a collection of highlight spans
50/// that define syntax highlighting regions, and the programming language.
51#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
52pub struct HighlightedCode {
53    /// The original source code text
54    pub original: String,
55    /// Collection of highlight spans defining syntax highlighting regions
56    pub spans: Vec<HighlightSpan>,
57    /// Programming language of the source code
58    pub language: String,
59}
60
61impl HighlightedCode {
62    /// Creates a new `HighlightedCode` instance.
63    ///
64    /// # Arguments
65    ///
66    /// * `original` - The original source code text
67    /// * `spans` - Collection of highlight spans defining syntax highlighting regions
68    /// * `language` - Programming language of the source code
69    ///
70    /// # Returns
71    ///
72    /// A new `HighlightedCode` instance with the provided values.
73    #[must_use]
74    pub fn new(original: String, spans: Vec<HighlightSpan>, language: String) -> Self {
75        Self {
76            original,
77            spans,
78            language,
79        }
80    }
81}

100/// Maps tree-sitter highlight token names to domain highlight categories.
101#[must_use]
102pub fn map_highlight_to_category(name: &str) -> HighlightCategory {
103    match name {
104        "keyword" => HighlightCategory::Keyword,
105        "string" => HighlightCategory::String,
106        "comment" => HighlightCategory::Comment,
107        "function" => HighlightCategory::Function,
108        "variable" | "constant" | "attribute" | "property" | "tag" => HighlightCategory::Variable,
109        "type" => HighlightCategory::Type,
110        "number" => HighlightCategory::Number,
111        "operator" => HighlightCategory::Operator,
112        "punctuation" | "punctuation.bracket" | "punctuation.delimiter" | "punctuation.special" => {
113            HighlightCategory::Punctuation
114        }
115        _ => HighlightCategory::Other,
116    }
117}
