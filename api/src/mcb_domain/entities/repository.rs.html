4use schemars::JsonSchema;
5use serde::{Deserialize, Serialize};
6
7/// Type of version control system.
8#[derive(
9    Debug,
10    Clone,
11    Copy,
12    PartialEq,
13    Eq,
14    Serialize,
15    Deserialize,
16    JsonSchema,
17    strum_macros::Display,
18    strum_macros::AsRefStr,
19    strum_macros::EnumString,
20)]
21#[serde(rename_all = "snake_case")]
22#[strum(serialize_all = "snake_case", ascii_case_insensitive)]
23pub enum VcsType {
24    /// Git repository.
25    #[strum(serialize = "git")]
26    Git,
27    /// Mercurial repository.
28    #[strum(serialize = "mercurial", serialize = "hg")]
29    Mercurial,
30    /// Subversion repository.
31    #[strum(serialize = "svn", serialize = "subversion")]
32    Svn,
33}
34
35impl VcsType {
36    /// Returns the string representation.
37    #[must_use]
38    pub fn as_str(&self) -> &str {
39        match self {
40            Self::Git => "git",
41            Self::Mercurial => "mercurial",
42            Self::Svn => "svn",
43        }
44    }
45}

49// ---------------------------------------------------------------------------
50
51crate::define_entity! {
52    /// A tracked repository registered in the platform.
53    #[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
54    pub struct Repository { id, org_id, project_id, created_at, updated_at } {
55        /// Display name of the repository.
56        pub name: String,
57        /// Remote URL of the repository (e.g. <https://github.com/user/repo.git>).
58        pub url: String,
59        /// Local path where the repository is checked out.
60        pub local_path: String,
61        /// Type of version control system used.
62        pub vcs_type: VcsType,
63    }
64}

70crate::define_entity! {
71    /// A tracked branch within a repository.

74        /// Repository this branch belongs to.
75        pub repository_id: String,
76        /// Branch name (e.g. "main", "feat/data-model-v2").
77        pub name: String,
