8use serde::{Deserialize, Serialize};
9
10use crate::value_objects::Language;
11
12/// Core Entity: Semantically Meaningful Code Segment
13///
14/// A code chunk represents a meaningful unit of code that has been extracted
15/// through AST-based parsing. This is the primary entity for semantic indexing
16/// and search operations.
17///
18/// ## Business Rules
19///
20/// - Each chunk must have a unique identifier
21/// - Content must be non-empty and meaningful
22/// - Language identification enables proper parsing
23/// - Metadata provides additional context for search
24///
25/// ## Example
26///
27/// ```rust
28/// use mcb_domain::entities::CodeChunk;
29/// use mcb_domain::value_objects::Language;
30///
31/// let chunk = CodeChunk {
32///     id: "chunk_001".to_string(),
33///     content: "fn authenticate(user: &str) -> bool { true }".to_string(),
34///     file_path: "src/auth.rs".to_string(),
35///     start_line: 10,
36///     end_line: 12,
37///     language: "rust".to_string(),
38///     metadata: serde_json::json!({"type": "function", "name": "authenticate"}),
39/// };
40/// ```
41#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
42pub struct CodeChunk {
43    /// Unique identifier for this code chunk
44    pub id: String,
45    /// The actual code content
46    pub content: String,
47    /// Path to the source file
48    pub file_path: String,
49    /// Starting line number in the source file
50    pub start_line: u32,
51    /// Ending line number in the source file
52    pub end_line: u32,
53    /// Programming language of the code
54    pub language: Language,
55    /// Additional metadata as JSON (context, AST info, etc.)
56    pub metadata: serde_json::Value,
57}
