4use schemars::JsonSchema;

8use super::types::{AgentSessionStatus, AgentType};

10/// Represents an agent session with execution metadata and results.
11///
12/// Tracks the lifecycle of an agent execution, including timing, status, resource usage,
13/// and summaries of the prompt and results.
14#[skip_serializing_none]
15#[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
16pub struct AgentSession {
17    /// Unique identifier for this agent session.
18    pub id: String,
19    /// Reference to the associated session summary.
20    pub session_summary_id: String,
21    /// Type of agent that executed this session.
22    pub agent_type: AgentType,
23    /// Model used by the agent (e.g., "claude-3-opus").
24    pub model: String,
25    /// Optional parent session ID if this is a nested/delegated session.
26    pub parent_session_id: Option<String>,
27    /// Unix timestamp (milliseconds) when the session started.
28    pub started_at: i64,
29    /// Unix timestamp (milliseconds) when the session ended, if completed.
30    pub ended_at: Option<i64>,
31    /// Total duration of the session in milliseconds.
32    pub duration_ms: Option<i64>,
33    /// Current status of the session (running, completed, failed, etc.).
34    pub status: AgentSessionStatus,
35    /// Brief summary of the input prompt.
36    pub prompt_summary: Option<String>,
37    /// Brief summary of the execution result.
38    pub result_summary: Option<String>,
39    /// Total tokens consumed during this session.
40    pub token_count: Option<i64>,
41    /// Number of tool calls made during this session.
42    pub tool_calls_count: Option<i64>,
43    /// Number of delegations made during this session.
44    pub delegations_count: Option<i64>,
45    /// Optional project ID this session belongs to.
46    pub project_id: Option<String>,
47    /// Optional worktree ID this session is working in.
48    pub worktree_id: Option<String>,
49}
