6use schemars::JsonSchema;
7use serde::{Deserialize, Serialize};
8
9crate::define_entity! {
10    /// An API key is a bearer credential scoped to a user within an
11    /// organization. Keys can be narrowed by JSON-encoded scopes and
12    /// optionally expire. Revocation is tracked via `revoked_at`.
13    #[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
14    pub struct ApiKey { id, org_id, created_at } {
15        /// User identifier this key is associated with.
16        pub user_id: String,
17        /// Hashed representation of the API key.
18        pub key_hash: String,
19        /// Readable name for the API key.
20        pub name: String,
21        /// JSON-encoded list of allowed scopes.
22        pub scopes_json: String,
23        /// Optional expiration timestamp.
24        pub expires_at: Option<i64>,
25        /// Optional revocation timestamp.
26        pub revoked_at: Option<i64>,
27    }
28}
