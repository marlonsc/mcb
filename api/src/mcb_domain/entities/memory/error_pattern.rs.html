7/// Categories for error patterns to classify the type of error encountered.
8#[derive(
9    Debug,
10    Clone,
11    PartialEq,
12    Eq,
13    Serialize,
14    Deserialize,
15    strum_macros::AsRefStr,
16    strum_macros::Display,
17    strum_macros::EnumString,
18)]
19#[strum(serialize_all = "lowercase", ascii_case_insensitive)]
20pub enum ErrorPatternCategory {
21    /// Compilation errors.
22    Compilation,
23    /// Runtime errors.
24    Runtime,
25    /// Test execution errors.
26    Test,
27    /// Linting errors.
28    Lint,
29    /// Build process errors.
30    Build,
31    /// Configuration errors.
32    Config,
33    /// Network-related errors.
34    Network,
35    /// Other error types.
36    Other,
37}
38
39crate::impl_as_str_from_as_ref!(ErrorPatternCategory);
40
41/// Represents a recurring error pattern detected in a project.
42///
43/// An error pattern captures the signature, category, and metadata of errors
44/// that occur repeatedly, enabling pattern recognition and solution tracking.
45#[skip_serializing_none]
46#[derive(Debug, Clone, Serialize, Deserialize)]
47pub struct ErrorPattern {
48    /// Unique identifier for this error pattern.
49    pub id: String,
50    /// Project ID this pattern belongs to.
51    pub project_id: String,
52    /// Signature/fingerprint of the error pattern for matching.
53    pub pattern_signature: String,
54    /// Human-readable description of the error pattern.
55    pub description: String,
56    /// Category of the error.
57    pub category: ErrorPatternCategory,
58    /// List of known solutions for this error pattern.
59    pub solutions: Vec<String>,
60    /// Files affected by this error pattern.
61    pub affected_files: Vec<String>,
62    /// Tags for categorizing and searching the pattern.
63    pub tags: Vec<String>,
64    /// Number of times this pattern has occurred.
65    pub occurrence_count: i64,
66    /// Timestamp when this pattern was first observed.
67    pub first_seen_at: i64,
68    /// Timestamp when this pattern was last observed.
69    pub last_seen_at: i64,
70    /// Optional embedding ID for semantic search.
71    pub embedding_id: Option<String>,
72}
73
74/// Represents a match between an observed error and a known error pattern.
75///
76/// Tracks when an error pattern is detected, the confidence level, and whether
77/// a solution was applied and successful.
78#[skip_serializing_none]
79#[derive(Debug, Clone, Serialize, Deserialize)]
80pub struct ErrorPatternMatch {
81    /// Unique identifier for this match record.
82    pub id: String,
83    /// ID of the error pattern that was matched.
84    pub pattern_id: String,
85    /// ID of the observation that triggered this match.
86    pub observation_id: String,
87    /// Confidence level of the match (0-100).
88    pub confidence: i64,
89    /// Index of the solution applied, if any.
90    pub solution_applied: Option<i32>,
91    /// Whether the applied solution was successful.
92    pub resolution_successful: Option<bool>,
93    /// Timestamp when the pattern was matched.
94    pub matched_at: i64,
95    /// Timestamp when the error was resolved, if applicable.
96    pub resolved_at: Option<i64>,
97}
