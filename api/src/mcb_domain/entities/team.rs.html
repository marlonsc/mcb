1//! Team and TeamMember entities — groups of users within an organization.
2//!
3//! **Documentation**: [docs/modules/domain.md](../../../../docs/modules/domain.md#core-entities)
4//!
5
6use schemars::JsonSchema;
7use serde::{Deserialize, Serialize};
8
9crate::define_entity! {
10    /// A team groups users within an organization for access control and
11    /// project assignment. Teams are used in the GitHub-like RBAC model:
12    /// Organization → Teams → Projects.
13    #[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
14    pub struct Team { id, org_id, created_at } {
15        /// Display name of the team.
16        pub name: String,
17    }
18}
19
20use crate::value_objects::ids::TeamMemberId;
21
22/// A membership link between a user and a team, with a role describing
23/// the user's authority within that team.
24#[derive(Debug, Clone, Serialize, Deserialize, JsonSchema, PartialEq, Eq)]
25pub struct TeamMember {
26    /// Unique identifier for the membership (composite of `team_id:user_id`).
27    #[serde(default)]
28    pub id: TeamMemberId,
29    /// Team the user belongs to.
30    pub team_id: String,
31    /// User who is a member.
32    pub user_id: String,
33    /// Role the user holds within the team.
34    pub role: TeamMemberRole,
35    /// Timestamp when the user joined the team (Unix epoch).
36    pub joined_at: i64,
37}
38
39/// Role a user holds within a specific team.
40#[derive(
41    Debug,
42    Clone,
43    PartialEq,
44    Eq,
45    Serialize,
46    Deserialize,
47    JsonSchema,
48    strum_macros::AsRefStr,
49    strum_macros::Display,
50    strum_macros::EnumString,
51)]
52#[strum(serialize_all = "lowercase", ascii_case_insensitive)]
53pub enum TeamMemberRole {
54    /// Team lead with management capabilities.
55    Lead,
56    /// Regular team member.
57    Member,
58}
59
60crate::impl_as_str_from_as_ref!(TeamMemberRole);
