9/// Categorizes the type of observation recorded.
10#[derive(
11    Debug,
12    Clone,
13    PartialEq,
14    Eq,
15    Serialize,
16    Deserialize,
17    strum_macros::AsRefStr,
18    strum_macros::Display,
19    strum_macros::EnumString,
20)]
21#[strum(serialize_all = "snake_case", ascii_case_insensitive)]
22pub enum ObservationType {
23    /// Represents a code snippet or file content.
24    Code,
25    /// Represents a recorded decision.
26    Decision,
27    /// Represents general project context or information.
28    Context,
29    /// Represents an error or exception.
30    Error,
31    /// Represents a summary or aggregation of data.
32    Summary,
33    /// Represents an execution trace or log.
34    Execution,
35    /// Represents a quality gate check result.
36    QualityGate,
37}
38
39crate::impl_as_str_from_as_ref!(ObservationType);

41/// Metadata associated with an observation.
42///
43/// Contains contextual information about where and when an observation was recorded,
44/// including session, repository, file, and git information.
45#[skip_serializing_none]
46#[derive(Clone, Serialize, Deserialize)]
47pub struct ObservationMetadata {
48    /// Unique identifier for the metadata.
49    pub id: String,
50    /// Identifier of the session where the observation occurred.
51    pub session_id: Option<String>,
52    /// Identifier of the repository related to the observation.
53    pub repo_id: Option<String>,
54    /// Path to the file related to the observation.
55    pub file_path: Option<String>,
56    /// Git branch name.
57    pub branch: Option<String>,
58    /// Git commit hash.
59    pub commit: Option<String>,
60    /// Details about the tool execution (if applicable).
61    pub execution: Option<ExecutionMetadata>,
62    /// Details about the quality gate result (if applicable).
63    pub quality_gate: Option<QualityGateResult>,
64    /// Contextual information about the origin of the observation.
65    pub origin_context: Option<OriginContext>,
66}
67
68impl std::fmt::Debug for ObservationMetadata {
69    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
70        f.debug_struct("ObservationMetadata")
71            .field("id", &self.id)
72            .field(
73                "session_id",
74                &if self.session_id.is_some() {
75                    "REDACTED"
76                } else {
77                    "NONE"
78                },
79            )
80            .field("repo_id", &self.repo_id)
81            .field("file_path", &self.file_path)
82            .field("branch", &self.branch)
83            .field("commit", &self.commit)
84            .field("execution", &self.execution)
85            .field("quality_gate", &self.quality_gate)
86            .field("origin_context", &self.origin_context)
87            .finish()
88    }
89}

91impl Default for ObservationMetadata {
92    /// Creates a new `ObservationMetadata` with a generated UUID and all optional fields set to `None`.
93    fn default() -> Self {

98            file_path: None,
99            branch: None,
100            commit: None,
101            execution: None,
102            quality_gate: None,
103            origin_context: None,

105    }
106}

108/// A recorded unit of knowledge or event in the system.
109///
110/// Observations are immutable records of code, decisions, context, errors, summaries,
111/// execution traces, or quality gate results. Each observation is uniquely identified,
112/// tagged for semantic search, and includes metadata about its origin and context.
113#[derive(Debug, Clone, Serialize, Deserialize)]
114pub struct Observation {
115    /// Unique identifier for the observation.
116    pub id: String,
117    /// Identifier of the project this observation belongs to.
118    pub project_id: String,
119    /// The actual content or payload of the observation.
120    pub content: String,
121    /// Hash of the content for deduplication.
122    pub content_hash: String,
123    /// List of semantic tags.
124    pub tags: Vec<String>,
125    /// Classification of the observation.
126    #[serde(rename = "type", alias = "observation_type")]
127    pub r#type: ObservationType,
128    /// Additional metadata.
129    pub metadata: ObservationMetadata,
130    /// Timestamp when the observation was created (Unix epoch).
131    pub created_at: i64,
132    /// Reference to vector embedding.
133    pub embedding_id: Option<String>,
134}
