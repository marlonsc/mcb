4use schemars::JsonSchema;
5use serde::{Deserialize, Serialize};
6
7crate::define_entity! {
8    /// Represents a user within the system.
9    ///
10    /// Users are associated with an organization and have specific roles that
11    /// determine their permissions.
12    #[derive(Debug, Clone, Serialize, Deserialize, JsonSchema)]
13    pub struct User { id, org_id, created_at, updated_at } {
14        /// Email address of the user.
15        pub email: String,
16        /// Name to be displayed for the user.
17        pub display_name: String,
18        /// Role assigned to the user within the organization.
19        pub role: UserRole,
20        /// Hashed API key for the user, if applicable.
21        pub api_key_hash: Option<String>,
22    }
23}
24
25/// Role a user holds within an organization.
26#[derive(
27    Debug,
28    Clone,
29    Default,
30    PartialEq,
31    Eq,
32    Serialize,
33    Deserialize,
34    JsonSchema,
35    strum_macros::AsRefStr,
36    strum_macros::Display,
37    strum_macros::EnumString,
38)]
39#[strum(serialize_all = "lowercase", ascii_case_insensitive)]
40pub enum UserRole {
41    /// Full administrative access.
42    #[default]
43    Admin,
44    /// Standard member with read/write access.
45    Member,
46    /// Read-only viewer.
47    Viewer,
48    /// Service account (API-only, used by agents).
49    Service,
50}
51
52crate::impl_as_str_from_as_ref!(UserRole);
