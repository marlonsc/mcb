1//! MCP JSON-RPC protocol types (domain-level)
2//!
3//! These are pure data types for the JSON-RPC-based MCP protocol.
4//! They live in the domain layer because they represent the protocol
5//! contract — no transport logic, no HTTP dependencies.
6//!
7//! **Documentation**: [docs/modules/domain.md](../../../../docs/modules/domain.md)
8
9use mcb_utils::constants::protocol::JSONRPC_VERSION;
10use serde::{Deserialize, Serialize};
11
12/// MCP request payload (JSON-RPC format).
13#[derive(Debug, Clone, Serialize, Deserialize)]
14pub struct McpRequest {
15    /// JSON-RPC version (e.g. "2.0")
16    #[serde(default = "default_jsonrpc")]
17    pub jsonrpc: String,
18    /// JSON-RPC method
19    pub method: String,
20    /// Request parameters
21    pub params: Option<serde_json::Value>,
22    /// Request ID
23    pub id: Option<serde_json::Value>,
24}
25
26/// MCP response payload (JSON-RPC format).
27#[derive(Debug, Clone, Serialize, Deserialize)]
28pub struct McpResponse {
29    /// JSON-RPC version
30    #[serde(default = "default_jsonrpc")]
31    pub jsonrpc: String,
32    /// Response result (if successful)
33    #[serde(skip_serializing_if = "Option::is_none")]
34    pub result: Option<serde_json::Value>,
35    /// Error (if failed)
36    #[serde(skip_serializing_if = "Option::is_none")]
37    pub error: Option<McpError>,
38    /// Request ID
39    pub id: Option<serde_json::Value>,
40}
41
42fn default_jsonrpc() -> String {
43    JSONRPC_VERSION.to_owned()
44}
45
46/// MCP error response (JSON-RPC format).
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct McpError {
49    /// Error code
50    pub code: i32,
51    /// Error message
52    pub message: String,
53}
54
55impl McpResponse {
56    /// Create a success response.
57    #[must_use]
58    pub fn from_success(id: Option<serde_json::Value>, result: serde_json::Value) -> Self {
59        Self {
60            jsonrpc: JSONRPC_VERSION.to_owned(),
61            result: Some(result),
62            error: None,
63            id,
64        }
65    }
66
67    /// Create an error response.
68    #[must_use]
69    pub fn from_error(
70        id: Option<serde_json::Value>,
71        code: i32,
72        message: impl Into<String>,
73    ) -> Self {
74        Self {
75            jsonrpc: JSONRPC_VERSION.to_owned(),
76            result: None,
77            error: Some(McpError {
78                code,
79                message: message.into(),
80            }),
81            id,
82        }
83    }
84}
