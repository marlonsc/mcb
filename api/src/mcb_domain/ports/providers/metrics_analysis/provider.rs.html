6use std::path::Path;
7
8use async_trait::async_trait;
9
10use crate::error::Result;
11use crate::value_objects::SupportedLanguage;
12
13use super::types::{FileMetrics, FunctionMetrics};
14
15#[async_trait]
16pub trait MetricsAnalysisProvider: Send + Sync {
17    fn provider_name(&self) -> &str;
18    fn supported_languages(&self) -> &[SupportedLanguage];
19
20    fn supports_language(&self, lang: SupportedLanguage) -> bool {
21        self.supported_languages().contains(&lang)
22    }
23
24    async fn analyze_file(&self, path: &Path) -> Result<FileMetrics>;
25
26    async fn analyze_code(
27        &self,
28        content: &[u8],
29        language: SupportedLanguage,
30        file_path: Option<&str>,
31    ) -> Result<FileMetrics>;
32
33    async fn analyze_functions(&self, path: &Path) -> Result<Vec<FunctionMetrics>>;
34
35    async fn analyze_function(
36        &self,
37        path: &Path,
38        function_name: &str,
39    ) -> Result<Option<FunctionMetrics>> {
40        let functions = self.analyze_functions(path).await?;
41        Ok(functions.into_iter().find(|f| f.name == function_name))
42    }
43
44    fn can_analyze(&self, path: &Path) -> bool {
45        SupportedLanguage::from_path(path).is_some_and(|lang| self.supports_language(lang))
46    }
47}
