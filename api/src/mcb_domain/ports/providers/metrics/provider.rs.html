6use std::collections::HashMap;
7use std::time::Duration;
8
9use async_trait::async_trait;
10
11use super::metric_defs::{MetricLabels, MetricsResult, labels_from};
12
13#[async_trait]
14pub trait MetricsProvider: Send + Sync {
15    fn name(&self) -> &str;
16    async fn increment(&self, name: &str, labels: &MetricLabels) -> MetricsResult<()>;
17    async fn increment_by(
18        &self,
19        name: &str,
20        value: f64,
21        labels: &MetricLabels,
22    ) -> MetricsResult<()>;
23    async fn gauge(&self, name: &str, value: f64, labels: &MetricLabels) -> MetricsResult<()>;
24    async fn histogram(&self, name: &str, value: f64, labels: &MetricLabels) -> MetricsResult<()>;
25
26    async fn record_index_time(&self, duration: Duration, collection: &str) -> MetricsResult<()> {
27        let labels = labels_from([("collection", collection)]);
28        self.histogram(
29            "mcb_index_duration_seconds",
30            duration.as_secs_f64(),
31            &labels,
32        )
33        .await
34    }
35
36    async fn record_search_latency(
37        &self,
38        duration: Duration,
39        collection: &str,
40    ) -> MetricsResult<()> {
41        let labels = labels_from([("collection", collection)]);
42        self.histogram(
43            "mcb_search_duration_seconds",
44            duration.as_secs_f64(),
45            &labels,
46        )
47        .await
48    }
49
50    async fn record_embedding_latency(
51        &self,
52        duration: Duration,
53        provider: &str,
54    ) -> MetricsResult<()> {
55        let labels = labels_from([("provider", provider)]);
56        self.histogram(
57            "mcb_embedding_duration_seconds",
58            duration.as_secs_f64(),
59            &labels,
60        )
61        .await
62    }
63
64    async fn increment_indexed_files(&self, collection: &str, count: u64) -> MetricsResult<()> {
65        let labels = labels_from([("collection", collection)]);
66        self.increment_by("mcb_indexed_files_total", count as f64, &labels)
67            .await
68    }
69
70    async fn increment_search_requests(&self, collection: &str) -> MetricsResult<()> {
71        let labels = labels_from([("collection", collection)]);
72        self.increment("mcb_search_requests_total", &labels).await
73    }
74
75    async fn set_active_indexing_jobs(&self, count: u64) -> MetricsResult<()> {
76        self.gauge("mcb_active_indexing_jobs", count as f64, &HashMap::new())
77            .await
78    }
79
80    async fn set_vector_store_size(&self, collection: &str, vectors: u64) -> MetricsResult<()> {
81        let labels = labels_from([("collection", collection)]);
82        self.gauge("mcb_vector_store_size", vectors as f64, &labels)
83            .await
84    }
85
86    async fn record_cache_access(&self, hit: bool, cache_type: &str) -> MetricsResult<()> {
87        let labels = labels_from([
88            ("cache_type", cache_type),
89            ("result", if hit { "hit" } else { "miss" }),
90        ]);
91        self.increment("mcb_cache_accesses_total", &labels).await
92    }
93}
