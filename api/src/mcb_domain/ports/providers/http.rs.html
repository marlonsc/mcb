3use std::time::Duration;

5use async_trait::async_trait;
6use serde::{Deserialize, Serialize};

8use crate::error::Result;

10/// HTTP client configuration.
11#[derive(Debug, Clone, Serialize, Deserialize)]
12pub struct HttpClientConfig {
13    /// Maximum number of idle connections per host.
14    pub max_idle_per_host: usize,
15    /// Maximum idle time for a connection.
16    pub idle_timeout: Duration,
17    /// Connection keep-alive interval.
18    pub keepalive: Duration,
19    /// Request timeout.
20    pub timeout: Duration,
21    /// User agent string for outgoing requests.
22    pub user_agent: String,
23}
24
25impl Default for HttpClientConfig {
26    fn default() -> Self {
27        Self {
28            max_idle_per_host: 10,
29            idle_timeout: Duration::from_secs(90),
30            keepalive: Duration::from_secs(60),
31            timeout: Duration::from_secs(30),
32            user_agent: "mcb/domain-client".to_owned(),
33        }
34    }
35}
36
37/// HTTP client provider trait.
38///
39/// Provides an abstract HTTP client interface without coupling to any
40/// concrete HTTP library. Implementations live in the provider or
41/// infrastructure layer and wrap a real HTTP client.
42#[async_trait]
43pub trait HttpClientProvider: Send + Sync {
44    /// Get the HTTP client configuration.
45    fn config(&self) -> &HttpClientConfig;
46
47    /// Execute a GET request and return the response body as bytes.
48    ///
49    /// # Errors
50    /// Returns an error if the request fails.
51    async fn get(&self, url: &str) -> Result<Vec<u8>>;
52
53    /// Execute a POST request with a raw byte body and return the response body.
54    ///
55    /// # Errors
56    /// Returns an error if the request fails.
57    async fn post(&self, url: &str, body: &[u8]) -> Result<Vec<u8>>;
58
59    /// Return true if the HTTP client is configured and enabled.
60    fn is_enabled(&self) -> bool;
61}
