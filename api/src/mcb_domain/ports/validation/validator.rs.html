3use super::config::{NamedCheck, ValidationConfig, run_checks};
4use super::language_id::LanguageId;
5use super::types::{ValidatorResult, Violation};
6
7/// Interface for implementing codebase validation rules.
8///
9/// Validators are responsible for checking certain aspects of the code (naming, SOLID, cleanliness)
10/// and returning a list of violations found.
11pub trait Validator: Send + Sync {
12    /// Returns the unique name of this validator.
13    fn name(&self) -> &'static str;
14
15    /// Returns a list of individual named checks for this validator.
16    ///
17    /// # Errors
18    /// Returns a `ValidatorResult` which may contain an error if check preparation fails.
19    fn checks<'a>(&'a self, _config: &'a ValidationConfig) -> ValidatorResult<Vec<NamedCheck<'a>>> {
20        Ok(Vec::new())
21    }
22
23    /// Performs the full validation scan for this validator.
24    ///
25    /// # Errors
26    /// Returns a `ValidatorResult` error if the validation process fails.
27    fn validate(&self, config: &ValidationConfig) -> ValidatorResult<Vec<Box<dyn Violation>>> {
28        run_checks(self.name(), self.checks(config)?)
29    }
30
31    /// Check if this validator should run by default in a standard scan.
32    fn enabled_by_default(&self) -> bool {
33        true
34    }
35
36    /// Get a human-readable description of what this validator checks.
37    fn description(&self) -> &'static str {
38        ""
39    }
40
41    /// Get the list of languages supported by this validator.
42    fn supported_languages(&self) -> &[LanguageId] {
43        &[LanguageId::Rust]
44    }
45}
