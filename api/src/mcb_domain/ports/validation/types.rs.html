8/// Opaque error type for validator failures.
9pub type ValidatorError = Box<dyn std::error::Error + Send + Sync>;
10/// Specialized result for validation operations.
11pub type ValidatorResult<T> = std::result::Result<T, ValidatorError>;
12
13/// Severity level of a code violation.
14#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
15#[serde(rename_all = "UPPERCASE")]
16pub enum Severity {
17    /// Blocking issue that must be fixed.
18    Error,
19    /// Non-blocking issue that should be fixed.
20    Warning,
21    /// Suggestion for improvement or informational note.
22    Info,
23}
24
25impl Display for Severity {
26    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
27        match self {
28            Self::Error => write!(f, "ERROR"),
29            Self::Warning => write!(f, "WARNING"),
30            Self::Info => write!(f, "INFO"),
31        }
32    }
33}

115/// Shared interface for all code violations.
116pub trait Violation: Display + Send + Sync + std::fmt::Debug {
117    /// Unique identifier for the violation type (e.g., "ARCH001").
118    fn id(&self) -> &str;
119    /// Category used for grouping violations in reports.
120    fn category(&self) -> ViolationCategory;
121    /// Importance of the violation.
122    fn severity(&self) -> Severity;
123    /// Path to the file containing the violation, if applicable.
124    fn file(&self) -> Option<&PathBuf>;
125    /// Line number where the violation occurs, if applicable.
126    fn line(&self) -> Option<usize>;
127
128    /// Human-readable description of the violation.
129    fn message(&self) -> String {
130        self.to_string()
131    }
132
133    /// Actionable advice on how to fix the violation.
134    fn suggestion(&self) -> Option<String> {
135        None
136    }
137
138    /// Helper to convert the violation into a trait object.
139    fn boxed(self) -> Box<dyn Violation>
140    where
141        Self: Sized + 'static,
142    {
143        Box::new(self)
144    }
145}
