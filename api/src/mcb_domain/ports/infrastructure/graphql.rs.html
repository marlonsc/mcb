3use std::sync::Arc;
4
5/// Domain port for GraphQL schema providers.
6///
7/// Implementations build a GraphQL schema from the database connection
8/// and return it as a type-erased `Box<dyn Any + Send + Sync>`.
9/// The concrete schema type (e.g. `async_graphql::dynamic::Schema`) lives
10/// in the provider layer; the domain remains free of GraphQL dependencies.
11pub trait GraphQLSchemaProvider: Send + Sync {
12    /// Build a GraphQL schema using the given opaque database connection.
13    ///
14    /// # Arguments
15    /// * `db` - Database connection as `Box<dyn Any + Send + Sync>`
16    /// * `depth` - Optional query depth limit
17    /// * `complexity` - Optional query complexity limit
18    ///
19    /// # Errors
20    /// Returns an error if the schema build fails.
21    fn build_schema(
22        &self,
23        db: Box<dyn std::any::Any + Send + Sync>,
24        depth: Option<usize>,
25        complexity: Option<usize>,
26    ) -> crate::error::Result<Box<dyn std::any::Any + Send + Sync>>;
27}
28
29/// Shared GraphQL schema provider for dependency injection.
30pub type SharedGraphQLSchemaProvider = Arc<dyn GraphQLSchemaProvider>;
