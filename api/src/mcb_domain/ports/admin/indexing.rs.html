1//! Indexing operation tracking ports.
2
3use std::collections::HashMap;
4
5use serde::{Deserialize, Serialize};
6
7use crate::value_objects::{CollectionId, OperationId};
8
9/// Status of an indexing operation
10#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
11pub enum IndexingOperationStatus {
12    /// Operation is starting
13    Starting,
14    /// Operation is in progress
15    InProgress,
16    /// Operation has completed successfully
17    Completed,
18    /// Operation has failed with an error message
19    Failed(String),
20}
21
22/// Data about an ongoing indexing operation
23#[derive(Debug, Clone, Serialize, Deserialize)]
24pub struct IndexingOperation {
25    /// Unique identifier for the operation
26    pub id: OperationId,
27    /// Target collection for indexing
28    pub collection: CollectionId,
29    /// Current status of the operation
30    pub status: IndexingOperationStatus,
31    /// Total number of files to index
32    pub total_files: usize,
33    /// Number of files processed so far
34    pub processed_files: usize,
35    /// Current file being processed, if any
36    pub current_file: Option<String>,
37    /// Timestamp when the operation started
38    pub started_at: i64,
39}
40
41/// Interface for tracking indexing operations
42pub trait IndexingOperationsInterface: Send + Sync {
43    /// Get all tracked indexing operations.
44    fn get_operations(&self) -> HashMap<OperationId, IndexingOperation>;
45    /// Start a new indexing operation.
46    fn start_operation(&self, collection: &CollectionId, total_files: usize) -> OperationId;
47    /// Update progress of an operation.
48    fn update_progress(
49        &self,
50        operation_id: &OperationId,
51        current_file: Option<String>,
52        processed: usize,
53    );
54    /// Mark an operation as completed.
55    fn complete_operation(&self, operation_id: &OperationId);
56}
