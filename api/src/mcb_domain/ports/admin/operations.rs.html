19/// Data structure for detailed performance metrics
20#[derive(Debug, Clone, Serialize, Deserialize, Default)]
21pub struct PerformanceMetricsData {
22    /// Total queries processed
23    pub total_queries: u64,
24    /// Total successful queries
25    pub successful_queries: u64,
26    /// Number of failed queries
27    pub failed_queries: u64,
28    /// Average response time in milliseconds
29    pub average_response_time_ms: f64,
30    /// Cache hit rate (0.0 to 1.0)
31    pub cache_hit_rate: f64,
32    /// Number of currently active connections
33    pub active_connections: usize,
34    /// Server uptime in seconds
35    pub uptime_seconds: u64,
36}
37
38/// Interface for tracking performance metrics
39pub trait PerformanceMetricsInterface: Send + Sync {
40    /// Get server uptime in seconds
41    fn uptime_secs(&self) -> u64;
42    /// Record a query execution
43    fn record_query(&self, response_time_ms: u64, success: bool, cache_hit: bool);
44    /// Update active connections count
45    fn update_active_connections(&self, delta: i64);
46    /// Get current performance metrics
47    fn get_performance_metrics(&self) -> PerformanceMetricsData;
48}

54/// Status of an indexing operation
55#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
56pub enum IndexingOperationStatus {
57    /// Indicates the operation is initializing.
58    Starting,
59    /// Indicates the operation is currently running.
60    InProgress,
61    /// Indicates the operation finished successfully.
62    Completed,
63    /// Indicates the operation failed with an error message.
64    Failed(String),
65}
66
67/// Data about an ongoing indexing operation
68#[derive(Debug, Clone, Serialize, Deserialize)]
69pub struct IndexingOperation {
70    /// Unique identifier for the operation.
71    pub id: OperationId,
72    /// Identifier of the collection being indexed.
73    pub collection: CollectionId,
74    /// Current status of the operation.
75    pub status: IndexingOperationStatus,
76    /// Total number of files to process.
77    pub total_files: usize,
78    /// Number of files processed so far.
79    pub processed_files: usize,
80    /// Path of the file currently being processed.
81    pub current_file: Option<String>,
82    /// Timestamp when the operation started (Unix epoch seconds).
83    pub started_at: i64,
84}
85
86/// Interface for tracking indexing operations
87pub trait IndexingOperationsInterface: Send + Sync {
88    /// Get all tracking operations
89    fn get_operations(&self) -> HashMap<OperationId, IndexingOperation>;
90    /// Start tracking a new indexing operation
91    fn start_operation(&self, collection: &CollectionId, total_files: usize) -> OperationId;
92    /// Update progress of an operation
93    fn update_progress(
94        &self,
95        operation_id: &OperationId,
96        current_file: Option<String>,
97        processed: usize,
98    );
99    /// Mark operation as completed
100    fn complete_operation(&self, operation_id: &OperationId);
101}

107/// Status of a validation operation
108#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
109pub enum ValidationStatus {
110    /// Indicates the operation is waiting to start.
111    Queued,
112    /// Indicates the operation is currently executing.
113    InProgress,
114    /// Indicates the operation finished.
115    Completed,
116    /// Indicates the operation failed with an error.
117    Failed(String),
118    /// Indicates the operation was manually stopped.
119    Canceled,
120}
121
122/// Result metadata for a completed validation operation
123#[derive(Debug, Clone, Serialize, Deserialize)]
124pub struct ValidationOperationResult {
125    /// Total number of rule violations found.
126    pub total_violations: usize,
127    /// Number of error-level violations.
128    pub errors: usize,
129    /// Number of warning-level violations.
130    pub warnings: usize,
131    /// Whether the validation passed acceptance criteria.
132    pub passed: bool,
133}
134
135/// Data about an ongoing validation operation
136#[derive(Debug, Clone, Serialize, Deserialize)]
137pub struct ValidationOperation {
138    /// Unique identifier for the operation.
139    pub id: OperationId,
140    /// Target workspace or directory being validated.
141    pub workspace: String,
142    /// Current status of the operation.
143    pub status: ValidationStatus,
144    /// List of validator names enabled for this operation.
145    pub validators: Vec<String>,
146    /// Progress percentage (0-100).
147    pub progress_percent: u8,
148    /// Path of the file currently being validated.
149    pub current_file: Option<String>,
150    /// Number of items processed so far.
151    pub processed_items: usize,
152    /// Total number of items to process.
153    pub total_items: usize,
154    /// Timestamp when the operation started (Unix epoch seconds).
155    pub started_at: i64,
156    /// Timestamp when the operation completed (Unix epoch seconds, if applicable).
157    pub completed_at: Option<i64>,
158    /// Final result of the validation (if completed).
159    pub result: Option<ValidationOperationResult>,
160}
161
162/// Interface for tracking validation operations
163pub trait ValidationOperationsInterface: Send + Sync {
164    /// Get all validation operations
165    fn get_operations(&self) -> HashMap<OperationId, ValidationOperation>;
166    /// Get a specific operation
167    fn get_operation(&self, operation_id: &OperationId) -> Option<ValidationOperation>;
168    /// Start a new validation operation
169    fn start_operation(&self, workspace: &str, validators: &[String]) -> OperationId;
170    /// Update progress
171    fn update_progress(
172        &self,
173        operation_id: &OperationId,
174        current_file: Option<String>,
175        processed: usize,
176        total: usize,
177    );
178    /// Mark as completed
179    fn complete_operation(&self, operation_id: &OperationId, result: ValidationOperationResult);
180    /// Cancel an operation
181    fn cancel_operation(&self, operation_id: &OperationId);
182    /// Check if in progress
183    fn is_in_progress(&self, operation_id: &OperationId) -> bool;
184}
