4//! Jobs Service Domain Ports
5//!
6//! Defines generic job management interfaces that abstract over specific
7//! operation types (indexing, validation, analysis, etc.).
8//!
9//! This is the unified task tracking system. Individual operation types
10//! (like `IndexingOperation`) map to `Job` instances with a specific `JobType`.

23/// Unique identifier for a job (wraps `OperationId` for domain consistency)
24pub type JobId = OperationId;
25
26/// The type of work a job performs
27#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash, Display)]
28pub enum JobType {
29    /// Codebase indexing operation
30    #[display("indexing")]
31    Indexing,
32    /// Architectural validation operation
33    #[display("validation")]
34    Validation,
35    /// Code analysis / complexity assessment
36    #[display("analysis")]
37    Analysis,
38    /// Custom job type with a user-defined label
39    #[display("custom:{_0}")]
40    Custom(String),
41}
42
43/// Lifecycle status of a job
44#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
45pub enum JobStatus {
46    /// Job is waiting to be picked up
47    Queued,
48    /// Job is currently executing
49    Running,
50    /// Job completed successfully
51    Completed,
52    /// Job terminated with an error
53    Failed(String),
54    /// Job was manually cancelled
55    Cancelled,
56}
57
58impl JobStatus {
59    /// Returns `true` if the job is in a terminal state
60    #[must_use]
61    pub fn is_terminal(&self) -> bool {
62        matches!(self, Self::Completed | Self::Failed(_) | Self::Cancelled)
63    }
64
65    /// Returns `true` if the job is actively running
66    #[must_use]
67    pub fn is_active(&self) -> bool {
68        matches!(self, Self::Queued | Self::Running)
69    }
70}
71
72/// Result metadata attached to a completed job
73#[derive(Debug, Clone, Serialize, Deserialize)]
74pub struct JobResult {
75    /// Summary message describing the outcome
76    pub summary: String,
77    /// Number of items successfully processed
78    pub items_processed: usize,
79    /// Number of items that failed processing
80    pub items_failed: usize,
81    /// Arbitrary key/value metadata
82    pub metadata: HashMap<String, serde_json::Value>,
83}

89/// A generic background job tracked by the system
90#[derive(Debug, Clone, Serialize, Deserialize)]
91pub struct Job {
92    /// Unique identifier for the job
93    pub id: JobId,
94    /// What kind of work this job performs
95    pub job_type: JobType,
96    /// Human-readable label for the job
97    pub label: String,
98    /// Current lifecycle status
99    pub status: JobStatus,
100    /// Progress as a percentage (0..=100)
101    pub progress_percent: u8,
102    /// Number of items processed so far
103    pub processed_items: usize,
104    /// Total number of items to process (0 = unknown)
105    pub total_items: usize,
106    /// Description of the item currently being processed
107    pub current_item: Option<String>,
108    /// When the job was created/queued (Unix epoch seconds)
109    pub created_at: i64,
110    /// When the job started running (Unix epoch seconds, if applicable)
111    pub started_at: Option<i64>,
112    /// When the job reached a terminal state (Unix epoch seconds, if applicable)
113    pub completed_at: Option<i64>,
114    /// Result metadata (populated on completion)
115    pub result: Option<JobResult>,
116}
117
118impl Job {
119    /// Create a new job in `Queued` status
120    pub fn new(id: JobId, job_type: JobType, label: impl Into<String>) -> Self {
121        Self {
122            id,
123            job_type,
124            label: label.into(),
125            status: JobStatus::Queued,
126            progress_percent: 0,
127            processed_items: 0,
128            total_items: 0,
129            current_item: None,
130            created_at: chrono::Utc::now().timestamp(),
131            started_at: None,
132            completed_at: None,
133            result: None,
134        }
135    }
136}

142/// Progress update payload for advancing a running job
143#[derive(Debug, Clone)]
144pub struct JobProgressUpdate {
145    /// Description of the current item being processed
146    pub current_item: Option<String>,
147    /// Number of items processed so far
148    pub processed_items: usize,
149    /// Total number of items to process
150    pub total_items: usize,
151}
152
153/// Interface for managing the lifecycle of background jobs.
154///
155/// Implementations track creation, progress, completion, and cancellation
156/// of jobs across all job types.
157pub trait JobManagerInterface: Send + Sync {
158    /// List all tracked jobs, optionally filtered by type
159    fn list_jobs(&self, job_type: Option<&JobType>) -> Vec<Job>;
160
161    /// Get a specific job by ID
162    fn get_job(&self, job_id: &JobId) -> Option<Job>;
163
164    /// Submit a new job and return its assigned ID
165    fn submit_job(&self, job_type: JobType, label: &str, total_items: usize) -> JobId;
166
167    /// Mark a queued job as running
168    fn start_job(&self, job_id: &JobId);
169
170    /// Update progress on a running job
171    fn update_progress(&self, job_id: &JobId, update: JobProgressUpdate);
172
173    /// Mark a job as successfully completed
174    fn complete_job(&self, job_id: &JobId, result: Option<JobResult>);
175
176    /// Mark a job as failed with an error message
177    fn fail_job(&self, job_id: &JobId, error: &str);
178
179    /// Cancel a queued or running job
180    fn cancel_job(&self, job_id: &JobId);
181
182    /// Get counts of jobs by status (for dashboard summaries)
183    fn job_counts(&self) -> JobCounts;
184}

186/// Summary counts of jobs grouped by status
187#[derive(Debug, Clone, Default, Serialize, Deserialize)]
188pub struct JobCounts {
189    /// Number of jobs waiting to start
190    pub queued: usize,
191    /// Number of actively running jobs
192    pub running: usize,
193    /// Number of successfully completed jobs
194    pub completed: usize,
195    /// Number of failed jobs
196    pub failed: usize,
197    /// Number of cancelled jobs
198    pub cancelled: usize,
199}
