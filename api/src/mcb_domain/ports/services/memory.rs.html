12/// Input payload for creating or updating a session summary.
13#[derive(Debug, Clone)]
14pub struct CreateSessionSummaryInput {
15    /// Project identifier owning this session summary.
16    pub project_id: String,
17    /// Organization identifier owning this session summary.
18    pub org_id: String,
19    /// Session identifier being summarized.
20    pub session_id: SessionId,
21    /// Main topics covered in the session.
22    pub topics: Vec<String>,
23    /// Concrete decisions taken during the session.
24    pub decisions: Vec<String>,
25    /// Actionable next steps produced by the session.
26    pub next_steps: Vec<String>,
27    /// Important files touched or discussed.
28    pub key_files: Vec<String>,
29    /// Optional origin context metadata.
30    pub origin_context: Option<OriginContext>,
31}

33/// Input payload describing an observation to store.
34#[derive(Debug, Clone)]
35pub struct StoreObservationInput {
36    /// Project identifier owning the observation.

38    /// Observation content body.
39    pub content: String,
40    /// Observation type/category.
41    pub r#type: ObservationType,
42    /// Tags associated with the observation.
43    pub tags: Vec<String>,
44    /// Observation metadata.
45    pub metadata: ObservationMetadata,
46}

48/// Manager for core observations.
49#[async_trait]
50pub trait ObservationManager: Send + Sync {
51    /// Store an observation with optional embedding for semantic search.
52    async fn store_observation(
53        &self,
54        input: StoreObservationInput,
55    ) -> Result<(ObservationId, bool)>;
56
57    /// Get an observation by ID.
58    async fn get_observation(&self, id: &ObservationId) -> Result<Option<Observation>>;
59
60    /// Performs the delete observation operation.
61    async fn delete_observation(&self, id: &ObservationId) -> Result<()>;
62
63    /// Get multiple observations by IDs.
64    async fn get_observations_by_ids(&self, ids: &[ObservationId]) -> Result<Vec<Observation>>;
65}

67/// Manager for error patterns.
68#[async_trait]
69pub trait ErrorPatternManager: Send + Sync {

72
73    /// Search for error patterns.
74    async fn search_error_patterns(

76        query: &str,
77        project_id: String,

82/// Manager for session summaries.
83#[async_trait]
84pub trait SessionSummaryManager: Send + Sync {
85    /// Get a session summary by session ID.

92/// Semantic text operations and memory search.
93#[async_trait]
94pub trait MemorySearcher: Send + Sync {
95    /// Search memories using semantic similarity.
96    async fn search_memories(
97        &self,
98        query: &str,
99        filter: Option<MemoryFilter>,
100        limit: usize,
101    ) -> Result<Vec<MemorySearchResult>>;
102
103    /// Generate embedding for content (for external use).
104    async fn embed_content(&self, content: &str) -> Result<Embedding>;
105
106    /// Get observations in timeline order around an anchor.
107    async fn get_timeline(
108        &self,
109        anchor_id: &ObservationId,
110        before: usize,
111        after: usize,
112        filter: Option<MemoryFilter>,
113    ) -> Result<Vec<Observation>>;
114
115    /// Token-efficient memory search - returns index only (no full content).
116    async fn memory_search(
117        &self,
118        query: &str,
119        filter: Option<MemoryFilter>,
120        limit: usize,
121    ) -> Result<Vec<MemorySearchIndex>>;
122}
