15/// Report containing validation results
16#[derive(Debug, Clone, serde::Serialize)]
17pub struct ValidationReport {
18    /// Total number of violations found
19    pub total_violations: usize,
20    /// Number of error-level violations
21    pub errors: usize,
22    /// Number of warning-level violations
23    pub warnings: usize,
24    /// Number of info-level violations
25    pub infos: usize,
26    /// All violations found
27    pub violations: Vec<ViolationEntry>,
28    /// Whether validation passed (no error-level violations)
29    pub passed: bool,
30}
31
32/// A single violation entry.
33///
34/// # Code Smells
35/// TODO(qlty): Found 16 lines of similar code with `crates/mcb-validate/src/generic_reporter.rs`.
36#[derive(Debug, Clone, serde::Serialize)]
37pub struct ViolationEntry {
38    /// Unique violation ID (e.g., "CA001", "SOLID002")
39    pub id: String,
40    /// Category (e.g., "`clean_architecture`", "solid", "quality")
41    pub category: String,
42    /// Severity level: "ERROR", "WARNING", or "INFO"
43    pub severity: String,
44    /// File path where violation was found (if applicable)
45    pub file: Option<String>,
46    /// Line number (if applicable)
47    pub line: Option<usize>,
48    /// Human-readable description of the violation
49    pub message: String,
50    /// Suggested fix (if available)
51    pub suggestion: Option<String>,
52}

54/// Information about a validation rule
55#[derive(Debug, Clone, serde::Serialize)]
56pub struct RuleInfo {
57    /// Rule ID (e.g., "CA001")
58    pub id: String,
59    /// Rule category
60    pub category: String,
61    /// Rule severity (error, warning, info)
62    pub severity: String,
63    /// Human-readable description
64    pub description: String,
65    /// Engine that executes this rule
66    pub engine: String,
67}

69/// Code complexity metrics report
70#[derive(Debug, Clone, serde::Serialize)]
71pub struct ComplexityReport {
72    /// File path
73    pub file: String,
74    /// Cyclomatic complexity
75    pub cyclomatic: f64,
76    /// Cognitive complexity
77    pub cognitive: f64,
78    /// Maintainability index (0-100)
79    pub maintainability_index: f64,
80    /// Source lines of code
81    pub sloc: usize,
82    /// Function-level metrics (if requested)
83    pub functions: Vec<FunctionComplexity>,
84}
85
86/// Function-level complexity metrics
87#[derive(Debug, Clone, serde::Serialize)]
88pub struct FunctionComplexity {
89    /// Function name
90    pub name: String,
91    /// Start line number
92    pub line: usize,
93    /// Cyclomatic complexity
94    pub cyclomatic: f64,
95    /// Cognitive complexity
96    pub cognitive: f64,
97    /// Source lines of code
98    pub sloc: usize,
99}

101/// Architecture Validation Service Interface
102#[async_trait]
103pub trait ValidationServiceInterface: Send + Sync {
104    /// Performs the validate operation.
105    async fn validate(
106        &self,
107        workspace_root: &Path,
108        validators: Option<&[String]>,
109        severity_filter: Option<&str>,
110    ) -> Result<ValidationReport>;
111
112    /// Performs the list validators operation.
113    async fn list_validators(&self) -> Result<Vec<String>>;
114
115    /// Performs the validate file operation.
116    async fn validate_file(
117        &self,
118        file_path: &Path,
119        validators: Option<&[String]>,
120    ) -> Result<ValidationReport>;
121
122    /// Performs the get rules operation.
123    async fn get_rules(&self, category: Option<&str>) -> Result<Vec<RuleInfo>>;
124
125    /// Performs the analyze complexity operation.
126    async fn analyze_complexity(
127        &self,
128        file_path: &Path,
129        include_functions: bool,
130    ) -> Result<ComplexityReport>;
131}
