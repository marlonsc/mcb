14/// Indexing Service Interface
15///
16/// Defines the contract for codebase indexing operations.
17#[async_trait]
18pub trait IndexingServiceInterface: Send + Sync {
19    /// Index a codebase at the given path
20    async fn index_codebase(
21        &self,
22        path: &Path,
23        collection: &CollectionId,
24    ) -> Result<IndexingResult>;
25
26    /// Get the current indexing status
27    fn get_status(&self) -> IndexingStatus;
28
29    /// Clear all indexed data from a collection
30    async fn clear_collection(&self, collection: &CollectionId) -> Result<()>;
31}
32
33/// Result of an indexing operation
34#[derive(Debug, Clone)]
35pub struct IndexingResult {
36    /// Number of files processed
37    pub files_processed: usize,
38    /// Number of chunks created
39    pub chunks_created: usize,
40    /// Number of files skipped
41    pub files_skipped: usize,
42    /// Any errors encountered (non-fatal)
43    pub errors: Vec<String>,
44    /// Operation ID for async tracking (None for synchronous operations)
45    pub operation_id: Option<OperationId>,
46    /// Status string: "started", "completed", "failed"
47    pub status: String,
48}
49
50/// Current indexing status
51#[derive(Debug, Clone, Default)]
52pub struct IndexingStatus {
53    /// Whether indexing is currently in progress
54    pub is_indexing: bool,
55    /// Current progress (0.0 to 1.0)
56    pub progress: f64,
57    /// Current file being processed
58    pub current_file: Option<String>,
59    /// Total files to process
60    pub total_files: usize,
61    /// Files processed so far
62    pub processed_files: usize,
63}
64
65/// Domain Service: Advanced Batch Indexing Operations
66///
67/// Extended interface for batch indexing services that handle
68/// incremental updates, rebuilds, and detailed statistics.
69#[async_trait]
70pub trait BatchIndexingServiceInterface: Send + Sync {
71    /// Index a batch of code chunks
72    async fn index_chunks(&self, chunks: &[CodeChunk]) -> Result<()>;
73
74    /// Index files from a directory
75    async fn index_directory(&self, path: &Path) -> Result<()>;
76
77    /// Process a synchronization batch
78    async fn process_sync_batch(&self, batch: &SyncBatch) -> Result<()>;
79
80    /// Rebuild index for a collection
81    async fn rebuild_index(&self, collection: &CollectionId) -> Result<()>;
82
83    /// Get indexing statistics
84    async fn get_indexing_stats(&self) -> Result<IndexingStats>;
85}
86
87/// Value Object: Indexing Statistics
88#[derive(Debug, Clone)]
89pub struct IndexingStats {
90    /// Total number of chunks indexed
91    pub total_chunks: u64,
92    /// Total number of collections
93    pub total_collections: u64,
94    /// Last indexing operation timestamp
95    pub last_indexed_at: Option<i64>,
96    /// Average indexing throughput (chunks per second)
97    pub avg_throughput: f64,
98}
