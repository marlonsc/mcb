1//! # GraphQL Schema Initializer
2//!
3//! Builds the GraphQL schema at startup via CA/DI resolution through
4//! [`mcb_domain::registry::graphql`]. No direct dependency on providers.
5
6use async_trait::async_trait;
7use axum::Router as AxumRouter;
8use loco_rs::prelude::*;
9
10use mcb_utils::constants::limits::{SCHEMA_COMPLEXITY, SCHEMA_DEPTH};
11
12/// Loco initializer that builds the GraphQL schema and stores it
13/// in [`AppContext::shared_store`] for use by the GraphQL controller.
14pub struct GraphQLInitializer;
15
16#[async_trait]
17impl Initializer for GraphQLInitializer {
18    fn name(&self) -> String {
19        "graphql".to_owned()
20    }
21
22    async fn after_routes(&self, router: AxumRouter, ctx: &AppContext) -> Result<AxumRouter> {
23        // Resolve the GraphQL schema provider via domain DI registry
24        let provider = mcb_domain::registry::graphql::resolve_graphql_schema_provider(
25            &mcb_domain::registry::graphql::GraphQLSchemaProviderConfig::new("seaography"),
26        )
27        .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
28
29        // Pass the database connection as type-erased Any
30        let db_any: Box<dyn std::any::Any + Send + Sync> = Box::new(ctx.db.clone());
31        let schema_any = provider
32            .build_schema(db_any, Some(SCHEMA_DEPTH), Some(SCHEMA_COMPLEXITY))
33            .map_err(|e| loco_rs::Error::string(&format!("GraphQL schema build failed: {e}")))?;
34
35        // Insert the schema as-is into shared_store. The GraphQL controller in
36        // mcb-server knows the concrete type and will downcast when needed.
37        mcb_server::graphql_store::insert_schema(&ctx.shared_store, schema_any)
38            .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
39
40        Ok(router)
41    }
42}
