27/// Build the embedding provider config from the resolved `AppConfig`.
28fn build_embedding_config(
29    app_config: &mcb_infrastructure::config::app::AppConfig,
30) -> EmbeddingProviderConfig {
31    let mut embed_cfg = EmbeddingProviderConfig::new(
32        app_config
33            .providers
34            .embedding
35            .provider
36            .as_deref()
37            .unwrap_or(mcb_utils::constants::DEFAULT_NULL_PROVIDER),
38    );
39    if let Some(ref v) = app_config.providers.embedding.cache_dir {
40        embed_cfg = embed_cfg.with_cache_dir(v.clone());
41    }
42    if let Some(ref v) = app_config.providers.embedding.model {
43        embed_cfg = embed_cfg.with_model(v.clone());
44    }
45    if let Some(ref v) = app_config.providers.embedding.base_url {
46        embed_cfg = embed_cfg.with_base_url(v.clone());
47    }
48    if let Some(ref v) = app_config.providers.embedding.api_key {
49        embed_cfg = embed_cfg.with_api_key(v.clone());
50    }
51    if let Some(d) = app_config.providers.embedding.dimensions {
52        embed_cfg = embed_cfg.with_dimensions(d);
53    }
54    embed_cfg
55}

176/// Protected routes — require admin API-key auth.
177///
178/// Captures `state`/`settings` clones for the admin-auth middleware closure so
179/// authorization does not depend on Extension-layer ordering.
180fn build_protected_routes(
181    state: mcb_server::McbState,
182    settings: Option<serde_json::Value>,
183) -> AxumRouter {
184    let admin_auth_middleware = axum::middleware::from_fn(
185        move |req: axum::http::Request<axum::body::Body>, next: axum::middleware::Next| {
186            let settings = settings.clone();
187            let state = state.clone();
188            async move {
189                if let Err(_e) = mcb_server::auth::authorize_admin_api_key(
190                    state.auth_repo.as_ref(),
191                    req.headers(),
192                    settings.as_ref(),
193                )
194                .await
195                {
196                    return Err(axum::http::StatusCode::UNAUTHORIZED);
197                }
198                Ok(next.run(req).await)
199            }
200        },
201    );
202
203    admin_route_table().layer(admin_auth_middleware)
204}

206/// Whether the MCP stdio transport should be started.

211/// Resolve and validate `AppConfig` from Loco settings via the config provider.
212fn resolve_app_config(ctx: &AppContext) -> Result<mcb_infrastructure::config::app::AppConfig> {
213    let settings = ctx
214        .config
215        .settings
216        .clone()
217        .ok_or_else(|| loco_rs::Error::string("missing loco settings for AppConfig"))?;
218
219    // Resolve config provider via CA/DI registry
220    let config_provider = resolve_config_provider(&ConfigProviderConfig::new(
221        mcb_utils::constants::DEFAULT_CONFIG_PROVIDER,
222    ))
223    .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
224
225    // Deserialize + validate via resolved provider (production path)
226    let app_config_any = config_provider
227        .deserialize_from_value(&settings)
228        .map_err(|e| loco_rs::Error::string(&format!("AppConfig: {e}")))?;
229
230    let app_config = *app_config_any
231        .downcast::<mcb_infrastructure::config::app::AppConfig>()
232        .map_err(|_| loco_rs::Error::string("ConfigProvider returned unexpected type"))?;
233
234    Ok(app_config)
235}

237/// Resolve event bus and provider adapters into a `ServiceResolutionContext`.
238fn build_resolution_ctx(
239    ctx: &AppContext,
240    app_config: mcb_infrastructure::config::app::AppConfig,
241) -> Result<ServiceResolutionContext> {
242    let event_bus = mcb_domain::registry::events::resolve_event_bus_provider(
243        &mcb_domain::registry::events::EventBusProviderConfig::new(
244            app_config
245                .system
246                .infrastructure
247                .event_bus
248                .provider
249                .provider_name(),
250        ),
251    )
252    .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
253
254    // Resolve providers via mcb-domain registries — no infrastructure helpers
255    let embedding_provider = resolve_embedding_provider(&build_embedding_config(&app_config))
256        .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
257
258    let vector_store_provider =
259        resolve_vector_store_provider(&build_vector_store_config(&app_config))
260            .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
261
262    Ok(ServiceResolutionContext {
263        db: Arc::new(ctx.db.clone()),
264        config: Arc::new(app_config),
265        event_bus,
266        embedding_provider,
267        vector_store_provider,
268    })
269}

271/// Resolve `AppConfig` from Loco settings and build the MCP server bootstrap.
272///
273/// Centralizes config-provider deserialization, provider resolution, and the
274/// bootstrap wiring so `after_routes` reads as a short orchestration. Returns
275/// the bootstrap plus whether the stdio transport should be started.
276fn build_bootstrap(ctx: &AppContext) -> Result<(mcb_server::state::McpServerBootstrap, bool)> {
277    let app_config = resolve_app_config(ctx)?;
278
279    let execution_flow = if app_config.mcp.stdio_only {
280        ExecutionFlow::StdioOnly
281    } else {
282        ExecutionFlow::ServerHybrid
283    };
284    let start_stdio = stdio_enabled(&app_config.mcp);
285
286    let resolution_ctx = build_resolution_ctx(ctx, app_config)?;
287
288    let hybrid_search: Arc<dyn mcb_domain::ports::HybridSearchProvider> =
289        mcb_domain::registry::hybrid_search::resolve_hybrid_search_provider(
290            &mcb_domain::registry::hybrid_search::HybridSearchProviderConfig::new(
291                mcb_utils::constants::DEFAULT_HYBRID_SEARCH_PROVIDER,
292            ),
293        )
294        .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
295
296    let bootstrap = build_mcp_server_bootstrap(
297        &resolution_ctx,
298        Arc::clone(&resolution_ctx.db),
299        Arc::clone(&resolution_ctx.embedding_provider),
300        Arc::clone(&resolution_ctx.vector_store_provider),
301        hybrid_search,
302        execution_flow,
303    )
304    .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
305    Ok((bootstrap, start_stdio))
306}

308/// Build the HTTP MCP streamable service from the resolved server handle.
309fn build_mcp_service(
310    mcp_server: Arc<mcb_server::McpServer>,
311) -> StreamableHttpService<mcb_server::McpServer, LocalSessionManager> {
312    let ct = CancellationToken::new();
313    // rmcp 1.x marks StreamableHttpServerConfig #[non_exhaustive]; build via Default.
314    let mut config = StreamableHttpServerConfig::default();
315    config.stateful_mode = false;
316    config.cancellation_token = ct.child_token();
317    StreamableHttpService::new(
318        move || {
319            let server = (*mcp_server).clone();
320            Ok(server)
321        },
322        LocalSessionManager::default().into(),
323        config,
324    )
325}

327/// Spawn the MCP stdio server, detaching the task.
328fn spawn_stdio_server(mcp_server: Arc<mcb_server::McpServer>) {
329    // Detached: handle intentionally dropped so the stdio server runs for the
330    // process lifetime. `let _ =` is rejected by clippy::let_underscore_future.
331    let _handle = tokio::spawn(async move {
332        let server = (*mcp_server).clone();
333        if let Err(e) = server.serve_stdio().await {
334            mcb_domain::error!("mcp_initializer", "MCP stdio server stopped", &e);
335        }
336    });
337}

339/// Loco initializer that builds the MCP server and injects `McbState` into the router.
340pub struct McpServerInitializer;
341
342#[async_trait]
343impl Initializer for McpServerInitializer {
344    fn name(&self) -> String {
345        "mcp_server".to_owned()
346    }
347
348    async fn after_routes(&self, router: AxumRouter, ctx: &AppContext) -> Result<AxumRouter> {
349        mcb_domain::infra::logging::set_log_fn(mcb_infrastructure::logging::tracing_log_fn);
350
351        let (bootstrap, start_stdio) = build_bootstrap(ctx)?;
352
353        if start_stdio {
354            spawn_stdio_server(Arc::clone(&bootstrap.mcp_server));
355        }
356
357        let mcb_state = bootstrap.into_mcb_state();
358        ctx.shared_store.insert(mcb_state.clone());
359
360        let mcp_service = build_mcp_service(Arc::clone(&mcb_state.mcp_server));
361
362        let protected_routes =
363            build_protected_routes(mcb_state.clone(), ctx.config.settings.clone());
364
365        // Merge public + protected routes, then apply Extension layer so all routes get McbState
366        let router = router
367            .merge(build_public_routes())
368            .merge(protected_routes)
369            .layer(Extension(mcb_state));
370        let mcp_routes = axum::Router::new().nest_service("/mcp", mcp_service);
371
372        // 404 fallback handler for unknown routes
373        let router = router
374            .merge(mcp_routes)
375            .fallback(axum::routing::get(|| async {
376                (
377                    axum::http::StatusCode::NOT_FOUND,
378                    axum::response::Html(mcb_server::controllers::web::not_found_html()),
379                )
380            }));
381
382        Ok(router)
383    }
