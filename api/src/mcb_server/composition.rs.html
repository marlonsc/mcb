1//! MCP server composition from a resolution context.
2//!
3//! Single composition root used by the Loco initializer and tests.
4//! All handler wiring goes through Loco; this function builds [`McpServerBootstrap`]
5//! from decomposed DI parts (database, providers, registry context).
6//!
7//! ## Pure Registry DI (ADR-050 + ADR-053)
8//!
9//! All services are resolved via the linkme registry. Shared providers (embedding,
10//! vector store) are pre-resolved at startup and passed as parameters.

38/// Build MCP server and dashboard/auth ports from decomposed DI parts.
39///
40/// Uses **pure registry DI** (ADR-050 + ADR-053): shared providers are pre-resolved
41/// at startup and passed in. All services are built via linkme registry resolution.
42/// Zero direct `::new()` construction of infrastructure services.
43///
44/// # Arguments
45///
46/// * `registry_ctx` - Opaque context for linkme service registry resolution (downcast internally).
47/// * `db_connection` - Database connection boxed as `Any` for registry database resolution.
48/// * `embedding_provider` - Shared embedding provider resolved at startup.
49/// * `vector_store_provider` - Shared vector store provider resolved at startup.
50/// * `hybrid_search` - Hybrid search provider for combined BM25/semantic search.
51/// * `execution_flow` - Whether to run in stdio-only or hybrid mode.
52///
53/// # Errors
54///
55/// Returns a domain error if any service or repository resolution fails.
56#[allow(clippy::too_many_arguments)]
57pub fn build_mcp_server_bootstrap(
58    registry_ctx: &dyn std::any::Any,
59    db_connection: Arc<dyn std::any::Any + Send + Sync>,
60    embedding_provider: Arc<dyn EmbeddingProvider>,
61    vector_store_provider: Arc<dyn VectorStoreProvider>,
62    hybrid_search: Arc<dyn HybridSearchProvider>,
63    execution_flow: ExecutionFlow,
64) -> mcb_domain::Result<McpServerBootstrap> {
65    // 1. Resolve DB repos
66    let repos = resolve_database_repositories(
67        DEFAULT_DATABASE_PROVIDER,
68        db_connection,
69        DEFAULT_NAMESPACE.to_owned(),
70    )?;
71
72    // 2. Create shared operation trackers for admin endpoints
73    let (indexing_ops, validation_ops) = resolve_admin_operation_trackers()?;
74
75    // 3. Build MCP services struct from registry-resolved services
76    let mcp_services = build_mcp_services(registry_ctx, &repos, hybrid_search)?;
77
78    let vcs_for_defaults = Arc::clone(&mcp_services.vcs);
79    let mcp_server = Arc::new(McpServer::new(
80        mcp_services,
81        &vcs_for_defaults,
82        Some(execution_flow),
83    ));
84
85    // 5. Build bootstrap with shared ports from context
86    Ok(McpServerBootstrap {
87        mcp_server,
88        dashboard: repos.dashboard,
89        auth_repo: repos.auth,
90        embedding_provider,
91        vector_store: vector_store_provider,
92        indexing_ops,
93        validation_ops,
94    })
95}

97/// Resolve the shared indexing/validation operation trackers used by admin endpoints.
98fn resolve_admin_operation_trackers() -> mcb_domain::Result<(
99    Arc<dyn IndexingOperationsInterface>,
100    Arc<dyn ValidationOperationsInterface>,
101)> {
102    let indexing_ops = resolve_indexing_operations_provider(
103        &IndexingOperationsProviderConfig::new(mcb_utils::constants::DEFAULT_INDEXING_OP_PROVIDER),
104    )?;
105    let validation_ops =
106        resolve_validation_operations_provider(&ValidationOperationsProviderConfig::new(
107            mcb_utils::constants::DEFAULT_VALIDATION_OP_PROVIDER,
108        ))?;
109    Ok((indexing_ops, validation_ops))
110}

112/// Resolve all MCP services via the linkme registry and assemble [`McpServices`].
113fn build_mcp_services(
114    registry_ctx: &dyn std::any::Any,
115    repos: &DatabaseRepositories,
116    hybrid_search: Arc<dyn HybridSearchProvider>,
117) -> mcb_domain::Result<McpServices> {
118    Ok(McpServices {
119        indexing: resolve_indexing_service(registry_ctx)?,
120        context: resolve_context_service(registry_ctx)?,
121        search: resolve_search_service(registry_ctx)?,
122        validation: resolve_validation_service(registry_ctx)?,
123        memory: resolve_memory_service(registry_ctx)?,
124        agent_session: resolve_agent_session_service(registry_ctx)?,
125        project: resolve_project_detection_service(&ProjectDetectionServiceConfig::new(
126            DEFAULT_LANGUAGE_PROVIDER,
127        ))?,
128        project_workflow: Arc::clone(&repos.project),
129        vcs: resolve_vcs_provider(&VcsProviderConfig::new(DEFAULT_VCS_PROVIDER))?,
130        hybrid_search,
131        entities: McpEntityRepositories {
132            vcs: Arc::clone(&repos.vcs_entity),
133            plan: Arc::clone(&repos.plan_entity),
134            issue: Arc::clone(&repos.issue_entity),
135            org: Arc::clone(&repos.org_entity),
136        },
137    })
138}
