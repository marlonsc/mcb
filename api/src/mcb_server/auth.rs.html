17// Support both direct app routes (`/alive`) and prefixed ingress rewrites (`/api/alive`).
18const ADMIN_AUTH_EXEMPT_PATHS: &[&str] = &["/alive", "/api/alive"];

20/// Authenticated admin principal.
21#[derive(Debug, Clone, PartialEq, Eq)]
22pub struct AdminPrincipal {
23    /// User ID from the `users` table.
24    pub user_id: String,
25    /// User email from the `users` table.
26    pub email: String,
27    /// User role from the `users` table.
28    pub role: String,
29}
30
31/// Authorize admin access using API key.
32///
33/// This checks `X-API-Key` (or `Authorization: Bearer <key>` fallback) against
34/// `users.api_key_hash` entries.
35///
36/// # Errors
37///
38/// Returns `Unauthorized` when the key is missing or invalid.
39pub async fn authorize_admin_api_key(
40    auth_repo: &dyn AuthRepositoryPort,
41    headers: &HeaderMap,
42    settings: Option<&serde_json::Value>,
43) -> Result<AdminPrincipal> {
44    let api_key_header = configured_api_key_header(settings);
45    let api_key = extract_api_key(headers, &api_key_header)?;
46
47    let users_with_keys = auth_repo
48        .find_active_api_key_candidates()
49        .await
50        .map_err(|e| {
51            mcb_domain::error!("auth", "auth repository lookup failed", &e);
52            Error::InternalServerError
53        })?;
54
55    for user_with_key in users_with_keys {
56        if verify_api_key(&user_with_key.api_key_hash, &api_key)? {
57            let user = user_with_key.user;
58            return Ok(AdminPrincipal {
59                user_id: user.id,
60                email: user.email,
61                role: user.role.to_string(),
62            });
63        }
64    }
65
66    Err(Error::Unauthorized("invalid api key".to_owned()))
67}

69pub(crate) fn configured_api_key_header(settings: Option<&serde_json::Value>) -> String {
70    // Try admin-specific header first (settings.auth.admin.header), then
71    // fall back to the general API-key header (settings.auth.api_key.header).
72    let auth = settings.and_then(|s| s.get("auth"));
73    auth.and_then(|a| a.get("admin"))
74        .and_then(|admin| admin.get("header"))
75        .and_then(serde_json::Value::as_str)
76        .or_else(|| {
77            auth.and_then(|a| a.get("api_key"))
78                .and_then(|api_key| api_key.get("header"))
79                .and_then(serde_json::Value::as_str)
80        })
81        .map_or_else(|| API_KEY_HEADER.to_owned(), str::to_ascii_lowercase)
82}

90/// Extract API key from headers, checking both custom header and Authorization bearer.
91///
92/// # Arguments
93/// * `headers` - HTTP headers to search
94/// * `header_name` - Name of the custom header to check first
95///
96/// # Errors
97///
98/// Returns `Unauthorized` when the key is missing or header value is invalid.
99pub fn extract_api_key(headers: &HeaderMap, header_name: &str) -> Result<String> {
100    if let Some(value) = headers.get(header_name) {
101        let key = value
102            .to_str()
103            .map_err(|_| Error::Unauthorized("invalid api key header value".to_owned()))?
104            .trim();
105        if !key.is_empty() {
106            return Ok(key.to_owned());
107        }
108    }
109
110    if let Some(value) = headers.get(HTTP_HEADER_AUTHORIZATION) {
111        let value = value
112            .to_str()
113            .map_err(|_| Error::Unauthorized("invalid authorization header value".to_owned()))?;
114        if let Some(token) = value.strip_prefix(BEARER_PREFIX) {
115            let key = token.trim();
116            if !key.is_empty() {
117                return Ok(key.to_owned());
118            }
119        }
120    }
121
122    Err(Error::Unauthorized(format!(
123        "missing api key header ({header_name} or authorization bearer)"
124    )))
125}

127fn verify_api_key(hash: &str, candidate: &str) -> Result<bool> {
128    if let Ok(parsed) = PasswordHash::new(hash) {
129        return Ok(Argon2::default()
130            .verify_password(candidate.as_bytes(), &parsed)
131            .is_ok());
132    }
133
134    if hash.starts_with("$2a$") || hash.starts_with("$2b$") || hash.starts_with("$2y$") {
135        return bcrypt::verify(candidate, hash).map_err(|e| {
136            mcb_domain::error!("auth", "bcrypt verification failed", &e);
137            Error::InternalServerError
138        });
139    }
140
141    tracing::warn!(hash_prefix = %hash.chars().take(4).collect::<String>(), "unrecognized password hash format");
142    Ok(false)
143}
