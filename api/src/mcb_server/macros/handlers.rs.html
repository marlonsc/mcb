9/// Extract a value from a `Result<T, CallToolResult>`, returning the error as `Ok(err)`.
10///
11/// The `require_*` helpers return `Result<T, CallToolResult>` where the `Err` variant
12/// is a tool-level error (not a protocol error). This macro converts the match:
13///
14/// ```ignore
15/// let value = match require_str(data, "field") {
16///     Ok(v) => v,
17///     Err(error_result) => return Ok(error_result),
18/// };
19/// ```
20///
21/// Into a single expression:
22///
23/// ```ignore
24/// let value = extract_field!(require_str(data, "field"));
25/// ```
26macro_rules! extract_field {
27    ($expr:expr) => {
28        match $expr {
29            Ok(value) => value,
30            Err(error_result) => return Ok(error_result),
31        }
32    };
33}

35/// Parse a string into an enum via `FromStr`, returning a tool error on failure.
36///
37/// Replaces the common pattern:
38///
39/// ```ignore
40/// let typed: MyEnum = match string_val.parse() {
41///     Ok(v) => v,
42///     Err(_) => return Ok(tool_error(format!("Invalid field: {string_val}"))),
43/// };
44/// ```
45///
46/// With:
47///
48/// ```ignore
49/// let typed: MyEnum = parse_enum!(string_val, "field");
50/// ```
51macro_rules! parse_enum {
52    ($str_val:expr, $label:expr) => {
53        match $str_val.parse() {
54            Ok(v) => v,
55            Err(_) => {
56                return Ok($crate::utils::mcp::tool_error(format!(
57                    "Invalid {}: {}",
58                    $label, $str_val
59                )));
60            }
61        }
62    };
63}

65/// Extract a required argument from an `Option<String>`, returning `McpError` on `None`.
66///
67/// Replaces the repeated pattern in entity handlers:
68///
69/// ```ignore
70/// let project_id = args.project_id.as_deref().ok_or_else(|| {
71///     McpError::invalid_params("project_id required for list", None)
72/// })?;
73/// ```
74///
75/// With:
76///
77/// ```ignore
78/// let project_id = require_arg!(args.project_id, "project_id required for list");
79/// ```
80macro_rules! require_arg {
81    ($opt:expr, $msg:literal) => {
82        $opt.as_deref()
83            .ok_or_else(|| McpError::invalid_params($msg, None))?
84    };
85}

87/// Extract a required `Option` field from `AdminState`, returning
88/// `AdminError::unavailable` when `None`.
89///
90/// ```ignore
91/// let cache = require_service!(state, cache, "Cache provider not available");
92/// ```
93macro_rules! require_service {
94    ($state:expr, $field:ident, $msg:literal) => {
95        match $state.$field {
96            Some(ref svc) => svc,
97            None => return Err($crate::admin::error::AdminError::unavailable($msg)),
98        }
99    };
100}

102/// Generate a simple template page handler that renders a Handlebars template
103/// with standard nav context.
104///
105/// ```ignore
106/// template_page!(config_page, "admin/config", "Configuration", "config");
107/// ```
108macro_rules! template_page {
109    ($fn_name:ident, $template:literal, $title:literal, $page:literal) => {
110        #[allow(missing_docs)]
111        pub async fn $fn_name() -> $crate::templates::Template {
112            mcb_domain::info!("web", concat!(stringify!($fn_name), " called"));
113            $crate::templates::Template::render(
114                $template,
115                context! {
116                    title: $title,
117                    current_page: $page,
118                    nav_groups: $crate::admin::web::view_model::nav_groups(),
119                },
120            )
121        }
122    };
123}

151/// Generate a handler constructor that extracts dependencies from `AppContext`.
152///
153/// This macro eliminates boilerplate `pub fn new(...)` constructors by automatically
154/// generating them from a list of field names and their types.
155///
156/// # Example
157///
158/// ```ignore
159/// handler_new!(MyHandler {
160///     repo: Arc<dyn MyRepository>,
161///     service: Arc<dyn MyService>,
162/// });
163/// ```
164///
165/// Expands to:
166///
167/// ```ignore
168/// impl MyHandler {
169///     pub fn new(repo: Arc<dyn MyRepository>, service: Arc<dyn MyService>) -> Self {
170///         Self {
171///             repo,
172///             service,
173///         }
174///     }
175/// }
176/// ```
177macro_rules! handler_new {
178    (
179        $handler_name:ident {
180            $($field:ident: $field_ty:ty),* $(,)?
181        }
182    ) => {
183        impl $handler_name {
184            /// Create a new handler with the given dependencies.
185            pub fn new($($field: $field_ty),*) -> Self {
186                Self {
187                    $($field),*
188                }
189            }
190        }
191    };
