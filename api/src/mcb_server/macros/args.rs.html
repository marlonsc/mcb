8/// Define a tool argument struct with auto-derives
9macro_rules! tool_schema {
10    ($(#[$meta:meta])* $vis:vis struct $name:ident { $($fields:tt)* }) => {
11        $(#[$meta])*
12
13        #[derive(Debug, Clone, Deserialize, JsonSchema, Validate)]
14        $vis struct $name {
15            $($fields)*
16        }
17    };
18}
19
20/// Define a tool enum with auto-derives and serde config
21macro_rules! tool_enum {
22    ($(#[$meta:meta])* $vis:vis enum $name:ident { $($variants:tt)* }) => {
23        $(#[$meta])*
24
25        #[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, JsonSchema)]
26        #[serde(rename_all = "snake_case")]
27        $vis enum $name {
28            $($variants)*
29        }
30    };
31}
32
33/// Define a standard CRUD action enum (Create, Get, Update, List, Delete) plus extras
34macro_rules! tool_crud_action_enum {
35    ($(#[$meta:meta])* $vis:vis enum $name:ident { $($extra:tt)* }) => {
36        tool_enum! {
37            $(#[$meta])*
38            $vis enum $name {
39                /// Create a new entity.
40                Create,
41                /// Get an entity by ID.
42                Get,
43                /// Update an existing entity.
44                Update,
45                /// List entities matching criteria.
46                List,
47                /// Delete an entity by ID.
48                Delete,
49                $($extra)*
50            }
51        }
52    };
53}

55/// Define a unified entity args schema with action and resource types
56macro_rules! entity_args_schema {
57    (
58        $(#[$meta:meta])*
59        $vis:vis struct $name:ident {
60            action: $action_ty:ty,
61            action_desc: $action_desc:literal,
62            resource: $resource_ty:ty,
63            resource_desc: $resource_desc:literal,
64            $(
65                $(#[$field_meta:meta])*
66                $field:ident: $field_ty:ty => $field_desc:literal,
67            )*
68        }
69    ) => {
70        tool_schema! {
71            $(#[$meta])*
72            $vis struct $name {
73                #[doc = $action_desc]
74                #[schemars(description = $action_desc)]
75                pub action: $action_ty,
76
77                #[doc = $resource_desc]
78                #[schemars(description = $resource_desc)]
79                pub resource: $resource_ty,
80
81                #[doc = "Resource ID (for get/update/delete)"]
82                #[schemars(description = "Resource ID (for get/update/delete)")]
83                pub id: Option<String>,
84
85                #[doc = "Organization ID (uses default if omitted)"]
86                #[schemars(description = "Organization ID (uses default if omitted)")]
87                pub org_id: Option<String>,
88
89                $(
90                    $(#[$field_meta])*
91                    #[schemars(description = $field_desc)]
92                    pub $field: $field_ty,
93                )*
94
95                #[doc = "Data payload for create/update (JSON object)"]
96                #[schemars(
97                    description = "Data payload for create/update (JSON object)",
98                    with = "serde_json::Value"
99                )]
100                pub data: Option<serde_json::Value>,
101            }
102        }
103    };
