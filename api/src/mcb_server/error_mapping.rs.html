10/// Logs the underlying error server-side and returns a generic internal error.
11///
12/// Use this instead of `McpError::internal_error(e.to_string(), None)` so that
13/// underlying error details are never leaked to MCP clients.
14pub fn safe_internal_error(context: &str, err: &dyn std::fmt::Display) -> McpError {
15    error!(context, "internal operation failed", err);
16    McpError::internal_error("internal server error", None)
17}
18
19/// Maps a domain error to an opaque MCP error.
20///
21/// # Behavior
22/// Returns client-fixable errors for `NotFound` and `InvalidArgument`; all
23/// other variants are converted to a generic internal error.
24///
25/// # Security
26/// Never exposes provider/internal details to external callers.
27#[must_use]
28pub fn to_opaque_mcp_error(e: &Error) -> McpError {
29    error!("McpError", "operation failed", e);
30    if matches!(e, Error::NotFound { .. } | Error::InvalidArgument { .. }) {
31        McpError::invalid_params(e.to_string(), None)
32    } else {
33        McpError::internal_error("internal server error", None)
34    }
35}

37/// Builds a contextual MCP tool error response from a domain error.
38///
39/// # Behavior
40/// Applies categorized mappers (client, provider, config, system, encoding,
41/// IO, generic) and returns the first matching sanitized message.
42///
43/// # Security
44/// Logs internal details server-side and returns safe text only.
45pub fn to_contextual_tool_error(e: impl Into<Error>) -> CallToolResult {
46    let error: Error = e.into();
47    let message = groups::map_error_message(&error).unwrap_or_else(|| {
48        error!("ErrorMapping", "unmapped error variant", &error);
49        "Internal error".to_owned()
50    });
51    CallToolResult::error(vec![Content::text(message)])
52}
