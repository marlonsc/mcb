14fn json_sort_key(v: &Value) -> String {
15    match v {
16        Value::String(s) => format!("s{}", s.to_lowercase()),
17        Value::Number(n) => {
18            if let Some(i) = n.as_i64() {
19                // Shift by i64::MIN so negative values sort correctly with zero-padding.
20                format!("n{:020}", (i as u64).wrapping_add(i64::MIN as u64))
21            } else if let Some(u) = n.as_u64() {
22                format!("n{:020}", u.wrapping_add(i64::MIN as u64))
23            } else {
24                format!("n{:+021.6e}", n.as_f64().unwrap_or(0.0))
25            }
26        }
27        Value::Bool(b) => format!("b{b}"),
28        Value::Null => String::new(),
29        other @ (Value::Array(_) | Value::Object(_)) => format!("x{other}"),
30    }
31}

33fn apply_search_filter(records: &mut Vec<Value>, search: Option<&str>) {
34    let Some(query) = search.filter(|value| !value.is_empty()) else {
35        return;
36    };
37
38    let query_lower = query.to_lowercase();
39    records.retain(|rec| {
40        if let Value::Object(map) = rec {
41            map.values().any(|v| match v {
42                Value::String(s) => s.to_lowercase().contains(&query_lower),
43                Value::Null
44                | Value::Bool(_)
45                | Value::Number(_)
46                | Value::Array(_)
47                | Value::Object(_) => v.to_string().to_lowercase().contains(&query_lower),
48            })
49        } else {
50            false
51        }
52    });
53}

55fn parse_date_range(params: &FilterParams) -> (Option<i64>, Option<i64>) {
56    let epoch_from = params.date_from.as_deref().and_then(|s| {
57        if s.is_empty() {
58            None
59        } else {
60            parse_iso_date_to_epoch(s)
61        }
62    });
63    let epoch_to = params.date_to.as_deref().and_then(|s| {
64        if s.is_empty() {
65            None
66        } else {
67            parse_iso_date_to_epoch_end(s)
68        }
69    });
70    (epoch_from, epoch_to)
71}

73fn record_matches_date_range(rec: &Value, epoch_from: Option<i64>, epoch_to: Option<i64>) -> bool {
74    if let Value::Object(map) = rec {
75        let mut has_any_ts = false;
76        let in_range = map
77            .iter()
78            .filter(|(k, _)| k.ends_with("_at"))
79            .filter_map(|(_, v)| match v {
80                Value::Number(n) => n.as_i64(),
81                Value::Null
82                | Value::Bool(_)
83                | Value::String(_)
84                | Value::Array(_)
85                | Value::Object(_) => None,
86            })
87            .any(|ts| {
88                has_any_ts = true;
89                let after = epoch_from.is_none_or(|from| ts >= from);
90                let before = epoch_to.is_none_or(|to| ts <= to);
91                after && before
92            });
93        in_range || !has_any_ts
94    } else {
95        true
96    }
97}

99fn apply_date_filter(records: &mut Vec<Value>, epoch_from: Option<i64>, epoch_to: Option<i64>) {
100    if epoch_from.is_none() && epoch_to.is_none() {
101        return;
102    }
103    records.retain(|rec| record_matches_date_range(rec, epoch_from, epoch_to));
104}

106fn apply_sort(records: &mut [Value], params: &FilterParams, valid_sort_fields: &HashSet<String>) {
107    let Some(field) = params.sort_field.as_ref() else {
108        return;
109    };
110    if !valid_sort_fields.contains(field.as_str()) {
111        return;
112    }
113
114    let desc = matches!(params.sort_order, Some(SortOrder::Desc));
115    records.sort_by(|a, b| {
116        let va = a.get(field).map(json_sort_key);
117        let vb = b.get(field).map(json_sort_key);
118        let cmp = va.cmp(&vb);
119        if desc { cmp.reverse() } else { cmp }
120    });
121}

123fn paginate(records: &[Value], params: &FilterParams) -> FilteredResult {
124    let total_count = records.len();
125    let per_page = if params.per_page == 0 {
126        20
127    } else {
128        params.per_page
129    };
130    let total_pages = if total_count == 0 {
131        0
132    } else {
133        total_count.div_ceil(per_page)
134    };
135    let page = if params.page == 0 { 1 } else { params.page };
136    let start = (page - 1) * per_page;
137    let page_records = if start >= total_count {
138        Vec::new()
139    } else {
140        let end = (start + per_page).min(total_count);
141        records[start..end].to_vec()
142    };
143
144    FilteredResult {
145        records: page_records,
146        total_count,
147        page,
148        per_page,
149        total_pages,
150    }
151}

153/// Apply in-memory filtering, sorting, and pagination to a pre-fetched record list.
154#[must_use]
155pub fn apply_filter_pipeline(
156    mut records: Vec<Value>,
157    params: &FilterParams,
158    valid_sort_fields: &HashSet<String>,
159) -> FilteredResult {
160    apply_search_filter(&mut records, params.search.as_deref());
161    let (epoch_from, epoch_to) = parse_date_range(params);
162    apply_date_filter(&mut records, epoch_from, epoch_to);
163    apply_sort(&mut records, params, valid_sort_fields);
164    paginate(&records, params)
165}
