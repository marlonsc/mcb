22fn find_or_404(slug: &str) -> Result<&'static crate::admin::registry::AdminEntityMeta, AdminError> {
23    AdminRegistry::find(slug)
24        .ok_or_else(|| AdminError::not_found(format!("Unknown entity slug: {slug}")))
25}

27/// Entity catalog page — lists all registered entities with field counts.
28pub async fn entities_index(State(state): State<AdminState>) -> Template {
29    let mut entities = Vec::new();
30    let mut total_records: usize = 0;
31
32    for entity in AdminRegistry::all() {
33        let record_count = match resolve_adapter(entity.slug, &state) {
34            Some(adapter) => match adapter.list_all().await {
35                Ok(v) => v.len(),
36                Err(e) => {
37                    warn!("AdminWeb", "list_all failed", &e);
38                    0
39                }
40            },
41            None => 0,
42        };
43        total_records += record_count;
44        entities.push(crate::admin::web::view_model::DashboardEntityCard {
45            slug: entity.slug.to_owned(),
46            title: entity.title.to_owned(),
47            group: entity.group.to_owned(),
48            field_count: entity.fields().iter().filter(|field| !field.hidden).count(),
49            record_count,
50        });
51    }
52
53    let entity_count = entities.len();
54    let group_count = nav_groups().len();
55
56    Template::render(
57        "admin/entity_index",
58        context! {
59            title: "Entities",
60            current_page: "entities",
61            entities: entities,
62            entity_count: entity_count,
63            group_count: group_count,
64            total_records: total_records,
65            nav_groups: nav_groups(),
66        },
67    )
68}

70/// Entity list page with filtering, sorting, and pagination.
71///
72/// # Errors
73/// Returns `404` when entity slug is not registered.
74pub async fn entities_list(
75    Path(slug): Path<String>,
76    Query(params): Query<FilterParams>,
77    State(state): State<AdminState>,
78) -> Result<Template, AdminError> {
79    let entity = find_or_404(&slug)?;
80
81    let fields = entity
82        .fields()
83        .into_iter()
84        .filter(|field| !field.hidden)
85        .collect::<Vec<_>>();
86    let field_names = fields
87        .iter()
88        .map(|field| field.name.clone())
89        .collect::<Vec<_>>();
90    let valid_sort_fields = field_names.iter().cloned().collect::<HashSet<_>>();
91
92    let result = match resolve_adapter(&slug, &state) {
93        Some(adapter) => adapter
94            .list_filtered(&params, &valid_sort_fields)
95            .await
96            .unwrap_or_default(),
97        None => crate::admin::web::filter::FilteredResult {
98            records: Vec::new(),
99            total_count: 0,
100            page: params.page,
101            per_page: params.per_page,
102            total_pages: 0,
103        },
104    };
105    let has_records = !result.records.is_empty();
106
107    Ok(Template::render(
108        "admin/entity_list",
109        context! {
110            title: entity.title,
111            current_page: entity.slug,
112            entity_slug: entity.slug,
113            entity_group: entity.group,
114            fields: fields,
115            field_names: field_names,
116            records: result.records,
117            has_records: has_records,
118            total_count: result.total_count,
119            page: result.page,
120            per_page: result.per_page,
121            total_pages: result.total_pages,
122            nav_groups: nav_groups(),
123        },
124    ))
125}

155/// Detail view — fetches a single record via service adapter when available.
156///
157/// # Errors

159pub async fn entities_detail(

174        },
175        None => (serde_json::Value::Null, false),
176    };
177
178    Ok(Template::render(
179        "admin/entity_detail",
180        context! {
181            title: entity.title,
182            current_page: entity.slug,
183            entity_slug: entity.slug,
184            entity_group: entity.group,
185            entity_id: id,
186            fields: fields,
187            record: record,
188            has_record: has_record,
189            nav_groups: nav_groups(),
190        },
191    ))
192}

272/// Create entity — persists via service adapter and redirects to the list page.
273///
274/// # Errors
275/// Returns `404` for unknown entity slugs, `400` for invalid form payloads, and `500` for persistence failures.
276pub async fn entities_create(
277    Path(slug): Path<String>,
278    State(state): State<AdminState>,
279    Form(form): Form<std::collections::HashMap<String, String>>,
280) -> Result<Redirect, AdminError> {
281    find_or_404(&slug)?;
282
283    if let Some(adapter) = resolve_adapter(&slug, &state) {
284        let data =
285            serde_json::to_value(form).map_err(|e| AdminError::bad_request(e.to_string()))?;
286        adapter
287            .create_from_json(data)
288            .await
289            .map_err(AdminError::internal)?;
290    }
291
292    let url = format!("/ui/entities/{slug}?toast=created");
293    Ok(Redirect::to(&url))
294}

296/// Update entity — persists via service adapter and redirects to the detail page.
297///
298/// # Errors
299/// Returns `404` for unknown entity slugs, `400` for invalid form payloads, and `500` for persistence failures.
300pub async fn entities_update(
301    Path((slug, id)): Path<(String, String)>,
302    State(state): State<AdminState>,
303    Form(form): Form<std::collections::HashMap<String, String>>,
304) -> Result<Redirect, AdminError> {
305    find_or_404(&slug)?;
306
307    if let Some(adapter) = resolve_adapter(&slug, &state) {
308        let mut map = form;
309        map.insert("id".to_owned(), id.clone());
310        let data = serde_json::to_value(map).map_err(|e| AdminError::bad_request(e.to_string()))?;
311        adapter
312            .update_from_json(data)
313            .await
314            .map_err(AdminError::internal)?;
315    }
316
317    let url = format!("/ui/entities/{slug}/{id}?toast=updated");
318    Ok(Redirect::to(&url))
319}

321/// Bulk delete — deletes multiple records by comma-separated IDs.
322///
323/// # Errors
324/// Returns `404` when entity slug is not registered.
325pub async fn entities_bulk_delete(
326    Path(slug): Path<String>,
327    State(state): State<AdminState>,
328    Form(form): Form<std::collections::HashMap<String, String>>,
329) -> Result<Redirect, AdminError> {
330    find_or_404(&slug)?;
331
332    let ids_raw = form.get("ids").map_or("", std::string::String::as_str);
333    let ids = ids_raw
334        .split(',')
335        .map(str::trim)
336        .filter(|s| !s.is_empty())
337        .collect::<Vec<_>>();
338
339    if let Some(adapter) = resolve_adapter(&slug, &state) {
340        let total = ids.len();
341        let mut failed = 0usize;
342        for id in &ids {
343            if adapter.delete_by_id(id).await.is_err() {
344                failed += 1;
345            }
346        }
347        let success = total - failed;
348        let toast = if failed == 0 {
349            format!("bulk_deleted&count={success}")
350        } else if success == 0 {
351            format!("bulk_error&count={failed}")
352        } else {
353            format!("bulk_partial&success={success}&failed={failed}")
354        };
355        let url = format!("/ui/entities/{slug}?toast={toast}");
356        return Ok(Redirect::to(&url));
357    }
358
359    let url = format!("/ui/entities/{slug}?toast=bulk_error&count=0");
360    Ok(Redirect::to(&url))
361}
