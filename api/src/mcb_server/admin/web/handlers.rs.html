89#[derive(Debug, Clone, Serialize)]
90struct RecentActivityItem {
91    entity_title: String,
92    record_count: usize,
93    timestamp: i64,
94}
95
96async fn render_dashboard_template(title: &str, state: Option<&AdminState>) -> Template {
97    let mut cards = Vec::<DashboardEntityCard>::new();
98    let mut recent_activity = Vec::<RecentActivityItem>::new();
99    let now_ts = chrono::Utc::now().timestamp();
100    let mut total_records = 0usize;
101
102    for entity in AdminRegistry::all() {
103        let record_count = match state.and_then(|s| resolve_adapter(entity.slug, s)) {
104            Some(adapter) => match adapter.list_all().await {
105                Ok(rows) => rows.len(),
106                Err(e) => {
107                    warn!("AdminWeb", "list_all failed", &e);
108                    0
109                }
110            },
111            None => 0,
112        };
113        total_records += record_count;
114
115        let field_count = entity.fields().iter().filter(|field| !field.hidden).count();
116        cards.push(DashboardEntityCard {
117            slug: entity.slug.to_owned(),
118            title: entity.title.to_owned(),
119            group: entity.group.to_owned(),
120            field_count,
121            record_count,
122        });
123
124        if record_count > 0 {
125            recent_activity.push(RecentActivityItem {
126                entity_title: entity.title.to_owned(),
127                record_count,
128                timestamp: now_ts,
129            });
130        }
131    }
132
133    recent_activity.truncate(10);
134    let entity_count = cards.len();
135    let active_entity_count = cards.iter().filter(|card| card.record_count > 0).count();
136
137    Template::render(
138        "admin/dashboard",
139        context! {
140            title: title,
141            current_page: "dashboard",
142            nav_groups: nav_groups(),
143            entity_cards: cards,
144            entity_count: entity_count,
145            active_entity_count: active_entity_count,
146            total_records: total_records,
147            recent_activity: recent_activity,
148        },
149    )
150}
