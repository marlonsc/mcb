6use serde::{Deserialize, Serialize};

8/// Column sort direction.
9#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, Default)]
10#[serde(rename_all = "lowercase")]
11pub enum SortOrder {
12    /// Ascending (A→Z, 0→9).
13    #[default]
14    Asc,
15    /// Descending (Z→A, 9→0).
16    Desc,
17}
18
19/// Query-string parameters for entity list pages.
20///
21/// Parsed from query parameters: `?q=&sort=&order=&page=&per_page=&parent_field=&parent_id=&date_from=&date_to=`.
22#[derive(Debug, Clone, Deserialize, Serialize)]
23pub struct FilterParams {
24    /// FK field name to scope by (e.g. `"team_id"`).
25    pub parent_field: Option<String>,
26    /// Value of the parent FK (e.g. the team's UUID).
27    pub parent_id: Option<String>,
28    /// Full-text search term (from query param "q").
29    #[serde(rename = "q", default)]
30    pub search: Option<String>,
31    /// Column to sort by (validated against `AdminFieldMeta` names, from query param "sort").
32    #[serde(rename = "sort", default)]
33    pub sort_field: Option<String>,
34    /// Sort direction (from query param "order").
35    #[serde(rename = "order", default)]
36    pub sort_order: Option<SortOrder>,
37    /// 1-based page number (default 1).
38    #[serde(default = "default_page")]
39    pub page: usize,
40    /// Records per page (default 20).
41    #[serde(default = "default_per_page")]
42    pub per_page: usize,
43    /// ISO date string lower bound for timestamp filtering (e.g. "2026-01-15").
44    pub date_from: Option<String>,
45    /// ISO date string upper bound for timestamp filtering (e.g. "2026-02-11").
46    pub date_to: Option<String>,
47}
48
49fn default_page() -> usize {
50    1
51}
52
53fn default_per_page() -> usize {
54    20
55}

57/// Parse an ISO date string ("YYYY-MM-DD") to a Unix epoch timestamp (start of day UTC).
58/// Returns `None` if the string is empty or unparseable.
59#[must_use]
60pub fn parse_iso_date_to_epoch(date_str: &str) -> Option<i64> {
61    chrono::NaiveDate::parse_from_str(date_str.trim(), "%Y-%m-%d")
62        .ok()
63        .and_then(|d| d.and_hms_opt(0, 0, 0))
64        .map(|dt| dt.and_utc().timestamp())
65}
66
67/// Parse an ISO date string to end-of-day epoch (23:59:59 UTC).
68#[must_use]
69pub fn parse_iso_date_to_epoch_end(date_str: &str) -> Option<i64> {
70    chrono::NaiveDate::parse_from_str(date_str.trim(), "%Y-%m-%d")
71        .ok()
72        .and_then(|d| d.and_hms_opt(23, 59, 59))
73        .map(|dt| dt.and_utc().timestamp())
74}

76/// Paginated result returned by `EntityCrudAdapter::list_filtered`.
77#[derive(Debug, Clone, Default, Serialize)]
78pub struct FilteredResult {
79    /// Records for the current page.
80    pub records: Vec<serde_json::Value>,
81    /// Total records matching filter (before pagination).
82    pub total_count: usize,
83    /// Current 1-based page.
84    pub page: usize,
85    /// Page size.
86    pub per_page: usize,
87    /// Computed ceiling of `total_count / per_page`.
88    pub total_pages: usize,
89}
