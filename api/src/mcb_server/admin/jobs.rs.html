18/// Jobs status response (unified job tracking)
19#[derive(Serialize)]
20pub struct JobsStatusResponse {
21    /// Total number of tracked jobs
22    pub total: usize,
23    /// Number of currently running jobs
24    pub running: usize,
25    /// Number of queued jobs
26    pub queued: usize,
27    /// Job details
28    pub jobs: Vec<Job>,
29}
30
31fn build_jobs_response(
32    indexing: &dyn mcb_domain::ports::IndexingOperationsInterface,
33) -> JobsStatusResponse {
34    let operations = indexing.get_operations();
35
36    let jobs = operations
37        .values()
38        .map(|op| {
39            let progress = if op.total_files > 0 {
40                ((op.processed_files as f64 / op.total_files as f64) * 100.0) as u8
41            } else {
42                0
43            };
44            Job {
45                id: op.id,
46                job_type: JobType::Indexing,
47                label: op.collection.to_string(),
48                status: JobStatus::Running,
49                progress_percent: progress,
50                processed_items: op.processed_files,
51                total_items: op.total_files,
52                current_item: op.current_file.clone(),
53                created_at: op.started_at,
54                started_at: Some(op.started_at),
55                completed_at: None,
56                result: None,
57            }
58        })
59        .collect::<Vec<_>>();
60
61    let running = jobs.len();
62    JobsStatusResponse {
63        total: running,
64        running,
65        queued: 0,
66        jobs,
67    }
68}
69
70/// Axum handler: get indexing job status.
71pub async fn get_jobs_status_axum(
72    AxumState(state): AxumState<Arc<AdminState>>,
73) -> AxumJson<JobsStatusResponse> {
74    info!("jobs", "get_jobs_status called");
75    AxumJson(build_jobs_response(state.indexing.as_ref()))
76}
