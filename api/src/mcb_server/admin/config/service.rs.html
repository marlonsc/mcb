14/// Internal error type for config update operations
15#[derive(Debug)]
16pub enum ConfigUpdateError {
17    /// Invalid configuration section
18    InvalidSection,
19    /// Configuration watcher is not running
20    WatcherUnavailable,
21    /// Configuration file path is not set
22    PathUnavailable,
23    /// Failed to read configuration file
24    ReadFailed(String),
25    /// Failed to parse configuration file
26    ParseFailed(String),
27    /// Invalid configuration value format
28    InvalidFormat,
29    /// Failed to serialize configuration
30    SerializeFailed(String),
31    /// Failed to write configuration file
32    WriteFailed(String),
33    /// Configuration reload failed after write
34    ReloadFailed(String),
35}

37/// Validate prerequisites for config update
38///
39/// # Errors
40/// Returns an error when section is invalid or watcher/path are unavailable.
41pub fn validate_update_prerequisites(
42    section: &str,
43    watcher: Option<&Arc<ConfigWatcher>>,
44    config_path: Option<&PathBuf>,
45) -> Result<(Arc<ConfigWatcher>, PathBuf), ConfigUpdateError> {
46    use super::is_valid_section;
47
48    if !is_valid_section(section) {
49        return Err(ConfigUpdateError::InvalidSection);
50    }
51
52    let watcher = watcher
53        .cloned()
54        .ok_or(ConfigUpdateError::WatcherUnavailable)?;
55    let config_path = config_path
56        .cloned()
57        .ok_or(ConfigUpdateError::PathUnavailable)?;
58
59    Ok((watcher, config_path))
60}

62/// Read, parse, and update the configuration
63///
64/// # Errors
65/// Returns an error when reading, parsing, or value conversion fails.
66pub fn read_update_config(
67    config_path: &Path,
68    section: &str,
69    values: &serde_json::Value,
70) -> Result<toml::Value, ConfigUpdateError> {
71    let content = std::fs::read_to_string(config_path)
72        .map_err(|e| ConfigUpdateError::ReadFailed(e.to_string()))?;
73
74    let mut config: toml::Value =
75        toml::from_str(&content).map_err(|e| ConfigUpdateError::ParseFailed(e.to_string()))?;
76
77    let toml_value = json_to_toml(values).ok_or(ConfigUpdateError::InvalidFormat)?;
78
79    if let Some(table) = config.as_table_mut() {
80        merge_section(table, section, toml_value);
81    }
82
83    Ok(config)
84}

86/// Merge new values into a config section
87fn merge_section(
88    table: &mut toml::map::Map<String, toml::Value>,
89    section: &str,
90    new_value: toml::Value,
91) {
92    let toml::Value::Table(new_table) = new_value else {
93        table.insert(section.to_owned(), new_value);
94        return;
95    };
96
97    let Some(existing) = table.get_mut(section).and_then(|v| v.as_table_mut()) else {
98        table.insert(section.to_owned(), toml::Value::Table(new_table));
99        return;
100    };
101
102    for (key, value) in new_table {
103        existing.insert(key, value);
104    }
105}

107/// Write config to file and reload
108///
109/// # Errors
110/// Returns an error when serialization, write, or reload fails.
111pub async fn write_and_reload_config(
112    config_path: &Path,
113    config: &toml::Value,
114    watcher: &ConfigWatcher,
115) -> Result<SanitizedConfig, ConfigUpdateError> {
116    let content = toml::to_string_pretty(config)
117        .map_err(|e| ConfigUpdateError::SerializeFailed(e.to_string()))?;
118
119    std::fs::write(config_path, content)
120        .map_err(|e| ConfigUpdateError::WriteFailed(e.to_string()))?;
121
122    let new_config = watcher
123        .reload()
124        .await
125        .map_err(|e| ConfigUpdateError::ReloadFailed(e.to_string()))?;
126
127    Ok(SanitizedConfig::from_app_config(&new_config))
128}
