6use std::collections::HashSet;
7
8use async_trait::async_trait;
9use serde_json::Value;
10
11use crate::admin::web::filter::{FilterParams, FilteredResult};
12use crate::admin::web::pipeline::apply_filter_pipeline;
13
14/// Async CRUD operations that map entity slugs to domain service calls.
15#[async_trait]
16pub trait EntityCrudAdapter: Send + Sync {
17    /// List all records for this entity.
18    async fn list_all(&self) -> Result<Vec<Value>, String>;
19    /// Get a single record by its primary key.
20    async fn get_by_id(&self, id: &str) -> Result<Value, String>;
21    /// Create a record from raw JSON form data.
22    async fn create_from_json(&self, data: Value) -> Result<Value, String>;
23    /// Update a record from raw JSON form data.
24    async fn update_from_json(&self, data: Value) -> Result<(), String>;
25    /// Delete a record by its primary key.
26    async fn delete_by_id(&self, id: &str) -> Result<(), String>;
27
28    /// List records with in-memory filtering, sorting, and pagination.
29    async fn list_filtered(
30        &self,
31        params: &FilterParams,
32        valid_sort_fields: &HashSet<String>,
33    ) -> Result<FilteredResult, String> {
34        match self.list_all().await {
35            Ok(records) => Ok(apply_filter_pipeline(records, params, valid_sort_fields)),
36            Err(e) => Err(e),
37        }
38    }
39}
