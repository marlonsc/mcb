4/// Resolves typed values from MCP request and context metadata.
5///
6/// This module extracts session IDs, project IDs, and other context values
7/// from the `Meta` maps attached to tool call requests, handling multiple
8/// key name conventions (`snake_case`, camelCase, x-prefixed).
9use rmcp::model::Meta;
10use serde_json::Value;
11
12fn meta_value_as_string(meta: &Meta, keys: &[&str]) -> Option<String> {
13    for key in keys {
14        let value = meta.get(*key)?;
15        let extracted = match value {
16            Value::String(v) => Some(v.clone()),
17            Value::Number(v) => Some(v.to_string()),
18            Value::Bool(v) => Some(v.to_string()),
19            Value::Null | Value::Array(_) | Value::Object(_) => None,
20        };
21        if extracted.is_some() {
22            return extracted;
23        }
24    }
25    None
26}
27
28pub(crate) fn resolve_context_value(
29    request_meta: Option<&Meta>,
30    context_meta: &Meta,
31    keys: &[&str],
32) -> Option<String> {
33    request_meta
34        .and_then(|meta| meta_value_as_string(meta, keys))
35        .or_else(|| meta_value_as_string(context_meta, keys))
36}
37
38fn meta_value_as_bool(meta: &Meta, keys: &[&str]) -> Option<bool> {
39    for key in keys {
40        let value = meta.get(*key)?;
41        let extracted = match value {
42            Value::Bool(v) => Some(*v),
43            Value::String(v) => match v.trim().to_ascii_lowercase().as_str() {
44                "true" | "1" | "yes" => Some(true),
45                "false" | "0" | "no" => Some(false),
46                _ => None,
47            },
48            Value::Null | Value::Number(_) | Value::Array(_) | Value::Object(_) => None,
49        };
50        if extracted.is_some() {
51            return extracted;
52        }
53    }
54    None
55}
56
57pub(crate) fn resolve_context_bool(
58    request_meta: Option<&Meta>,
59    context_meta: &Meta,
60    keys: &[&str],
61) -> Option<bool> {
62    request_meta
63        .and_then(|meta| meta_value_as_bool(meta, keys))
64        .or_else(|| meta_value_as_bool(context_meta, keys))
