1//! Jobs API controller — returns indexing and validation operation status as JSON.
2
3use std::collections::HashMap;
4
5use crate::state::McbState;
6use axum::extract::Extension;
7use loco_rs::prelude::*;
8use mcb_domain::ports::{
9    IndexingOperation, IndexingOperationStatus, ValidationOperation, ValidationStatus,
10};
11use mcb_domain::value_objects::ids::OperationId;
12
13/// Returns a summary of all indexing and validation operations.
14///
15/// Calls `IndexingOperationsInterface::get_operations()` (sync) and
16/// `ValidationOperationsInterface::get_operations()` (sync) on the shared
17/// trackers from `McbState`.
18///
19/// # Errors
20///
21/// Fails when operations cannot be serialized.
22pub async fn jobs(Extension(state): Extension<McbState>) -> Result<Response> {
23    let indexing_ops = state.indexing_ops.get_operations();
24    let validation_ops = state.validation_ops.get_operations();
25
26    let indexing_running = indexing_ops
27        .values()
28        .filter(|op| {
29            matches!(
30                op.status,
31                IndexingOperationStatus::Starting | IndexingOperationStatus::InProgress
32            )
33        })
34        .count();
35    let validation_running = validation_ops
36        .values()
37        .filter(|op| {
38            matches!(
39                op.status,
40                ValidationStatus::Queued | ValidationStatus::InProgress
41            )
42        })
43        .count();
44
45    let total = indexing_ops.len() + validation_ops.len();
46    let running = indexing_running + validation_running;
47    let queued = count_queued(&indexing_ops, &validation_ops);
48
49    // Combine all operations into a single jobs array for API consumers
50    let mut jobs: Vec<serde_json::Value> = indexing_ops
51        .values()
52        .map(|op| serde_json::to_value(op).unwrap_or_default())
53        .collect();
54    jobs.extend(
55        validation_ops
56            .values()
57            .map(|op| serde_json::to_value(op).unwrap_or_default()),
58    );
59
60    format::json(serde_json::json!({
61        "total": total,
62        "running": running,
63        "queued": queued,
64        "jobs": jobs,
65        "indexing_operations": indexing_ops.values().collect::<Vec<_>>(),
66        "validation_operations": validation_ops.values().collect::<Vec<_>>(),
67    }))
68}
69
70/// Count operations in a queued/starting state across both trackers.
71fn count_queued(
72    indexing_ops: &HashMap<OperationId, IndexingOperation>,
73    validation_ops: &HashMap<OperationId, ValidationOperation>,
74) -> usize {
75    validation_ops
76        .values()
77        .filter(|op| matches!(op.status, ValidationStatus::Queued))
78        .count()
79        + indexing_ops
80            .values()
81            .filter(|op| matches!(op.status, IndexingOperationStatus::Starting))
82            .count()
83}
84
85/// Registers jobs API routes.
86#[must_use]
87pub fn routes() -> Routes {
88    Routes::new().prefix("jobs").add("/", get(jobs))
89}
