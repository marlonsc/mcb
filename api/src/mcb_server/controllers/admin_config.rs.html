1use loco_rs::prelude::Result;
2
3/// Default config directory when `MCB_PRO_ADMIN_CONFIG_DIR` is not set.
4const DEFAULT_PRO_ADMIN_CONFIG_DIR: &str = "config/pro_admin";
5
6/// Loads admin config from sea-orm-pro and returns it as JSON.
7///
8/// The config directory is read from the `MCB_PRO_ADMIN_CONFIG_DIR` environment
9/// variable, falling back to `config/pro_admin` for backward compatibility.
10///
11/// # Errors
12///
13/// Fails when config cannot be loaded or serialized to JSON.
14pub fn load_admin_config() -> Result<serde_json::Value> {
15    let config_root = std::env::var("MCB_PRO_ADMIN_CONFIG_DIR")
16        .unwrap_or_else(|_| DEFAULT_PRO_ADMIN_CONFIG_DIR.to_owned());
17    let cfg = sea_orm_pro::ConfigParser::new()
18        .load_config(&config_root)
19        .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
20    serde_json::to_value(cfg).map_err(|e| loco_rs::Error::string(&e.to_string()))
21}
