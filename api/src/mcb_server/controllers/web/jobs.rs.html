14/// Jobs page handler.
15///
16/// # Errors
17///
18/// Fails when job data cannot be loaded.
19pub async fn jobs_page(Extension(state): Extension<McbState>) -> Result<Response> {
20    let idx_ops = state.indexing_ops.get_operations();
21    let val_ops = state.validation_ops.get_operations();
22
23    let running = idx_ops
24        .values()
25        .filter(|o| {
26            matches!(
27                o.status,
28                IndexingOperationStatus::Starting | IndexingOperationStatus::InProgress
29            )
30        })
31        .count()
32        + val_ops
33            .values()
34            .filter(|o| {
35                matches!(
36                    o.status,
37                    ValidationStatus::Queued | ValidationStatus::InProgress
38                )
39            })
40            .count();
41    let total = idx_ops.len() + val_ops.len();
42    let status_str = if running > 0 { "running" } else { "idle" };
43
44    let idx_rows = render_indexing_rows(&idx_ops);
45    let val_rows = render_validation_rows(&val_ops);
46    let all_rows = format!("{idx_rows}{val_rows}");
47
48    let body = format!(
49        r#"<h1>Jobs</h1>
50<div class="dashboard-grid">{}{}{}{}</div>
51<div class="card"><h3>Operations</h3>{}</div>"#,
52        metric_card!("Status", status_badge!(status_str, status_str), ""),
53        metric_card!("Total", total, ""),
54        metric_card!("Running", running, ""),
55        metric_card!("Complete", total - running, ""),
56        data_table!(
57            "<th>ID</th><th>Type</th><th>Status</th><th>Files</th>",
58            all_rows,
59            "No operations recorded."
60        ),
61    );
62    html_page!("Jobs", body)
63}

65/// Render indexing operations as `Operations` table rows.
66fn render_indexing_rows(ops: &HashMap<OperationId, IndexingOperation>) -> String {
67    ops.iter()
68        .map(|(id, op)| {
69            let cls = match op.status {
70                IndexingOperationStatus::Starting | IndexingOperationStatus::InProgress => {
71                    "running"
72                }
73                IndexingOperationStatus::Completed => "completed",
74                _ => "error",
75            };
76            format!(
77                "<tr><td>{}</td><td>indexing</td><td>{}</td><td class=\"num\">{}</td></tr>",
78                html_escape(&id.to_string()),
79                status_badge!(cls, format!("{:?}", op.status)),
80                op.total_files,
81            )
82        })
83        .collect::<Vec<_>>()
84        .join("\n")
85}
86
87/// Render validation operations as `Operations` table rows.
88fn render_validation_rows(ops: &HashMap<OperationId, ValidationOperation>) -> String {
89    ops.iter()
90        .map(|(id, op)| {
91            let cls = match op.status {
92                ValidationStatus::Queued | ValidationStatus::InProgress => "running",
93                ValidationStatus::Completed => "completed",
94                ValidationStatus::Failed(_) | ValidationStatus::Canceled => "error",
95            };
96            format!(
97                "<tr><td>{}</td><td>validation</td><td>{}</td><td class=\"num\">—</td></tr>",
98                html_escape(&id.to_string()),
99                status_badge!(cls, format!("{:?}", op.status)),
100            )
101        })
102        .collect::<Vec<_>>()
103        .join("\n")
104}
