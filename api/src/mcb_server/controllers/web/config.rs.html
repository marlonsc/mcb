9/// Helper to extract a nested JSON string or return "—".
10fn cfg_str<'a>(config: &'a serde_json::Value, path: &[&str]) -> &'a str {
11    let mut v = config;
12    for key in path {
13        v = match v.get(*key) {
14            Some(next) => next,
15            None => return "—",
16        };
17    }
18    v.as_str().unwrap_or("—")
19}
20
21/// Helper to extract a nested JSON i64 or return "—".
22fn cfg_i64(config: &serde_json::Value, path: &[&str]) -> String {
23    let mut v = config;
24    for key in path {
25        v = match v.get(*key) {
26            Some(next) => next,
27            None => return "—".to_owned(),
28        };
29    }
30    v.as_i64().map_or_else(|| "—".to_owned(), |n| n.to_string())
31}

33/// Configuration page handler.
34///
35/// # Errors
36///
37/// Fails when config cannot be loaded.
38pub async fn config_page(Extension(state): Extension<McbState>) -> Result<Response> {
39    let provider = state.embedding_provider.provider_name();
40    let dims = state.embedding_provider.dimensions();
41    let cfg = load_admin_config()?;
42
43    let body = format!(
44        r#"<h1>Configuration</h1>
45<div class="config-grid">
46  <div class="card"><h3>Embedding Provider</h3>
47    <table class="data-table">
48      <tr><td>Provider</td><td><strong>{provider}</strong></td></tr>
49      <tr><td>Dimensions</td><td>{dims}</td></tr>
50    </table>
51  </div>
52  <div class="card"><h3>Vector Store</h3>
53    <table class="data-table">
54      <tr><td>Provider</td><td><strong>{vp}</strong></td></tr>
55      <tr><td>Host</td><td>{vh}</td></tr>
56      <tr><td>Port</td><td>{vport}</td></tr>
57    </table>
58  </div>
59  <div class="card"><h3>Database</h3>
60    <table class="data-table">
61      <tr><td>URI</td><td>{db}</td></tr>
62    </table>
63  </div>
64</div>"#,
65        vp = cfg_str(&cfg, &["providers", "vector_store", "provider"]),
66        vh = cfg_str(&cfg, &["providers", "vector_store", "host"]),
67        vport = cfg_i64(&cfg, &["providers", "vector_store", "port"]),
68        db = cfg_str(&cfg, &["database", "uri"]),
69    );
70    html_page!("Configuration", body)
71}
