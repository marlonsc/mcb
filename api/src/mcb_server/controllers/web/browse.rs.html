1//! Browse page — vector store collection viewer with keyboard navigation.
2
3use super::{html_escape, html_page};
4use crate::state::McbState;
5use axum::extract::Extension;
6use loco_rs::prelude::*;
7use mcb_domain::value_objects::CollectionId;
8
9/// Browse page handler.
10///
11/// # Errors
12///
13/// Fails when collection data cannot be loaded.
14pub async fn browse_page(Extension(state): Extension<McbState>) -> Result<Response> {
15    let collections = state
16        .vector_store
17        .list_collections()
18        .await
19        .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
20
21    let mut chunks: Vec<mcb_domain::value_objects::SearchResult> = Vec::new();
22    for col in &collections {
23        let id = CollectionId::from_string(&col.name);
24        let vecs = state
25            .vector_store
26            .list_vectors(&id, mcb_utils::constants::DEFAULT_BROWSE_LIMIT)
27            .await
28            .map_err(|e| loco_rs::Error::string(&e.to_string()))?;
29        chunks.extend(vecs);
30    }
31
32    let grid = if chunks.is_empty() {
33        r#"<p class="no-chunks">No collections indexed yet. Use <code>index_repo</code> to index a codebase.</p>"#.to_owned()
34    } else {
35        chunks
36            .iter()
37            .enumerate()
38            .map(|(i, c)| chunk_card(i, c))
39            .collect::<Vec<_>>()
40            .join("\n")
41    };
42
43    let body = format!(
44        r#"<h1>Browse</h1>
45<div id="collections-grid" class="collections-grid">{grid}</div>
46<script>{KEYBOARD_NAV_JS}</script>"#
47    );
48    html_page!("Browse", body)
49}
50
51/// Render a single code chunk card.
52fn chunk_card(idx: usize, c: &mcb_domain::value_objects::SearchResult) -> String {
53    let lang = html_escape(&c.language.to_lowercase());
54    let content = html_escape(&c.content);
55    let file = html_escape(&c.file_path);
56    let id = html_escape(&c.id);
57    format!(
58        r#"<div class="code-chunk" data-chunk-id="{id}" data-index="{idx}" data-language="{lang}" tabindex="0">
59  <div class="chunk-header">
60    <span class="chunk-file">{file}</span>
61    <span class="chunk-lang">{lang}</span>
62    <span class="chunk-lines">:{line}</span>
63  </div>
64  <pre class="chunk-content"><code>{content}</code></pre>
65</div>"#,
66        line = c.start_line,
67    )
68}

70/// Vim-style keyboard navigation (j/k/g/G/End/c/Esc).
71const KEYBOARD_NAV_JS: &str = r#"(function(){
72  function q(s){return Array.from(document.querySelectorAll(s));}
73  function ai(ch){var a=document.querySelector('[data-active="true"]');
74    if(a)return parseInt(a.dataset.index||'0',10);
75    var f=document.activeElement;
76    if(f&&f.hasAttribute('data-chunk-id'))return parseInt(f.dataset.index||'0',10);
77    return -1;}
78  function sa(ch,i){ch.forEach(function(c){c.removeAttribute('data-active');});
79    var t=ch[i];if(t){t.setAttribute('data-active','true');t.focus();
80    t.scrollIntoView({behavior:'smooth',block:'nearest'});}}
81  var pg=false;
82  document.addEventListener('keydown',function(e){
83    var ch=q('[data-chunk-id]');if(!ch.length)return;
84    var c=ai(ch);if(c<0)c=0;
85    if(e.key==='j'){e.preventDefault();sa(ch,Math.min(c+1,ch.length-1));pg=false;}
86    else if(e.key==='k'){e.preventDefault();sa(ch,Math.max(c-1,0));pg=false;}
87    else if(e.key==='g'&&!e.shiftKey){e.preventDefault();
88      if(pg){sa(ch,0);pg=false;}else{pg=true;setTimeout(function(){pg=false;},500);}}
89    else if(e.key==='G'||(e.shiftKey&&(e.key==='g'||e.key==='G'))){e.preventDefault();sa(ch,ch.length-1);pg=false;}
90    else if(e.key==='End'){e.preventDefault();sa(ch,ch.length-1);pg=false;}
91    else if(e.key==='c'){e.preventDefault();var x=ch[c>=0?c:0];
92      if(x){navigator.clipboard.writeText((x.textContent||'').trim()).catch(function(){});}pg=false;}
93    else if(e.key==='Escape'){e.preventDefault();var a=document.querySelector('[data-active="true"]');
94      if(a)a.removeAttribute('data-active');pg=false;}
95    else{pg=false;}
96  });
97})()"#;
