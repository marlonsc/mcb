4use schemars::JsonSchema;

8use crate::args::schema_helpers::ObjectDataSchema;

10tool_enum! {
11/// Actions available for project resource management
12pub enum ProjectAction {
13    /// Create a new resource.
14    Create,
15    /// Get an existing resource.
16    Get,
17    /// Update an existing resource.
18    Update,
19    /// List resources.
20    List,
21    /// Delete a resource.
22    Delete,
23}
24}
25
26tool_enum! {
27/// Types of project resources that can be managed
28pub enum ProjectResource {
29    /// Project metadata.
30    Project,
31    /// Project phase.
32    Phase,
33    /// Project issue.
34    Issue,
35    /// Issue dependency.
36    Dependency,
37    /// Project decision.
38    Decision,
39}
40}
41
42tool_schema! {
43/// Arguments for project resource management operations
44pub struct ProjectArgs {
45    /// Action: create, update, list, delete
46    #[schemars(description = "Action: create, update, list, delete")]
47    pub action: ProjectAction,
48
49    /// Resource type: phase, issue, dependency, decision
50    #[schemars(description = "Resource type: phase, issue, dependency, decision")]
51    pub resource: ProjectResource,
52
53    /// Project ID
54    #[schemars(description = "Project ID")]
55    pub project_id: Option<String>,
56
57    /// Resource ID (for get/delete on phase, issue, dependency, decision)
58    #[schemars(description = "Resource ID (for get/delete operations)")]
59    pub id: Option<String>,
60
61    /// Issue ID (for dependency operations)
62    #[schemars(description = "Issue ID (for dependency list)")]
63    pub issue_id: Option<String>,
64
65    /// Data payload for create/update (JSON object)
66    #[schemars(
67        description = "Data payload for create/update (JSON object)",
68        with = "ObjectDataSchema"
69    )]
70    pub data: Option<serde_json::Value>,
71
72    /// Additional filters for list action
73    #[schemars(
74        description = "Additional filters for list action",
75        with = "ObjectDataSchema"
76    )]
77    pub filters: Option<serde_json::Value>,
78
79    /// Organization ID (auto-injected).
80    #[schemars(skip)]
81    pub org_id: Option<String>,
82}
83}
