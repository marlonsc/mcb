4use schemars::JsonSchema;
5use serde::{Deserialize, Serialize};
6use validator::Validate;
7
8tool_enum! {
9/// Actions available for version control system operations
10pub enum VcsAction {
11    /// List all repositories for the project
12    ListRepositories,
13    /// Index a specific repository
14    IndexRepository,
15    /// Compare two branches (diff/impact)
16    CompareBranches,
17    /// Search for a branch name
18    SearchBranch,
19    /// Analyze impact of changes
20    AnalyzeImpact,
21}
22}
23
24tool_schema! {
25/// Arguments for version control system operations
26pub struct VcsArgs {
27    /// Action to perform
28    #[schemars(
29        description = "Action: list_repositories, index_repository, compare_branches, search_branch, analyze_impact"
30    )]
31    pub action: VcsAction,
32
33    /// Organization identifier
34    #[schemars(description = "Organization ID (uses default if omitted)")]
35    pub org_id: Option<String>,
36
37    /// Repository identifier
38    #[schemars(description = "Repository identifier", with = "String")]
39    pub repo_id: Option<String>,
40
41    /// Local file system path to the repository
42    #[schemars(description = "Repository path on disk", with = "String")]
43    #[validate(custom(function = "super::validate_file_path", message = "Invalid file path"))]
44    pub repo_path: Option<String>,
45
46    /// Base branch for comparison
47    #[schemars(description = "Base branch name", with = "String")]
48    pub base_branch: Option<String>,
49
50    /// Target branch for comparison
51    #[schemars(description = "Compare/target branch name", with = "String")]
52    pub target_branch: Option<String>,
53
54    /// Search query string
55    #[schemars(description = "Search query for branch search", with = "String")]
56    pub query: Option<String>,
57
58    /// Specific branches to index
59    #[schemars(
60        description = "Branches to index (default: repo default branch)",
61        with = "Vec<String>"
62    )]
63    pub branches: Option<Vec<String>>,
64
65    /// Whether to index commit history
66    #[schemars(
67        description = "Whether to include commit history when indexing",
68        with = "bool"
69    )]
70    pub include_commits: Option<bool>,
71
72    /// Depth of history to index
73    #[schemars(
74        description = "Commit history depth (default: 50 from config, or 1000 if no config)",
75        with = "usize"
76    )]
77    pub depth: Option<usize>,
78
79    /// Result limit for list/search operations
80    #[schemars(description = "Limit for search or list actions", with = "u32")]
81    pub limit: Option<u32>,
82}
83}
