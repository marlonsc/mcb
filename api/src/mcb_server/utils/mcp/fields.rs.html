1//! JSON field extraction helpers for freeform MCP data payloads.
2
3use rmcp::model::CallToolResult;
4use serde_json::{Map, Value};
5
6use super::tool_error;
7
8/// Requires a string value from a JSON object, returning an error if missing.
9///
10/// # Errors
11/// Returns an error when the key is missing or value is not a string.
12pub fn require_str(data: &Map<String, Value>, key: &str) -> Result<String, CallToolResult> {
13    data.get(key)
14        .and_then(Value::as_str)
15        .map(str::to_owned)
16        .ok_or_else(|| tool_error(format!("Missing required field: {key}")))
17}
18

29/// Requires an i32 value from a JSON object, returning an error if missing or out of range.
30///
31/// # Errors
32/// Returns an error when the key is missing, value is not an integer, or exceeds i32 range.
33pub fn require_i32(data: &Map<String, Value>, key: &str) -> Result<i32, CallToolResult> {
34    data.get(key)
35        .and_then(Value::as_i64)
36        .and_then(|value| value.try_into().ok())
37        .ok_or_else(|| tool_error(format!("Missing required field: {key}")))
38}

50/// Extracts an optional string value from a JSON object.
51pub fn opt_str(data: &Map<String, Value>, key: &str) -> Option<String> {
52    data.get(key).and_then(Value::as_str).map(str::to_owned)
53}

55/// Extracts an optional boolean value from a JSON object.
56pub fn opt_bool(data: &Map<String, Value>, key: &str) -> Option<bool> {
57    data.get(key).and_then(Value::as_bool)
58}

60/// Extracts a string array from a JSON object, defaulting to empty if missing.
61pub fn str_vec(data: &Map<String, Value>, key: &str) -> Vec<String> {
62    data.get(key)
63        .and_then(Value::as_array)
64        .map(|items| {
65            items
66                .iter()
67                .filter_map(|item| item.as_str().map(str::to_owned))
68                .collect()
69        })
70        .unwrap_or_default()
71}
