4use std::path::Path;
5use std::sync::Arc;
6
7use mcb_domain::ports::VcsProvider;
8use rmcp::ErrorData as McpError;
9use rmcp::model::CallToolResult;
10
11use super::responses::{ImpactFile, ImpactResponse, ImpactSummary, repo_path};
12use crate::args::VcsArgs;
13use crate::constants::git::GIT_REF_HEAD;
14use crate::constants::vcs::{
15    IMPACT_CHANGE_COUNT_WEIGHT, IMPACT_FILE_COUNT_WEIGHT, MAX_IMPACT_SCORE,
16};
17use crate::error_mapping::to_contextual_tool_error;
18use crate::formatter::ResponseFormatter;
19
20/// Analyzes the impact of changes between branches.
21#[tracing::instrument(skip_all)]
22pub async fn analyze_impact(
23    vcs_provider: &Arc<dyn VcsProvider>,
24    args: &VcsArgs,
25) -> Result<CallToolResult, McpError> {
26    let path = match repo_path(args) {
27        Ok(p) => p,
28        Err(error_result) => return Ok(error_result),
29    };
30    let repo = match vcs_provider.open_repository(Path::new(&path)).await {
31        Ok(repo) => repo,
32        Err(e) => {
33            return Ok(to_contextual_tool_error(e));
34        }
35    };
36    let base_ref = args
37        .base_branch
38        .clone()
39        .unwrap_or_else(|| repo.default_branch().to_owned());
40    let head_ref = args
41        .target_branch
42        .clone()
43        .unwrap_or_else(|| GIT_REF_HEAD.to_owned());
44    let diff = match vcs_provider.diff_refs(&repo, &base_ref, &head_ref).await {
45        Ok(diff) => diff,
46        Err(e) => {
47            return Ok(to_contextual_tool_error(e));
48        }
49    };
50    let mut added = 0;
51    let mut modified = 0;
52    let mut deleted = 0;
53    let mut impacted_files = Vec::new();
54    for file in diff.files.iter() {
55        let status = file.status.to_string();
56        match status.as_str() {
57            "added" => added += 1,
58            "deleted" => deleted += 1,
59            _ => modified += 1,
60        }
61        impacted_files.push(ImpactFile {
62            path: file.path.to_str().unwrap_or_default().to_owned(),
63            status: status.clone(),
64            impact: file.additions + file.deletions,
65        });
66    }
67    let total_changes = diff.total_additions + diff.total_deletions;
68    let impact_score = ((diff.files.len() as f64).ln_1p() * IMPACT_FILE_COUNT_WEIGHT
69        + (total_changes as f64).ln_1p() * IMPACT_CHANGE_COUNT_WEIGHT)
70        .min(MAX_IMPACT_SCORE);
71    let result = ImpactResponse {
72        base_ref,
73        head_ref,
74        impact_score,
75        summary: ImpactSummary {
76            total_files: diff.files.len(),
77            added,
78            modified,
79            deleted,
80            total_changes,
81        },
82        impacted_files,
83    };
84    ResponseFormatter::json_success(&result)
85}
