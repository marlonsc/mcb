6use std::sync::Arc;
7
8use mcb_domain::ports::AgentSessionServiceInterface;
9use mcb_domain::ports::MemoryServiceInterface;
10use rmcp::ErrorData as McpError;
11use rmcp::handler::server::wrapper::Parameters;
12use rmcp::model::CallToolResult;
13use validator::Validate;
14
15use super::{create, get, list, summarize, update};
16use crate::args::{SessionAction, SessionArgs};
17
18/// Handler for agent session MCP tool operations.
19///
20/// Supports creating, updating, listing, and summarizing agent sessions.
21#[derive(Clone)]
22pub struct SessionHandler {
23    agent_service: Arc<dyn AgentSessionServiceInterface>,
24    memory_service: Arc<dyn MemoryServiceInterface>,
25}
26
27handler_new!(SessionHandler {
28    agent_service: Arc<dyn AgentSessionServiceInterface>,
29    memory_service: Arc<dyn MemoryServiceInterface>,
30});
31
32impl SessionHandler {
33    /// Handle a session tool request.
34    ///
35    /// # Errors
36    /// Returns an error when argument validation fails.
37    #[tracing::instrument(skip_all)]
38    pub async fn handle(
39        &self,
40        Parameters(args): Parameters<SessionArgs>,
41    ) -> Result<CallToolResult, McpError> {
42        args.validate()
43            .map_err(|_| McpError::invalid_params("invalid arguments", None))?;
44
45        match args.action {
46            SessionAction::Create => create::create_session(&self.agent_service, &args).await,
47            SessionAction::Get => get::get_session(&self.agent_service, &args).await,
48            SessionAction::Update => update::update_session(&self.agent_service, &args).await,
49            SessionAction::List => list::list_sessions(&self.agent_service, &args).await,
50            SessionAction::Summarize => {
51                summarize::summarize_session(&self.memory_service, &args).await
52            }
53        }
54    }
55}
