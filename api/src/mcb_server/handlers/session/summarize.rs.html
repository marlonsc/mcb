4use std::sync::Arc;
5
6use mcb_domain::ports::{CreateSessionSummaryInput, MemoryServiceInterface};
7use rmcp::ErrorData as McpError;
8use rmcp::model::CallToolResult;
9
10use super::common::{json_map, str_vec};
11use crate::args::SessionArgs;
12use crate::error_mapping::to_contextual_tool_error;
13use crate::formatter::ResponseFormatter;
14use crate::utils::mcp::{OriginPayloadFields, resolve_origin_context, tool_error};
15
16/// Creates or retrieves a session summary.
17#[tracing::instrument(skip_all)]
18pub async fn summarize_session(
19    memory_service: &Arc<dyn MemoryServiceInterface>,
20    args: &SessionArgs,
21) -> Result<CallToolResult, McpError> {
22    let session_id = match args.session_id.as_ref() {
23        Some(id) => id,
24        None => {
25            return Ok(tool_error("Missing session_id"));
26        }
27    };
28    if let Some(data) = json_map(&args.data) {
29        let session_id_str = session_id.to_string();
30        let topics = str_vec(data, "topics");
31        let decisions = str_vec(data, "decisions");
32        let next_steps = str_vec(data, "next_steps");
33        let key_files = str_vec(data, "key_files");
34        let payload = OriginPayloadFields::extract(data);
35        let mut input = payload.to_input();
36        input.org_id = args.org_id.as_deref();
37        input.project_id_args = args.project_id.as_deref();
38        input.session_from_args = Some(&session_id_str);
39        input.parent_session_from_args = args.parent_session_id.as_deref();
40        input.tool_name_args = Some("session");
41        input.worktree_id_args = args.worktree_id.as_deref();
42        input.require_project_id = true;
43        let origin_context = resolve_origin_context(&input)?;
44        let project_id = origin_context.project_id.clone().ok_or_else(|| {
45            McpError::invalid_params("project_id is required for summarize", None)
46        })?;
47        match memory_service
48            .create_session_summary(CreateSessionSummaryInput {
49                project_id,
50                org_id: args
51                    .org_id
52                    .clone()
53                    .unwrap_or(mcb_domain::constants::keys::DEFAULT_ORG_ID.to_owned()),
54                session_id: *session_id,
55                topics,
56                decisions,
57                next_steps,
58                key_files,
59                origin_context: Some(origin_context),
60            })
61            .await
62        {
63            Ok(summary_id) => ResponseFormatter::json_success(&serde_json::json!({
64                "summary_id": summary_id,
65                "session_id": session_id_str,
66            })),
67            Err(e) => Ok(to_contextual_tool_error(e)),
68        }
69    } else {
70        match memory_service.get_session_summary(session_id).await {
71            Ok(Some(summary)) => ResponseFormatter::json_success(&serde_json::json!({
72                "session_id": summary.session_id,
73                "topics": summary.topics,
74                "decisions": summary.decisions,
75                "next_steps": summary.next_steps,
76                "key_files": summary.key_files,
77                "created_at": summary.created_at,
78            })),
79            Ok(None) => Ok(tool_error("Session summary not found")),
80            Err(e) => Ok(to_contextual_tool_error(e)),
81        }
82    }
83}
