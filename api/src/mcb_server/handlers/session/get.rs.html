19/// Retrieves an agent session by ID.
20#[tracing::instrument(skip_all)]
21pub async fn get_session(
22    agent_service: &Arc<dyn AgentSessionServiceInterface>,
23    args: &SessionArgs,
24) -> Result<CallToolResult, McpError> {
25    let session_id = match require_session_id_str(args) {
26        Ok(id) => id,
27        Err(error_result) => return Ok(error_result),
28    };
29    match agent_service.get_session(&session_id).await {
30        Ok(Some(session)) => ResponseFormatter::json_success(&serde_json::json!({
31            schema::ID: session.id,
32            schema::SESSION_SUMMARY_ID: session.session_summary_id,
33            schema::AGENT_TYPE: session.agent_type.as_str(),
34            schema::MODEL: session.model,
35            schema::PARENT_SESSION_ID: session.parent_session_id,
36            schema::STARTED_AT: session.started_at,
37            schema::ENDED_AT: session.ended_at,
38            schema::DURATION_MS: session.duration_ms,
39            schema::STATUS: session.status.as_str(),
40            schema::PROMPT_SUMMARY: session.prompt_summary,
41            schema::RESULT_SUMMARY: session.result_summary,
42            schema::TOKEN_COUNT: session.token_count,
43            schema::TOOL_CALLS_COUNT: session.tool_calls_count,
44            schema::DELEGATIONS_COUNT: session.delegations_count,
45        })),
46        Ok(None) => Ok(tool_error("Agent session not found")),
47        Err(e) => {
48            error!("get_session", "Failed to get agent session", &e);
49            Ok(to_contextual_tool_error(e))
50        }
51    }
52}
