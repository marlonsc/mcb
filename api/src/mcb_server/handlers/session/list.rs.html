4use std::sync::Arc;
5
6use mcb_domain::ports::AgentSessionQuery;
7use mcb_domain::ports::AgentSessionServiceInterface;
8use rmcp::ErrorData as McpError;
9use rmcp::model::CallToolResult;
10
11use crate::args::SessionArgs;
12use crate::constants::fields::{FIELD_COUNT, FIELD_SESSIONS};
13use crate::constants::limits::DEFAULT_SESSION_LIST_LIMIT;
14use crate::error_mapping::to_contextual_tool_error;
15use crate::formatter::ResponseFormatter;
16
17/// Lists agent sessions based on filters.
18#[tracing::instrument(skip_all)]
19pub async fn list_sessions(
20    agent_service: &Arc<dyn AgentSessionServiceInterface>,
21    args: &SessionArgs,
22) -> Result<CallToolResult, McpError> {
23    let query = AgentSessionQuery {
24        session_summary_id: None,
25        parent_session_id: args.parent_session_id.clone(),
26        agent_type: args
27            .agent_type
28            .as_ref()
29            .and_then(|value| value.parse().ok()),
30        status: args
31            .status
32            .as_ref()
33            .filter(|value| !value.is_empty())
34            .map(|value| value.parse())
35            .transpose()
36            .map_err(|_| McpError::invalid_params("Invalid status", None))?,
37        project_id: args.project_id.clone(),
38        worktree_id: args.worktree_id.clone(),
39        limit: Some(args.limit.unwrap_or(DEFAULT_SESSION_LIST_LIMIT as u32) as usize),
40    };
41    match agent_service.list_sessions(query).await {
42        Ok(sessions) => {
43            let items: Vec<_> = sessions
44                .iter()
45                .map(|session| {
46                    serde_json::json!({
47                        "id": session.id,
48                        "parent_session_id": session.parent_session_id,
49                        "agent_type": session.agent_type.as_str(),
50                        "status": session.status.as_str(),
51                        "started_at": session.started_at,
52                        "duration_ms": session.duration_ms,
53                    })
54                })
55                .collect();
56            ResponseFormatter::json_success(&serde_json::json!({
57                (FIELD_SESSIONS): items,
58                (FIELD_COUNT): items.len(),
59            }))
60        }
61        Err(e) => Ok(to_contextual_tool_error(e)),
62    }
63}
