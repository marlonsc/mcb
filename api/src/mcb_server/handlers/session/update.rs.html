23/// Updates an existing agent session.
24#[tracing::instrument(skip_all)]
25pub async fn update_session(
26    agent_service: &Arc<dyn AgentSessionServiceInterface>,
27    args: &SessionArgs,
28) -> Result<CallToolResult, McpError> {
29    let session_id = match require_session_id_str(args) {
30        Ok(id) => id,
31        Err(error_result) => return Ok(error_result),
32    };
33    let data = json_map(&args.data);
34    let status = parse_status(args, data)?;
35    match agent_service.get_session(&session_id).await {
36        Ok(Some(mut session)) => {
37            apply_resolved_identifier(
38                &mut session.project_id,
39                schema::PROJECT_ID,
40                args.project_id.as_deref(),
41                payload_str(data, schema::PROJECT_ID).as_deref(),
42            )?;
43            apply_resolved_identifier(
44                &mut session.worktree_id,
45                schema::WORKTREE_ID,
46                args.worktree_id.as_deref(),
47                payload_str(data, schema::WORKTREE_ID).as_deref(),
48            )?;
49
50            if let Some(status) = status {
51                session.status = status;
52            }
53            if let Some(data) = data {
54                apply_session_updates(&mut session, data);
55            }
56            let status = session.status.as_str().to_owned();
57            match agent_service.update_session(session).await {
58                Ok(_) => ResponseFormatter::json_success(&serde_json::json!({
59                    schema::ID: session_id,
60                    schema::STATUS: status,
61                    (FIELD_UPDATED): true,
62                })),
63                Err(e) => {
64                    error!("update_session", "Failed to update agent session", &e);
65                    Ok(to_contextual_tool_error(e))
66                }
67            }
68        }
69        Ok(None) => Ok(tool_error("Agent session not found")),
70        Err(e) => {
71            error!(
72                "update_session",
73                "Failed to update agent session (get failed)", &e
74            );
75            Ok(to_contextual_tool_error(e))
76        }
77    }
78}

80fn parse_status(
81    args: &SessionArgs,
82    data: Option<&Map<String, Value>>,
83) -> Result<Option<AgentSessionStatus>, McpError> {
84    let status_value = args.status.clone().or_else(|| {
85        data.and_then(|d| {
86            d.get(schema::STATUS)
87                .and_then(Value::as_str)
88                .map(str::to_owned)
89        })
90    });
91
92    status_value
93        .map(|status| {
94            status
95                .parse()
96                .map_err(|e: String| McpError::invalid_params(e, None))
97        })
98        .transpose()
99}

101fn payload_str(data: Option<&Map<String, Value>>, key: &str) -> Option<String> {
102    data.and_then(|d| opt_str(d, key))
103}
104
105fn apply_resolved_identifier(
106    session_value: &mut Option<String>,
107    field_name: &str,
108    args_value: Option<&str>,
109    payload_value: Option<&str>,
110) -> Result<(), McpError> {
111    let resolved = resolve_identifier_precedence(field_name, args_value, payload_value)?;
112    if let Some(value) = resolved {
113        if let Some(existing) = session_value.as_deref()
114            && existing != value
115        {
116            return Err(McpError::invalid_params(
117                format!("conflicting {field_name}: args/data='{value}', session='{existing}'"),
118                None,
119            ));
120        }
121        *session_value = Some(value);
122    }
123    Ok(())
124}
125
126fn apply_session_updates(session: &mut AgentSession, data: &Map<String, Value>) {
127    session.result_summary = data
128        .get(schema::RESULT_SUMMARY)
129        .and_then(Value::as_str)
130        .map(str::to_owned)
131        .or(session.result_summary.clone());
132    session.token_count = data
133        .get(schema::TOKEN_COUNT)
134        .and_then(Value::as_i64)
135        .or(session.token_count);
136    session.tool_calls_count = data
137        .get(schema::TOOL_CALLS_COUNT)
138        .and_then(Value::as_i64)
139        .or(session.tool_calls_count);
140    session.delegations_count = data
141        .get(schema::DELEGATIONS_COUNT)
142        .and_then(Value::as_i64)
143        .or(session.delegations_count);
144}
