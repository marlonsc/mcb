24/// Validated execution data extracted from JSON payload
25struct ValidatedExecutionData {
26    command: String,
27    exit_code: i32,
28    duration_ms: i64,
29    success: bool,
30    execution_type: mcb_domain::entities::memory::ExecutionType,
31}
32
33impl ValidatedExecutionData {
34    /// Validate and extract all required fields from JSON data
35    fn validate(data: &serde_json::Map<String, serde_json::Value>) -> Result<Self, CallToolResult> {
36        let command = require_str(data, "command")?;
37        let exit_code = require_i32(data, "exit_code")?;
38        let duration_ms = require_i64(data, "duration_ms")?;
39        let success = require_bool(data, "success")?;
40        let execution_type_str = require_str(data, "execution_type")?;
41
42        let execution_type = execution_type_str
43            .parse()
44            .map_err(|_| tool_error(format!("Invalid execution_type: {execution_type_str}")))?;
45
46        Ok(Self {
47            command,
48            exit_code,
49            duration_ms,
50            success,
51            execution_type,
52        })
53    }
54}
55
56/// Assemble [`ExecutionMetadata`] from validated fields plus optional payload extras.
57fn build_execution_metadata(
58    validated: &ValidatedExecutionData,
59    data: &serde_json::Map<String, serde_json::Value>,
60) -> ExecutionMetadata {
61    ExecutionMetadata {
62        id: domain_id::generate().to_string(),
63        command: validated.command.clone(),
64        exit_code: Some(validated.exit_code),
65        duration_ms: Some(validated.duration_ms),
66        success: validated.success,
67        execution_type: validated.execution_type.clone(),
68        coverage: data
69            .get("coverage")
70            .and_then(Value::as_f64)
71            .map(|value| value as f32),
72        files_affected: str_vec(data, "files_affected"),
73        output_summary: data
74            .get("output_summary")
75            .and_then(Value::as_str)
76            .map(str::to_owned),
77        warnings_count: data
78            .get("warnings_count")
79            .and_then(Value::as_i64)
80            .and_then(|value| value.try_into().ok()),
81        errors_count: data
82            .get("errors_count")
83            .and_then(Value::as_i64)
84            .and_then(|value| value.try_into().ok()),
85    }
86}

88/// Human-readable summary line for an execution observation.
89fn format_execution_content(validated: &ValidatedExecutionData) -> String {
90    format!(
91        "Execution: {} (exit_code={}, success={})",
92        validated.command, validated.exit_code, validated.success
93    )
94}

96/// Tags describing an execution observation (kind plus success/failure).
97fn build_execution_tags(
98    validated: &ValidatedExecutionData,
99    metadata: &ExecutionMetadata,
100) -> Vec<String> {
101    let outcome = if validated.success {
102        TAG_SUCCESS
103    } else {
104        TAG_FAILURE
105    };
106    vec![
107        TAG_EXECUTION.to_owned(),
108        metadata.execution_type.as_str().to_owned(),
109        outcome.to_owned(),
110    ]
111}

113/// Store an execution observation in memory
114#[tracing::instrument(skip_all)]
115pub async fn store_execution(
116    memory_service: &Arc<dyn MemoryServiceInterface>,
117    args: &MemoryArgs,
118) -> Result<CallToolResult, McpError> {
119    let data = extract_field!(require_data_map(
120        &args.data,
121        "Missing data payload for execution store"
122    ));
123    let validated = extract_field!(ValidatedExecutionData::validate(data));
124    let metadata = build_execution_metadata(&validated, data);
125    let content = format_execution_content(&validated);
126    let tags = build_execution_tags(&validated, &metadata);
127    let payload_execution_id = opt_str(data, "execution_id");
128    let generated_execution_id = metadata.id.clone();
129
130    let origin = resolve_memory_origin_context(
131        args,
132        data,
133        &MemoryOriginOptions {
134            execution_from_args: Some(generated_execution_id.as_str()),
135            execution_from_data: payload_execution_id.as_deref(),
136            file_path_payload: None,
137            timestamp: None,
138        },
139    )
140    .map_err(|err| {
141        McpError::invalid_params(
142            format!("failed to resolve execution origin context: {err}"),
143            None,
144        )
145    })?;
146
147    let obs_metadata = build_observation_metadata(
148        origin.canonical_session_id,
149        origin.origin_context,
150        Some(metadata),
151        None,
152    );
153
154    persist_execution_observation(
155        memory_service,
156        origin.project_id,
157        content,
158        tags,
159        obs_metadata,
160    )
161    .await
162}

164/// Store the execution observation and format the MCP response.
165async fn persist_execution_observation(
166    memory_service: &Arc<dyn MemoryServiceInterface>,
167    project_id: String,
168    content: String,
169    tags: Vec<String>,
170    obs_metadata: mcb_domain::entities::memory::ObservationMetadata,
171) -> Result<CallToolResult, McpError> {
172    match memory_service
173        .store_observation(StoreObservationInput {
174            project_id,
175            content,
176            r#type: ObservationType::Execution,
177            tags,
178            metadata: obs_metadata,
179        })
180        .await
181    {
182        Ok((observation_id, deduplicated)) => ResponseFormatter::json_success(&serde_json::json!({
183            FIELD_OBSERVATION_ID: observation_id,
184            "deduplicated": deduplicated,
185        })),
186        Err(e) => {
187            mcb_domain::error!("store_execution", "Failed to store execution", &e);
188            Ok(tool_error("Failed to store execution"))
189        }
190    }
191}

193/// Retrieve execution observations filtered by session and repo
194#[tracing::instrument(skip_all)]
195pub async fn get_executions(
196    memory_service: &Arc<dyn MemoryServiceInterface>,
197    args: &MemoryArgs,
198) -> Result<CallToolResult, McpError> {
199    search_memories_as_json(
200        memory_service,
201        args,
202        SearchMemoriesJsonSpec {
203            query: "execution",
204            obs_type: ObservationType::Execution,
205            result_key: "executions",
206            mapper: |result: &MemorySearchResult| {
207                let execution = result.observation.metadata.execution.as_ref()?;
208                Some(serde_json::json!({
209                    FIELD_OBSERVATION_ID: result.observation.id,
210                    "command": execution.command,
211                    "exit_code": execution.exit_code,
212                    "duration_ms": execution.duration_ms,
213                    "success": execution.success,
214                    "execution_type": execution.execution_type.as_str(),
215                    "coverage": execution.coverage,
216                    "files_affected": execution.files_affected,
217                    "output_summary": execution.output_summary,
218                    "warnings_count": execution.warnings_count,
219                    "errors_count": execution.errors_count,
220                    "created_at": result.observation.created_at,
221                }))
222            },
223        },
224    )
225    .await
