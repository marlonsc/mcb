20/// Injects semantic memory context into the MCP tool result based on the provided filter.
21#[tracing::instrument(skip_all)]
22pub async fn inject_context(
23    memory_service: &Arc<dyn MemoryServiceInterface>,
24    args: &MemoryArgs,
25) -> Result<CallToolResult, McpError> {
26    let filter = build_memory_filter(args, None, None);
27    let limit = args.limit.unwrap_or(DEFAULT_MEMORY_LIMIT as u32) as usize;
28    let max_tokens = args.max_tokens.unwrap_or(DEFAULT_MAX_CONTEXT_TOKENS);
29    let vcs_context = capture_vcs_context();
30    match memory_service
31        .search_memories("", Some(filter), limit)
32        .await
33    {
34        Ok(results) => {
35            let mut context = String::new();
36            let mut observation_ids = Vec::new();
37            let max_chars = max_tokens * CHARS_PER_TOKEN_ESTIMATE;
38            for result in results {
39                observation_ids.push(result.observation.id.clone());
40                let entry = format!(
41                    "[{}] {}: {}\n\n",
42                    result.observation.r#type.as_str().to_uppercase(),
43                    result.observation.id,
44                    result.observation.content
45                );
46                if context.len() + entry.len() > max_chars {
47                    break;
48                }
49                context.push_str(&entry);
50            }
51            ResponseFormatter::json_success(&serde_json::json!({
52                "observation_count": observation_ids.len(),
53                "observation_ids": observation_ids,
54                "context": context,
55                "estimated_tokens": context.len() / CHARS_PER_TOKEN_ESTIMATE,
56                "vcs_context": {
57                    "branch": vcs_context.branch,
58                    "commit": vcs_context.commit,
59                }
60            }))
61        }
62        Err(e) => {
63            error!("inject_context", "Failed to inject context", &e);
64            Ok(tool_error("Failed to inject context"))
65        }
66    }
67}
