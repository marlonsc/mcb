6use std::sync::Arc;
7
8use mcb_domain::info;
9use mcb_domain::ports::ProjectRepository;
10use rmcp::handler::server::wrapper::Parameters;
11use rmcp::model::{CallToolResult, ErrorData as McpError};
12
13use crate::args::{ProjectAction, ProjectArgs, ProjectResource};
14use crate::utils::mcp::{map_opaque_error, ok_json, resolve_org_id};
15
16/// Handler for the consolidated `project` MCP tool.
17pub struct ProjectHandler {
18    repo: Arc<dyn ProjectRepository>,
19}
20
21handler_new!(ProjectHandler {
22    repo: Arc<dyn ProjectRepository>,
23});
24
25impl ProjectHandler {
26    /// Route an incoming `project` tool call to the appropriate operation.
27    ///
28    /// # Errors
29    /// Returns an error when required identifiers are missing or action/resource is unsupported.
30    #[tracing::instrument(skip_all)]
31    pub async fn handle(
32        &self,
33        Parameters(args): Parameters<ProjectArgs>,
34    ) -> Result<CallToolResult, McpError> {
35        let project_id = &args.project_id;
36        let org_id = resolve_org_id(None);
37
38        if project_id.trim().is_empty() && !matches!(args.action, ProjectAction::List) {
39            return Err(McpError::invalid_params("project_id is required", None));
40        }
41
42        info!(
43            "ProjectHandler",
44            "project request",
45            &format!(
46                "action={:?} resource={:?} project_id={project_id}",
47                args.action, args.resource
48            )
49        );
50
51        match (args.action, args.resource) {
52            (ProjectAction::Get, ProjectResource::Project) => ok_json(&map_opaque_error(
53                self.repo.get_by_id(org_id.as_str(), project_id).await,
54            )?),
55            (ProjectAction::List, ProjectResource::Project) => {
56                ok_json(&map_opaque_error(self.repo.list(org_id.as_str()).await)?)
57            }
58
59            _ => Err(McpError::invalid_params(
60                format!(
61                    "Unsupported action {:?} for resource {:?}",
62                    args.action, args.resource
63                ),
64                None,
65            )),
66        }
67    }
68}
