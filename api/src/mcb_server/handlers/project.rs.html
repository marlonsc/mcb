30impl ProjectHandler {
31    /// Route an incoming `project` tool call to the appropriate operation.
32    ///
33    /// # Errors
34    /// Returns an error when required identifiers are missing or action/resource is unsupported.
35    #[tracing::instrument(skip_all)]
36    pub async fn handle(
37        &self,
38        Parameters(args): Parameters<ProjectArgs>,
39    ) -> Result<CallToolResult, McpError> {
40        let project_id = args.project_id.clone().unwrap_or_default();
41        let org_id = resolve_org_id(None);
42
43        Self::ensure_project_id_present(&project_id, args.action, args.resource)?;
44
45        info!(
46            "ProjectHandler",
47            "project request",
48            &format!(
49                "action={:?} resource={:?} project_id={project_id}",
50                args.action, args.resource
51            )
52        );
53
54        match args.resource {
55            ProjectResource::Project => self.handle_project(&org_id, &project_id, args).await,
56            ProjectResource::Phase => self.handle_phase(&project_id, args).await,
57            ProjectResource::Issue => self.handle_issue(&org_id, &project_id, args).await,
58            ProjectResource::Dependency => self.handle_dependency(args).await,
59            ProjectResource::Decision => self.handle_decision(&project_id, args).await,
60        }
61    }
62
63    /// Reject requests that omit `project_id` for operations that require it.
64    fn ensure_project_id_present(
65        project_id: &str,
66        action: ProjectAction,
67        resource: ProjectResource,
68    ) -> Result<(), McpError> {
69        if !project_id.trim().is_empty() {
70            return Ok(());
71        }
72        let exempt = matches!(
73            (action, resource),
74            (ProjectAction::List, ProjectResource::Project)
75        ) || matches!(
76            (action, resource),
77            (
78                ProjectAction::Get | ProjectAction::Delete,
79                ProjectResource::Phase | ProjectResource::Decision | ProjectResource::Dependency
80            )
81        );
82        if exempt {
83            Ok(())
84        } else {
85            Err(McpError::invalid_params(
86                "project_id is required (not resolved from context)",
87                None,
88            ))
89        }
90    }

92    async fn handle_project(
93        &self,
94        org_id: &str,
95        project_id: &str,
96        args: ProjectArgs,
97    ) -> Result<CallToolResult, McpError> {
98        match args.action {
99            ProjectAction::Get => ResponseFormatter::json_success(&map_opaque_error(
100                self.repo.get_by_id(org_id, project_id).await,
101            )?),
102            ProjectAction::List => {
103                ResponseFormatter::json_success(&map_opaque_error(self.repo.list(org_id).await)?)
104            }
105            ProjectAction::Create => self.create_project(org_id, project_id, &args).await,
106            ProjectAction::Update => self.update_project(org_id, project_id, &args).await,
107            ProjectAction::Delete => self.delete_project(org_id, project_id).await,
108        }
109    }
110
111    async fn handle_phase(
112        &self,
113        project_id: &str,
114        args: ProjectArgs,
115    ) -> Result<CallToolResult, McpError> {
116        match args.action {
117            ProjectAction::Create => {
118                let mut phase: ProjectPhase = require_data(args.data, "data required")?;
119                if phase.project_id.is_empty() {
120                    phase.project_id = project_id.to_owned();
121                }
122                map_opaque_error(self.repo.create_phase(&phase).await)?;
123                ResponseFormatter::json_success(&phase)
124            }
125            ProjectAction::Get => {
126                let id = require_id(&args.id)?;
127                ResponseFormatter::json_success(&map_opaque_error(self.repo.get_phase(&id).await)?)
128            }
129            ProjectAction::List => ResponseFormatter::json_success(&map_opaque_error(
130                self.repo.list_phases(project_id).await,
131            )?),
132            ProjectAction::Update => {
133                let phase: ProjectPhase = require_data(args.data, "data required")?;
134                map_opaque_error(self.repo.update_phase(&phase).await)?;
135                ok_text("updated")
136            }
137            ProjectAction::Delete => {
138                let id = require_id(&args.id)?;
139                map_opaque_error(self.repo.delete_phase(&id).await)?;
140                ok_text("deleted")
141            }
142        }
143    }

145    async fn handle_issue(
146        &self,
147        org_id: &str,
148        project_id: &str,
149        args: ProjectArgs,
150    ) -> Result<CallToolResult, McpError> {
151        match args.action {
152            ProjectAction::Create => {
153                let mut issue: ProjectIssue = require_data(args.data, "data required")?;
154                if issue.project_id.is_empty() {
155                    issue.project_id = project_id.to_owned();
156                }
157                issue.org_id = org_id.to_owned();
158                map_opaque_error(self.repo.create_issue(&issue).await)?;
159                ResponseFormatter::json_success(&issue)
160            }
161            ProjectAction::Get => {
162                let id = require_id(&args.id)?;
163                ResponseFormatter::json_success(&map_opaque_error(
164                    self.repo.get_issue(org_id, &id).await,
165                )?)
166            }
167            ProjectAction::List => {
168                if let Some(filters) = &args.filters {
169                    let filter: IssueFilter = serde_json::from_value(filters.clone())
170                        .map_err(|e| McpError::invalid_params(format!("bad filters: {e}"), None))?;
171                    ResponseFormatter::json_success(&map_opaque_error(
172                        self.repo.list_issues_filtered(org_id, &filter).await,
173                    )?)
174                } else {
175                    ResponseFormatter::json_success(&map_opaque_error(
176                        self.repo.list_issues(org_id, project_id).await,
177                    )?)
178                }
179            }
180            ProjectAction::Update => {
181                let mut issue: ProjectIssue = require_data(args.data, "data required")?;
182                issue.org_id = org_id.to_owned();
183                map_opaque_error(self.repo.update_issue(&issue).await)?;
184                ok_text("updated")
185            }
186            ProjectAction::Delete => {
187                let id = require_id(&args.id)?;
188                map_opaque_error(self.repo.delete_issue(org_id, &id).await)?;
189                ok_text("deleted")
190            }
191        }
192    }

194    async fn handle_dependency(&self, args: ProjectArgs) -> Result<CallToolResult, McpError> {
195        match args.action {
196            ProjectAction::Create => {
197                let dep: ProjectDependency = require_data(args.data, "data required")?;
198                map_opaque_error(self.repo.create_dependency(&dep).await)?;
199                ResponseFormatter::json_success(&dep)
200            }
201            ProjectAction::List => {
202                let issue_id = args.issue_id.as_deref().ok_or_else(|| {
203                    McpError::invalid_params("issue_id required for dependency list", None)
204                })?;
205                ResponseFormatter::json_success(&map_opaque_error(
206                    self.repo.list_dependencies(issue_id).await,
207                )?)
208            }
209            ProjectAction::Delete => {
210                let id = require_id(&args.id)?;
211                map_opaque_error(self.repo.delete_dependency(&id).await)?;
212                ok_text("deleted")
213            }
214            action @ (ProjectAction::Get | ProjectAction::Update) => Err(McpError::invalid_params(
215                format!("Unsupported action {action:?} for resource Dependency"),
216                None,
217            )),
218        }
219    }

221    async fn handle_decision(
222        &self,
223        project_id: &str,
224        args: ProjectArgs,
225    ) -> Result<CallToolResult, McpError> {
226        match args.action {
227            ProjectAction::Create => {
228                let mut decision: ProjectDecision = require_data(args.data, "data required")?;
229                if decision.project_id.is_empty() {
230                    decision.project_id = project_id.to_owned();
231                }
232                map_opaque_error(self.repo.create_decision(&decision).await)?;
233                ResponseFormatter::json_success(&decision)
234            }
235            ProjectAction::Get => {
236                let id = require_id(&args.id)?;
237                ResponseFormatter::json_success(&map_opaque_error(
238                    self.repo.get_decision(&id).await,
239                )?)
240            }
241            ProjectAction::List => ResponseFormatter::json_success(&map_opaque_error(
242                self.repo.list_decisions(project_id).await,
243            )?),
244            ProjectAction::Update => {
245                let decision: ProjectDecision = require_data(args.data, "data required")?;
246                map_opaque_error(self.repo.update_decision(&decision).await)?;
247                ok_text("updated")
248            }
249            ProjectAction::Delete => {
250                let id = require_id(&args.id)?;
251                map_opaque_error(self.repo.delete_decision(&id).await)?;
252                ok_text("deleted")
253            }
254        }
255    }

257    /// Creates a new project from the provided MCP arguments.
258    async fn create_project(
259        &self,
260        org_id: &str,
261        project_id: &str,
262        args: &ProjectArgs,
263    ) -> Result<CallToolResult, McpError> {
264        let data = args
265            .data
266            .as_ref()
267            .ok_or_else(|| McpError::invalid_params("data is required for create", None))?;
268        let name = data
269            .get("name")
270            .and_then(|v| v.as_str())
271            .ok_or_else(|| McpError::invalid_params("name is required in data", None))?;
272        let path = data
273            .get("path")
274            .and_then(|v| v.as_str())
275            .ok_or_else(|| McpError::invalid_params("path is required in data", None))?;
276        let now = mcb_utils::utils::time::epoch_secs_i64()
277            .map_err(|e| safe_internal_error("resolve timestamp", &e))?;
278        let project = Project {
279            id: project_id.to_owned(),
280            org_id: org_id.to_owned(),
281            name: name.to_owned(),
282            path: path.to_owned(),
283            created_at: now,
284            updated_at: now,
285        };
286        map_opaque_error(self.repo.create(&project).await)?;
287        ResponseFormatter::json_success(&project)
288    }

290    /// Updates an existing project with optional field changes from data payload.
291    async fn update_project(
292        &self,
293        org_id: &str,
294        project_id: &str,
295        args: &ProjectArgs,
296    ) -> Result<CallToolResult, McpError> {
297        let mut project = map_opaque_error(self.repo.get_by_id(org_id, project_id).await)?;
298        if let Some(data) = &args.data {
299            if let Some(name) = data.get("name").and_then(|v| v.as_str()) {
300                project.name = name.to_owned();
301            }
302            if let Some(path) = data.get("path").and_then(|v| v.as_str()) {
303                project.path = path.to_owned();
304            }
305        }
306        let now = mcb_utils::utils::time::epoch_secs_i64()
307            .map_err(|e| safe_internal_error("resolve timestamp", &e))?;
308        project.updated_at = now;
309        map_opaque_error(self.repo.update(&project).await)?;
310        ResponseFormatter::json_success(&project)
311    }

313    /// Deletes a project by ID.
314    async fn delete_project(
315        &self,
316        org_id: &str,
317        project_id: &str,
318    ) -> Result<CallToolResult, McpError> {
319        map_opaque_error(self.repo.delete(org_id, project_id).await)?;
320        ResponseFormatter::json_success(&serde_json::json!({
321            "deleted": true,
322            "id": project_id,
323        }))
324    }
