21/// Handler for the consolidated `issue_entity` MCP tool.
22pub struct IssueEntityHandler {
23    repo: Arc<dyn IssueEntityRepository>,
24}
25
26handler_new!(IssueEntityHandler {
27    repo: Arc<dyn IssueEntityRepository>,
28});
29
30impl IssueEntityHandler {
31    /// Route an incoming `issue_entity` tool call to the appropriate CRUD operation.
32    ///
33    /// # Errors
34    /// Returns an error when required identifiers or payload fields are missing.
35    #[tracing::instrument(skip_all)]
36    pub async fn handle(
37        &self,
38        Parameters(args): Parameters<IssueEntityArgs>,
39    ) -> Result<CallToolResult, McpError> {
40        let org_id = resolve_org_id(args.org_id.as_deref());
41        match args.resource {
42            IssueEntityResource::Issue => self.handle_issue(&org_id, args).await,
43            IssueEntityResource::Comment => self.handle_comment(args).await,
44            IssueEntityResource::Label => self.handle_label(&org_id, args).await,
45            IssueEntityResource::LabelAssignment => self.handle_label_assignment(args).await,
46        }
47    }
48
49    /// Dispatch CRUD actions for the `Issue` resource.
50    async fn handle_issue(
51        &self,
52        org_id: &str,
53        args: IssueEntityArgs,
54    ) -> Result<CallToolResult, McpError> {
55        crate::entity_crud_dispatch! {
56            action = args.action,
57            resource = args.resource,
58            {
59            (IssueEntityAction::Create, IssueEntityResource::Issue) => {
60                let mut issue: ProjectIssue = require_data(args.data, "data required for create")?;
61                issue.project_id = require_resolved_identifier(
62                    "project_id",
63                    args.project_id.as_deref(),
64                    Some(issue.project_id.as_str()),
65                    "project_id required for issue create",
66                )?;
67                issue.org_id = org_id.to_owned();
68                map_opaque_error(self.repo.create_issue(&issue).await)?;
69                ResponseFormatter::json_success(&issue)
70            }
71            (IssueEntityAction::Get, IssueEntityResource::Issue) => {
72                let id = require_id(&args.id)?;
73                ResponseFormatter::json_success(&map_opaque_error(self.repo.get_issue(org_id, &id).await)?)
74            }
75            (IssueEntityAction::List, IssueEntityResource::Issue) => {
76                let project_id = require_arg!(args.project_id, "project_id required for list");
77                ResponseFormatter::json_success(&map_opaque_error(self.repo.list_issues(org_id, project_id).await)?)
78            }
79            (IssueEntityAction::Update, IssueEntityResource::Issue) => {
80                let mut issue: ProjectIssue = require_data(args.data, "data required for update")?;
81                issue.org_id = org_id.to_owned();
82                map_opaque_error(self.repo.update_issue(&issue).await)?;
83                ok_text("updated")
84            }
85            (IssueEntityAction::Delete, IssueEntityResource::Issue) => {
86                let id = require_id(&args.id)?;
87                map_opaque_error(self.repo.delete_issue(org_id, &id).await)?;
88                ok_text("deleted")
89            }
90            }
91        }
92    }
