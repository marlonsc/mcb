11pub(super) fn build_search_response_message(
12    query: &str,
13    results: &[SearchResult],
14    duration: Duration,
15    limit: usize,
16) -> String {
17    let mut message = "🔍 **Semantic Code Search Results**\n\n".to_owned();
18    let _ = writeln!(message, "**Query:** \"{query}\" ");
19    let _ = writeln!(
20        message,
21        "**Search completed in:** {:.2}s",
22        duration.as_secs_f64()
23    );
24    let _ = write!(message, "**Results found:** {}\n\n", results.len());
25
26    if results.is_empty() {
27        append_empty_search_response(&mut message);
28    } else {
29        append_search_results(&mut message, results, limit, duration);
30    }
31
32    message
33}
34
35fn append_empty_search_response(message: &mut String) {
36    message.push_str("❌ **No Results Found**\n\n");
37    message.push_str("**Possible Reasons:**\n");
38    message.push_str("• Codebase not indexed yet (run `index` action=start first)\n");
39    message.push_str("• Query terms not present in the codebase\n");
40    message.push_str("• Try different keywords or more general terms\n\n");
41    message.push_str("**Search Tips:**\n");
42    message.push_str("• Use natural language: \"find error handling\", \"authentication logic\"\n");
43    message.push_str("• Be specific: \"HTTP request middleware\" > \"middleware\"\n");
44    message.push_str("• Include technologies: \"React component state management\"\n");
45    message.push_str("• Try synonyms: \"validate\" instead of \"check\"\n");
46}

48fn append_search_results(
49    message: &mut String,
50    results: &[SearchResult],
51    limit: usize,
52    duration: Duration,
53) {
54    message.push_str("📊 **Search Results:**\n\n");
55
56    for (i, result) in results.iter().enumerate() {
57        let _ = writeln!(
58            message,
59            "**{}.** 📁 `{}` (line {})",
60            i + 1,
61            result.file_path,
62            result.start_line
63        );
64
65        append_code_preview(message, result);
66        let _ = write!(message, "🎯 **Relevance Score:** {:.3}\n\n", result.score);
67    }
68
69    if results.len() == limit {
70        let _ = writeln!(
71            message,
72            "💡 **Showing top {limit} results.** For more results, try:"
73        );
74        message.push_str("• More specific search terms\n");
75        message.push_str("• Different query formulations\n");
76        message.push_str("• Breaking complex queries into simpler ones\n");
77    }
78
79    if duration.as_millis() > SEARCH_SLOW_THRESHOLD_MS {
80        let _ = write!(
81            message,
82            "\n⚠️ **Performance Note:** Search took {:.2}s. Consider using more specific queries for faster results.\n",
83            duration.as_secs_f64()
84        );
85    }
86}

88fn append_code_preview(message: &mut String, result: &SearchResult) {
89    let lines: Vec<&str> = result.content.lines().collect();
90    let preview_lines = if lines.len() > CODE_PREVIEW_MAX_LINES {
91        lines
92            .iter()
93            .take(CODE_PREVIEW_MAX_LINES)
94            .cloned()
95            .collect::<Vec<_>>()
96            .join("\n")
97    } else {
98        result.content.clone()
99    };
100
101    let file_ext = Path::new(&result.file_path)
102        .extension()
103        .and_then(|ext| ext.to_str())
104        .unwrap_or("");
105
106    let lang_hint = get_language_hint(file_ext, &result.language);
107
108    if lang_hint.is_empty() {
109        let _ = write!(message, "```\n{preview_lines}\n```\n");
110    } else {
111        let _ = write!(message, "``` {lang_hint}\n{preview_lines}\n```\n");
112    }
113}

115fn get_language_hint<'a>(file_ext: &str, default_lang: &'a str) -> &'a str {
116    match file_ext {
117        "rs" => "rust",
118        "py" => "python",
119        "js" => "javascript",
120        "ts" => "typescript",
121        "go" => "go",
122        "java" => "java",
123        "cpp" | "cc" | "cxx" => "cpp",
124        "c" => "c",
125        "cs" => "csharp",
126        _ => default_lang,
127    }
128}
