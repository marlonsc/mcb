1//! Tool call routing and handler dispatch.
2//!
3//! **Documentation**: [docs/modules/server.md](../../../../docs/modules/server.md)
4//!
5//! Routes incoming tool call requests to the appropriate handlers.
6//! This module provides a centralized dispatch mechanism for MCP tool calls.
7
8use std::sync::Arc;
9
10use rmcp::ErrorData as McpError;
11use rmcp::model::{CallToolRequestParams, CallToolResult};
12
13use crate::handlers::{
14    AgentHandler, EntityHandler, IndexHandler, IssueEntityHandler, MemoryHandler, OrgEntityHandler,
15    PlanEntityHandler, ProjectHandler, SearchHandler, SessionHandler, ValidateHandler,
16    VcsEntityHandler, VcsHandler,
17};
18use crate::hooks::HookProcessor;
19use crate::tools::context::ToolExecutionContext;
20use crate::tools::dispatch_tool_call;
21use crate::tools::validation::{trigger_post_tool_use_hook, validate_execution_context};
22/// Handler references for tool routing
23#[derive(Clone)]
24pub struct ToolHandlers {
25    /// Handler for coding implementation tasks.
26    pub index: Arc<IndexHandler>,
27    /// Handler for search operations.
28    pub search: Arc<SearchHandler>,
29    /// Handler for validation operations.
30    pub validate: Arc<ValidateHandler>,
31    /// Handler for memory operations.
32    pub memory: Arc<MemoryHandler>,
33    /// Handler for session management.
34    pub session: Arc<SessionHandler>,
35    /// Handler for agent operations.
36    pub agent: Arc<AgentHandler>,
37    /// Handler for project management.
38    pub project: Arc<ProjectHandler>,
39    /// Handler for VCS operations.
40    pub vcs: Arc<VcsHandler>,
41    /// Handler for VCS entity CRUD.
42    pub vcs_entity: Arc<VcsEntityHandler>,
43    /// Handler for plan entity CRUD.
44    pub plan_entity: Arc<PlanEntityHandler>,
45    /// Handler for issue entity CRUD.
46    pub issue_entity: Arc<IssueEntityHandler>,
47    /// Handler for org entity CRUD.
48    pub org_entity: Arc<OrgEntityHandler>,
49    /// Handler for unified entity CRUD.
50    pub entity: Arc<EntityHandler>,
51    /// Processor for tool execution hooks.
52    pub hook_processor: Arc<HookProcessor>,
53}
54
55/// Route a tool call request to the appropriate handler
56///
57/// Parses the request arguments and delegates to the matching handler.
58/// After tool execution, automatically triggers `PostToolUse` hook for memory operations.
59///
60/// # Errors
61/// Returns an error when execution context validation or tool dispatch fails.
62pub async fn route_tool_call(
63    request: CallToolRequestParams,
64    handlers: &ToolHandlers,
65    execution_context: ToolExecutionContext,
66) -> Result<CallToolResult, McpError> {
67    validate_execution_context(request.name.as_ref(), &execution_context)?;
68
69    let result = dispatch_tool_call(&request, handlers).await?;
70
71    if let Err(e) = trigger_post_tool_use_hook(
72        request.name.as_ref(),
73        &result,
74        &handlers.hook_processor,
75        &execution_context,
76    )
77    .await
78    {
79        mcb_domain::warn!("ToolRouter", "PostToolUse hook failed (non-fatal)", &e);
80    }
81
82    Ok(result)
83}
