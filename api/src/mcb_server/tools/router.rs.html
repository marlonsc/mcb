58/// Execution context extracted at transport boundary and propagated to hooks.
59pub struct ToolExecutionContext {
60    /// Canonical session identifier for the current tool call.
61    pub session_id: Option<String>,
62    /// Optional parent session identifier for delegated/subagent calls.
63    pub parent_session_id: Option<String>,
64    /// Optional project identifier associated with this execution.
65    pub project_id: Option<String>,
66    /// Optional worktree identifier associated with this execution.
67    pub worktree_id: Option<String>,
68    /// Optional repository identifier associated with this execution.
69    pub repo_id: Option<String>,
70    /// Optional repository/workspace path associated with this execution.
71    pub repo_path: Option<String>,
72    /// Optional operator/user identifier for this execution.
73    pub operator_id: Option<String>,
74    /// Optional machine/host fingerprint for this execution.
75    pub machine_id: Option<String>,
76    /// Optional agent program/IDE identifier for this execution.
77    pub agent_program: Option<String>,
78    /// Optional model identifier for this execution.
79    pub model_id: Option<String>,
80    /// Optional delegated flag for this execution.
81    pub delegated: Option<bool>,
82    /// Execution timestamp (Unix timestamp in seconds).
83    pub timestamp: Option<i64>,
84    /// Optional execution flow identifier for tracing.
85    pub execution_flow: Option<String>,
86}

99        insert_argument_if_missing(request, "repo_id", self.repo_id.as_deref());
100        insert_argument_if_missing(request, "repo_path", self.repo_path.as_deref());
101        insert_argument_if_missing(request, "operator_id", self.operator_id.as_deref());
102        insert_argument_if_missing(request, "machine_id", self.machine_id.as_deref());
103        insert_argument_if_missing(request, "agent_program", self.agent_program.as_deref());
104        insert_argument_if_missing(request, "model_id", self.model_id.as_deref());
105        insert_bool_argument_if_missing(request, "delegated", self.delegated);

107        insert_argument_if_missing(request, "execution_flow", self.execution_flow.as_deref());

111fn insert_i64_argument_if_missing(
112    request: &mut CallToolRequestParams,
113    key: &'static str,
114    value: Option<i64>,
115) {
116    let Some(value) = value else {
117        return;
118    };
119
120    let arguments = request.arguments.get_or_insert_with(Default::default);
121    arguments
122        .entry(key.to_owned())
123        .or_insert_with(|| Value::Number(serde_json::Number::from(value)));
124}

156/// Route a tool call request to the appropriate handler
157///
158/// Parses the request arguments and delegates to the matching handler.
159/// After tool execution, automatically triggers `PostToolUse` hook for memory operations.
160///
161/// # Errors
162/// Returns an error when execution context validation or tool dispatch fails.
163pub async fn route_tool_call(
164    request: CallToolRequestParams,
165    handlers: &ToolHandlers,
166    execution_context: ToolExecutionContext,
167) -> Result<CallToolResult, McpError> {
168    validate_execution_context(request.name.as_ref(), &execution_context)?;
169
170    let result = dispatch_tool_call(&request, handlers).await?;
171
172    if let Err(e) = trigger_post_tool_use_hook(
173        request.name.as_ref(),
174        &result,
175        &handlers.hook_processor,
176        &execution_context,
177    )
178    .await
179    {
180        warn!("ToolRouter", "PostToolUse hook failed (non-fatal)", &e);
181    }
182
183    Ok(result)
184}

186fn validate_execution_context(
187    tool_name: &str,
188    execution_context: &ToolExecutionContext,
189) -> Result<(), McpError> {
190    validate_operation_mode_matrix(tool_name, execution_context)?;
191
192    let requires_provenance = matches!(tool_name, "index" | "search" | "memory");
193    if !requires_provenance {
194        return Ok(());
195    }
196
197    let mut missing = Vec::new();
198    if is_missing_text(&execution_context.session_id) {
199        missing.push("session_id");
200    }
201    if is_missing_text(&execution_context.repo_id) {
202        missing.push("repo_id");
203    }
204    if is_missing_text(&execution_context.repo_path) {
205        missing.push("repo_path");
206    }
207    if is_missing_text(&execution_context.operator_id) {
208        missing.push("operator_id");
209    }
210    if is_missing_text(&execution_context.machine_id) {
211        missing.push("machine_id");
212    }
213    if is_missing_text(&execution_context.agent_program) {
214        missing.push("agent_program");
215    }
216    if is_missing_text(&execution_context.model_id) {
217        missing.push("model_id");
218    }
219    if execution_context.delegated.is_none() {
220        missing.push("delegated");
221    }
222    if execution_context.timestamp.is_none() {
223        missing.push("timestamp");
224    }
225    if execution_context.delegated == Some(true)
226        && is_missing_text(&execution_context.parent_session_id)
227    {
228        missing.push("parent_session_id");
229    }
230
231    if missing.is_empty() {
232        Ok(())
233    } else {
234        Err(McpError::invalid_params(
235            format!(
236                "Missing execution provenance for '{tool_name}': {}",
237                missing.join(", ")
238            ),
239            None,
240        ))
241    }
242}
243
244fn is_missing_text(value: &Option<String>) -> bool {
245    value.as_deref().is_none_or(|s| s.trim().is_empty())
246}

248fn validate_operation_mode_matrix(
249    tool_name: &str,
250    execution_context: &ToolExecutionContext,
251) -> Result<(), McpError> {
252    let flow = normalize_execution_flow(execution_context.execution_flow.as_deref())?;
253
254    let allowed = if matches!(tool_name, "validate") {
255        &["stdio-only", "client-hybrid"][..]
256    } else {
257        &["stdio-only", "client-hybrid", "server-hybrid"][..]
258    };
259
260    if allowed.contains(&flow) {
261        Ok(())
262    } else {
263        Err(McpError::invalid_params(
264            format!(
265                "Operation mode matrix violation for '{tool_name}': flow '{flow}' is not allowed. Allowed flows: {}",
266                allowed.join(", ")
267            ),
268            None,
269        ))
270    }
271}
272
273fn normalize_execution_flow(flow: Option<&str>) -> Result<&'static str, McpError> {
274    let normalized = flow.unwrap_or("stdio-only").trim().to_ascii_lowercase();
275    match normalized.as_str() {
276        "stdio-only" => Ok("stdio-only"),
277        "client-hybrid" => Ok("client-hybrid"),
278        "server-hybrid" => Ok("server-hybrid"),
279        _ => Err(McpError::invalid_params(
280            format!(
281                "Invalid execution_flow '{normalized}'. Expected one of: stdio-only, client-hybrid, server-hybrid"
282            ),
283            None,
284        )),
285    }
286}

288async fn trigger_post_tool_use_hook(
289    tool_name: &str,
290    result: &CallToolResult,
291    hook_processor: &HookProcessor,
292    execution_context: &ToolExecutionContext,
293) -> Result<(), String> {
294    let mut context =
295        PostToolUseContext::new(tool_name.to_owned(), result.is_error.unwrap_or(false));
296
297    if let Some(session_id) = &execution_context.session_id {
298        context = context.with_session_id(SessionId::from_string(session_id));
299    }
300    if let Some(parent_session_id) = &execution_context.parent_session_id {
301        context = context.with_metadata("parent_session_id", parent_session_id.as_str());
302    }
303    if let Some(project_id) = &execution_context.project_id {
304        context = context.with_metadata("project_id", project_id.as_str());
305    }
306    if let Some(worktree_id) = &execution_context.worktree_id {
307        context = context.with_metadata("worktree_id", worktree_id.as_str());
308    }
309    if let Some(repo_id) = &execution_context.repo_id {
310        context = context.with_metadata("repo_id", repo_id.as_str());
311    }
312    if let Some(repo_path) = &execution_context.repo_path {
313        context = context.with_metadata("repo_path", repo_path.as_str());
314    }
315    if let Some(operator_id) = &execution_context.operator_id {
316        context = context.with_metadata("operator_id", operator_id.as_str());
317    }
318    if let Some(machine_id) = &execution_context.machine_id {
319        context = context.with_metadata("machine_id", machine_id.as_str());
320    }
321    if let Some(agent_program) = &execution_context.agent_program {
322        context = context.with_metadata("agent_program", agent_program.as_str());
323    }
324    if let Some(model_id) = &execution_context.model_id {
325        context = context.with_metadata("model_id", model_id.as_str());
326    }
327    if let Some(delegated) = execution_context.delegated {
328        context = context.with_metadata("delegated", delegated.to_string());
329    }
330    if let Some(timestamp) = execution_context.timestamp {
331        context = context.with_metadata("timestamp", timestamp.to_string());
332    }
333
334    hook_processor
335        .process_post_tool_use(context)
336        .await
337        .map_err(|e| e.to_string())
338}

362    #[test]
363    fn rejects_blank_provenance_scope_for_search() {
364        let mut context = valid_context();
365        context.operator_id = Some("   ".to_owned());
366
367        let validation = validate_execution_context("search", &context);
368        assert!(validation.is_err(), "blank operator_id must be rejected");
369        let error = match validation {
370            Ok(()) => return,
371            Err(error) => error,
372        };
373        assert!(error.message.contains("operator_id"));
374    }
