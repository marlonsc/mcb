1//! Tool execution context extraction and resolution.
2//!
3//! Provides the `ToolExecutionContext` struct for capturing execution provenance
4//! at the transport boundary and resolving context values from request/context metadata.
5
6use std::collections::HashMap;
7
8use rmcp::model::{CallToolRequestParams, Meta};
9use serde_json::Value;
10
11use crate::tools::defaults::RuntimeDefaults;
12use crate::tools::field_aliases::{
13    BOOL_FIELD_ALIASES, STRING_FIELD_ALIASES, field_aliases, resolve_override_bool,
14    resolve_override_value,
15};

17#[derive(Debug, Clone, Default)]
18/// Execution context extracted at transport boundary and propagated to hooks.
19pub struct ToolExecutionContext {
20    /// Canonical session identifier for the current tool call.
21    pub session_id: Option<String>,
22    /// Optional parent session identifier for delegated/subagent calls.
23    pub parent_session_id: Option<String>,
24    /// Optional organization identifier associated with this execution.
25    pub org_id: Option<String>,
26    /// Optional project identifier associated with this execution.
27    pub project_id: Option<String>,
28    /// Optional worktree identifier associated with this execution.
29    pub worktree_id: Option<String>,
30    /// Optional repository identifier associated with this execution.
31    pub repo_id: Option<String>,
32    /// Optional repository/workspace path associated with this execution.
33    pub repo_path: Option<String>,
34    /// Optional operator/user identifier for this execution.
35    pub operator_id: Option<String>,
36    /// Optional machine/host fingerprint for this execution.
37    pub machine_id: Option<String>,
38    /// Optional agent program/IDE identifier for this execution.
39    pub agent_program: Option<String>,
40    /// Optional model identifier for this execution.
41    pub model_id: Option<String>,
42    /// Optional delegated flag for this execution.
43    pub delegated: Option<bool>,
44    /// Execution timestamp (Unix timestamp in seconds).
45    pub timestamp: Option<i64>,
46    /// Optional execution flow identifier for tracing.
47    pub execution_flow: Option<String>,
48}
49
50impl ToolExecutionContext {
51    /// Collect request/context metadata into canonical override keys.
52    #[must_use]
53    pub fn metadata_overrides(
54        request_meta: Option<&Meta>,
55        context_meta: &Meta,
56    ) -> HashMap<String, String> {
57        let mut overrides = HashMap::new();
58        for &(canonical, aliases) in STRING_FIELD_ALIASES {
59            crate::tools::field_aliases::insert_override(
60                &mut overrides,
61                canonical,
62                resolve_context_value(request_meta, context_meta, aliases),
63            );
64        }
65        for &(canonical, aliases) in BOOL_FIELD_ALIASES {
66            if let Some(val) = resolve_context_bool(request_meta, context_meta, aliases) {
67                overrides.insert(canonical.to_owned(), val.to_string());
68            }
69        }
70        overrides
71    }
72
73    /// Resolve execution context from explicit overrides and runtime defaults.
74    #[must_use]
75    pub fn resolve(defaults: &RuntimeDefaults, overrides: &HashMap<String, String>) -> Self {
76        let session_id = resolve_override_value(overrides, field_aliases("session_id"))
77            .or_else(|| defaults.session_id.clone());
78        let parent_session_id =
79            resolve_override_value(overrides, field_aliases("parent_session_id"));
80        let org_id = resolve_override_value(overrides, field_aliases("org_id"))
81            .or_else(|| defaults.org_id.clone());
82        let project_id = resolve_override_value(overrides, field_aliases("project_id"))
83            .or_else(|| defaults.project_id.clone());
84        let worktree_id = resolve_override_value(overrides, field_aliases("worktree_id"));
85        let repo_id = resolve_override_value(overrides, field_aliases("repo_id"))
86            .or_else(|| defaults.repo_id.clone());
87        let repo_path = resolve_override_value(overrides, field_aliases("repo_path"))
88            .or_else(|| resolve_override_value(overrides, field_aliases("workspace_root")))
89            .or_else(|| defaults.repo_path.clone())
90            .or_else(|| defaults.workspace_root.clone());
91        let operator_id = resolve_override_value(overrides, field_aliases("operator_id"))
92            .or_else(|| defaults.operator_id.clone());
93        let machine_id = resolve_override_value(overrides, field_aliases("machine_id"))
94            .or_else(|| defaults.machine_id.clone());
95        let agent_program = resolve_override_value(overrides, field_aliases("agent_program"))
96            .or_else(|| defaults.agent_program.clone());
97        let model_id = resolve_override_value(overrides, field_aliases("model_id"))
98            .or_else(|| defaults.model_id.clone());
99        let delegated = resolve_override_bool(overrides, field_aliases("delegated"))
100            .or(Some(parent_session_id.is_some()));
101        let execution_flow = resolve_override_value(overrides, field_aliases("execution_flow"))
102            .or_else(|| defaults.execution_flow.map(|f| f.to_string()));
103
104        Self {
105            session_id,
106            parent_session_id,
107            org_id,
108            project_id,
109            worktree_id,
110            repo_id,
111            repo_path,
112            operator_id,
113            machine_id,
114            agent_program,
115            model_id,
116            delegated,
117            timestamp: mcb_utils::utils::time::epoch_secs_i64().ok(),
118            execution_flow,
119        }
120    }
121
122    /// Inject execution context into tool arguments when those keys are missing.
123    pub fn apply_to_request_if_missing(&self, request: &mut CallToolRequestParams) {
124        let args = request.arguments.get_or_insert_with(Default::default);
125        for (key, value) in self.context_entries() {
126            if let Some(v) = value {
127                args.entry(key.to_owned()).or_insert(v);
128            }
129        }
130    }
131
132    /// Build the `(key, optional JSON value)` pairs representing this execution context.
133    fn context_entries(&self) -> [(&'static str, Option<Value>); 14] {
134        let str_value = crate::tools::field_aliases::str_value;
135        [
136            ("session_id", self.session_id.as_deref().map(str_value)),
137            (
138                "parent_session_id",
139                self.parent_session_id.as_deref().map(str_value),
140            ),
141            ("org_id", self.org_id.as_deref().map(str_value)),
142            ("project_id", self.project_id.as_deref().map(str_value)),
143            ("worktree_id", self.worktree_id.as_deref().map(str_value)),
144            ("repo_id", self.repo_id.as_deref().map(str_value)),
145            ("repo_path", self.repo_path.as_deref().map(str_value)),
146            ("operator_id", self.operator_id.as_deref().map(str_value)),
147            ("machine_id", self.machine_id.as_deref().map(str_value)),
148            (
149                "agent_program",
150                self.agent_program.as_deref().map(str_value),
151            ),
152            ("model_id", self.model_id.as_deref().map(str_value)),
153            ("delegated", self.delegated.map(Value::Bool)),
154            ("timestamp", self.timestamp.map(|v| Value::Number(v.into()))),
155            (
156                "execution_flow",
157                self.execution_flow.as_deref().map(str_value),
158            ),
159        ]
160    }
161}

163/// Extract a string value from metadata by checking all aliases.
164fn meta_value_as_string(meta: &Meta, keys: &[&str]) -> Option<String> {
165    for key in keys {
166        let Some(value) = meta.get(*key) else {
167            continue;
168        };
169
170        let extracted = match value {
171            Value::String(v) => {
172                let trimmed = v.trim();
173                if trimmed.is_empty() {
174                    None
175                } else {
176                    Some(trimmed.to_owned())
177                }
178            }
179            Value::Number(v) => Some(v.to_string()),
180            Value::Bool(v) => Some(v.to_string()),
181            Value::Null | Value::Array(_) | Value::Object(_) => None,
182        };
183
184        if extracted.is_some() {
185            return extracted;
186        }
187    }
188
189    None
190}

192/// Resolve a string value from request or context metadata.
193fn resolve_context_value(
194    request_meta: Option<&Meta>,
195    context_meta: &Meta,
196    keys: &[&str],
197) -> Option<String> {
198    request_meta
199        .and_then(|meta| meta_value_as_string(meta, keys))
200        .or_else(|| meta_value_as_string(context_meta, keys))
201}
202
203/// Extract a boolean value from metadata by checking all aliases.
204fn meta_value_as_bool(meta: &Meta, keys: &[&str]) -> Option<bool> {
205    for key in keys {
206        let Some(value) = meta.get(*key) else {
207            continue;
208        };
209
210        let extracted = match value {
211            Value::Bool(v) => Some(*v),
212            Value::String(v) => match v.trim().to_ascii_lowercase().as_str() {
213                "true" | "1" | "yes" => Some(true),
214                "false" | "0" | "no" => Some(false),
215                _other => {
216                    mcb_domain::trace!("context", "Unrecognized boolean mapping", v);
217                    None
218                }
219            },
220            Value::Null | Value::Number(_) | Value::Array(_) | Value::Object(_) => None,
221        };
222
223        if extracted.is_some() {
224            return extracted;
225        }
226    }
227
228    None
229}

231/// Resolve a boolean value from request or context metadata.
232fn resolve_context_bool(
233    request_meta: Option<&Meta>,
234    context_meta: &Meta,
235    keys: &[&str],
236) -> Option<bool> {
237    request_meta
238        .and_then(|meta| meta_value_as_bool(meta, keys))
239        .or_else(|| meta_value_as_bool(context_meta, keys))
