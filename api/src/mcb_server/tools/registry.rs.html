1//! Registry-backed tool definitions and dispatch for MCP protocol.
2//!
3//! # Safety
4//!
5//! [`linkme::distributed_slice`] uses `#[link_section]` internally which
6//! requires `unsafe_code`. The macro generates only a static array of
7//! [`ToolDescriptor`] references — no raw pointer dereference or mutable
8//! aliasing occurs. Every entry is produced by [`register_tool!`] which emits
9//! safe, read-only descriptors. This is the same pattern used in `mcb-domain`
10//! registry macros. See: <https://docs.rs/linkme/latest/linkme/attr.distributed_slice.html>
11//!
12//! The `#![allow(unsafe_code)]` is scoped to this module file alone — it is
13//! the single blessed location for tool registration in the server crate.
14
15// Allow unsafe in this module only — linkme::distributed_slice requires it,
16// and there is no safe alternative for compile-time registration.

41/// Async future returned by a descriptor-based tool call.
42pub type ToolCallFuture<'a> =
43    Pin<Box<dyn Future<Output = Result<CallToolResult, McpError>> + Send + 'a>>;
44/// Function pointer signature used by registry-backed dispatch.
45pub type ToolCallFn = for<'a> fn(&'a CallToolRequestParams, &'a ToolHandlers) -> ToolCallFuture<'a>;
46
47/// Single source-of-truth descriptor for both tool listing and dispatch.
48pub struct ToolDescriptor {
49    /// Unique tool name exposed via MCP.
50    pub name: &'static str,
51    /// Human-readable tool description.
52    pub description: &'static str,
53    /// Factory that produces the JSON schema for tool arguments.
54    pub schema: fn() -> schemars::Schema,
55    /// Dispatch function for tool invocation.
56    pub call: ToolCallFn,
57}

65fn parse_args<T>(request: &CallToolRequestParams) -> Result<T, McpError>
66where
67    T: serde::de::DeserializeOwned + Validate,
68{
69    let args_value = match &request.arguments {
70        Some(map) => serde_json::Value::Object(map.clone()),
71        None => serde_json::Value::Object(Default::default()),
72    };
73    let args: T = serde_json::from_value(args_value)
74        .map_err(|e| McpError::invalid_params(format!("Failed to parse arguments: {e}"), None))?;
75    args.validate()
76        .map_err(|e| McpError::invalid_params(format!("Argument validation failed: {e}"), None))?;
77    Ok(args)
78}

335fn create_tool_from_descriptor(descriptor: &ToolDescriptor) -> Result<Tool, McpError> {
336    let schema_value = serde_json::to_value((descriptor.schema)())
337        .map_err(|e| safe_internal_error("serialize tool schema", &e))?;
338    let input_schema = schema_value
339        .as_object()
340        .ok_or_else(|| {
341            safe_internal_error(
342                "validate tool schema shape",
343                &format_args!("schema for '{}' is not an object", descriptor.name),
344            )
345        })?
346        .clone();
347    // rmcp 1.x marks Tool #[non_exhaustive]; build via its constructor.
348    Ok(Tool::new(
349        Cow::Borrowed(descriptor.name),
350        Cow::Borrowed(descriptor.description),
351        Arc::new(input_schema),
352    ))
353}

355fn validate_registry_unique_tool_names() -> Result<(), McpError> {
356    let mut names = HashSet::new();
357    for descriptor in TOOL_DESCRIPTORS {
358        if !names.insert(descriptor.name) {
359            return Err(safe_internal_error(
360                "validate tool registry",
361                &format_args!("duplicate tool descriptor: {}", descriptor.name),
362            ));
363        }
364    }
365    Ok(())
366}

368fn descriptor_by_name(name: &str) -> Option<&'static ToolDescriptor> {
369    TOOL_DESCRIPTORS
370        .iter()
371        .find(|descriptor| descriptor.name == name)
372}

374/// Resolve a tool definition by name from the descriptor registry.
375///
376/// # Errors
377///
378/// Returns an error if the tool name is unknown or the registry contains duplicates.
379pub fn tool_by_name(name: &str) -> Result<Tool, McpError> {
380    validate_registry_unique_tool_names()?;
381    let descriptor = descriptor_by_name(name)
382        .ok_or_else(|| McpError::invalid_params(format!("Unknown tool: {name}"), None))?;
383    create_tool_from_descriptor(descriptor)
384}

386/// Create the complete list of available tools from the shared registry.
387///
388/// # Errors
389///
390/// Returns an error if the registry contains duplicates or a schema fails to serialize.
391pub fn create_tool_list() -> Result<Vec<Tool>, McpError> {
392    validate_registry_unique_tool_names()?;
393    TOOL_DESCRIPTORS
394        .iter()
395        .map(create_tool_from_descriptor)
396        .collect()
397}

399/// Dispatch to the tool call function from the shared descriptor registry.
400///
401/// # Errors
402///
403/// Returns an error if the tool name is unknown or the handler fails.
404pub async fn dispatch_tool_call(
405    request: &CallToolRequestParams,
406    handlers: &ToolHandlers,
407) -> Result<CallToolResult, McpError> {
408    validate_registry_unique_tool_names()?;
409    let descriptor = descriptor_by_name(request.name.as_ref())
410        .ok_or_else(|| McpError::invalid_params(format!("Unknown tool: {}", request.name), None))?;
411    (descriptor.call)(request, handlers).await
412}
