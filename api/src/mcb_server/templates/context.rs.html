15pub(crate) fn mime_for_extension(ext: &str) -> Option<String> {
16    let mime = match ext {
17        "html" | "htm" => "text/html",
18        "xml" => "application/xml",
19        "json" => "application/json",
20        "css" => "text/css",
21        "js" | "mjs" => "application/javascript",
22        "txt" => "text/plain",
23        "csv" => "text/csv",
24        "svg" => "image/svg+xml",
25        "md" | "markdown" => "text/markdown",
26        "yaml" | "yml" => "application/yaml",
27        "toml" => "application/toml",
28        _ => return None,
29    };
30    Some(mime.to_owned())
31}

44impl Context {
45    pub fn initialize(root: &Path, callback: &Callback) -> Option<Context> {
46        fn is_file_with_ext(entry: &walkdir::DirEntry, ext: &str) -> bool {
47            let is_file = entry.file_type().is_file();
48            let has_ext = entry.path().extension().is_some_and(|e| e == ext);
49            is_file && has_ext
50        }
51
52        let root = match root.normalize() {
53            Ok(root) => root.into_path_buf(),
54            Err(_) => {
55                mcb_domain::info!(
56                    "Template",
57                    "Directory not found, using embedded templates",
58                    &root.display()
59                );
60                return Self::initialize_embedded(callback);
61            }
62        };
63
64        let mut templates: HashMap<String, TemplateInfo> = HashMap::new();
65        for &ext in Engines::ENABLED_EXTENSIONS {
66            for entry in walkdir::WalkDir::new(&root).follow_links(true) {
67                let entry = match entry {
68                    Ok(entry) if is_file_with_ext(&entry, ext) => entry,
69                    Ok(_) | Err(_) => continue,
70                };
71
72                let (name, data_type_str) = match split_path(&root, entry.path()) {
73                    Ok(parts) => parts,
74                    Err(error) => {
75                        let detail = format!(
76                            "path={}, root={}, error={error}",
77                            entry.path().display(),
78                            root.display()
79                        );
80                        mcb_domain::warn!("Template", "Failed to split template path", &detail);
81                        continue;
82                    }
83                };
84                if let Some(info) = templates.get(&*name) {
85                    let existing = match info.path {
86                        Some(ref path) => format!("path: {path:?}"),
87                        None => format!("type: {}", info.data_type),
88                    };
89                    let detail = format!(
90                        "'{name}' duplicate (existing {existing}, additional {:?}) â€” keeping existing",
91                        entry.path()
92                    );
93                    mcb_domain::warn!("Template", "Template name not unique", &detail);
94                    continue;
95                }
96
97                let data_type = data_type_str
98                    .as_ref()
99                    .and_then(|ext| mime_for_extension(ext))
100                    .unwrap_or_else(|| "text/plain".to_owned());
101
102                templates.insert(
103                    name,
104                    TemplateInfo {
105                        path: Some(entry.into_path()),
106                        engine_ext: ext,
107                        data_type,
108                    },
109                );
110            }
111        }
112
113        let mut engines = Engines::init(&templates)?;
114        if let Err(e) = callback(&mut engines) {
115            mcb_domain::error!("Template", "Customization callback failed", &e);
116            return None;
117        }
118
119        for (name, engine_ext) in engines.templates() {
120            if !templates.contains_key(name) {
121                let data_type = Path::new(name)
122                    .extension()
123                    .and_then(|osstr| osstr.to_str())
124                    .and_then(mime_for_extension)
125                    .unwrap_or_else(|| "text/plain".to_owned());
126
127                let info = TemplateInfo {
128                    path: None,
129                    engine_ext,
130                    data_type,
131                };
132                templates.insert(name.to_owned(), info);
133            }
134        }
135
136        Some(Context {
137            root,
138            templates,
139            engines,
140        })
141    }
142
143    fn initialize_embedded(callback: &Callback) -> Option<Context> {
144        let mut engines = Engines::init_embedded()?;
145        if let Err(e) = callback(&mut engines) {
146            mcb_domain::error!("Template", "Customization callback failed", &e);
147            return None;
148        }
149
150        let mut templates = HashMap::new();
151        for &(name, _) in embedded::EMBEDDED_TEMPLATES {
152            let data_type = Path::new(name)
153                .extension()
154                .and_then(|osstr| osstr.to_str())
155                .and_then(mime_for_extension)
156                .unwrap_or_else(|| "text/html".to_owned());
157
158            templates.insert(
159                name.to_owned(),
160                TemplateInfo {
161                    path: None,
162                    engine_ext: "hbs",
163                    data_type,
164                },
165            );
166        }
167
168        for (name, engine_ext) in engines.templates() {
169            if !templates.contains_key(name) {
170                let data_type = Path::new(name)
171                    .extension()
172                    .and_then(|osstr| osstr.to_str())
173                    .and_then(mime_for_extension)
174                    .unwrap_or_else(|| "text/plain".to_owned());
175
176                templates.insert(
177                    name.to_owned(),
178                    TemplateInfo {
179                        path: None,
180                        engine_ext,
181                        data_type,
182                    },
183                );
184            }
185        }
186
187        Some(Context {
188            root: PathBuf::from("<embedded>"),
189            templates,
190            engines,
191        })
192    }
193}

219mod manager {
220    use std::ops::Deref;
221    use std::sync::mpsc::{Receiver, channel};
222    use std::sync::{Mutex, RwLock};
223
224    use notify::{Event, RecommendedWatcher, RecursiveMode, Result as NotifyResult, Watcher};
225
226    use super::Context;
227
228    pub(crate) struct ContextManager {
229        context: RwLock<Context>,
230        watcher: Option<(RecommendedWatcher, Mutex<Receiver<NotifyResult<Event>>>)>,
231    }
232
233    impl ContextManager {
234        pub fn new(ctxt: Context) -> ContextManager {
235            let (tx, rx) = channel();
236            let watcher = notify::recommended_watcher(move |res: NotifyResult<Event>| {
237                let _ = tx.send(res);
238            })
239            .and_then(|mut watcher| {
240                watcher.watch(&ctxt.root.canonicalize()?, RecursiveMode::Recursive)?;
241                Ok(watcher)
242            });
243
244            let watcher = match watcher {
245                Ok(watcher) => Some((watcher, Mutex::new(rx))),
246                Err(e) => {
247                    mcb_domain::warn!("Template", "Live reloading unavailable", &e);
248                    None
249                }
250            };
251
252            ContextManager {
253                watcher,
254                context: RwLock::new(ctxt),
255            }
256        }
257
258        pub fn context(&self) -> impl Deref<Target = Context> + '_ {
259            self.context.read().unwrap_or_else(|poisoned| {
260                mcb_domain::warn!("Template", "RwLock poisoned, continuing with inner value");
261                poisoned.into_inner()
262            })
263        }
264
265        pub fn is_reloading(&self) -> bool {
266            self.watcher.is_some()
267        }
268    }
269}

271fn remove_extension(path: &Path) -> PathBuf {
272    let stem = match path.file_stem() {
273        Some(stem) => stem,
274        None => return path.to_path_buf(),
275    };
276
277    match path.parent() {
278        Some(parent) => parent.join(stem),
279        None => PathBuf::from(stem),
280    }
281}

283fn split_path(root: &Path, path: &Path) -> mcb_domain::error::Result<(String, Option<String>)> {
284    let rel_path = domain_path::strict_strip_prefix(path, root)?;
285    let path_no_ext = remove_extension(&rel_path);
286    let data_type = path_no_ext.extension();
287    let mut name = domain_path::path_to_utf8_string(&remove_extension(&path_no_ext))?;
288
289    if cfg!(windows) {
290        name = name.replace("\\", "/");
291    }
292
293    let data_type = data_type
294        .map(|d| domain_path::path_to_utf8_string(Path::new(d)))
295        .transpose()?;
296
297    Ok((name, data_type))
298}
