66impl AxumRouter {

81/// Builds the Axum router with middleware and routes.
82pub fn build_router(state: &Arc<AppState>) -> Router {
83    let cors = CorsLayer::new()
84        .allow_origin(Any)
85        .allow_methods([Method::GET, Method::POST, Method::PATCH, Method::OPTIONS])
86        .allow_headers(Any);
87
88    let mut app = Router::new()
89        .route("/health", get(health_handler))
90        .route("/ready", get(readiness_handler))
91        .route("/live", get(liveness_handler))
92        .with_state(Arc::clone(state));
93
94    // Mount config routes when admin state is available
95    if let (Some(admin_state), Some(auth_config)) =
96        (state.admin_state.clone(), state.auth_config.clone())
97    {
98        let admin_routes = Router::new()
99            .route("/metrics", get(get_metrics_axum))
100            .route("/health/extended", get(extended_health_check_axum))
101            .route("/config", get(get_config_axum))
102            .route("/config/reload", post(reload_config_axum))
103            .route("/config/{section}", patch(update_config_section_axum))
104            .route("/shutdown", post(shutdown_axum))
105            .route("/cache/stats", get(get_cache_stats_axum))
106            .route("/services", get(list_services_axum))
107            .route("/services/health", get(services_health_axum))
108            .route("/services/{name}/start", post(start_service_axum))
109            .route("/services/{name}/stop", post(stop_service_axum))
110            .route("/services/{name}/restart", post(restart_service_axum))
111            .route("/browse/projects", get(list_browse_projects))
112            .route("/browse/repositories", get(list_browse_repositories))
113            .route("/browse/plans", get(list_browse_plans))
114            .route("/browse/issues", get(list_browse_issues))
115            .route("/browse/organizations", get(list_browse_organizations))
116            .layer(axum::middleware::from_fn(axum_admin_auth_layer))
117            .layer(Extension(auth_config))
118            .with_state(Arc::clone(&admin_state));
119
120        let public_admin_routes = Router::new()
121            .route("/jobs", get(get_jobs_status_axum))
122            .route("/events", get(events_stream))
123            .with_state(Arc::clone(&admin_state));
124
125        app = app
126            .merge(admin_routes)
127            .merge(public_admin_routes)
128            .merge(web_router_with_state((*admin_state).clone()));
129
130        if let Some(browse_state) = state.browse_state.clone() {
131            let browse_routes = Router::new()
132                .route("/collections", get(list_collections_axum))
133                .route("/collections/{name}/files", get(list_collection_files_axum))
134                .route(
135                    "/collections/{name}/chunks/{*path}",
136                    get(get_file_chunks_axum),
137                )
138                .route("/collections/{name}/tree", get(get_collection_tree_axum))
139                .layer(axum::middleware::from_fn(axum_admin_auth_layer))
140                .layer(Extension(state.auth_config.clone().unwrap_or_default()))
141                .with_state(browse_state);
142            app = app.merge(browse_routes);
143        }
144    }
145
146    app.layer(TraceLayer::new_for_http()).layer(cors)
147}

149/// Binds an address and serves the Axum router.
150///
151/// # Errors
152/// Returns an error if binding or serving fails.
153pub async fn run_axum_server(
154    addr: SocketAddr,
155    state: Arc<AppState>,
156) -> Result<(), Box<dyn std::error::Error + Send + Sync>> {
157    let listener = tokio::net::TcpListener::bind(addr).await?;
158    let router = AxumRouter::new(&state).into_inner();
159    let local_addr = listener.local_addr()?;
160    info!("AxumTransport", "Axum transport listening", &local_addr);
161    axum::serve(listener, router).await?;
162
163    Ok(())
164}

166async fn health_handler(State(state): State<Arc<AppState>>) -> Json<AdminHealthResponse> {
167    let metrics = state.metrics.get_performance_metrics();
168    let operations = state.indexing.get_operations();
169
170    Json(AdminHealthResponse {
171        status: "healthy",
172        uptime_seconds: metrics.uptime_seconds,
173        active_indexing_operations: operations.len(),
174    })
175}

177async fn readiness_handler(State(state): State<Arc<AppState>>) -> impl IntoResponse {
178    let metrics = state.metrics.get_performance_metrics();
179    let ready = metrics.uptime_seconds >= 1;
180    let status = if ready {
181        StatusCode::OK
182    } else {
183        StatusCode::SERVICE_UNAVAILABLE
184    };
185    (
186        status,
187        Json(ReadinessResponse {
188            ready,
189            uptime_seconds: metrics.uptime_seconds,
190        }),
191    )
192}
193
194async fn liveness_handler(State(state): State<Arc<AppState>>) -> impl IntoResponse {
195    let metrics = state.metrics.get_performance_metrics();
196    (
197        StatusCode::OK,
198        Json(crate::admin::models::LivenessResponse {
199            alive: true,
200            uptime_seconds: metrics.uptime_seconds,
201        }),
202    )
