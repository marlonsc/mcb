4use std::net::SocketAddr;
5
6use crate::utils::config::load_startup_config_or_default;
7
8/// Configuration for the HTTP transport layer.
9#[derive(Debug, Clone)]
10pub struct HttpTransportConfig {
11    /// Bind address (e.g. `"127.0.0.1"`).
12    pub host: String,
13    /// Bind port (e.g. `8080`).
14    pub port: u16,
15    /// Whether CORS headers are attached to responses.
16    pub enable_cors: bool,
17}
18
19impl Default for HttpTransportConfig {
20    fn default() -> Self {
21        let config = load_startup_config_or_default();
22        Self {
23            host: config.server.network.host,
24            port: config.server.network.port,
25            enable_cors: config.server.cors.cors_enabled,
26        }
27    }
28}
29
30impl HttpTransportConfig {
31    /// Build a config bound to the configured host with a custom port.
32    #[must_use]
33    pub fn localhost(port: u16) -> Self {
34        let config = load_startup_config_or_default();
35        Self {
36            host: config.server.network.host,
37            port,
38            enable_cors: config.server.cors.cors_enabled,
39        }
40    }
41
42    /// Resolve the configured host and port into a [`SocketAddr`].
43    ///
44    /// # Errors
45    /// Returns an error if the host/port combination is not a valid socket address.
46    pub fn socket_addr(&self) -> Result<SocketAddr, std::net::AddrParseError> {
47        format!("{}:{}", self.host, self.port).parse()
48    }
49}
