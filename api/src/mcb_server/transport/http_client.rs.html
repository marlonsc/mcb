57/// HTTP client transport
58///
59/// Bridges stdio (for Claude Code) to HTTP (for MCB server).
60/// Each request is forwarded to the server over JSON-RPC.
61pub struct HttpClientTransport {
62    config: McpClientConfig,
63    client: reqwest::Client,
64}
65
66impl HttpClientTransport {
67    /// Create a new HTTP client transport with explicit session source values.
68    ///
69    /// Used by tests to validate session source precedence without mutating process env.
70    ///
71    /// # Errors
72    /// Returns an error when secure transport validation, session initialization, or client construction fails.
73    pub fn new_with_session_source(
74        server_url: String,
75        session_prefix: Option<String>,
76        timeout: Duration,
77        session_id_override: Option<String>,
78        session_file_override: Option<String>,
79    ) -> Result<Self, Box<dyn std::error::Error>> {
80        Self::require_secure_transport(&server_url)?;
81
82        let public_session_id = Self::generate_session_id(session_prefix.clone());
83        Self::initialize_session_state(session_prefix, session_id_override, session_file_override)?;
84
85        let workspace_root = std::env::current_dir()
86            .ok()
87            .and_then(|p| p.to_str().map(String::from));
88
89        let config = McpClientConfig {
90            server_url,
91            client_instance_id: domain_id::generate().to_string(),
92            public_session_id,
93            timeout,
94            workspace_root: workspace_root.clone(),
95            repo_path: workspace_root,
96        };
97
98        let client = reqwest::Client::builder()
99            .timeout(timeout)
100            .build()
101            .map_err(|e| format!("Failed to create HTTP client: {e}"))?;
102
103        Ok(Self { config, client })
104    }

106    fn initialize_session_state(
107        session_prefix: Option<String>,
108        session_id_override: Option<String>,
109        session_file_override: Option<String>,
110    ) -> Result<(), Box<dyn std::error::Error>> {
111        if session_id_override
112            .and_then(Self::normalize_env_value)
113            .is_some()
114        {
115            return Ok(());
116        }
117
118        if let Some(session_file) = session_file_override.and_then(Self::normalize_env_value) {
119            let path = Path::new(&session_file);
120
121            if path.exists() {
122                let existing = std::fs::read_to_string(path)?;
123                if Self::normalize_env_value(existing).is_some() {
124                    return Ok(());
125                }
126            }
127
128            let generated = Self::generate_session_id(session_prefix);
129            if let Some(parent) = path.parent()
130                && !parent.as_os_str().is_empty()
131            {
132                std::fs::create_dir_all(parent)?;
133            }
134            std::fs::write(path, format!("{generated}\n"))?;
135            return Ok(());
136        }
137
138        Ok(())
139    }
140
141    fn generate_session_id(session_prefix: Option<String>) -> String {
142        match session_prefix {
143            Some(prefix) => format!("{}_{}", prefix, domain_id::generate()),
144            None => domain_id::generate().to_string(),
145        }
146    }

148    /// Reject cleartext HTTP for non-loopback hosts.
149    ///
150    /// HTTPS is always accepted. Plain HTTP is only permitted when the host is
151    /// a loopback address (`127.0.0.1`, `localhost`, `[::1]`), since the
152    /// traffic never leaves the local machine. Any other combination is
153    /// rejected to prevent cleartext transmission of sensitive data.
154    ///
155    /// # Errors
156    ///
157    /// Returns `Err` when the URL uses plain HTTP with a non-loopback host.
158    pub fn require_secure_transport(url: &str) -> Result<(), String> {
159        let lower = url.to_ascii_lowercase();
160
161        if lower.starts_with("https://") {
162            return Ok(());
163        }
164
165        if let Some(after_scheme) = lower.strip_prefix("http://") {
166            let host = if after_scheme.starts_with('[') {
167                match after_scheme.find(']') {
168                    Some(end) => &after_scheme[..=end],
169                    None => after_scheme,
170                }
171            } else {
172                after_scheme.split([':', '/']).next().unwrap_or("")
173            };
174
175            return match host {
176                "127.0.0.1" | "localhost" | "[::1]" => Ok(()),
177                _ => Err(format!(
178                    "Cleartext HTTP is only allowed for loopback addresses \
179                     (127.0.0.1, localhost, [::1]). \
180                     Use HTTPS for remote host: {host}"
181                )),
182            };
183        }
184
185        Err(format!("Unsupported URL scheme in: {url}"))
186    }

188    fn normalize_env_value(value: impl AsRef<str>) -> Option<String> {
189        let normalized = value.as_ref().trim();
190        if normalized.is_empty() {
191            None
192        } else {
193            Some(normalized.to_owned())
194        }
195    }

197    /// Run the client transport
198    ///
199    /// Main loop that:
200    /// 1. Reads JSON-RPC requests from stdin
201    /// 2. Forwards them to the MCB server via HTTP
202    /// 3. Writes responses to stdout
203    ///
204    /// Runs until stdin is closed (EOF).
205    ///
206    /// # Errors
207    /// Returns an error when writing responses to stdout fails.
208    pub async fn run(&self) -> Result<(), Box<dyn std::error::Error>> {
209        info!(
210            "HttpClient",
211            "MCB client transport started",
212            &format!(
213                "server_url={} client_instance_id={}",
214                self.config.server_url,
215                mask_id(&self.config.client_instance_id)
216            )
217        );
218
219        let stdin = io::stdin();
220        let mut stdout = io::stdout();
221
222        for line in stdin.lock().lines() {
223            let line = match line {
224                Ok(l) => l,
225                Err(e) => {
226                    if e.kind() == io::ErrorKind::UnexpectedEof {
227                        info!("HttpClient", "stdin closed, shutting down");
228                        break;
229                    }
230                    error!("HttpClient", "Error reading from stdin", &e);
231                    continue;
232                }
233            };
234            self.process_line(&line, &mut stdout).await?;
235        }
236
237        info!("HttpClient", "MCB client transport finished");
238        Ok(())
239    }

241    /// Parse a single stdin line, forward it to the server, and write the response.
242    async fn process_line(
243        &self,
244        line: &str,
245        stdout: &mut io::Stdout,
246    ) -> Result<(), Box<dyn std::error::Error>> {
247        if line.trim().is_empty() {
248            return Ok(());
249        }
250
251        debug!("HttpClient", "Received request from stdin", &line.len());
252
253        let request: McpRequest = match serde_json::from_str(line) {
254            Ok(req) => req,
255            Err(e) => {
256                warn!(
257                    "HttpClient",
258                    "Failed to parse request",
259                    &format!("error={e} len={}", line.len())
260                );
261                let error_response = Self::create_parse_error(&e);
262                return Self::write_response(stdout, &error_response);
263            }
264        };
265
266        let response = self.forward_request(&request).await;
267        Self::write_response(stdout, &response)
268    }
269
270    /// Send a request to the MCB server
271    async fn send_request(&self, request: &McpRequest) -> Result<McpResponse, reqwest::Error> {
272        let url = format!("{}{MCP_ENDPOINT_PATH}", self.config.server_url);
273
274        debug!(
275            "HttpClient",
276            "Sending request to server",
277            &format!(
278                "url={url} method={} client_instance_id={}",
279                request.method,
280                mask_id(&self.config.client_instance_id)
281            )
282        );
283
284        let response = post_mcp_request(&self.client, &url, request, &self.config).await?;
285
286        let status = response.status();
287        debug!("HttpClient", "Received response from server", &status);
288
289        if !status.is_success() {
290            warn!("HttpClient", "Server returned non-success status", &status);
291        }
292
293        response.json::<McpResponse>().await
294    }

296    /// Get the server URL

308    /// Forward a request to the server, handling errors
309    async fn forward_request(&self, request: &McpRequest) -> McpResponse {
310        match self.send_request(request).await {
311            Ok(resp) => resp,
312            Err(e) => {
313                error!("HttpClient", "Failed to send request to server", &e);
314                Self::create_server_error(&e, request.id.clone())
315            }
316        }
317    }
318
319    /// Create a JSON-RPC parse error response
320    fn create_parse_error(e: &serde_json::Error) -> McpResponse {
321        McpResponse {
322            jsonrpc: JSONRPC_VERSION.to_owned(),
323            result: None,
324            error: Some(super::types::McpError {
325                code: JSONRPC_PARSE_ERROR,
326                message: format!("Parse error: {e}"),
327            }),
328            id: None,
329        }
330    }
331
332    /// Create a JSON-RPC server error response
333    fn create_server_error(e: &reqwest::Error, id: Option<serde_json::Value>) -> McpResponse {
334        McpResponse {
335            jsonrpc: JSONRPC_VERSION.to_owned(),
336            result: None,
337            error: Some(super::types::McpError {
338                code: JSONRPC_INTERNAL_ERROR,
339                message: format!("Server communication error: {e}"),
340            }),
341            id,
342        }
343    }
344
345    /// Write a response to stdout
346    fn write_response(
347        stdout: &mut io::Stdout,
348        response: &McpResponse,
349    ) -> Result<(), Box<dyn std::error::Error>> {
350        let response_json = serde_json::to_string(response)?;
351        debug!("HttpClient", "Sending response to stdout", &response_json);
352        writeln!(stdout, "{response_json}")?;
353        stdout.flush()?;
354        Ok(())
355    }

357
358async fn post_mcp_request(
359    client: &reqwest::Client,
360    url: &str,
361    request: &McpRequest,
362    config: &McpClientConfig,
363) -> Result<reqwest::Response, reqwest::Error> {
364    let mut builder = client
365        .post(url)
366        .header(HTTP_HEADER_CONTENT_TYPE, CONTENT_TYPE_JSON)
367        .header(HTTP_HEADER_EXECUTION_FLOW, EXECUTION_FLOW_HYBRID);
368
369    if let Some(ref ws) = config.workspace_root {
370        builder = builder.header(HEADER_WORKSPACE_ROOT, ws);
371    }
372    if let Some(ref rp) = config.repo_path {
373        builder = builder.header(HEADER_REPO_PATH, rp);
374    }
375    builder = builder.header(HEADER_SESSION_ID, &config.public_session_id);
376
377    if let Ok(user) = std::env::var("USER") {
378        builder = builder.header(HEADER_OPERATOR_ID, user);
379    }
380
381    let machine_id = hostname::get()
382        .ok()
383        .and_then(|h| h.into_string().ok())
384        .or_else(|| std::env::var("HOSTNAME").ok())
385        .unwrap_or_else(|| FALLBACK_UNKNOWN.to_owned());
386    builder = builder.header(HEADER_MACHINE_ID, machine_id);
387
388    builder = builder.header(HEADER_AGENT_PROGRAM, IDE_MCB_CLIENT);
389    builder = builder.header(HEADER_MODEL_ID, FALLBACK_UNKNOWN);
390    builder = builder.header(HEADER_DELEGATED, "false");
391
392    builder.json(request).send().await
393}
