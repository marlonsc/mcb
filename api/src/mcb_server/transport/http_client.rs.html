4//! HTTP Client Transport
5//!
6//! MCP client that connects to a remote MCB server via HTTP.
7//! Reads MCP requests from stdin, forwards them to the server,
8//! and writes responses to stdout.
9//!
10//! This enables MCB to run in "client mode" where it acts as a
11//! stdio-to-HTTP bridge for Claude Code integration.

51/// HTTP client transport
52///
53/// Bridges stdio (for Claude Code) to HTTP (for MCB server).
54/// Each request is forwarded to the server over JSON-RPC.
55pub struct HttpClientTransport {
56    config: McpClientConfig,
57    client: reqwest::Client,
58}
59
60impl HttpClientTransport {
61    /// Create a new HTTP client transport with explicit session source values.
62    ///
63    /// Used by tests to validate session source precedence without mutating process env.
64    ///
65    /// # Errors
66    /// Returns an error when secure transport validation, session initialization, or client construction fails.
67    pub fn new_with_session_source(
68        server_url: String,
69        session_prefix: Option<String>,
70        timeout: Duration,
71        session_id_override: Option<String>,
72        session_file_override: Option<String>,
73    ) -> Result<Self, Box<dyn std::error::Error>> {
74        Self::require_secure_transport(&server_url)?;
75
76        let public_session_id = Self::generate_session_id(session_prefix.clone());
77        Self::initialize_session_state(session_prefix, session_id_override, session_file_override)?;
78
79        let workspace_root = std::env::current_dir()
80            .ok()
81            .and_then(|p| p.to_str().map(String::from));
82
83        let config = McpClientConfig {
84            server_url,
85            client_instance_id: domain_id::generate().to_string(),
86            public_session_id,
87            timeout,
88            workspace_root: workspace_root.clone(),
89            repo_path: workspace_root,
90        };
91
92        let client = reqwest::Client::builder()
93            .timeout(timeout)
94            .build()
95            .map_err(|e| format!("Failed to create HTTP client: {e}"))?;
96
97        Ok(Self { config, client })
98    }

100    fn initialize_session_state(
101        session_prefix: Option<String>,
102        session_id_override: Option<String>,
103        session_file_override: Option<String>,
104    ) -> Result<(), Box<dyn std::error::Error>> {
105        if session_id_override
106            .and_then(Self::normalize_env_value)
107            .is_some()
108        {
109            return Ok(());
110        }
111
112        if let Some(session_file) = session_file_override.and_then(Self::normalize_env_value) {
113            let path = Path::new(&session_file);
114
115            if path.exists() {
116                let existing = std::fs::read_to_string(path)?;
117                if Self::normalize_env_value(existing).is_some() {
118                    return Ok(());
119                }
120            }
121
122            let generated = Self::generate_session_id(session_prefix);
123            if let Some(parent) = path.parent()
124                && !parent.as_os_str().is_empty()
125            {
126                std::fs::create_dir_all(parent)?;
127            }
128            std::fs::write(path, format!("{generated}\n"))?;
129            return Ok(());
130        }
131
132        Ok(())
133    }
134
135    fn generate_session_id(session_prefix: Option<String>) -> String {
136        match session_prefix {
137            Some(prefix) => format!("{}_{}", prefix, domain_id::generate()),
138            None => domain_id::generate().to_string(),
139        }
140    }

142    /// Reject cleartext HTTP for non-loopback hosts.
143    ///
144    /// HTTPS is always accepted. Plain HTTP is only permitted when the host is
145    /// a loopback address (`127.0.0.1`, `localhost`, `[::1]`), since the
146    /// traffic never leaves the local machine. Any other combination is
147    /// rejected to prevent cleartext transmission of sensitive data.
148    fn require_secure_transport(url: &str) -> Result<(), String> {
149        let lower = url.to_ascii_lowercase();
150
151        if lower.starts_with("https://") {
152            return Ok(());
153        }
154
155        if let Some(after_scheme) = lower.strip_prefix("http://") {
156            let host = if after_scheme.starts_with('[') {
157                match after_scheme.find(']') {
158                    Some(end) => &after_scheme[..=end],
159                    None => after_scheme,
160                }
161            } else {
162                after_scheme.split([':', '/']).next().unwrap_or("")
163            };
164
165            return match host {
166                "127.0.0.1" | "localhost" | "[::1]" => Ok(()),
167                _ => Err(format!(
168                    "Cleartext HTTP is only allowed for loopback addresses \
169                     (127.0.0.1, localhost, [::1]). \
170                     Use HTTPS for remote host: {host}"
171                )),
172            };
173        }
174
175        Err(format!("Unsupported URL scheme in: {url}"))
176    }
177
178    fn normalize_env_value(value: impl AsRef<str>) -> Option<String> {
179        let normalized = value.as_ref().trim();
180        if normalized.is_empty() {
181            None
182        } else {
183            Some(normalized.to_owned())
184        }
185    }

187    /// Run the client transport
188    ///
189    /// Main loop that:
190    /// 1. Reads JSON-RPC requests from stdin
191    /// 2. Forwards them to the MCB server via HTTP
192    /// 3. Writes responses to stdout
193    ///
194    /// Runs until stdin is closed (EOF).
195    ///
196    /// # Errors
197    /// Returns an error when writing responses to stdout fails.
198    pub async fn run(&self) -> Result<(), Box<dyn std::error::Error>> {
199        info!(
200            "HttpClient",
201            "MCB client transport started",
202            &format!(
203                "server_url={} client_instance_id={}",
204                self.config.server_url,
205                mask_id(&self.config.client_instance_id)
206            )
207        );
208
209        let stdin = io::stdin();
210        let mut stdout = io::stdout();
211
212        for line in stdin.lock().lines() {
213            let line = match line {
214                Ok(l) => l,
215                Err(e) => {
216                    if e.kind() == io::ErrorKind::UnexpectedEof {
217                        info!("HttpClient", "stdin closed, shutting down");
218                        break;
219                    }
220                    error!("HttpClient", "Error reading from stdin", &e);
221                    continue;
222                }
223            };
224
225            // Skip empty lines
226            if line.trim().is_empty() {
227                continue;
228            }
229
230            debug!("HttpClient", "Received request from stdin", &line.len());
231
232            // Parse the request
233            let request: McpRequest = match serde_json::from_str(&line) {
234                Ok(req) => req,
235                Err(e) => {
236                    warn!(
237                        "HttpClient",
238                        "Failed to parse request",
239                        &format!("error={e} len={}", line.len())
240                    );
241                    let error_response = Self::create_parse_error(&e);
242                    Self::write_response(&mut stdout, &error_response)?;
243                    continue;
244                }
245            };
246
247            // Forward to server and handle response
248            let response = self.forward_request(&request).await;
249            Self::write_response(&mut stdout, &response)?;
250        }
251
252        info!("HttpClient", "MCB client transport finished");
253        Ok(())
254    }

256    /// Send a request to the MCB server
257    async fn send_request(&self, request: &McpRequest) -> Result<McpResponse, reqwest::Error> {
258        let url = format!("{}{MCP_ENDPOINT_PATH}", self.config.server_url);
259
260        debug!(
261            "HttpClient",
262            "Sending request to server",
263            &format!(
264                "url={url} method={} client_instance_id={}",
265                request.method,
266                mask_id(&self.config.client_instance_id)
267            )
268        );
269
270        let response = post_mcp_request(&self.client, &url, request, &self.config).await?;
271
272        let status = response.status();
273        debug!("HttpClient", "Received response from server", &status);
274
275        if !status.is_success() {
276            warn!("HttpClient", "Server returned non-success status", &status);
277        }
278
279        response.json::<McpResponse>().await
280    }

294    /// Forward a request to the server, handling errors
295    async fn forward_request(&self, request: &McpRequest) -> McpResponse {
296        match self.send_request(request).await {
297            Ok(resp) => resp,
298            Err(e) => {
299                error!("HttpClient", "Failed to send request to server", &e);
300                Self::create_server_error(&e, request.id.clone())
301            }
302        }
303    }
304
305    /// Create a JSON-RPC parse error response
306    fn create_parse_error(e: &serde_json::Error) -> McpResponse {
307        McpResponse {
308            jsonrpc: JSONRPC_VERSION.to_owned(),
309            result: None,
310            error: Some(super::types::McpError {
311                code: JSONRPC_PARSE_ERROR,
312                message: format!("Parse error: {e}"),
313            }),
314            id: None,
315        }
316    }
317
318    /// Create a JSON-RPC server error response
319    fn create_server_error(e: &reqwest::Error, id: Option<serde_json::Value>) -> McpResponse {
320        McpResponse {
321            jsonrpc: JSONRPC_VERSION.to_owned(),
322            result: None,
323            error: Some(super::types::McpError {
324                code: JSONRPC_INTERNAL_ERROR,
325                message: format!("Server communication error: {e}"),
326            }),
327            id,
328        }
329    }
330
331    /// Write a response to stdout
332    fn write_response(
333        stdout: &mut io::Stdout,
334        response: &McpResponse,
335    ) -> Result<(), Box<dyn std::error::Error>> {
336        let response_json = serde_json::to_string(response)?;
337        debug!("HttpClient", "Sending response to stdout", &response_json);
338        writeln!(stdout, "{response_json}")?;
339        stdout.flush()?;
340        Ok(())
341    }

343
344async fn post_mcp_request(
345    client: &reqwest::Client,
346    url: &str,
347    request: &McpRequest,
348    config: &McpClientConfig,
349) -> Result<reqwest::Response, reqwest::Error> {
350    let mut builder = client
351        .post(url)
352        .header("Content-Type", CONTENT_TYPE_JSON)
353        .header(HTTP_HEADER_EXECUTION_FLOW, EXECUTION_FLOW_HYBRID);
354
355    if let Some(ref ws) = config.workspace_root {
356        builder = builder.header("X-Workspace-Root", ws);
357    }
358    if let Some(ref rp) = config.repo_path {
359        builder = builder.header("X-Repo-Path", rp);
360    }
361    builder = builder.header("X-Session-Id", &config.public_session_id);
362
363    if let Ok(user) = std::env::var("USER") {
364        builder = builder.header("X-Operator-Id", user);
365    }
366
367    let machine_id = hostname::get()
368        .ok()
369        .and_then(|h| h.into_string().ok())
370        .or_else(|| std::env::var("HOSTNAME").ok())
371        .unwrap_or_else(|| "unknown".to_owned());
372    builder = builder.header("X-Machine-Id", machine_id);
373
374    builder = builder.header("X-Agent-Program", "mcb-client");
375    builder = builder.header("X-Model-Id", "unknown");
376    builder = builder.header("X-Delegated", "false");
377
378    builder.json(request).send().await
379}

388    fn session_id_override_takes_precedence_over_file() {
389        let temp_dir = match tempfile::tempdir() {
390            Ok(dir) => dir,
391            Err(_) => return,
392        };
393        let session_file = temp_dir.path().join("session.id");
394
395        let session_file_value = match session_file.to_str() {
396            Some(value) => value.to_owned(),
397            None => return,
398        };
399
400        let client = match HttpClientTransport::new_with_session_source(
401            "http://127.0.0.1:18080".to_owned(),
402            Some("prefix".to_owned()),
403            Duration::from_secs(10),
404            Some("explicit-session-id".to_owned()),
405            Some(session_file_value),
406        ) {
407            Ok(client) => client,
408            Err(_) => return,
409        };
