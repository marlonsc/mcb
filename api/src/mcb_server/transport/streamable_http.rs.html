1//! HTTP header extraction utilities for workspace provenance.
2//!
3//! **Documentation**: [docs/modules/server.md](../../../docs/modules/server.md)
4//!
5//! Provides utilities for extracting and mapping custom HTTP headers to
6//! execution context overrides for workspace provenance enforcement.
7
8use std::collections::HashMap;
9
10use axum::http::HeaderMap;
11use mcb_utils::constants::headers::PROVENANCE_HEADER_MAPPINGS;
12use mcb_utils::constants::protocol::HTTP_HEADER_EXECUTION_FLOW;
13
14/// Extract a single header value, trimming whitespace.
15pub fn extract_override(headers: &HeaderMap, header_name: &str) -> Option<String> {
16    headers
17        .get(header_name)
18        .and_then(|value| value.to_str().ok())
19        .map(str::trim)
20        .filter(|value| !value.is_empty())
21        .map(ToOwned::to_owned)
22}
23
24/// Build a `HashMap` of header overrides from HTTP headers.
25///
26/// Maps custom headers to their corresponding context keys using the
27/// centralized `PROVENANCE_HEADER_MAPPINGS` table plus execution flow.
28#[must_use]
29pub fn build_overrides(headers: &HeaderMap) -> HashMap<String, String> {
30    let mut overrides = HashMap::new();
31
32    for &(header_name, key) in PROVENANCE_HEADER_MAPPINGS {
33        if let Some(value) = extract_override(headers, header_name) {
34            overrides.insert(key.to_owned(), value);
35        }
36    }
37
38    // Execution flow is transport-level, not provenance, but still overridden via header.
39    if let Some(value) = extract_override(headers, HTTP_HEADER_EXECUTION_FLOW) {
40        overrides.insert("execution_flow".to_owned(), value);
41    }
42
43    overrides
44}
