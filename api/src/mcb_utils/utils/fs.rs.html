6use crate::error::UtilsError;
7use std::path::{Path, PathBuf};

9fn read_entries(dir: &Path) -> Result<std::fs::ReadDir, UtilsError> {
10    std::fs::read_dir(dir).map_err(|e| {
11        UtilsError::Filesystem(format!("Failed to read directory {}: {}", dir.display(), e))
12    })
13}
14
15fn read_file_type(entry: &std::fs::DirEntry, path: &Path) -> Result<std::fs::FileType, UtilsError> {
16    entry.file_type().map_err(|e| {
17        UtilsError::Filesystem(format!(
18            "Failed to get file type for {}: {}",
19            path.display(),
20            e
21        ))
22    })
23}
24
25fn has_matching_extension(path: &Path, extensions: &[&str]) -> bool {
26    path.extension()
27        .and_then(|e| e.to_str())
28        .is_some_and(|ext| extensions.iter().any(|&e| e.eq_ignore_ascii_case(ext)))
29}

31/// Collect all files recursively from a directory that match specific extensions.
32///
33/// # Errors
34///
35/// Returns an error if directory traversal fails.
36pub fn find_files_by_extensions(
37    root: &Path,
38    extensions: &[&str],
39) -> Result<Vec<PathBuf>, UtilsError> {
40    if !root.exists() {
41        return Ok(Vec::new());
42    }
43
44    let mut files = Vec::new();
45    let mut stack = vec![root.to_path_buf()];
46
47    while let Some(dir) = stack.pop() {
48        for entry in read_entries(&dir)? {
49            classify_entry(&dir, entry, extensions, &mut stack, &mut files)?;
50        }
51    }
52
53    Ok(files)
54}
55
56/// Route one directory entry: push subdirectories onto `stack`, collect matching files.
57fn classify_entry(
58    dir: &Path,
59    entry: std::io::Result<std::fs::DirEntry>,
60    extensions: &[&str],
61    stack: &mut Vec<PathBuf>,
62    files: &mut Vec<PathBuf>,
63) -> Result<(), UtilsError> {
64    let entry = entry.map_err(|e| {
65        UtilsError::Filesystem(format!(
66            "Failed to read directory entry in {}: {}",
67            dir.display(),
68            e
69        ))
70    })?;
71    let path = entry.path();
72    let file_type = read_file_type(&entry, &path)?;
73
74    if file_type.is_dir() {
75        stack.push(path);
76    } else if file_type.is_file() && has_matching_extension(&path, extensions) {
77        files.push(path);
78    }
79
80    Ok(())
81}
