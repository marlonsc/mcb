6/// Errors produced by utility functions in mcb-utils.
7///
8/// Each variant maps to a category of failure that utility code can encounter.
9/// Domain crates implement `From<UtilsError>` to convert into their own error types.
10#[derive(Debug, thiserror::Error)]
11pub enum UtilsError {
12    /// Standard I/O errors (file open, read, write).
13    #[error("IO error: {0}")]
14    Io(#[from] std::io::Error),
15
16    /// Invalid or non-UTF-8 path, strip-prefix failure, canonicalization failure.
17    #[error("Invalid path: {0}")]
18    InvalidPath(String),
19
20    /// Generic parse failure (regex compilation, format issues).
21    #[error("Parse error: {0}")]
22    Parse(String),
23
24    /// System clock or time conversion errors.
25    #[error("Time error: {0}")]
26    Time(String),
27
28    /// Hash computation or digest errors.
29    #[error("Hash error: {0}")]
30    Hash(String),
31
32    /// Base64, hex, or other encoding/decoding errors.
33    #[error("Encoding error: {0}")]
34    Encoding(String),
35
36    /// Directory traversal, file-type detection, or recursive scan errors.
37    #[error("Filesystem error: {0}")]
38    Filesystem(String),
39
40    /// Regex compilation or matching errors.
41    #[error("Regex error: {0}")]
42    Regex(String),
43
44    /// Catch-all for errors that don't fit other categories.
45    #[error("{0}")]
46    Other(String),
47}
48
49/// Convenience alias used throughout mcb-utils.
50pub type Result<T> = std::result::Result<T, UtilsError>;
