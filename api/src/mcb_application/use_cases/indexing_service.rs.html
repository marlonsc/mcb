4//! Indexing Service Use Case
5//!
6//! # Overview
7//! The `IndexingService` manages the ingestion and processing of code assets into the semantic
8//! context system. It handles the full lifecycle from file discovery to vector storage, ensuring
9//! that the system's understanding of the codebase remains up-to-date.
10//!
11//! # Responsibilities
12//! - **File Discovery**: Recursively scanning workspace directories while respecting ignore patterns.
13//! - **Language-Aware Chunking**: Splitting code files into semantic chunks using AST-based strategies.
14//! - **Incremental Indexing**: Optimizing ingestion by only processing changed files (via hash tracking).
15//! - **Async Processing**: Executing long-running indexing tasks in the background to maintain responsiveness.
16//! - **Event Publishing**: Notifying the system of indexing progress and completion.
17//!
18//! # Architecture
19//! Implements `IndexingServiceInterface` and acts as a coordinator between:
20//! - `LanguageChunkingProvider`: For parsing and splitting code.
21//! - `ContextService`: For embedding and storing chunks.
22//! - `FileHashRepository`: For change detection.
23//! - `EventBusProvider`: For system-wide notifications.

42/// Accumulator for indexing progress and operational metrics.
43///
44/// Tracks the state of an active indexing operation, including success counts,
45/// skipped files, and encountered errors for final reporting.
46struct IndexingProgress {
47    files_processed: usize,
48    chunks_created: usize,
49    files_skipped: usize,
50    errors: Vec<String>,
51}
52
53impl IndexingProgress {
54    fn new() -> Self {
55        Self {
56            files_processed: 0,
57            chunks_created: 0,
58            files_skipped: 0,
59            errors: Vec::new(),
60        }
61    }
62
63    fn with_counts(
64        files_processed: usize,
65        chunks_created: usize,
66        files_skipped: usize,
67        errors: Vec<String>,
68    ) -> Self {
69        Self {
70            files_processed,
71            chunks_created,
72            files_skipped,
73            errors,
74        }
75    }
76
77    fn record_error(&mut self, context: &str, path: &Path, error: impl std::fmt::Display) {
78        self.errors
79            .push(format!("{} {}: {}", context, path.display(), error));
80    }
81
82    /// Build final `IndexingResult` (used by sync path and tests).
83    fn into_result(self, operation_id: Option<OperationId>, status: &str) -> IndexingResult {
84        IndexingResult {
85            files_processed: self.files_processed,
86            chunks_created: self.chunks_created,
87            files_skipped: self.files_skipped,
88            errors: self.errors,
89            operation_id,
90            status: status.to_owned(),
91        }
92    }
93}

173    fn normalize_supported_extensions(extensions: Vec<String>) -> Vec<String> {
174        extensions
175            .into_iter()
176            .map(|ext| ext.trim().trim_start_matches('.').to_ascii_lowercase())
177            .filter(|ext| !ext.is_empty())
178            .collect()
179    }
180
181    /// Discover files recursively from a path
182    async fn discover_files(
183        &self,
184        path: &Path,
185        progress: &mut IndexingProgress,
186    ) -> Vec<std::path::PathBuf> {
187        let mut files = Vec::new();
188        let walker = WalkBuilder::new(path)
189            .hidden(false)
190            .filter_entry(|entry| {
191                if !entry.file_type().is_some_and(|ft| ft.is_dir()) {
192                    return true;
193                }
194
195                entry
196                    .file_name()
197                    .to_str()
198                    .is_none_or(|name| !SKIP_DIRS.contains(&name))
199            })
200            .build();
201
202        for entry_result in walker {
203            match entry_result {
204                Ok(entry) => {
205                    if entry.file_type().is_some_and(|ft| ft.is_file())
206                        && self.is_supported_file(entry.path())
207                    {
208                        files.push(entry.path().to_path_buf());
209                    }
210                }
211                Err(e) => {
212                    progress.record_error("Failed to read directory entry", path, e);
213                }
214            }
215        }
216
217        files
218    }
219
220    /// Check if file has a supported extension
221    fn is_supported_file(&self, path: &Path) -> bool {
222        path.extension()
223            .and_then(|ext| ext.to_str())
224            .is_some_and(|ext| {
225                self.supported_extensions
226                    .iter()
227                    .any(|supported| supported == &ext.to_ascii_lowercase())
228            })
229    }
230}

232/// Result of processing a single file during indexing.
233#[derive(Debug, Clone)]
234enum ProcessResult {
235    /// File was successfully indexed with the given number of chunks.
236    Processed { chunks: usize },
237    /// File was skipped because it hasn't changed.
238    Skipped,
239}

241#[async_trait::async_trait]
242impl IndexingServiceInterface for IndexingServiceImpl {
243    /// # Errors
244    ///
245    /// Returns an error if collection initialization fails.
246    async fn index_codebase(
247        &self,
248        path: &Path,
249        collection: &CollectionId,
250    ) -> Result<IndexingResult> {
251        // Initialize collection
252        self.context_service.initialize(collection).await?;
253
254        // Discover files first (quick operation)
255        let mut progress = IndexingProgress::new();
256        let files = self.discover_files(path, &mut progress).await;
257        let total_files = files.len();
258
259        info!(
260            "Starting indexing: {} files in {}",
261            total_files,
262            path.display()
263        );
264
265        // Start tracking operation
266        let operation_id = self.indexing_ops.start_operation(collection, total_files);
267
268        // Publish IndexingStarted event
269        if let Err(e) = self
270            .event_bus
271            .publish_event(DomainEvent::IndexingStarted {
272                collection: collection.to_string(),
273                total_files,
274            })
275            .await
276        {
277            warn!("Failed to publish IndexingStarted event: {}", e);
278        }
279
280        // Clone service for the background task
281        // IndexingServiceImpl is cheap to clone (Arc-based)
282        let service = self.clone();
283        let collection_id = *collection;
284        let op_id = operation_id;
285        let workspace_root = path.to_path_buf();
286
287        // Fire-and-forget: caller gets operation_id immediately, polling for completion.
288        // Sync execution path available via run_indexing_task() directly in tests.
289        let _handle = tokio::spawn(async move {
290            Self::run_indexing_task(service, files, workspace_root, collection_id, op_id).await;
291        });
292
293        // Return immediately with operation_id
294        Ok(IndexingResult {
295            files_processed: 0,
296            chunks_created: 0,
297            files_skipped: 0,
298            errors: vec![],
299            operation_id: Some(operation_id),
300            status: INDEXING_STATUS_STARTED.to_owned(),
301        })
302    }
303
304    fn get_status(&self) -> IndexingStatus {
305        let ops = self.indexing_ops.get_operations();
306        // Get first active operation if any - use if-let to avoid expect()
307        if let Some((_, op)) = ops.iter().next() {
308            let total = op.total_files.max(1);
309            IndexingStatus {
310                is_indexing: true,
311                progress: op.processed_files as f64 / total as f64,
312                current_file: op.current_file.clone(),
313                total_files: op.total_files,
314                processed_files: op.processed_files,
315            }
316        } else {
317            IndexingStatus::default()
318        }
319    }
320
321    /// # Errors
322    ///
323    /// Returns an error if the context service fails to clear the collection.
324    async fn clear_collection(&self, collection: &CollectionId) -> Result<()> {
325        self.context_service.clear_collection(collection).await
326    }
327}

329impl IndexingServiceImpl {
330    /// Background task that performs the actual indexing work.
331    async fn run_indexing_task(
332        service: IndexingServiceImpl,
333        files: Vec<PathBuf>,
334        workspace_root: PathBuf,
335        collection: CollectionId,
336        operation_id: OperationId,
337    ) {
338        let start = Instant::now();
339        let total = files.len();
340        let mut chunks_created = 0;
341        let mut files_processed = 0;
342        let mut failed_files: Vec<String> = Vec::new();
343
344        for (i, file_path) in files.iter().enumerate() {
345            match service
346                .process_file(file_path, &workspace_root, &collection, &operation_id, i)
347                .await
348            {
349                Ok(ProcessResult::Processed { chunks }) => {
350                    files_processed += 1;
351                    chunks_created += chunks;
352                }
353                Ok(ProcessResult::Skipped) => {
354                    // File hasn't changed, increment skip count but don't record as processed
355                }
356                Err(e) => {
357                    warn!(file = %file_path.display(), error = %e, "Failed to process file during indexing");
358                    failed_files.push(file_path.display().to_string());
359                }
360            }
361        }
362
363        service
364            .indexing_ops
365            .update_progress(&operation_id, None, total);
366
367        service.indexing_ops.complete_operation(&operation_id);
368
369        let duration_ms = start.elapsed().as_millis() as u64;
370        let files_skipped = total.saturating_sub(files_processed);
371
372        let result = IndexingProgress::with_counts(
373            files_processed,
374            chunks_created,
375            files_skipped,
376            failed_files.clone(),
377        )
378        .into_result(Some(operation_id), INDEXING_STATUS_COMPLETED);
379
380        if let Err(e) = service
381            .event_bus
382            .publish_event(DomainEvent::IndexingCompleted {
383                collection: collection.to_string(),
384                chunks: result.chunks_created,
385                duration_ms,
386            })
387            .await
388        {
389            warn!("Failed to publish IndexingCompleted event: {}", e);
390        }
391
392        let error_count = failed_files.len();
393        if error_count > 0 {
394            error!(
395                files_processed = files_processed,
396                chunks_created = chunks_created,
397                errors = error_count,
398                duration_ms = duration_ms,
399                "Indexing completed with errors"
400            );
401        } else {
402            info!(
403                files_processed = files_processed,
404                chunks_created = chunks_created,
405                duration_ms = duration_ms,
406                "Indexing completed successfully"
407            );
408        }
409    }
410
411    /// Process a single file: check for changes, chunk it, and store results.
412    async fn process_file(
413        &self,
414        file_path: &Path,
415        workspace_root: &Path,
416        collection: &CollectionId,
417        operation_id: &OperationId,
418        index: usize,
419    ) -> Result<ProcessResult> {
420        let relative_path = Self::workspace_relative_path(file_path, workspace_root)?;
421
422        // Update progress tracking
423        self.indexing_ops
424            .update_progress(operation_id, Some(relative_path.clone()), index);
425
426        // Read file content
427        let content = std::fs::read_to_string(file_path)
428            .map_err(|e| mcb_domain::error::Error::internal(format!("Failed to read file: {e}")))?;
429
430        // Incremental check using file hashes
431        let current_hash = mcb_domain::utils::compute_content_hash(&content);
432        #[allow(clippy::collapsible_if)]
433        if let Some(repo) = &self.file_hash_repository {
434            if !repo
435                .has_changed(&collection.to_string(), &relative_path, &current_hash)
436                .await?
437            {
438                return Ok(ProcessResult::Skipped);
439            }
440        }
441
442        // Generate semantic chunks
443        let chunks = self.language_chunker.chunk(&content, &relative_path);
444        let chunk_count = chunks.len();
445
446        // Store chunks in context storage
447        if !chunks.is_empty() {
448            self.context_service
449                .store_chunks(collection, &chunks)
450                .await?;
451        }
452
453        // Update hash repository to reflect success
454        if let Some(repo) = &self.file_hash_repository {
455            repo.upsert_hash(&collection.to_string(), &relative_path, &current_hash)
456                .await?;
457        }
458
459        Ok(ProcessResult::Processed {
460            chunks: chunk_count,
461        })
462    }
463}

465#[cfg(test)]
466mod tests {

470    #[test]
471    fn workspace_relative_path_normalizes_within_workspace() {
472        let workspace = Path::new("/repo");
473        let file = Path::new("/repo/src/main.rs");
474
475        let relative =
476            IndexingServiceImpl::workspace_relative_path(file, workspace).expect("relative path");
477
478        assert_eq!(relative, "src/main.rs");
479    }
480
481    #[test]
482    fn workspace_relative_path_rejects_outside_workspace() {
483        let workspace = Path::new("/repo");
484        let file = Path::new("/other/main.rs");
485
486        let err = IndexingServiceImpl::workspace_relative_path(file, workspace)
487            .expect_err("outside path must fail");
488
489        assert!(err.to_string().contains("is not under root"));
490    }
