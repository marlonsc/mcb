3//!
4//! Dependency graph construction and layer-boundary analysis.
5//!
6//! Builds a `petgraph` directed graph from extracted [`Fact`]s, enabling
7//! cycle detection and inter-layer dependency checks.
8
9use crate::extractor::{Fact, FactType};
10use petgraph::Direction;
11use petgraph::algo::tarjan_scc;
12use petgraph::graph::{DiGraph, NodeIndex};
13use std::collections::{HashMap, HashSet};
14
15/// Type of edge in the dependency graph.
16#[derive(Debug, Clone, PartialEq, Eq)]
17pub enum EdgeType {
18    /// The source fact depends on the target fact (e.g. via an import).
19    DependsOn,
20    /// The source fact contains the target fact (e.g. a module containing a struct).
21    Contains,
22}
23
24/// A directed dependency graph built from code facts.
25#[derive(Debug, Default)]
26pub struct DependencyGraph {
27    graph: DiGraph<Fact, EdgeType>,
28    node_map: HashMap<String, NodeIndex>,
29}
30
31impl DependencyGraph {
32    /// Create an empty dependency graph.
33    #[must_use]
34    pub fn new() -> Self {
35        Self {
36            graph: DiGraph::new(),
37            node_map: HashMap::new(),
38        }
39    }
40
41    /// Populate the graph from extracted facts.
42    ///
43    /// Performs two passes:
44    /// 1. Adds all facts as nodes.
45    /// 2. Adds `Contains` edges (parent → child) and `DependsOn` edges
46    ///    (import → target module) using heuristic name matching.
47    pub fn build(&mut self, facts: &[Fact]) {
48        self.add_nodes(facts);
49        let module_indices = self.module_indices();
50        for fact in facts {
51            self.add_edges_for_fact(fact, &module_indices);
52        }
53    }
54
55    /// Detect circular dependencies using Tarjan's algorithm.
56    ///
57    /// Returns a list of strongly-connected components with more than one
58    /// member, plus any self-loops.
59    #[must_use]
60    pub fn check_cycles(&self) -> Vec<Vec<String>> {
61        let sccs = tarjan_scc(&self.graph);
62        let mut cycles = Vec::new();
63        for scc in sccs {
64            if scc.len() > 1 {
65                cycles.push(scc.iter().map(|idx| self.graph[*idx].id.clone()).collect());
66            } else if scc.len() == 1 {
67                // Check for self-loop
68                if let Some(idx) = scc.first().copied()
69                    && self.graph.contains_edge(idx, idx)
70                {
71                    cycles.push(vec![self.graph[idx].id.clone()]);
72                }
73            }
74        }
75        cycles
76    }
77
78    /// Check for forbidden dependencies between architectural layers.
79    ///
80    /// An edge from a node whose name contains `source_pattern` to a node
81    /// whose name contains `target_pattern` via the `Contains → DependsOn`
82    /// two-hop path is reported as a violation.
83    #[must_use]
84    pub fn check_layer_violation(
85        &self,
86        source_pattern: &str,
87        target_pattern: &str,
88    ) -> Vec<(String, String)> {
89        let source_nodes: Vec<NodeIndex> = self
90            .graph
91            .node_indices()
92            .filter(|idx| self.graph[*idx].name.contains(source_pattern))
93            .collect();
94        let target_nodes: HashSet<NodeIndex> = self
95            .graph
96            .node_indices()
97            .filter(|idx| self.graph[*idx].name.contains(target_pattern))
98            .collect();
99
100        source_nodes
101            .iter()
102            .flat_map(|source| self.violations_for_source(*source, &target_nodes))
103            .collect()
104    }
105
106    fn add_nodes(&mut self, facts: &[Fact]) {
107        for fact in facts {
108            if self.node_map.contains_key(&fact.id) {
109                continue;
110            }
111
112            let idx = self.graph.add_node(fact.clone());
113            self.node_map.insert(fact.id.clone(), idx);
114        }
115    }
116
117    fn module_indices(&self) -> Vec<NodeIndex> {
118        self.node_map
119            .values()
120            .copied()
121            .filter(|idx| self.graph[*idx].fact_type == FactType::Module)
122            .collect()
123    }
124
125    fn add_edges_for_fact(&mut self, fact: &Fact, module_indices: &[NodeIndex]) {
126        let Some(source_idx) = self.node_map.get(&fact.id).copied() else {
127            return;
128        };
129
130        self.add_contains_edge(source_idx, fact);
131        self.add_import_dep_edges(source_idx, fact, module_indices);
132    }
133
134    fn add_contains_edge(&mut self, source_idx: NodeIndex, fact: &Fact) {
135        let Some(parent_id) = fact.parent_id.as_deref() else {
136            return;
137        };
138        let Some(parent_idx) = self.node_map.get(parent_id).copied() else {
139            return;
140        };
141
142        self.graph
143            .add_edge(parent_idx, source_idx, EdgeType::Contains);
144    }
145
146    fn add_import_dep_edges(
147        &mut self,
148        source_idx: NodeIndex,
149        fact: &Fact,
150        module_indices: &[NodeIndex],
151    ) {
152        if fact.fact_type != FactType::Import {
153            return;
154        }
155
156        self.add_exact_import_dependency(source_idx, &fact.name);
157        self.add_prefix_import_dependencies(source_idx, &fact.name, module_indices);
158    }
159
160    fn add_exact_import_dependency(&mut self, source_idx: NodeIndex, import_path: &str) {
161        let Some(target_idx) = self.node_map.get(import_path).copied() else {
162            return;
163        };
164
165        self.graph
166            .add_edge(source_idx, target_idx, EdgeType::DependsOn);
167    }
168
169    fn add_prefix_import_dependencies(
170        &mut self,
171        source_idx: NodeIndex,
172        import_path: &str,
173        module_indices: &[NodeIndex],
174    ) {
175        let normalized_import = import_path.replace('-', "_");
176        for idx in module_indices {
177            let normalized_module = self.graph[*idx].name.replace('-', "_");
178            if normalized_import.starts_with(&normalized_module) {
179                self.graph.add_edge(source_idx, *idx, EdgeType::DependsOn);
180            }
181        }
182    }
183
184    fn violations_for_source(
185        &self,
186        source_idx: NodeIndex,
187        target_nodes: &HashSet<NodeIndex>,
188    ) -> Vec<(String, String)> {
189        self.contains_neighbors(source_idx)
190            .flat_map(|child_idx| self.depends_on_neighbors(child_idx))
191            .filter(|dep_idx| target_nodes.contains(dep_idx))
192            .map(|dep_idx| {
193                (
194                    self.graph[source_idx].name.clone(),
195                    self.graph[dep_idx].name.clone(),
196                )
197            })
198            .collect()
199    }
200
201    fn contains_neighbors(&self, source_idx: NodeIndex) -> impl Iterator<Item = NodeIndex> + '_ {
202        self.graph
203            .neighbors_directed(source_idx, Direction::Outgoing)
204            .filter(move |neighbor_idx| {
205                self.graph
206                    .find_edge(source_idx, *neighbor_idx)
207                    .is_some_and(|edge| self.graph[edge] == EdgeType::Contains)
208            })
209    }
210
211    fn depends_on_neighbors(&self, source_idx: NodeIndex) -> impl Iterator<Item = NodeIndex> + '_ {
212        self.graph
213            .neighbors_directed(source_idx, Direction::Outgoing)
214            .filter(move |neighbor_idx| {
215                self.graph
216                    .find_edge(source_idx, *neighbor_idx)
217                    .is_some_and(|edge| self.graph[edge] == EdgeType::DependsOn)
218            })
219    }
