4//! Rust-specific AST fact extractor.
5//!
6//! Uses `rust-code-analysis` (which wraps `tree-sitter`) to parse Rust source
7//! files and extract [`Fact`]s for modules, imports, structs, and functions.
8
9use super::fact::{Fact, FactType, Location};
10use crate::Result;
11use rust_code_analysis::{Node, ParserTrait, RustParser};
12use std::fs;
13use std::path::Path;
14
15/// Extracts code facts from Rust source files via AST analysis.
16pub struct RustExtractor;
17
18impl RustExtractor {
19    fn location_for(path: &Path, node: &Node<'_>) -> Location {
20        Location {
21            file_path: path.to_path_buf(),
22            start_line: node.start_row() + 1,
23            end_line: node.end_row() + 1,
24            start_column: node.start_position().1 + 1,
25            end_column: node.end_position().1 + 1,
26        }
27    }
28
29    /// Parse a Rust file at `path` and return all extracted [`Fact`]s.
30    ///
31    /// Currently extracts:
32    /// - One `Module` fact per file (derived from the file stem).
33    /// - `Import` facts for every `use` declaration.
34    /// - `Struct` facts for every `struct` item.
35    /// - `Function` facts for every `fn` item.
36    /// # Errors
37    ///
38    /// Returns an error if the file cannot be read or parsed.
39    pub fn extract_facts(&self, path: &Path) -> Result<Vec<Fact>> {
40        let code = fs::read(path)?;
41        // Use RustParser which is a public type alias for Parser<RustCode>
42        let parser = RustParser::new(code, path, None);
43        let root = parser.get_root();
44        let code_ref = parser.get_code();
45
46        let mut facts = Vec::new();
47
48        // Detect module name from path
49        let module_name = path
50            .file_stem()
51            .and_then(|s| s.to_str())
52            .unwrap_or(mcb_utils::constants::FALLBACK_UNKNOWN)
53            .to_owned();
54
55        let module_id = format!("module::{module_name}");
56
57        // Add Module Fact
58        facts.push(Fact::new(
59            module_name.clone(),
60            FactType::Module,
61            Self::location_for(path, &root),
62            None,
63        ));
64
65        let ctx = FactCtx {
66            path,
67            code_ref,
68            module_id: &module_id,
69        };
70        Self::collect_imports(&ctx, &root, &mut facts);
71        Self::collect_named(&ctx, &root, "struct_item", &FactType::Struct, &mut facts);
72        Self::collect_named(
73            &ctx,
74            &root,
75            "function_item",
76            &FactType::Function,
77            &mut facts,
78        );
79
80        Ok(facts)
81    }
82
83    /// Collect every AST node of `kind` reachable from `root`.
84    fn find_by_kind<'a>(root: &Node<'a>, kind: &str) -> Vec<Node<'a>> {
85        let mut results = Vec::new();
86        let mut stack = vec![root.0];
87
88        while let Some(n) = stack.pop() {
89            if n.kind() == kind {
90                results.push(Node(n));
91            }
92            let mut cursor = n.walk();
93            for child in n.children(&mut cursor) {
94                stack.push(child);
95            }
96        }
97        results
98    }
99
100    /// Append an [`FactType::Import`] fact for each `use` declaration.
101    fn collect_imports(ctx: &FactCtx, root: &Node, facts: &mut Vec<Fact>) {
102        for import in Self::find_by_kind(root, "use_declaration") {
103            if let Some(text) = import.utf8_text(ctx.code_ref) {
104                let import_path = text.trim_start_matches("use ").trim_end_matches(';').trim();
105                facts.push(Fact::new(
106                    import_path.to_owned(),
107                    FactType::Import,
108                    Self::location_for(ctx.path, &import),
109                    Some(ctx.module_id.to_owned()),
110                ));
111            }
112        }
113    }
114
115    /// Append a named fact of `fact_type` for each named node of `kind`.
116    fn collect_named(
117        ctx: &FactCtx,
118        root: &Node,
119        kind: &str,
120        fact_type: &FactType,
121        facts: &mut Vec<Fact>,
122    ) {
123        for node in Self::find_by_kind(root, kind) {
124            if let Some(name_node) = node.0.child_by_field_name("name")
125                && let Ok(name) = name_node.utf8_text(ctx.code_ref)
126            {
127                facts.push(Fact::new(
128                    name.to_owned(),
129                    fact_type.clone(),
130                    Self::location_for(ctx.path, &node),
131                    Some(ctx.module_id.to_owned()),
132                ));
133            }
134        }
135    }
136}
137
138/// Shared context for fact collection within a single source file.
139struct FactCtx<'a> {
140    path: &'a Path,
141    code_ref: &'a [u8],
142    module_id: &'a str,
143}
