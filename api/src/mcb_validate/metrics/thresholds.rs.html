8use std::collections::HashMap;
9
10use crate::constants::rules::{METRICS_FIELD_MAX, METRICS_FIELD_SEVERITY};
11use crate::constants::severities::{SEVERITY_ERROR, SEVERITY_INFO};
12use crate::traits::violation::Severity;

14/// Types of metrics we can measure
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
16pub enum MetricType {
17    /// Cognitive complexity - how hard code is to understand
18    CognitiveComplexity,
19    /// Cyclomatic complexity - number of linearly independent paths
20    CyclomaticComplexity,
21    /// Number of lines/statements in a function
22    FunctionLength,
23    /// Maximum nesting depth (if/for/while/match)
24    NestingDepth,
25}
26
27impl MetricType {
28    /// Get the human-readable name
29    #[must_use]
30    pub fn name(&self) -> &'static str {
31        match self {
32            Self::CognitiveComplexity
33            | Self::CyclomaticComplexity
34            | Self::FunctionLength
35            | Self::NestingDepth => "Function",
36        }
37    }
38
39    /// Get metric description
40    #[must_use]
41    pub fn description(&self) -> &'static str {
42        match self {
43            Self::CognitiveComplexity => "cognitive complexity",
44            Self::CyclomaticComplexity => "cyclomatic complexity",
45            Self::FunctionLength => "length",
46            Self::NestingDepth => "nesting depth",
47        }
48    }
49
50    /// Get suggestion for fixing
51    #[must_use]
52    pub fn suggestion(&self) -> String {
53        match self {
54            Self::CognitiveComplexity => {
55                "Consider breaking this function into smaller, focused functions. \
56                 Extract complex conditions into named functions or early returns."
57                    .to_owned()
58            }
59            Self::CyclomaticComplexity => {
60                "Reduce the number of decision points (if/else, switch, loops). \
61                 Consider using polymorphism or strategy pattern instead of conditionals."
62                    .to_owned()
63            }
64            Self::FunctionLength => {
65                "Consider extracting helper functions or using the Extract Method refactoring. \
66                 Functions should ideally do one thing well."
67                    .to_owned()
68            }
69            Self::NestingDepth => {
70                "Consider using early returns, guard clauses, or extracting nested logic \
71                 into separate functions to reduce nesting."
72                    .to_owned()
73            }
74        }
75    }
76}

78/// A single metric threshold configuration
79#[derive(Debug, Clone)]
80pub struct MetricThreshold {
81    /// Maximum allowed value
82    pub max_value: u32,
83    /// Severity when threshold is exceeded
84    pub severity: Severity,
85}
86
87/// Configuration for all metric thresholds
88#[derive(Debug, Clone)]
89pub struct MetricThresholds {
90    thresholds: HashMap<MetricType, MetricThreshold>,
91}
92
93impl Default for MetricThresholds {
94    fn default() -> Self {
95        Self::new()
96            // Default thresholds based on common industry standards
97            .with_threshold(MetricType::CognitiveComplexity, 15, Severity::Warning)
98            .with_threshold(MetricType::CyclomaticComplexity, 10, Severity::Warning)
99            .with_threshold(MetricType::FunctionLength, 50, Severity::Warning)
100            .with_threshold(MetricType::NestingDepth, 4, Severity::Warning)
101    }
102}
103
104impl MetricThresholds {
105    /// Create empty thresholds
106    #[must_use]
107    pub fn new() -> Self {
108        Self {
109            thresholds: HashMap::new(),
110        }
111    }
112
113    /// Add or update a threshold
114    #[must_use]
115    pub fn with_threshold(
116        mut self,
117        metric: MetricType,
118        max_value: u32,
119        severity: Severity,
120    ) -> Self {
121        self.thresholds.insert(
122            metric,
123            MetricThreshold {
124                max_value,
125                severity,
126            },
127        );
128        self
129    }
130
131    /// Get threshold for a metric type
132    #[must_use]
133    pub fn get(&self, metric: MetricType) -> Option<&MetricThreshold> {
134        self.thresholds.get(&metric)
135    }
136
137    fn severity_from_str(s: Option<&str>) -> Severity {
138        match s {
139            Some(SEVERITY_ERROR) => Severity::Error,
140            Some(SEVERITY_INFO) => Severity::Info,
141            _ => Severity::Warning,
142        }
143    }
144
145    fn to_u32(val: u64) -> u32 {
146        u32::try_from(val).unwrap_or(u32::MAX)
147    }
148
149    fn parse_metric(
150        obj: &serde_json::Map<String, serde_json::Value>,
151        key: &str,
152    ) -> Option<(u32, Severity)> {
153        let section = obj.get(key)?;
154        let max = section.get(METRICS_FIELD_MAX)?.as_u64()?;
155        let severity_str = section.get(METRICS_FIELD_SEVERITY).and_then(|v| v.as_str());
156        Some((Self::to_u32(max), Self::severity_from_str(severity_str)))
157    }
158
159    fn ensure_defaults(mut self) -> Self {
160        const DEFAULTS: [(MetricType, u32, Severity); 4] = [
161            (MetricType::CognitiveComplexity, 15, Severity::Warning),
162            (MetricType::CyclomaticComplexity, 10, Severity::Warning),
163            (MetricType::FunctionLength, 50, Severity::Warning),
164            (MetricType::NestingDepth, 4, Severity::Warning),
165        ];
166
167        for (metric, max, severity) in &DEFAULTS {
168            if self.get(*metric).is_none() {
169                self = self.with_threshold(*metric, *max, *severity);
170            }
171        }
172        self
173    }
174
175    /// Parse thresholds from YAML rule config
176    #[must_use]
177    pub fn from_yaml(config: &serde_json::Value) -> Self {
178        let mut thresholds = Self::new();
179
180        if let Some(obj) = config.as_object() {
181            if let Some((max, sev)) = Self::parse_metric(obj, "cognitive_complexity") {
182                thresholds = thresholds.with_threshold(MetricType::CognitiveComplexity, max, sev);
183            }
184            if let Some((max, sev)) = Self::parse_metric(obj, "function_length") {
185                thresholds = thresholds.with_threshold(MetricType::FunctionLength, max, sev);
186            }
187            if let Some((max, sev)) = Self::parse_metric(obj, "cyclomatic_complexity") {
188                thresholds = thresholds.with_threshold(MetricType::CyclomaticComplexity, max, sev);
189            }
190            if let Some((max, sev)) = Self::parse_metric(obj, "nesting_depth") {
191                thresholds = thresholds.with_threshold(MetricType::NestingDepth, max, sev);
192            }
193        }
194
195        thresholds.ensure_defaults()
196    }
197
198    /// Create thresholds from a `MetricsConfig` struct (from `ValidatedRule`)
199    #[must_use]
200    pub fn from_metrics_config(config: &crate::rules::yaml_loader::MetricsConfig) -> Self {
201        let mut thresholds = Self::new();
202
203        if let Some(cc) = &config.cognitive_complexity {
204            let sev = Self::severity_from_str(cc.severity.as_deref());
205            thresholds = thresholds.with_threshold(MetricType::CognitiveComplexity, cc.max, sev);
206        }
207
208        if let Some(cyc) = &config.cyclomatic_complexity {
209            let sev = Self::severity_from_str(cyc.severity.as_deref());
210            if config.cognitive_complexity.is_none() {
211                thresholds =
212                    thresholds.with_threshold(MetricType::CognitiveComplexity, cyc.max, sev);
213            }
214        }
215
216        if let Some(fl) = &config.function_length {
217            let sev = Self::severity_from_str(fl.severity.as_deref());
218            thresholds = thresholds.with_threshold(MetricType::FunctionLength, fl.max, sev);
219        }
220
221        if let Some(nd) = &config.nesting_depth {
222            let sev = Self::severity_from_str(nd.severity.as_deref());
223            thresholds = thresholds.with_threshold(MetricType::NestingDepth, nd.max, sev);
224        }
225
226        thresholds
227    }
