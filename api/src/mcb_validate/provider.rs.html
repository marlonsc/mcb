38#[async_trait]
39impl ValidationProvider for McbValidateProvider {
40    fn provider_name(&self) -> &str {
41        "mcb-validate"
42    }
43
44    fn description(&self) -> &str {
45        "Architecture and code quality validation engine"
46    }
47
48    fn list_validators(&self) -> Vec<ValidatorInfo> {
49        let workspace_root = std::env::current_dir().unwrap_or_else(|_| PathBuf::from("."));
50        let registry = ValidatorRegistry::standard_for(workspace_root);
51
52        let descriptions: HashMap<&'static str, &'static str> = registry
53            .validators()
54            .iter()
55            .map(|validator| (validator.name(), validator.description()))
56            .collect();
57
58        let mut rule_counts: HashMap<String, usize> = HashMap::new();
59        for rule in self.get_rules(None) {
60            *rule_counts.entry(rule.category).or_insert(0) += 1;
61        }
62
63        ValidatorRegistry::standard_validator_names()
64            .iter()
65            .map(|name| ValidatorInfo {
66                id: (*name).to_owned(),
67                name: (*name).replace('_', " "),
68                description: descriptions
69                    .get(name)
70                    .copied()
71                    .unwrap_or("No description available")
72                    .to_owned(),
73                rule_count: *rule_counts.get(*name).unwrap_or(&0),
74                categories: vec![(*name).to_owned()],
75            })
76            .collect()
77    }
78
79    fn get_rules(&self, category: Option<&str>) -> Vec<RuleInfo> {
80        crate::utils::yaml::get_validation_rules(category)
81    }
82
83    async fn validate(
84        &self,
85        workspace_root: &Path,
86        options: ValidationOptions,
87    ) -> Result<ValidationReport> {
88        run_validation(workspace_root, &options)
89    }
90
91    async fn validate_file(
92        &self,
93        file_path: &Path,
94        options: ValidationOptions,
95    ) -> Result<ValidationReport> {
96        run_file_validation(file_path, &options)
97    }
98
99    fn can_validate(&self, path: &Path) -> bool {
100        if path.is_file() {
101            return path
102                .extension()
103                .and_then(std::ffi::OsStr::to_str)
104                .is_some_and(|ext| ext == "rs");
105        }
106
107        path.is_dir() && path.join("Cargo.toml").exists()
108    }
109
110    fn supported_extensions(&self) -> &[&str] {
111        &["rs"]
112    }
113}

115fn run_validation(workspace_root: &Path, options: &ValidationOptions) -> Result<ValidationReport> {
116    let mut config = ValidationConfig::new(workspace_root);
117    if let Some(patterns) = &options.exclude_patterns {
118        for pattern in patterns {
119            config = config.with_exclude_pattern(pattern.clone());
120        }
121    }
122
123    let registry = ValidatorRegistry::standard_for(workspace_root);
124
125    let report = if let Some(names) = options.validators.as_deref() {
126        let names_ref: Vec<&str> = names.iter().map(String::as_str).collect();
127        let violations = registry
128            .validate_named(&config, &names_ref)
129            .map_err(|e| Error::internal(e.to_string()))?;
130        GenericReporter::create_report(&violations, workspace_root.to_path_buf())
131    } else {
132        let violations = registry
133            .validate_all(&config)
134            .map_err(|e| Error::internal(e.to_string()))?;
135        GenericReporter::create_report(&violations, workspace_root.to_path_buf())
136    };
137
138    Ok(convert_report(
139        report,
140        options.severity_filter.as_deref(),
141        options.include_suggestions,
142    ))
143}
144
145fn run_file_validation(file_path: &Path, options: &ValidationOptions) -> Result<ValidationReport> {
146    let workspace_root = find_workspace_root_from(file_path).unwrap_or_else(|| {
147        file_path
148            .parent()
149            .map_or_else(|| file_path.to_path_buf(), Path::to_path_buf)
150    });
151
152    let full_report = run_validation(&workspace_root, options)?;
153    let file_str = file_path.to_string_lossy().into_owned();
154
155    let file_violations: Vec<ViolationEntry> = full_report
156        .violations
157        .into_iter()
158        .filter(|violation| {
159            violation
160                .file
161                .as_ref()
162                .is_some_and(|file| file.contains(&file_str))
163        })
164        .collect();
165
166    Ok(crate::utils::validation_report::from_violations(
167        file_violations,
168    ))
169}

171fn convert_report(
172    report: GenericReport,
173    severity_filter: Option<&str>,
174    include_suggestions: bool,
175) -> ValidationReport {
176    crate::utils::validation_report::from_generic_report(
177        report,
178        severity_filter,
179        include_suggestions,
180    )
181}
