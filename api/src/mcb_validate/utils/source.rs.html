33/// Generic helper: iterate over all Rust files in crate source directories.
34///
35/// # Errors
36///
37/// Returns an error if directory enumeration or file reading fails.
38pub fn for_each_rust_file<F>(config: &ValidationConfig, mut visitor: F) -> Result<()>
39where
40    F: FnMut(PathBuf, Vec<&str>) -> Result<()>,
41{
42    for crate_dir in config.get_source_dirs()? {
43        let src_dir = crate_dir.join("src");
44        if !src_dir.exists() {
45            continue;
46        }
47
48        for_each_file_under_root(config, &src_dir, Some(LanguageId::Rust), |entry| {
49            let path = &entry.absolute_path;
50            let content = std::fs::read_to_string(path)?;
51            let lines: Vec<&str> = content.lines().collect();
52            visitor(path.clone(), lines)
53        })?;
54    }
55    Ok(())
56}

60/// Iterate source lines, skipping comments and `#[cfg(test)]` modules.
61/// Yields `(1-based line number, trimmed line)`.
62#[must_use]
63pub fn source_lines(content: &str) -> Vec<(usize, &str)> {
64    let mut result = Vec::new();
65    let mut in_test_module = false;
66    for (idx, line) in content.lines().enumerate() {
67        let trimmed = line.trim();
68        if trimmed.starts_with(COMMENT_PREFIX) {
69            continue;
70        }
71        if trimmed.contains(CFG_TEST_MARKER) {
72            in_test_module = true;
73            continue;
74        }
75        if in_test_module {
76            continue;
77        }
78        result.push((idx + 1, trimmed));
79    }
80    result
81}

83/// Filter out lines that belong to `#[cfg(test)]` regions.
84/// Returns `(original 0-based index, trimmed line)` pairs.
85#[must_use]
86pub fn non_test_lines<'a>(lines: &[&'a str]) -> Vec<(usize, &'a str)> {
87    let mut result = Vec::new();
88    let mut in_test_module = false;
89    for (i, line) in lines.iter().enumerate() {
90        let trimmed = line.trim();
91        if trimmed.contains(CFG_TEST_MARKER) {
92            in_test_module = true;
93            continue;
94        }
95        if in_test_module {
96            continue;
97        }
98        result.push((i, trimmed));
99    }
100    result
101}

105/// Track function name from a regex pattern match.
106pub fn track_fn_name(fn_pattern: Option<&Regex>, trimmed: &str, name: &mut String) {
107    if let Some(re) = fn_pattern
108        && let Some(cap) = re.captures(trimmed)
109    {
110        *name = cap
111            .get(1)
112            .map(|m| m.as_str().to_owned())
113            .unwrap_or_default();
114    }
115}

117/// Compile `(pattern_id, description)` pairs into `(Regex, &str)`.
118///
119/// # Errors
120///
121/// Returns an error if any requested pattern ID is missing.
122pub fn compile_pattern_pairs<'a>(
123    ids: &[(&str, &'a str)],
124) -> Result<Vec<(&'static Regex, &'a str)>> {
125    ids.iter()
126        .map(|(id, desc)| required_pattern(id).map(|r| (r, *desc)))
127        .collect()
128}
129
130/// Compile an iterator of pattern IDs into a `Vec<&Regex>`.
131///
132/// # Errors
133///
134/// Returns an error if any requested pattern ID is missing.
135pub fn required_patterns<'a>(ids: impl Iterator<Item = &'a str>) -> Result<Vec<&'static Regex>> {
136    ids.map(required_pattern).collect()
137}

139/// Check if a line is a function signature or standalone brace.
140#[must_use]
141pub fn is_fn_signature_or_brace(line: &str) -> bool {
142    const FN_PREFIXES: [&str; 4] = ["fn ", "pub fn ", "async fn ", "pub async fn "];
143    matches!(line, "{" | "}") || FN_PREFIXES.iter().any(|prefix| line.starts_with(prefix))
144}

150/// A parsed function with its body and metadata.
151pub struct FunctionInfo {
152    /// Function name.
153    pub name: String,
154    /// 1-based start line in the source file.
155    pub start_line: usize,
156    /// All non-empty, non-comment body lines.
157    pub body_lines: Vec<String>,
158    /// Body lines excluding braces and `fn` signatures.
159    pub meaningful_body: Vec<String>,
160    /// Whether the body contains control-flow keywords.
161    pub has_control_flow: bool,
162}
