13/// Visit lines inside a brace-balanced block starting at `start_line`.
14pub fn within_block<F>(lines: &[&str], start_line: usize, mut visitor: F)
15where
16    F: FnMut(&str, usize) -> bool,
17{
18    let mut start_offset = 0;
19    let slice = &lines[start_line..];
20    while start_offset < slice.len() && !slice[start_offset].contains('{') {
21        start_offset += 1;
22    }
23
24    if start_offset < slice.len()
25        && let Some(count) =
26            mcb_domain::utils::analysis::count_balanced_block_lines(slice, usize::MAX)
27    {
28        for (offset, line) in slice[start_offset..count].iter().enumerate() {
29            if !visitor(line, start_offset + offset) {
30                break;
31            }
32        }
33    }
34}

36/// Count lines in a brace-balanced block.
37#[must_use]
38pub fn count_block_lines(lines: &[&str], start_line: usize) -> usize {
39    let mut start_offset = 0;
40    let slice = &lines[start_line..];
41    while start_offset < slice.len() && !slice[start_offset].contains('{') {
42        start_offset += 1;
43    }
44    if let Some(count) = mcb_domain::utils::analysis::count_balanced_block_lines(slice, usize::MAX)
45        && count > start_offset
46    {
47        return count - start_offset;
48    }
49    0
50}

52/// Count regex matches inside a brace-balanced block.
53#[must_use]
54pub fn count_matches_in_block(lines: &[&str], start_line: usize, pattern: &Regex) -> usize {
55    let mut count = 0;
56    within_block(lines, start_line, |line, _| {
57        if pattern.is_match(line) {
58            count += 1;
59        }
60        true
61    });
62    count
63}

65/// Count match arms in a `match` block.
66///
67/// # Errors
68/// Returns an error if the required arm-pattern regex is unavailable.
69pub fn count_match_arms(lines: &[&str], start_line: usize) -> Result<usize> {
70    let arrow_pattern = required_pattern("SOLID003.match_arrow")?;
71
72    let mut count = 0;
73    let mut brace_depth = 0;
74
75    within_block(lines, start_line, |line, _| {
76        brace_depth += line.chars().filter(|c| *c == '{').count();
77        brace_depth -= line.chars().filter(|c| *c == '}').count();
78
79        if arrow_pattern.is_match(line) && brace_depth >= 1 {
80            count += 1;
81        }
82        true
83    });
84    Ok(count)
85}

87/// Scan declarations in Rust files and emit violations when member count exceeds threshold.
88///
89/// # Errors
90/// Returns an error if file traversal fails.
91pub fn scan_decl_blocks<F, V>(
92    config: &ValidationConfig,
93    scan_config: &DeclScanConfig<'_>,
94    make_violation: F,
95) -> Result<Vec<V>>
96where
97    F: Fn(PathBuf, usize, &str, usize, usize) -> V,
98{
99    let mut violations = Vec::new();
100
101    for_each_rust_file(config, |path, lines| {
102        for (line_num, line) in lines.iter().enumerate() {
103            if let Some(cap) = scan_config.decl_pattern.captures(line) {
104                let name = cap.get(1).map_or("", |m| m.as_str());
105                let method_count =
106                    (scan_config.count_fn)(&lines, line_num, scan_config.member_fn_pattern);
107
108                if method_count > scan_config.max_allowed {
109                    violations.push(make_violation(
110                        path.clone(),
111                        line_num + 1,
112                        name,
113                        method_count,
114                        scan_config.max_allowed,
115                    ));
116                }
117            }
118        }
119        Ok(())
120    })?;
121
122    Ok(violations)
123}

125/// Configuration for declaration-block scans.
126pub struct DeclScanConfig<'a> {
127    /// Regex that captures declaration name in group 1.
128    pub decl_pattern: &'a Regex,
129    /// Regex used to count declaration members.
130    pub member_fn_pattern: &'a Regex,
131    /// Counting strategy for block members.
132    pub count_fn: fn(&[&str], usize, &Regex) -> usize,
133    /// Allowed maximum members before creating a violation.
134    pub max_allowed: usize,
135}
