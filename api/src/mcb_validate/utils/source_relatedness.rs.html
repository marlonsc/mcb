4use crate::validators::solid::constants::{
5    MAX_AFFIX_LENGTH, MIN_AFFIX_LENGTH, MIN_NAMES_FOR_RELATION_CHECK,

9pub(super) fn structs_seem_related_impl(names: &[String]) -> bool {
10    if names.len() < MIN_NAMES_FOR_RELATION_CHECK {
11        return true;
12    }
13
14    let checks = [
15        has_common_prefix,
16        has_common_suffix,
17        has_purpose_suffix,
18        has_shared_keyword,
19        has_common_words,
20    ];
21
22    checks.iter().any(|check| check(names))
23}
24
25fn has_common_prefix(names: &[String]) -> bool {
26    let first = &names[0];
27    for len in (MIN_AFFIX_LENGTH..=first.len().min(MAX_AFFIX_LENGTH)).rev() {
28        let prefix = &first[..len];
29        if names.iter().all(|n| n.starts_with(prefix)) {
30            return true;
31        }
32    }
33    false
34}
35
36fn has_common_suffix(names: &[String]) -> bool {
37    let first = &names[0];
38    for len in (MIN_AFFIX_LENGTH..=first.len().min(MAX_AFFIX_LENGTH)).rev() {
39        let suffix = &first[first.len().saturating_sub(len)..];
40        if names.iter().all(|n| n.ends_with(suffix)) {
41            return true;
42        }
43    }
44    false
45}

130fn has_common_words(names: &[String]) -> bool {
131    let words: Vec<Vec<&str>> = names
132        .iter()
133        .map(|n| crate::utils::naming::split_camel_case(n))
134        .collect();
135
136    if let Some(first_words) = words.first() {
137        for word in first_words {
138            if word.len() >= MIN_WORD_LENGTH_FOR_COMPARISON {
139                let count = words.iter().filter(|w| w.contains(word)).count();
140                if count > names.len() / 2 {
141                    return true;
142                }
143            }
144        }
145    }
146    false
