10use serde::{Deserialize, Serialize};
11
12/// Validation thresholds (configurable)
13///
14/// Contains all numeric limits used by architecture validators.
15/// Defaults are based on common code quality standards.
16#[derive(Debug, Clone, Serialize, Deserialize)]
17pub struct ValidationThresholds {
18    // ========================================================================
19    // SOLID Thresholds
20    // ========================================================================
21    /// Maximum methods per trait (Interface Segregation)
22    pub max_trait_methods: usize,
23
24    /// Maximum lines per struct definition
25    pub max_struct_lines: usize,
26
27    /// Maximum methods per impl block
28    pub max_impl_methods: usize,
29
30    /// Maximum arms in a match expression
31    pub max_match_arms: usize,
32
33    // ========================================================================
34    // KISS Thresholds
35    // ========================================================================
36    /// Maximum fields per struct
37    pub max_struct_fields: usize,
38
39    /// Maximum parameters per function
40    pub max_function_params: usize,
41
42    /// Maximum fields in a builder pattern
43    pub max_builder_fields: usize,
44
45    /// Maximum fields in DI container (exception for catalog structs)
46    pub max_di_container_fields: usize,
47
48    /// Maximum nesting depth for control structures
49    pub max_nesting_depth: usize,
50
51    // ========================================================================
52    // Quality Thresholds
53    // ========================================================================
54    /// Maximum lines per file
55    pub max_file_lines: usize,
56
57    /// Maximum lines per function
58    pub max_function_lines: usize,
59
60    // ========================================================================
61    // Complexity Thresholds
62    // ========================================================================
63    /// Maximum cyclomatic complexity per function
64    pub max_cyclomatic_complexity: u32,
65
66    /// Maximum cognitive complexity per function
67    pub max_cognitive_complexity: u32,
68}
69
70impl Default for ValidationThresholds {
71    fn default() -> Self {
72        Self {
73            // SOLID
74            max_trait_methods: 10,
75            max_struct_lines: 700,
76            max_impl_methods: 15,
77            max_match_arms: 15,
78
79            // KISS
80            max_struct_fields: 16,
81            max_function_params: 5,
82            max_builder_fields: 7,
83            max_di_container_fields: 25,
84            max_nesting_depth: 3,
85
86            // Quality
87            max_file_lines: 500,
88            max_function_lines: 50,
89
90            // Complexity
91            max_cyclomatic_complexity: 10,
92            max_cognitive_complexity: 15,
93        }
94    }
95}
96
97impl ValidationThresholds {
98    /// Create thresholds with default values
99    #[must_use]
100    pub fn new() -> Self {
101        Self::default()
102    }
103}
104
105// ============================================================================
106// Global Singleton (Thread-Safe)
107// ============================================================================
108
109use std::sync::OnceLock;
110
111static THRESHOLDS: OnceLock<ValidationThresholds> = OnceLock::new();
112
113/// Get the global validation thresholds
114///
115/// Returns a reference to the global thresholds singleton.
116/// Initializes with defaults on first access.
117pub fn thresholds() -> &'static ValidationThresholds {
118    THRESHOLDS.get_or_init(ValidationThresholds::default)
119}
