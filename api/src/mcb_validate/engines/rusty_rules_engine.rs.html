36/// Rusty rule definition with composition support
37#[derive(Debug, Clone)]
38pub struct RustyRule {
39    /// The type of rule (e.g., "`cargo_dependencies`", "`ast_pattern`").
40    pub rule_type: String,
41    /// The condition logic to evaluate.
42    pub condition: Condition,
43    /// The action to take if the condition is met.
44    pub action: Action,
45}
46
47/// Conditions with composition (all/any/not)
48#[derive(Debug, Clone)]
49pub enum Condition {
50    /// All conditions must be true
51    All(Vec<Condition>),
52    /// Any condition must be true
53    Any(Vec<Condition>),
54    /// Negate condition
55    Not(Box<Condition>),
56    /// Simple condition
57    Simple {
58        /// The type of fact being checked.
59        fact_type: String,
60        /// The field of the fact to check.
61        field: String,
62        /// The operator to use for comparison.
63        operator: String,
64        /// The value to compare against.
65        value: Value,
66    },
67}
68
69/// Actions to execute when condition matches
70#[derive(Debug, Clone)]
71pub enum Action {
72    /// Report a standard violation.
73    Violation {
74        /// The violation message.
75        message: String,
76        /// The severity of the violation.
77        severity: Severity,
78    },
79    /// Execute a custom action string.
80    Custom(String),
81}

89impl RustyRulesEngineWrapper {
90    /// Creates a new, empty `RustyRulesEngineWrapper`.
91    #[must_use]
92    pub fn new() -> Self {
93        Self {
94            rule_definitions: HashMap::new(),
95        }
96    }
97
98    /// Parse rule definition from JSON
99    ///
100    /// # Errors
101    ///
102    /// Returns an error if the rule definition cannot be parsed.
103    pub fn parse_rule_definition(&mut self, rule_id: String, definition: &Value) -> Result<()> {
104        let rule = Self::parse_rule_from_json(definition)?;
105        self.rule_definitions.insert(rule_id, rule);
106        Ok(())
107    }
108
109    fn parse_rule_from_json(definition: &Value) -> Result<RustyRule> {
110        let rule_type = json_str(definition, YAML_FIELD_FIX_TYPE, GENERIC).to_owned();
111        let condition = Self::parse_optional_condition(definition)?;
112        let action = Self::parse_optional_action(definition);
113
114        Ok(RustyRule {
115            rule_type,
116            condition,
117            action,
118        })
119    }
120
121    fn parse_optional_condition(definition: &Value) -> Result<Condition> {
122        definition
123            .get(YAML_FIELD_CONDITION)
124            .map(Self::parse_condition_value)
125            .transpose()
126            .map(|condition| condition.unwrap_or_else(|| Condition::All(vec![])))
127    }
128
129    fn parse_optional_action(definition: &Value) -> Action {
130        definition.get(YAML_FIELD_ACTION).map_or(
131            Action::Violation {
132                message: DEFAULT_VIOLATION_MESSAGE.to_owned(),
133                severity: Severity::Warning,
134            },
135            Self::parse_action,
136        )
137    }
138
139    fn parse_condition_value(condition_json: &Value) -> Result<Condition> {
140        if let Some(all_conditions) = condition_json.get(RUSTY_FIELD_ALL)
141            && let Some(conditions_array) = all_conditions.as_array()
142        {
143            let conditions = conditions_array
144                .iter()
145                .map(Self::parse_condition_value)
146                .collect::<Result<Vec<_>>>()?;
147            return Ok(Condition::All(conditions));
148        }
149
150        if let Some(any_conditions) = condition_json.get(RUSTY_FIELD_ANY)
151            && let Some(conditions_array) = any_conditions.as_array()
152        {
153            let conditions = conditions_array
154                .iter()
155                .map(Self::parse_condition_value)
156                .collect::<Result<Vec<_>>>()?;
157            return Ok(Condition::Any(conditions));
158        }
159
160        if let Some(not_condition) = condition_json.get(RUSTY_FIELD_NOT) {
161            let condition = Self::parse_condition_value(not_condition)?;
162            return Ok(Condition::Not(Box::new(condition)));
163        }
164
165        // Simple condition
166        let fact_type = json_str(condition_json, RUSTY_FIELD_FACT_TYPE, GENERIC).to_owned();
167
168        let field = json_str(condition_json, RUSTY_FIELD_FIELD, RUSTY_DEFAULT_FIELD).to_owned();
169
170        let operator =
171            json_str(condition_json, RUSTY_FIELD_OPERATOR, RUSTY_DEFAULT_OPERATOR).to_owned();
172
173        let value = condition_json
174            .get(RUSTY_FIELD_VALUE)
175            .cloned()
176            .unwrap_or(Value::Null);
177
178        Ok(Condition::Simple {
179            fact_type,
180            field,
181            operator,
182            value,
183        })
184    }
185
186    fn parse_action(action_json: &Value) -> Action {
187        if let Some(violation) = action_json.get(RUSTY_FIELD_VIOLATION) {
188            let message =
189                json_str(violation, YAML_FIELD_MESSAGE, DEFAULT_VIOLATION_MESSAGE).to_owned();
190
191            let severity = json_opt_str(violation, YAML_FIELD_SEVERITY)
192                .map_or(Severity::Warning, Self::parse_severity);
193
194            return Action::Violation { message, severity };
195        }
196
197        Action::Custom(RUSTY_CUSTOM_ACTION_DEFAULT.to_owned())
198    }
199
200    fn parse_severity(raw: &str) -> Severity {
201        match raw {
202            SEVERITY_ERROR => Severity::Error,
203            SEVERITY_INFO => Severity::Info,
204            _ => Severity::Warning,
205        }
206    }
207
208    fn has_forbidden_dependency(pattern: &str, context: &RuleContext) -> bool {
209        let trimmed_pattern = pattern.trim_matches('"');
210        let pattern_prefix = trimmed_pattern.trim_end_matches('*');
211
212        if workspace_has_forbidden_cargo_dependency(&context.workspace_root, pattern_prefix) {
213            return true;
214        }
215
216        for (path, content) in &context.file_contents {
217            if path.ends_with("Cargo.toml") && dependency_matches(content, pattern_prefix) {
218                return true;
219            }
220        }
221
222        false
223    }
224}
225
226fn workspace_has_forbidden_cargo_dependency(
227    workspace_root: &std::path::Path,
228    pattern_prefix: &str,
229) -> bool {
230    let mut stack = vec![workspace_root.to_path_buf()];
231
232    while let Some(dir) = stack.pop() {
233        let Ok(entries) = std::fs::read_dir(&dir) else {
234            continue;
235        };
236
237        for entry in entries.flatten() {
238            let path = entry.path();
239            if path.is_dir() {
240                stack.push(path);
241            } else if cargo_toml_matches(&path, pattern_prefix) {
242                return true;
243            }
244        }
245    }
246
247    false
248}
249
250/// Returns true when `path` is a `Cargo.toml` whose dependencies match `pattern_prefix`.
251fn cargo_toml_matches(path: &std::path::Path, pattern_prefix: &str) -> bool {
252    if path.file_name().and_then(std::ffi::OsStr::to_str) != Some("Cargo.toml") {
253        return false;
254    }
255
256    std::fs::read_to_string(path)
257        .is_ok_and(|content| dependency_matches(content.as_ref(), pattern_prefix))
258}
259
260fn dependency_matches(content: &str, pattern_prefix: &str) -> bool {
261    content
262        .parse::<toml::Value>()
263        .ok()
264        .is_some_and(|toml_value| toml_dependencies_match(&toml_value, pattern_prefix))
265        || dependencies_match_by_line(content, pattern_prefix)
266}
267
268fn toml_dependencies_match(toml_value: &toml::Value, pattern_prefix: &str) -> bool {
269    let Some(dependencies) = toml_value.get("dependencies") else {
270        return false;
271    };
272    let Some(deps_table) = dependencies.as_table() else {
273        return false;
274    };
275
276    deps_table
277        .keys()
278        .any(|dep_name| dep_name.starts_with(pattern_prefix))
279}
280
281fn dependencies_match_by_line(content: &str, pattern_prefix: &str) -> bool {
282    let mut in_dependencies = false;
283
284    for line in content.lines() {
285        let trimmed = line.trim();
286        if trimmed.is_empty() || trimmed.starts_with('#') {
287            continue;
288        }
289
290        if trimmed.starts_with('[') && trimmed.ends_with(']') {
291            in_dependencies = trimmed == "[dependencies]";
292            continue;
293        }
294
295        if in_dependencies && dependency_line_matches(trimmed, pattern_prefix) {
296            return true;
297        }
298    }
299
300    false
301}
302
303/// Returns true when a `[dependencies]` line declares a key starting with `pattern_prefix`.
304fn dependency_line_matches(trimmed: &str, pattern_prefix: &str) -> bool {
305    let Some((key, _)) = trimmed.split_once('=') else {
306        return false;
307    };
308    let dep_name = key.trim().trim_matches('"').trim_matches('\'');
309    dep_name.starts_with(pattern_prefix)
310}

312#[async_trait]
313impl RuleEngine for RustyRulesEngineWrapper {
314    async fn execute(
315        &self,
316        rule_definition: &Value,
317        context: &RuleContext,
318    ) -> Result<Vec<RuleViolation>> {
319        let Some(rule_type) = json_opt_str(rule_definition, YAML_FIELD_FIX_TYPE) else {
320            return Ok(vec![]);
321        };
322
323        self.dispatch_rule(rule_type, rule_definition, context)
324            .await
325    }
326}
327
328impl RustyRulesEngineWrapper {
329    async fn dispatch_rule(
330        &self,
331        rule_type: &str,
332        rule_definition: &Value,
333        context: &RuleContext,
334    ) -> Result<Vec<RuleViolation>> {
335        match rule_type {
336            RUSTY_RULE_TYPE_CARGO_DEPENDENCIES => {
337                self.execute_cargo_dependency_rule(rule_definition, context)
338                    .await
339            }
340            RUSTY_RULE_TYPE_FILE_SIZE => {
341                self.execute_file_size_rule(rule_definition, context).await
342            }
343            RUSTY_RULE_TYPE_AST_PATTERN => {
344                self.execute_ast_pattern_rule(rule_definition, context)
345                    .await
346            }
347            _ => Ok(vec![]),
348        }
349    }
350
351    async fn execute_cargo_dependency_rule(
352        &self,
353        rule_definition: &Value,
354        context: &RuleContext,
355    ) -> Result<Vec<RuleViolation>> {
356        let mut violations = Vec::new();
357        let condition = json_str(rule_definition, YAML_FIELD_CONDITION, NOT_EXISTS);
358        let Some(forbidden_pattern) = json_opt_str(rule_definition, YAML_FIELD_PATTERN) else {
359            return Ok(violations);
360        };
361
362        let has_forbidden = Self::has_forbidden_dependency(forbidden_pattern, context);
363        let should_report = match condition {
364            NOT_EXISTS => has_forbidden,
365            RUSTY_CONDITION_EXISTS => !has_forbidden,
366            _ => false,
367        };
368        if !should_report {
369            return Ok(violations);
370        }
371
372        let message = if condition == RUSTY_CONDITION_EXISTS {
373            RUSTY_CARGO_DEP_MISSING_MSG
374        } else {
375            RUSTY_CARGO_DEP_FORBIDDEN_MSG
376        };
377        violations.push(
378            RuleViolation::new(
379                RUSTY_CARGO_DEP_VIOLATION_ID,
380                ViolationCategory::Architecture,
381                Severity::Error,
382                message,
383            )
384            .with_context(format!("Pattern: {forbidden_pattern}")),
385        );
386
387        Ok(violations)
388    }
389
390    async fn execute_ast_pattern_rule(
391        &self,
392        rule_definition: &Value,
393        context: &RuleContext,
394    ) -> Result<Vec<RuleViolation>> {
395        let mut violations = Vec::new();
396        for pattern in forbidden_patterns(rule_definition) {
397            violations.extend(ast_pattern_violations(context, pattern));
398        }
399
400        Ok(violations)
401    }
402
403    async fn execute_file_size_rule(
404        &self,
405        rule_definition: &Value,
406        context: &RuleContext,
407    ) -> Result<Vec<RuleViolation>> {
408        let mut violations = Vec::new();
409        let condition = json_str(
410            rule_definition,
411            YAML_FIELD_CONDITION,
412            RUSTY_DEFAULT_FILE_SIZE_CONDITION,
413        );
414        let pattern = json_str(
415            rule_definition,
416            YAML_FIELD_PATTERN,
417            RUSTY_DEFAULT_FILE_SIZE_PATTERN,
418        );
419        let message = json_str(
420            rule_definition,
421            YAML_FIELD_MESSAGE,
422            "File exceeds size limit",
423        );
424
425        if condition != RUSTY_DEFAULT_FILE_SIZE_CONDITION {
426            return Ok(violations);
427        }
428
429        let max_lines = DEFAULT_MAX_FILE_LINES;
430        for (file_path, content) in &context.file_contents {
431            if !file_path.ends_with(pattern) || Self::is_size_check_excluded(file_path) {
432                continue;
433            }
434
435            let line_count = content.lines().count();
436            if line_count > max_lines {
437                violations.push(file_size_violation(
438                    file_path, line_count, max_lines, message,
439                ));
440            }
441        }
442
443        Ok(violations)
444    }
445
446    fn is_size_check_excluded(file_path: &str) -> bool {
447        file_path.contains(TEST_DIR_FRAGMENT)
448            || file_path.contains(RUSTY_TARGET_DIR_FRAGMENT)
449            || file_path.ends_with(TEST_FILE_SUFFIX)
450    }
451}

453/// Build a `QUAL006` violation for a file exceeding the line limit.
454fn file_size_violation(
455    file_path: &str,
456    line_count: usize,
457    max_lines: usize,
458    message: &str,
459) -> RuleViolation {
460    RuleViolation::new(
461        "QUAL006",
462        ViolationCategory::Quality,
463        Severity::Warning,
464        format!("{message}: {line_count} lines (max: {max_lines})"),
465    )
466    .with_file(std::path::PathBuf::from(file_path))
467    .with_context(format!("File: {file_path}, Lines: {line_count}"))
468}
469
470fn forbidden_patterns(rule_definition: &Value) -> Vec<&str> {
471    rule_definition
472        .get("forbidden")
473        .and_then(|v| v.as_array())
474        .map(|forbidden| {
475            forbidden
476                .iter()
477                .filter_map(serde_json::Value::as_str)
478                .collect()
479        })
480        // INTENTIONAL: Filter string collection; empty vec is safe if no filters
481        .unwrap_or_default()
482}
483
484fn json_str<'a>(obj: &'a Value, key: &str, default: &'a str) -> &'a str {
485    obj.get(key).and_then(Value::as_str).unwrap_or(default)
486}
487
488fn json_opt_str<'a>(obj: &'a Value, key: &str) -> Option<&'a str> {
489    obj.get(key).and_then(Value::as_str)
490}
491
492fn ast_pattern_violations(context: &RuleContext, pattern: &str) -> Vec<RuleViolation> {
493    context
494        .file_contents
495        .iter()
496        .filter(|(_, content)| content.contains(pattern))
497        .map(|(file_path, _)| {
498            RuleViolation::new(
499                RUSTY_AST_PATTERN_VIOLATION_ID,
500                ViolationCategory::Quality,
501                Severity::Error,
502                format!("Found forbidden pattern: {pattern}"),
503            )
504            .with_file(std::path::PathBuf::from(file_path))
505            .with_context(format!("Pattern: {pattern}"))
506        })
507        .collect()
508}
