22/// Registry of compiled regex patterns and configurations loaded from YAML rules
23pub struct PatternRegistry {
24    patterns: HashMap<String, Regex>,
25    configs: HashMap<String, serde_yaml::Value>,

28impl PatternRegistry {
29    /// Create an empty registry
30    #[must_use]
31    pub fn new() -> Self {
32        Self {
33            patterns: HashMap::new(),
34            configs: HashMap::new(),
35        }

38    /// Load patterns from all YAML rules in a directory, using config for template variables
39    ///
40    /// # Errors
41    ///
42    /// Returns an error if rule file enumeration fails.
43    pub fn load_from_rules(
44        rules_dir: &Path,
45        naming_config: &crate::config::NamingRulesConfig,
46        project_prefix: &str,
47    ) -> Result<Self> {
48        let mut registry = Self::new();
49
50        let rule_files = crate::utils::fs::collect_yaml_files(rules_dir)?;
51        for path in rule_files.into_iter().filter(|p| !is_template_path(p)) {
52            if let Err(e) = registry.load_rule_file(&path, naming_config, project_prefix) {
53                warn!(
54                    path = %path.display(),
55                    error = %e,
56                    "Failed to load patterns/config"
57                );
58            }
59        }
60
61        Ok(registry)
62    }
63
64    /// Load patterns and configurations from a single YAML rule file
65    fn load_rule_file(
66        &mut self,
67        path: &Path,
68        naming_config: &crate::config::NamingRulesConfig,
69        project_prefix: &str,
70    ) -> Result<()> {
71        let content = std::fs::read_to_string(path)?;
72        let mut yaml: serde_yaml::Value = serde_yaml::from_str(&content)?;
73
74        // Build template variables from configuration (no hardcoded crate names)
75        let mut variables = serde_yaml::Mapping::new();
76        variables.insert(
77            serde_yaml::Value::String("project_prefix".to_owned()),
78            serde_yaml::Value::String(project_prefix.to_owned()),
79        );
80
81        // Map each layer key to (crate_name, module_name) from NamingRulesConfig
82        let crates: [(&str, &str); 6] = [
83            ("domain", &naming_config.domain_crate),
84            ("application", &naming_config.application_crate),
85            ("providers", &naming_config.providers_crate),
86            ("infrastructure", &naming_config.infrastructure_crate),
87            ("server", &naming_config.server_crate),
88            ("validate", &naming_config.validate_crate),
89        ];
90
91        for (key, crate_name) in crates {
92            let module_name = crate_name.replace('-', "_");
93            variables.insert(
94                serde_yaml::Value::String(format!("{key}_crate")),
95                serde_yaml::Value::String(crate_name.to_owned()),
96            );
97            variables.insert(
98                serde_yaml::Value::String(format!("{key}_module")),
99                serde_yaml::Value::String(module_name),
100            );
101        }
102
103        let engine = TemplateEngine::new();
104        let variables_value = serde_yaml::Value::Mapping(variables);
105        if let Err(e) = engine.substitute_variables(&mut yaml, &variables_value) {
106            warn!(
107                path = %path.display(),
108                error = %e,
109                "Failed to substitute variables"
110            );
111        }
112
113        // Get rule ID for namespacing
114        let rule_id = yaml
115            .get(YAML_FIELD_ID)
116            .and_then(|v| v.as_str())
117            .unwrap_or("unknown");
118
119        // Load patterns from "patterns" section
120        for (name, pattern) in yaml
121            .get(YAML_FIELD_PATTERNS)
122            .and_then(|v| v.as_mapping())
123            .into_iter()
124            .flat_map(|patterns| patterns.iter())
125        {
126            if let (Some(name_str), Some(pattern_str)) = (name.as_str(), pattern.as_str()) {
127                let pattern_id = format!("{rule_id}.{name_str}");
128                self.register_pattern(&pattern_id, pattern_str)?;
129            }
130        }
131
132        // Load patterns from "selectors" section (for AST patterns)
133        for (i, selector) in yaml
134            .get(YAML_FIELD_SELECTORS)
135            .and_then(|v| v.as_sequence())
136            .into_iter()
137            .flat_map(|selectors| selectors.iter())
138            .enumerate()
139        {
140            if let Some(pattern) = selector.get(YAML_FIELD_REGEX).and_then(|v| v.as_str()) {
141                let pattern_id = format!("{rule_id}.selector_{i}");
142                self.register_pattern(&pattern_id, pattern)?;
143            }
144        }
145
146        // Load generic configuration from "config" section
147        if let Some(config) = yaml.get(YAML_FIELD_CONFIG) {
148            self.configs.insert(rule_id.to_owned(), config.clone());
149        }
150
151        // Also load top-level crate_name and allowed_dependencies if present (shorthand for dependency rules)
152        if let Some(crate_name) = yaml.get(YAML_FIELD_CRATE_NAME) {
153            let mut map = serde_yaml::Mapping::new();
154            map.insert(
155                serde_yaml::Value::from(YAML_FIELD_CRATE_NAME),
156                crate_name.clone(),
157            );
158            if let Some(allowed) = yaml.get(YAML_FIELD_ALLOWED_DEPS) {
159                map.insert(
160                    serde_yaml::Value::from(YAML_FIELD_ALLOWED_DEPS),
161                    allowed.clone(),
162                );
163            }
164
165            // Merge into config for this rule if it doesn't already have one, or extend it
166            let entry = self
167                .configs
168                .entry(rule_id.to_owned())
169                .or_insert_with(|| serde_yaml::Value::Mapping(serde_yaml::Mapping::new()));
170            if let Some(config_map) = entry.as_mapping_mut() {
171                for (k, v) in map {
172                    config_map.insert(k, v);
173                }
174            }
175        }
176
177        Ok(())
178    }

180    /// Register a pattern with the given ID
181    ///
182    /// # Errors
183    ///
184    /// Returns an error if the regex pattern is invalid.
185    pub fn register_pattern(&mut self, id: &str, pattern: &str) -> Result<()> {
186        let regex = Regex::new(pattern).map_err(|e| {
187            crate::ValidationError::Config(format!("Invalid regex pattern '{id}': {e}"))
188        })?;
189        self.patterns.insert(id.to_owned(), regex);
190        Ok(())
191    }
192
193    /// Get a pattern by ID
194    #[must_use]
195    pub fn get(&self, pattern_id: &str) -> Option<&Regex> {
196        self.patterns.get(pattern_id)
197    }
198
199    /// Get a configuration by rule ID
200    #[must_use]
201    pub fn get_config(&self, rule_id: &str) -> Option<&serde_yaml::Value> {
202        self.configs.get(rule_id)
203    }
204
205    /// Get a list of strings from configuration
206    #[must_use]
207    pub fn get_config_list(&self, rule_id: &str, key: &str) -> Vec<String> {
208        self.get_config(rule_id)
209            .and_then(|v| v.get(key))
210            .and_then(|v| v.as_sequence())
211            .map(|seq| {
212                seq.iter()
213                    .filter_map(|v| v.as_str().map(str::to_owned))
214                    .collect()
215            })
216            .unwrap_or_default()
217    }
218
219    /// Check if a pattern exists
220    #[must_use]
221    pub fn contains(&self, pattern_id: &str) -> bool {
222        self.patterns.contains_key(pattern_id)
223    }
224
225    /// Get all pattern IDs
226    pub fn pattern_ids(&self) -> impl Iterator<Item = &String> {
227        self.patterns.keys()
228    }
229
230    /// Get the number of registered patterns
231    #[must_use]
232    pub fn len(&self) -> usize {
233        self.patterns.len()
234    }
235
236    /// Check if the registry is empty
237    #[must_use]
238    pub fn is_empty(&self) -> bool {
239        self.patterns.is_empty()
240    }

243fn is_template_path(path: &Path) -> bool {
244    path.components()
245        .any(|component| component.as_os_str() == "templates")

254/// Get the default rules directory.
255///
256/// Resolution order (all workspace-relative unless overridden via env):
257/// 1. `MCB_RULES_DIR` environment variable (explicit override)
258/// 2. `CARGO_MANIFEST_DIR/rules` (building mcb-validate directly)
259/// 3. Workspace root `crates/mcb-validate/rules` (used as dependency)
260/// 4. CWD-relative `crates/mcb-validate/rules` (running from workspace root)
261/// 5. CWD-relative `rules/` fallback
262#[must_use]
263pub fn default_rules_dir() -> PathBuf {
264    // 1. Explicit override via environment variable
265    if let Ok(rules_dir) = std::env::var("MCB_RULES_DIR") {
266        let path = PathBuf::from(rules_dir);
267        if path.exists() {
268            return path;
269        }
270    }
271
272    // 2. Try CARGO_MANIFEST_DIR (works when building mcb-validate directly)
273    if let Ok(manifest_dir) = std::env::var("CARGO_MANIFEST_DIR") {
274        let rules_dir = PathBuf::from(&manifest_dir).join("rules");
275        if rules_dir.exists() {
276            return rules_dir;
277        }
278
279        // 3. When used as dependency, CARGO_MANIFEST_DIR points to consumer crate
280        // Try to find mcb-validate/rules relative to workspace root
281        if let Some(workspace_root) = PathBuf::from(&manifest_dir)
282            .ancestors()
283            .find(|p| p.join(CARGO_TOML_FILENAME).exists() && p.join("crates").exists())
284        {
285            let validate_rules = workspace_root.join("crates/mcb-validate/rules");
286            if validate_rules.exists() {
287                return validate_rules;
288            }
289        }
290    }
291
292    // 4. Try relative to current directory (works when running from workspace root)
293    let cwd_rules = PathBuf::from("crates/mcb-validate/rules");
294    if cwd_rules.exists() {
295        return cwd_rules;
296    }
297
298    // 5. Fallback to CWD-relative rules/
299    PathBuf::from("rules")
300}

302/// Global pattern registry, lazy-loaded from YAML rules and configuration
303pub static PATTERNS: std::sync::LazyLock<PatternRegistry> = std::sync::LazyLock::new(|| {
304    let rules_dir = default_rules_dir();
305    // Load config to get project-specific crate names for template substitution
306    let file_config = crate::config::FileConfig::load(".");
307    let naming_config = &file_config.rules.naming;
308    let project_prefix = &file_config.general.project_prefix;
309    PatternRegistry::load_from_rules(&rules_dir, naming_config, project_prefix).unwrap_or_else(
310        |e| {
311            error!(error = %e, "Failed to load pattern registry");
312            PatternRegistry::new()
313        },
314    )
315});
