18/// A fingerprint represents a hash of a code fragment
19#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
20pub struct Fingerprint(u64);

22impl Fingerprint {

27    }

31    pub fn value(&self) -> u64 {

49/// A match between two fingerprinted locations

51pub struct FingerprintMatch {

60/// Token fingerprinter using Rabin-Karp rolling hash
61///
62/// Uses a sliding window approach to generate fingerprints for code fragments.
63/// Fingerprints are stored in a hash map for O(1) duplicate lookup.
64pub struct TokenFingerprinter {
65    /// Window size in tokens
66    window_size: usize,
67    /// Base for the polynomial hash
68    base: u64,
69    /// Modulus for the hash (large prime)
70    modulus: u64,
71    /// Precomputed base^(window_size-1) mod modulus
72    base_power: u64,
73    /// Map from fingerprint to locations
74    fingerprint_map: HashMap<Fingerprint, Vec<FingerprintLocation>>,
75}
76
77impl TokenFingerprinter {
78    /// Create a new fingerprinter with the given window size
79    #[must_use]
80    pub fn new(window_size: usize) -> Self {
81        let base: u64 = RABIN_KARP_BASE;
82        let modulus: u64 = RABIN_KARP_MODULUS;
83
84        // Precompute base^(window_size-1) mod modulus
85        let base_power = Self::mod_pow(base, window_size.saturating_sub(1) as u64, modulus);
86
87        Self {
88            window_size,
89            base,
90            modulus,
91            base_power,
92            fingerprint_map: HashMap::new(),
93        }
94    }
95
96    /// Modular exponentiation: base^exp mod modulus
97    fn mod_pow(mut base: u64, mut exp: u64, modulus: u64) -> u64 {
98        let mut result: u64 = 1;
99        base %= modulus;
100        while exp > 0 {
101            if exp % 2 == 1 {
102                result = result.wrapping_mul(base) % modulus;
103            }
104            exp /= 2;
105            base = base.wrapping_mul(base) % modulus;
106        }
107        result
108    }
109
110    /// Convert a token to a numeric value for hashing
111    fn token_value(token: &str) -> u64 {
112        let mut hasher = std::collections::hash_map::DefaultHasher::new();
113        token.hash(&mut hasher);
114        hasher.finish()
115    }
116
117    /// Compute the initial hash for a window of tokens
118    fn initial_hash(&self, tokens: &[&str]) -> u64 {
119        let mut hash: u64 = 0;
120        for token in tokens.iter().take(self.window_size) {
121            let val = Self::token_value(token) % self.modulus;
122            hash = (hash.wrapping_mul(self.base) + val) % self.modulus;
123        }
124        hash
125    }
126
127    /// Compute the rolling hash by removing old token and adding new token
128    fn rolling_hash(&self, current_hash: u64, old_token: &str, new_token: &str) -> u64 {
129        let old_val = Self::token_value(old_token) % self.modulus;
130        let new_val = Self::token_value(new_token) % self.modulus;
131
132        // Remove old token contribution: hash - old_val * base^(n-1)
133        let mut hash = current_hash;
134        let old_contribution = old_val.wrapping_mul(self.base_power) % self.modulus;
135        hash = (hash + self.modulus - old_contribution) % self.modulus;
136
137        // Shift and add new token: hash * base + new_val
138        hash = (hash.wrapping_mul(self.base) + new_val) % self.modulus;
139        hash
140    }
141
142    /// Fingerprint a file's tokens and add to the map
143    pub fn fingerprint_file(&mut self, file: &std::path::Path, tokens: &[Token]) {
144        if tokens.len() < self.window_size {
145            return;
146        }
147
148        let token_strs: Vec<&str> = tokens.iter().map(|t| t.text.as_str()).collect();
149
150        // Compute initial hash
151        let mut hash = self.initial_hash(&token_strs[..self.window_size]);
152
153        // Store first fingerprint
154        let fingerprint = Fingerprint::new(hash);
155        let location = FingerprintLocation {
156            file: file.to_path_buf(),
157            start_line: tokens[0].line,
158            end_line: tokens[self.window_size - 1].line,
159            token_count: self.window_size,
160        };
161        self.fingerprint_map
162            .entry(fingerprint)
163            .or_default()
164            .push(location);
165
166        // Roll through remaining tokens
167        for i in 1..=(tokens.len() - self.window_size) {
168            hash = self.rolling_hash(
169                hash,
170                token_strs[i - 1],
171                token_strs[i + self.window_size - 1],
172            );
173
174            let fingerprint = Fingerprint::new(hash);
175            let location = FingerprintLocation {
176                file: file.to_path_buf(),
177                start_line: tokens[i].line,
178                end_line: tokens[i + self.window_size - 1].line,
179                token_count: self.window_size,
180            };
181            self.fingerprint_map
182                .entry(fingerprint)
183                .or_default()
184                .push(location);
185        }
186    }
187
188    /// Find all duplicate fingerprints
189    #[must_use]
190    pub fn find_duplicates(&self) -> Vec<FingerprintMatch> {
191        let mut matches = Vec::new();
192
193        for (fingerprint, locations) in &self.fingerprint_map {
194            if locations.len() < 2 {
195                continue;
196            }
197
198            // Generate pairs of duplicates
199            for i in 0..locations.len() {
200                for j in (i + 1)..locations.len() {
201                    let loc1 = &locations[i];
202                    let loc2 = &locations[j];
203
204                    // Skip if in the same file and overlapping
205                    if loc1.file == loc2.file {
206                        let overlap = lines_overlap(
207                            loc1.start_line,
208                            loc1.end_line,
209                            loc2.start_line,
210                            loc2.end_line,
211                        );
212                        if overlap {
213                            continue;
214                        }
215                    }
216
217                    matches.push(FingerprintMatch {
218                        location1: loc1.clone(),
219                        location2: loc2.clone(),
220                        fingerprint: *fingerprint,
221                    });
222                }
223            }
224        }
225
226        matches
227    }
228
229    /// Clear all stored fingerprints
230    pub fn clear(&mut self) {
231        self.fingerprint_map.clear();
232    }
233
234    /// Get statistics about stored fingerprints
235    pub fn stats(&self) -> FingerprintStats {
236        let total_fingerprints = self.fingerprint_map.len();
237        let total_locations: usize = self.fingerprint_map.values().map(std::vec::Vec::len).sum();
238        let duplicates: usize = self
239            .fingerprint_map
240            .values()
241            .filter(|v| v.len() > 1)
242            .count();
243
244        FingerprintStats {
245            total_fingerprints,
246            total_locations,
247            unique_fingerprints: total_fingerprints - duplicates,
248            duplicate_fingerprints: duplicates,
249        }
250    }
251}

253/// A simple token representation
254#[derive(Debug, Clone)]
255pub struct Token {
256    /// Token text
257    pub text: String,
258    /// Line number (1-based)
259    pub line: usize,
260    /// Column number (1-based)
261    pub column: usize,
262    /// Token type (for normalized comparison)
263    pub token_type: TokenType,
264}
265
266/// Token type for normalization
267#[derive(Debug, Clone, Copy, PartialEq, Eq)]
268pub enum TokenType {
269    /// Identifier (variable, function, type name)
270    Identifier,
271    /// Keyword (if, for, while, etc.)
272    Keyword,
273    /// Literal (string, number, etc.)
274    Literal,
275    /// Operator (+, -, *, etc.)
276    Operator,
277    /// Punctuation ({, }, (, ), etc.)
278    Punctuation,
279    /// Comment
280    Comment,
281    /// Whitespace
282    Whitespace,
283    /// Other/unknown
284    Other,
285}
286
287impl Token {
288    /// Create a new token
289    #[must_use]
290    pub fn new(text: String, line: usize, column: usize, token_type: TokenType) -> Self {
291        Self {
292            text,
293            line,
294            column,
295            token_type,
296        }
297    }
298
299    /// Normalize the token for Type 2 (renamed) clone detection
300    ///
301    /// Replaces identifiers with a placeholder while keeping structure
302    #[must_use]
303    pub fn normalized_text(&self) -> &str {
304        match self.token_type {
305            TokenType::Identifier => NORMALIZED_IDENTIFIER,
306            TokenType::Literal => NORMALIZED_LITERAL,
307            TokenType::Keyword
308            | TokenType::Operator
309            | TokenType::Punctuation
310            | TokenType::Comment
311            | TokenType::Whitespace
312            | TokenType::Other => &self.text,
313        }
314    }
315}
