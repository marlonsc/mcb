4//! Clean Architecture Validation
5//!
6//! Validates strict Clean Architecture compliance:
7//! - Domain layer contains only traits and types (minimal implementations)
8//! - Handlers use dependency injection (no direct service creation)
9//! - Port implementations have linkme provider registration
10//! - Entities have identity fields
11//! - Value objects are immutable
12//! - Server layer boundaries are respected
13
14use std::path::PathBuf;
15
16use crate::Severity;
17use crate::define_violations;
18use crate::traits::violation::ViolationCategory;
19
20define_violations! {
21    dynamic_severity,
22    ViolationCategory::Architecture,
23    pub enum CleanArchitectureViolation {
24        /// Domain layer contains implementation logic
25        #[violation(
26            id = "CA001",
27            severity = Warning,
28            message = "Domain layer contains {impl_type} at {file}:{line}",
29            suggestion = "Move implementation logic to providers or infrastructure layer"
30        )]
31        DomainContainsImplementation {
32            file: PathBuf,
33            line: usize,
34            impl_type: String,
35            severity: Severity,
36        },
37        /// Handler creates service directly instead of using DI
38        #[violation(
39            id = "CA002",
40            severity = Warning,
41            message = "Handler creates {service_name} directly at {file}:{line} - {context}",
42            suggestion = "Inject service via constructor injection instead of creating directly"
43        )]
44        HandlerCreatesService {
45            file: PathBuf,
46            line: usize,
47            service_name: String,
48            context: String,
49            severity: Severity,
50        },
51        /// Port implementation missing linkme provider registration
52        #[violation(
53            id = "CA003",
54            severity = Warning,
55            message = "{struct_name} implements {trait_name} but missing proper DI interface registration at {file}:{line}",
56            suggestion = "Add proper DI component registration for {trait_name}"
57        )]
58        PortMissingComponentDerive {
59            file: PathBuf,
60            line: usize,
61            struct_name: String,
62            trait_name: String,
63            severity: Severity,
64        },
65        /// Entity missing identity field
66        #[violation(
67            id = "CA004",
68            severity = Warning,
69            message = "Entity {entity_name} missing id/uuid field at {file}:{line}",
70            suggestion = "Add id: Uuid or similar identity field to entity"
71        )]
72        EntityMissingIdentity {
73            file: PathBuf,
74            line: usize,
75            entity_name: String,
76            severity: Severity,
77        },
78        /// Value object has mutable method
79        #[violation(
80            id = "CA005",
81            severity = Warning,
82            message = "Value object {vo_name} has mutable method {method_name} at {file}:{line}",
83            suggestion = "Value objects should be immutable - return new instance instead"
84        )]
85        ValueObjectMutable {
86            file: PathBuf,
87            line: usize,
88            vo_name: String,
89            method_name: String,
90            severity: Severity,
91        },
92        /// Server imports provider directly
93        #[violation(
94            id = "CA006",
95            severity = Warning,
96            message = "Server imports provider directly: {import_path} at {file}:{line}",
97            suggestion = "Import providers through infrastructure re-exports"
98        )]
99        ServerImportsProviderDirectly {
100            file: PathBuf,
101            line: usize,
102            import_path: String,
103            severity: Severity,
104        },
105        /// Infrastructure layer imports concrete service from Application
106        ///
107        /// CA007: Infrastructure should only import trait interfaces, not concrete types.
108        #[violation(
109            id = "CA007",
110            severity = Error,
111            message = "CA007: Infrastructure imports concrete service {concrete_type} at {file}:{line}",
112            suggestion = "Import only trait interfaces from Application, not concrete implementations"
113        )]
114        InfrastructureImportsConcreteService {
115            file: PathBuf,
116            line: usize,
117            import_path: String,
118            concrete_type: String,
119            severity: Severity,
120        },
121        /// Application layer imports ports from wrong location
122        ///
123        /// CA008: Application should import ports from the domain crate, not locally.
124        #[violation(
125            id = "CA008",
126            severity = Error,
127            message = "CA008: Application imports from {import_path} but should import from {should_be} at {file}:{line}",
128            suggestion = "Import ports from {should_be} instead"
129        )]
130        ApplicationWrongPortImport {
131            file: PathBuf,
132            line: usize,
133            import_path: String,
134            should_be: String,
135            severity: Severity,
136        },
137        /// Infrastructure layer imports from Application layer
138        ///
139        /// CA009: Infrastructure should NOT depend on Application layer.
140        /// Per Clean Architecture, the dependency flow is:
141        /// Server -> Infrastructure -> Domain
142        ///               |                  ^
143        ///               v                  |
144        ///          Providers ---------> Application
145        ///
146        /// Infrastructure importing from Application creates circular dependencies.
147        #[violation(
148            id = "CA009",
149            severity = Error,
150            message = "CA009: Infrastructure imports from Application layer: {import_path} at {file}:{line} - violates Clean Architecture dependency direction",
151            suggestion = "{suggestion}"
152        )]
153        InfrastructureImportsApplication {
154            file: PathBuf,
155            line: usize,
156            import_path: String,
157            suggestion: String,
158            severity: Severity,
159        },
160    }
161}
