4use std::path::Path;
5
6use regex::Regex;
7
8use super::super::violation::NamingViolation;
9use crate::constants::common::COMMENT_PREFIX;
10use crate::traits::violation::Severity;
11use crate::utils::naming::is_camel_case;
12
13pub fn validate_type_names(
14    path: &Path,
15    content: &str,
16    struct_pattern: &Regex,
17    enum_pattern: &Regex,
18    trait_pattern: &Regex,
19) -> Vec<NamingViolation> {
20    let mut violations = Vec::new();
21    let type_patterns = [struct_pattern, enum_pattern, trait_pattern];
22
23    for (line_num, line) in content.lines().enumerate() {
24        let trimmed = line.trim();
25        if trimmed.starts_with(COMMENT_PREFIX) {
26            continue;
27        }
28
29        for pattern in type_patterns {
30            if let Some(cap) = pattern.captures(line) {
31                let name = cap.get(1).map_or("", |m| m.as_str());
32                if is_camel_case(name) {
33                    continue;
34                }
35                violations.push(NamingViolation::BadTypeName {
36                    file: path.to_path_buf(),
37                    line: line_num + 1,
38                    name: name.to_owned(),
39                    expected_case: "CamelCase".to_owned(),
40                    severity: Severity::Warning,
41                });
42            }
43        }
44    }
45    violations
46}
