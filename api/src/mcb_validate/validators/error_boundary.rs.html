4//! Error Boundary Validation

27define_violations! {
28    dynamic_severity,
29    ViolationCategory::ErrorBoundary,
30    pub enum ErrorBoundaryViolation {
31        /// Error crossing layer without context
32        #[violation(
33            id = "ERR001",
34            severity = Warning,
35            message = "Missing error context: {file}:{line} - {error_pattern} ({suggestion})",
36            suggestion = "{suggestion}"
37        )]
38        MissingErrorContext {
39            file: PathBuf,
40            line: usize,
41            error_pattern: String,
42            suggestion: String,
43            severity: Severity,
44        },
45        /// Infrastructure error type used in domain layer
46        #[violation(
47            id = "ERR002",
48            severity = Warning,
49            message = "Wrong layer error: {file}:{line} - {error_type} in {layer}",
50            suggestion = "Wrap {error_type} in a domain error type instead of using it directly in {layer}"
51        )]
52        WrongLayerError {
53            file: PathBuf,
54            line: usize,
55            error_type: String,
56            layer: String,
57            severity: Severity,
58        },
59        /// Internal error details leaked to external API
60        #[violation(
61            id = "ERR003",
62            severity = Error,
63            message = "Leaked internal error: {file}:{line} - {pattern}",
64            suggestion = "Replace {pattern} with a sanitized error response that doesn't expose internal details"
65        )]
66        LeakedInternalError {
67            file: PathBuf,
68            line: usize,
69            pattern: String,
70            severity: Severity,
71        },
72    }
73}

87impl ErrorBoundaryValidator {
88    fn scan_relevant_lines<FileFilter, LineHandler>(
89        config: &ValidationConfig,
90        file_filter: FileFilter,
91        mut line_handler: LineHandler,
92    ) -> Result<()>
93    where
94        FileFilter: Fn(&Path, &str) -> bool,
95        LineHandler: FnMut(&PathBuf, usize, &str, &str),
96    {
97        for_each_scan_file(config, Some(LanguageId::Rust), false, |entry, _src_dir| {
98            let path = &entry.absolute_path;
99            let Some(path_str) = path.to_str() else {
100                return Ok(());
101            };
102            let normalized_path = path_str.replace('\\', "/");
103
104            if normalized_path.contains(TEST_DIR_FRAGMENT)
105                || !file_filter(path, normalized_path.as_str())
106            {
107                return Ok(());
108            }
109
110            let content = std::fs::read_to_string(path)?;
111            let mut in_test_module = false;
112
113            for (line_num, line) in content.lines().enumerate() {
114                let trimmed = line.trim();
115
116                if trimmed.starts_with(COMMENT_PREFIX) {
117                    continue;
118                }
119
120                if trimmed.contains(CFG_TEST_MARKER) {
121                    in_test_module = true;
122                    continue;
123                }
124
125                if in_test_module {
126                    continue;
127                }
128
129                line_handler(path, line_num + 1, line, trimmed);
130            }
131
132            Ok(())
133        })
134    }

136    /// Detects error propagation without context (missing `.context()` or `.map_err()`)
137    ///
138    /// # Errors
139    ///
140    /// Returns an error if regex compilation or source file reading fails.
141    pub fn validate_error_context(
142        config: &ValidationConfig,
143    ) -> Result<Vec<ErrorBoundaryViolation>> {
144        let mut violations = Vec::new();
145
146        // Pattern: ? operator without .context() or .with_context()
147        // This is a heuristic - we look for lines with ? but no context method
148        let question_mark_pattern = compile_regex(r"\?\s*;?\s*$")?;
149        let context_pattern = compile_regex(r"\.(context|with_context|map_err|ok_or_else)\s*\(")?;
150
151        // Files that are likely error boundary crossing points
152        let boundary_paths = [ARCH_PATH_HANDLERS, ARCH_PATH_ADAPTERS, ARCH_PATH_SERVICES];
153
154        Self::scan_relevant_lines(
155            config,
156            |_path, path_str| boundary_paths.iter().any(|p| path_str.contains(p)),
157            |path, line_num, _line, trimmed| {
158                if question_mark_pattern.is_match(trimmed)
159                    && !context_pattern.is_match(trimmed)
160                    && !trimmed.starts_with("return ")
161                    && !trimmed.contains("Ok(")
162                {
163                    violations.push(ErrorBoundaryViolation::MissingErrorContext {
164                        file: path.clone(),
165                        line: line_num,
166                        error_pattern: trimmed.chars().take(SHORT_PREVIEW_LENGTH).collect(),
167                        suggestion: "Add .context() or .map_err() for better error messages"
168                            .to_owned(),
169                        severity: Severity::Info,
170                    });
171                }
172            },
173        )?;
174
175        Ok(violations)
176    }

178    /// Detects infrastructure error types used in domain layer (layer boundary violation)
179    ///
180    /// # Errors
181    ///
182    /// Returns an error if regex compilation or source file reading fails.
183    pub fn validate_layer_error_types(
184        config: &ValidationConfig,
185    ) -> Result<Vec<ErrorBoundaryViolation>> {
186        let mut violations = Vec::new();
187
188        // Infrastructure error types that shouldn't appear in domain
189        let infra_errors = [
190            ("std::io::Error", "std::io::Error"),
191            ("reqwest::Error", "reqwest::Error"),
192            ("sqlx::Error", "sqlx::Error"),
193            ("tokio::.*Error", "tokio Error"),
194            ("hyper::Error", "hyper::Error"),
195            ("serde_json::Error", "serde_json::Error"),
196        ];
197
198        let compiled_errors = compile_regex_pairs(&infra_errors)?;
199
200        Self::scan_relevant_lines(
201            config,
202            |path, path_str| {
203                if !path_str.contains(ARCH_PATH_DOMAIN) {
204                    return false;
205                }
206                let file_name = path.file_name().and_then(|n| n.to_str()).unwrap_or("");
207                file_name != ERROR_MODULE_FILE && !file_name.starts_with(ERROR_FILE_PREFIX)
208            },
209            |path, line_num, line, _trimmed| {
210                for (pattern, desc) in &compiled_errors {
211                    if pattern.is_match(line) {
212                        violations.push(ErrorBoundaryViolation::WrongLayerError {
213                            file: path.clone(),
214                            line: line_num,
215                            error_type: desc.to_string(),
216                            layer: "domain".to_owned(),
217                            severity: Severity::Warning,
218                        });
219                    }
220                }
221            },
222        )?;
223
224        Ok(violations)
225    }

227    /// Detects internal error details leaked to API responses (information disclosure)
228    ///
229    /// # Errors
230    ///
231    /// Returns an error if regex compilation or source file reading fails.
232    pub fn validate_leaked_errors(
233        config: &ValidationConfig,
234    ) -> Result<Vec<ErrorBoundaryViolation>> {
235        let mut violations = Vec::new();
236
237        // Patterns that indicate internal errors being exposed
238        let leak_patterns = [
239            (
240                r#"format!\s*\(\s*"\{\:?\?\}""#,
241                "Debug formatting in response",
242            ),
243            (
244                r"\b(?:err|error|e)\b\.to_string\(\)",
245                "Error .to_string() in response",
246            ),
247            (
248                r#"serde_json::json!\s*\(\s*\{\s*"error"\s*:\s*format!"#,
249                "Internal error in JSON response",
250            ),
251        ];
252
253        let compiled_leaks = compile_regex_pairs(&leak_patterns)?;
254
255        Self::scan_relevant_lines(
256            config,
257            |_path, path_str| {
258                path_str.contains(ARCH_PATH_HANDLERS) || path_str.contains(HANDLER_FILE_SUFFIX)
259            },
260            |path, line_num, line, _trimmed| {
261                for (pattern, desc) in &compiled_leaks {
262                    if pattern.is_match(line) {
263                        violations.push(ErrorBoundaryViolation::LeakedInternalError {
264                            file: path.clone(),
265                            line: line_num,
266                            pattern: desc.to_string(),
267                            severity: Severity::Info,
268                        });
269                    }
270                }
271            },
272        )?;
273
274        Ok(violations)
275    }
