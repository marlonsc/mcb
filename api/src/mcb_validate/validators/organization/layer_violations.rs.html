19/// Checks for violations of Clean Architecture layer boundaries.
20///
21/// Detects issues such as:
22/// - Server layer code directly instantiating services (bypassing DI).
23/// - Application layer code importing from the server layer (dependency inversion violation).
24///
25/// # Errors
26///
27/// Returns an error if file scanning or reading fails.
28pub fn validate_layer_violations(config: &ValidationConfig) -> Result<Vec<OrganizationViolation>> {
29    let mut violations = Vec::new();
30
31    // Patterns for detecting layer violations
32    let arc_new_service_pattern = compile_regex(ARC_NEW_SERVICE_REGEX)?;
33    let server_import_pattern = compile_regex(SERVER_IMPORT_REGEX)?;
34
35    for_each_scan_file(config, Some(LanguageId::Rust), true, |entry, _src_dir| {
36        let path = &entry.absolute_path;
37        let Some(path_str) = path.to_str() else {
38            return Ok(());
39        };
40        if is_test_path(path_str) {
41            return Ok(());
42        }
43
44        let content = std::fs::read_to_string(path)?;
45        scan_layer_file(
46            path,
47            path_str,
48            &content,
49            (&arc_new_service_pattern, &server_import_pattern),
50            &mut violations,
51        );
52        Ok(())
53    })?;
54
55    Ok(violations)
56}

58/// Scan one file for layer-boundary violations: server-layer service creation
59/// and application/infrastructure imports from the server layer.
60fn scan_layer_file(
61    path: &Path,
62    path_str: &str,
63    content: &str,
64    patterns: (&Regex, &Regex),
65    violations: &mut Vec<OrganizationViolation>,
66) {
67    let (arc_new_service_pattern, server_import_pattern) = patterns;
68    let is_server_layer = path_str.contains(SERVER_LAYER_PATH);
69    let is_application_layer = path_str.contains(APPLICATION_LAYER_PATH);
70    let is_infrastructure_layer = path_str.contains(INFRASTRUCTURE_LAYER_PATH);
71
72    let file_name = path.file_name().and_then(|n| n.to_str()).unwrap_or("");
73    let bypass_server_creation = SERVICE_CREATION_BYPASS_FILES
74        .iter()
75        .any(|f| file_name.contains(f));
76
77    crate::validators::for_each_non_test_non_comment_line(content, |line_num, line, trimmed| {
78        if is_server_layer
79            && !bypass_server_creation
80            && let Some(cap) = arc_new_service_pattern.captures(line)
81        {
82            let service_name = cap.get(1).map_or("", |m| m.as_str());
83            violations.push(OrganizationViolation::ServerCreatingServices {
84                file: path.to_path_buf(),
85                line: line_num + 1,
86                service_name: service_name.to_owned(),
87                suggestion: "Use DI container to resolve services".to_owned(),
88                severity: Severity::Warning,
89            });
90        }
91
92        if (is_application_layer || is_infrastructure_layer)
93            && server_import_pattern.is_match(line)
94            && !trimmed.contains(PUB_USE_PREFIX)
95        {
96            violations.push(OrganizationViolation::ApplicationImportsServer {
97                file: path.to_path_buf(),
98                line: line_num + 1,
99                import_statement: trimmed.to_owned(),
100                severity: Severity::Warning,
101            });
102        }
103    });
104}
