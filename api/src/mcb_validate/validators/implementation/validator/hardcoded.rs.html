4use std::path::PathBuf;
5
6use regex::Regex;
7
8use super::super::violation::ImplementationViolation;
9use crate::Result;
10use crate::traits::violation::Severity;
11use crate::utils::source::{
12    compile_pattern_pairs, extract_functions, is_fn_signature_or_brace, non_test_lines,
13};
14use crate::validators::implementation::constants::{
15    HARDCODED_RETURN_PATTERNS, STUB_SKIP_FILE_KEYWORDS,
16};
17
18/// Detect hardcoded return values
19pub fn validate_hardcoded_returns(
20    files: &[(PathBuf, String)],
21    fn_pattern: &Regex,
22) -> Result<Vec<ImplementationViolation>> {
23    let compiled = compile_pattern_pairs(HARDCODED_RETURN_PATTERNS)?;
24    let mut violations = Vec::new();
25
26    for (file_path, content) in files {
27        let fname = file_path.file_name().and_then(|n| n.to_str()).unwrap_or("");
28        if STUB_SKIP_FILE_KEYWORDS.iter().any(|k| fname.contains(k)) {
29            continue;
30        }
31
32        let non_test_lines = non_test_lines(&content.lines().collect::<Vec<_>>());
33
34        for func in extract_functions(Some(fn_pattern), &non_test_lines)
35            .into_iter()
36            .filter(|func| !func.has_control_flow)
37        {
38            for line in func
39                .body_lines
40                .iter()
41                .filter(|line| !is_fn_signature_or_brace(line))
42            {
43                for (_pattern, desc) in compiled
44                    .iter()
45                    .filter(|(pattern, _)| pattern.is_match(line))
46                {
47                    violations.push(ImplementationViolation::HardcodedReturnValue {
48                        file: file_path.clone(),
49                        line: func.start_line,
50                        method_name: func.name.clone(),
51                        return_value: desc.to_string(),
52                        severity: Severity::Warning,
53                    });
54                }
55            }
56        }
57    }
58    Ok(violations)
59}
