4use std::path::PathBuf;
5
6use super::super::violation::ImplementationViolation;
7use crate::Result;
8use crate::traits::violation::Severity;
9use crate::utils::source::{required_patterns, source_lines};
10
11pub fn validate_empty_catch_alls(
12    files: &[(PathBuf, String)],
13) -> Result<Vec<ImplementationViolation>> {
14    let catchall_ids = [
15        "IMPL001.catchall_empty",
16        "IMPL001.catchall_unit",
17        "IMPL001.catchall_ok_unit",
18        "IMPL001.catchall_none",
19        "IMPL001.catchall_continue",
20    ];
21
22    let compiled = required_patterns(catchall_ids.iter().copied())?;
23    let mut violations = Vec::new();
24
25    for (file_path, content) in files {
26        for (line_num, trimmed) in source_lines(content) {
27            for pattern in &compiled {
28                if pattern.is_match(trimmed) {
29                    violations.push(ImplementationViolation::EmptyCatchAll {
30                        file: file_path.clone(),
31                        line: line_num,
32                        context: trimmed.to_owned(),
33                        severity: Severity::Warning,
34                    });
35                }
36            }
37        }
38    }
39    Ok(violations)
40}
