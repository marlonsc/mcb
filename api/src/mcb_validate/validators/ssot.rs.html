15    name: mcb_utils::constants::validate::VALIDATOR_SSOT,
16    description: "Detects duplicate declarations and forbidden legacy schema/repository references",
17
18    /// Validator for single-source-of-truth invariants.
19    pub struct SsotValidator {
20        config: ValidationConfig,
21    }
22
23    violations: dynamic_severity, ViolationCategory::Organization,
24    pub enum SsotViolation {
25        #[doc = "Duplicate declaration of the same public type or trait."]
26        #[violation(
27            id = "SSOT001",
28            severity = Error,
29            message = "Duplicate port declaration '{declaration_name}' found across files: {locations}",
30            suggestion = "Keep a single declaration under the canonical ports tree and remove duplicates"
31        )]
32        DuplicatePortDeclaration {
33            file: PathBuf,
34            line: usize,
35            declaration_name: String,
36            locations: String,
37            severity: Severity,
38        },
39        #[doc = "Forbidden import of the legacy repositories compatibility path."]
40        #[violation(
41            id = "SSOT002",
42            severity = Error,
43            message = "Forbidden legacy import: {file}:{line} uses '{import_path}'",
44            suggestion = "Replace with mcb_domain::ports::{TypeName} imports"
45        )]
46        ForbiddenLegacyImport {
47            file: PathBuf,
48            line: usize,
49            import_path: String,
50            severity: Severity,
51        },
52        #[doc = "Forbidden legacy schema symbol usage."]
53        #[violation(
54            id = "SSOT003",
55            severity = Error,
56            message = "Forbidden legacy schema symbol: {file}:{line} uses '{symbol_name}'",
57            suggestion = "Remove legacy schema symbols and use the canonical schema SSOT APIs"
58        )]
59        ForbiddenLegacySchemaSymbol {
60            file: PathBuf,
61            line: usize,
62            symbol_name: String,
63            severity: Severity,
64        },
65        #[doc = "Forbidden macro path referencing `schema::memory` internals."]
66        #[violation(
67            id = "SSOT004",
68            severity = Error,
69            message = "Forbidden schema macro path: {file}:{line} uses '{macro_path}'",
70            suggestion = "Stop referencing $crate::schema::memory::* in macros"
71        )]
72        ForbiddenSchemaMemoryMacroPath {
73            file: PathBuf,
74            line: usize,
75            macro_path: String,
76            severity: Severity,
77        },
78        #[doc = "Forbidden legacy schema import path."]
79        #[violation(
80            id = "SSOT005",
81            severity = Error,
82            message = "Forbidden legacy schema import: {file}:{line} uses '{import_path}'",
83            suggestion = "Remove imports of MemorySchema/ProjectSchema and migrate to canonical schema modules"
84        )]
85        ForbiddenLegacySchemaImport {
86            file: PathBuf,
87            line: usize,
88            import_path: String,
89            severity: Severity,
90        },
91        #[doc = "Forbidden root-level schema path usage."]
92        #[violation(
93            id = "SSOT006",
94            severity = Error,
95            message = "Forbidden root schema path: {file}:{line} uses '{path}'",
96            suggestion = "Use canonical mcb_domain::schema::* paths instead of root-level schema symbols"
97        )]
98        ForbiddenRootSchemaPath {
99            file: PathBuf,
100            line: usize,
101            path: String,
102            severity: Severity,
103        },
104        #[doc = "Forbidden raw ID field type in domain entities/value objects."]
105        #[violation(
106            id = "SSOT007",
107            severity = Error,
108            message = "Forbidden raw ID field type: {file}:{line} uses '{field_type}' for '{field_name}'",
109            suggestion = "Use a strong-typed ID from mcb_domain::value_objects::ids via define_id!"
110        )]
111        ForbiddenRawIdFieldType {
112            file: PathBuf,
113            line: usize,
114            field_name: String,
115            field_type: String,
116            severity: Severity,
117        },
118    }
119
120    checks: [
121        run_all_checks => Self::analyze_workspace
122    ]
123}

125const DECLARATION_PATTERN: &str = r"\bpub\s+(?:trait|struct)\s+([A-Z][A-Za-z0-9_]*)\b";
126const LEGACY_IMPORT_PATTERN: &str = r"\b(?:pub\s+)?use\s+mcb_domain::repositories::";
127const FORBIDDEN_SCHEMA_SYMBOL: &str = r"\b(ProjectSchema|MemorySchema|MemorySchemaDdlGenerator)\b";
128const FORBIDDEN_SCHEMA_MACRO_PATH: &str = r"\$crate::schema::memory::";
129const FORBIDDEN_SCHEMA_IMPORT_PATTERN: &str =
130    r"\b(?:pub\s+)?use\s+[^;]*\b(ProjectSchema|MemorySchema)\b";
131const FORBIDDEN_ROOT_SCHEMA_PATH_PATTERN: &str = r"\bmcb_domain::(Schema|SchemaDdlGenerator|ColumnDef|ColumnType|TableDef|IndexDef|FtsDef|ForeignKeyDef|UniqueConstraintDef|COL_OBSERVATION_TYPE)\b";
132const FORBIDDEN_ROOT_SCHEMA_IMPORT_PATTERN: &str = r"\b(?:pub\s+)?use\s+mcb_domain::\{[^}]*\b(Schema|SchemaDdlGenerator|ColumnDef|ColumnType|TableDef|IndexDef|FtsDef|ForeignKeyDef|UniqueConstraintDef|COL_OBSERVATION_TYPE)\b[^}]*\}";
133const FORBIDDEN_RAW_ID_FIELD_PATTERN: &str = r"\bpub\s+([a-z_][a-z0-9_]*)\s*:\s*(String|Uuid)\b";

137impl SsotValidator {
138    /// Overrides the generated `validate_all` method because it needs custom file gathering.
139    ///
140    /// # Errors
141    /// Returns an error if workspace scanning or regex compilation fails.
142    pub fn analyze_workspace(&self) -> Result<Vec<SsotViolation>> {
143        let mut files = Vec::new();
144
145        for_each_scan_file(
146            &self.config,
147            Some(LanguageId::Rust),
148            false,
149            |entry, _candidate_src_dir| {
150                let content = std::fs::read_to_string(&entry.absolute_path)?;
151                files.push((entry.absolute_path.clone(), content));
152                Ok(())
153            },
154        )?;
155
156        Self::analyze_files(files)
157    }
158
159    /// Validate an in-memory synthetic file map (used by unit tests).
160    ///
161    /// # Errors
162    /// Returns an error if regex compilation fails.
163    pub fn validate_synthetic_files(files: &HashMap<String, String>) -> Result<Vec<SsotViolation>> {
164        let synthetic_files = files
165            .iter()
166            .map(|(path, content)| (PathBuf::from(path), content.clone()))
167            .collect::<Vec<_>>();
168        Self::analyze_files(synthetic_files)
169    }
170
171    fn analyze_files(mut files: Vec<(PathBuf, String)>) -> Result<Vec<SsotViolation>> {
172        let patterns = SsotPatterns::compile()?;
173        files.sort_by(|a, b| a.0.cmp(&b.0));
174
175        let mut declaration_locations: BTreeMap<String, Vec<(PathBuf, usize)>> = BTreeMap::new();
176        let mut violations = Vec::new();
177
178        for (path, content) in &files {
179            for (line_index, line) in content.lines().enumerate() {
180                Self::analyze_line(
181                    &patterns,
182                    path,
183                    line,
184                    line_index + 1,
185                    &mut declaration_locations,
186                    &mut violations,
187                );
188            }
189        }
190
191        push_duplicate_declarations(&mut violations, declaration_locations);
192
193        Ok(violations)
194    }
195
196    /// Apply every SSOT pattern to one line, recording declarations and pushing
197    /// any forbidden-pattern violations.
198    fn analyze_line(
199        patterns: &SsotPatterns,
200        path: &Path,
201        line: &str,
202        line_number: usize,
203        declaration_locations: &mut BTreeMap<String, Vec<(PathBuf, usize)>>,
204        violations: &mut Vec<SsotViolation>,
205    ) {
206        record_declarations(
207            declaration_locations,
208            &patterns.declaration,
209            line,
210            path,
211            line_number,
212        );
213
214        push_line_match(violations, &patterns.legacy_import, line, |import_path| {
215            SsotViolation::ForbiddenLegacyImport {
216                file: path.to_path_buf(),
217                line: line_number,
218                import_path,
219                severity: Severity::Error,
220            }
221        });
222
223        Self::check_schema_patterns(patterns, path, line, line_number, violations);
224        Self::check_root_schema_patterns(patterns, path, line, line_number, violations);
225
226        if is_domain_model_file(path) {
227            push_raw_id_field_violations(
228                violations,
229                &patterns.forbidden_raw_id_field,
230                path,
231                line,
232                line_number,
233            );
234        }
235    }
236
237    /// Flag legacy schema symbol/macro/import usages on a single line.
238    fn check_schema_patterns(
239        patterns: &SsotPatterns,
240        path: &Path,
241        line: &str,
242        line_number: usize,
243        violations: &mut Vec<SsotViolation>,
244    ) {
245        push_capture_group(
246            violations,
247            &patterns.forbidden_schema_symbol,
248            line,
249            1,
250            |symbol_name| SsotViolation::ForbiddenLegacySchemaSymbol {
251                file: path.to_path_buf(),
252                line: line_number,
253                symbol_name: symbol_name.to_owned(),
254                severity: Severity::Error,
255            },
256        );
257
258        push_line_match(
259            violations,
260            &patterns.forbidden_schema_macro_path,
261            line,
262            |macro_path| SsotViolation::ForbiddenSchemaMemoryMacroPath {
263                file: path.to_path_buf(),
264                line: line_number,
265                macro_path,
266                severity: Severity::Error,
267            },
268        );
269
270        push_line_match(
271            violations,
272            &patterns.forbidden_schema_import,
273            line,
274            |import_path| SsotViolation::ForbiddenLegacySchemaImport {
275                file: path.to_path_buf(),
276                line: line_number,
277                import_path,
278                severity: Severity::Error,
279            },
280        );
281    }
282
283    /// Flag root-schema path references on a single line.
284    fn check_root_schema_patterns(
285        patterns: &SsotPatterns,
286        path: &Path,
287        line: &str,
288        line_number: usize,
289        violations: &mut Vec<SsotViolation>,
290    ) {
291        push_line_match(
292            violations,
293            &patterns.forbidden_root_schema_import,
294            line,
295            |path_text| SsotViolation::ForbiddenRootSchemaPath {
296                file: path.to_path_buf(),
297                line: line_number,
298                path: path_text,
299                severity: Severity::Error,
300            },
301        );
302
303        push_capture_group(
304            violations,
305            &patterns.forbidden_root_schema_path,
306            line,
307            0,
308            |path_text| SsotViolation::ForbiddenRootSchemaPath {
309                file: path.to_path_buf(),
310                line: line_number,
311                path: path_text.to_owned(),
312                severity: Severity::Error,
313            },
314        );
315    }
316}

318/// Compiled regexes for the SSOT scan.
319struct SsotPatterns {
320    declaration: regex::Regex,
321    legacy_import: regex::Regex,
322    forbidden_schema_symbol: regex::Regex,
323    forbidden_schema_macro_path: regex::Regex,
324    forbidden_schema_import: regex::Regex,
325    forbidden_root_schema_path: regex::Regex,
326    forbidden_root_schema_import: regex::Regex,
327    forbidden_raw_id_field: regex::Regex,
328}
329
330impl SsotPatterns {
331    /// Compile every SSOT pattern.
332    ///
333    /// # Errors
334    ///
335    /// Returns an error if any pattern fails to compile.
336    fn compile() -> Result<Self> {
337        Ok(Self {
338            declaration: compile_regex(DECLARATION_PATTERN)?,
339            legacy_import: compile_regex(LEGACY_IMPORT_PATTERN)?,
340            forbidden_schema_symbol: compile_regex(FORBIDDEN_SCHEMA_SYMBOL)?,
341            forbidden_schema_macro_path: compile_regex(FORBIDDEN_SCHEMA_MACRO_PATH)?,
342            forbidden_schema_import: compile_regex(FORBIDDEN_SCHEMA_IMPORT_PATTERN)?,
343            forbidden_root_schema_path: compile_regex(FORBIDDEN_ROOT_SCHEMA_PATH_PATTERN)?,
344            forbidden_root_schema_import: compile_regex(FORBIDDEN_ROOT_SCHEMA_IMPORT_PATTERN)?,
345            forbidden_raw_id_field: compile_regex(FORBIDDEN_RAW_ID_FIELD_PATTERN)?,
346        })
347    }
348}

350fn is_domain_model_file(path: &std::path::Path) -> bool {
351    let path = path.to_string_lossy().replace('\\', "/");
352    path.contains("mcb-domain/src/entities/") || path.contains("mcb-domain/src/value_objects/")
353}
354
355fn push_line_match(
356    violations: &mut Vec<SsotViolation>,
357    pattern: &regex::Regex,
358    line: &str,
359    make: impl Fn(String) -> SsotViolation,
360) {
361    if pattern.is_match(line) {
362        violations.push(make(line.trim().to_owned()));
363    }
364}
365
366fn push_capture_group(
367    violations: &mut Vec<SsotViolation>,
368    pattern: &regex::Regex,
369    line: &str,
370    group: usize,
371    make: impl Fn(&str) -> SsotViolation,
372) {
373    for cap in pattern.captures_iter(line) {
374        if let Some(m) = cap.get(group) {
375            violations.push(make(m.as_str()));
376        }
377    }
378}

380fn push_duplicate_declarations(
381    violations: &mut Vec<SsotViolation>,
382    declaration_locations: BTreeMap<String, Vec<(PathBuf, usize)>>,
383) {
384    for (declaration_name, locations) in declaration_locations {
385        let unique_paths = locations
386            .iter()
387            .map(|(path, _line)| path.clone())
388            .collect::<HashSet<_>>();
389
390        if unique_paths.len() < 2 {
391            continue;
392        }
393
394        let mut sorted_locations = locations;
395        sorted_locations.sort_by(|a, b| a.0.cmp(&b.0).then(a.1.cmp(&b.1)));
396
397        let Some((file, line)) = sorted_locations.first().cloned() else {
398            continue;
399        };
400
401        let locations_text = sorted_locations
402            .iter()
403            .map(|(path, location_line)| format!("{}:{location_line}", path.display()))
404            .collect::<Vec<_>>()
405            .join(", ");
406
407        violations.push(SsotViolation::DuplicatePortDeclaration {
408            file,
409            line,
410            declaration_name,
411            locations: locations_text,
412            severity: Severity::Error,
413        });
414    }
415}
416
417fn record_declarations(
418    declaration_locations: &mut BTreeMap<String, Vec<(PathBuf, usize)>>,
419    declaration_pattern: &regex::Regex,
420    line: &str,
421    path: &Path,
422    line_number: usize,
423) {
424    for cap in declaration_pattern.captures_iter(line) {
425        if let Some(name_match) = cap.get(1) {
426            declaration_locations
427                .entry(name_match.as_str().to_owned())
428                .or_default()
429                .push((path.to_path_buf(), line_number));
430        }
431    }
432}
433
434fn push_raw_id_field_violations(
435    violations: &mut Vec<SsotViolation>,
436    forbidden_raw_id_field_pattern: &regex::Regex,
437    path: &Path,
438    line: &str,
439    line_number: usize,
440) {
441    for cap in forbidden_raw_id_field_pattern.captures_iter(line) {
442        let Some(field_name_match) = cap.get(1) else {
443            continue;
444        };
445        let Some(field_type_match) = cap.get(2) else {
446            continue;
447        };
448
449        if !field_name_match.as_str().ends_with("id") {
450            continue;
451        }
452
453        violations.push(SsotViolation::ForbiddenRawIdFieldType {
454            file: path.to_path_buf(),
455            line: line_number,
456            field_name: field_name_match.as_str().to_owned(),
457            field_type: field_type_match.as_str().to_owned(),
458            severity: Severity::Error,
459        });
460    }
