14/// Category of violation for grouping in reports
15#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize)]
16pub enum ViolationCategory {
17    /// Architecture and layer boundaries
18    Architecture,
19    /// Code quality (unwrap, expect, panic)
20    Quality,
21    /// File organization and placement
22    Organization,
23    /// SOLID principles
24    Solid,
25    /// Dependency injection patterns
26    DependencyInjection,
27    /// Configuration patterns
28    Configuration,
29    /// Web framework patterns
30    WebFramework,
31    /// Performance patterns
32    Performance,
33    /// Asynchronous programming patterns
34    Async,
35    /// Documentation quality
36    Documentation,
37    /// Test hygiene and quality
38    Testing,
39    /// Naming conventions
40    Naming,
41    /// KISS principle (simplicity)
42    Kiss,
43    /// Refactoring completeness
44    Refactoring,
45    /// Error handling boundaries
46    ErrorBoundary,
47    /// Implementation details
48    Implementation,
49    /// Project Management and Tracking integration
50    Pmat,
51}

53impl Display for ViolationCategory {
54    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
55        match self {
56            Self::Architecture => write!(f, "Architecture"),
57            Self::Quality => write!(f, "Quality"),
58            Self::Organization => write!(f, "Organization"),
59            Self::Solid => write!(f, "SOLID"),
60            Self::DependencyInjection => write!(f, "DI/linkme"),
61            Self::Configuration => write!(f, "Configuration"),
62            Self::WebFramework => write!(f, "Web Framework"),
63            Self::Performance => write!(f, "Performance"),
64            Self::Async => write!(f, "Async"),
65            Self::Documentation => write!(f, "Documentation"),
66            Self::Testing => write!(f, "Testing"),
67            Self::Naming => write!(f, "Naming"),
68            Self::Kiss => write!(f, "KISS"),
69            Self::Refactoring => write!(f, "Refactoring"),
70            Self::ErrorBoundary => write!(f, "Error Boundary"),
71            Self::Implementation => write!(f, "Implementation"),
72            Self::Pmat => write!(f, "PMAT"),
73        }
74    }
75}

77/// Generic violation trait - all violations implement this
78///
79/// This trait provides a unified interface for handling violations
80/// across all validator types, enabling generic reporting and processing.
81pub trait Violation: Display + Send + Sync {
82    /// Unique violation ID (e.g., "DEP001", "QUAL002")
83    fn id(&self) -> &str;
84
85    /// Category for grouping in reports
86    fn category(&self) -> ViolationCategory;
87
88    /// Severity level
89    fn severity(&self) -> Severity;
90
91    /// File where violation occurred (if applicable)
92    fn file(&self) -> Option<&PathBuf>;
93
94    /// Line number where violation occurred (if applicable)
95    fn line(&self) -> Option<usize>;
96
97    /// Human-readable message describing the violation
98    fn message(&self) -> String {
99        self.to_string()
100    }
101
102    /// Suggested fix for the violation (if applicable)
103    fn suggestion(&self) -> Option<String> {
104        None
105    }
106
107    /// Convert to a boxed trait object for dynamic dispatch
108    fn boxed(self) -> Box<dyn Violation>
109    where
110        Self: Sized + 'static,
111    {
112        Box::new(self)
113    }
114}
