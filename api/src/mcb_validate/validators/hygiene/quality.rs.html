38/// Validates test quality by checking for trivial assertions, unwrap-only tests, and comment-only tests.
39///
40/// # Errors
41///
42/// Returns an error if regex compilation, directory enumeration, or file reading fails.
43pub fn validate_test_quality(config: &ValidationConfig) -> Result<Vec<HygieneViolation>> {
44    let mut violations = Vec::new();
45
46    let patterns = QualityPatterns {
47        mock_type: compile_regex(r"\bMock[A-Za-z0-9_]+\b")?,
48        skip_message: compile_regex("skipping:")?,
49        todo: compile_regex(r"\btodo!\(")?,
50        unimplemented: compile_regex(r"\bunimplemented!\(")?,
51    };
52
53    // Trivial assertion patterns
54    let trivial_patterns = [
55        (r"assert!\s*\(\s*true\s*\)", "assert!(true)"),
56        (r"assert!\s*\(\s*!false\s*\)", "assert!(!false)"),
57        (
58            r"assert_eq!\s*\(\s*true\s*,\s*true\s*\)",
59            "assert_eq!(true, true)",
60        ),
61        (r"assert_eq!\s*\(\s*1\s*,\s*1\s*\)", "assert_eq!(1, 1)"),
62        (r"assert_ne!\s*\(\s*1\s*,\s*2\s*\)", "assert_ne!(1, 2)"),
63        (
64            r"assert_ne!\s*\(\s*true\s*,\s*false\s*\)",
65            "assert_ne!(true, false)",
66        ),
67    ];
68
69    let compiled_trivial = compile_regex_pairs(&trivial_patterns)?;
70
71    let test_attr_pattern = compile_regex(r"#\[(?:tokio::)?test\]")?;
72    let fn_pattern = compile_regex(r"(?:async\s+)?fn\s+([a-z_][a-z0-9_]*)\s*\(")?;
73    // Match common assertion macros - allow leading whitespace for indented code
74    // The pattern checks for assertions at the start of a line (with optional whitespace)
75    // or preceded by whitespace/punctuation to avoid false positives like "some_assert!"
76    let real_assert_pattern = compile_regex(
77        r"(?:^|\s)(assert!|assert_eq!|assert_ne!|assert_matches!|debug_assert!|debug_assert_eq!|debug_assert_ne!|panic!)",
78    )?;
79    let unwrap_pattern = compile_regex(r"\.unwrap\(|\.expect\(")?;
80    let scan_input = QualityScanInput {
81        test_attr_pattern: &test_attr_pattern,
82        fn_pattern: &fn_pattern,
83        real_assert_pattern: &real_assert_pattern,
84        unwrap_pattern: &unwrap_pattern,
85        compiled_trivial: &compiled_trivial,
86    };
87
88    for crate_dir in config.get_source_dirs()? {
89        let tests_dir = crate_dir.join("tests");
90        if !tests_dir.exists() {
91            continue;
92        }
93
94        for_each_file_under_root(config, &tests_dir, Some(LanguageId::Rust), |entry| {
95            let path = &entry.absolute_path;
96            if path.to_str().is_some_and(|s| s.contains("/fixtures/")) {
97                return Ok(());
98            }
99
100            let content = std::fs::read_to_string(path)?;
101            let lines: Vec<&str> = content.lines().collect();
102
103            violations.extend(process_quality_file(path, &lines, &patterns, &scan_input));
104            Ok(())
105        })?;
106    }
107
108    Ok(violations)
109}

111fn find_next_test_fn(
112    lines: &[&str],
113    start_idx: usize,
114    fn_pattern: &Regex,
115) -> Option<(usize, String)> {
116    lines
117        .iter()
118        .enumerate()
119        .skip(start_idx)
120        .find_map(|(line_idx, candidate)| {
121            let captures = fn_pattern.captures(candidate)?;
122            let fn_name = captures.get(1).map_or("", |m| m.as_str());
123            Some((line_idx, fn_name.to_owned()))
124        })
125}

127fn process_quality_file(
128    path: &std::path::Path,
129    lines: &[&str],
130    patterns: &QualityPatterns,
131    scan_input: &QualityScanInput<'_>,
132) -> Vec<HygieneViolation> {
133    let mut violations = Vec::new();
134    check_forbidden_patterns(path, lines, patterns, &mut violations);
135
136    for (line_idx, line) in lines.iter().enumerate() {
137        if line.trim().starts_with(MODULE_DOC_PREFIX)
138            || !scan_input.test_attr_pattern.is_match(line)
139        {
140            continue;
141        }
142
143        let Some((fn_line_idx, fn_name)) =
144            find_next_test_fn(lines, line_idx + 1, scan_input.fn_pattern)
145        else {
146            continue;
147        };
148
149        let input = TestBodyInput {
150            path,
151            lines,
152            fn_line_idx,
153            fn_name: &fn_name,
154            real_assert_pattern: scan_input.real_assert_pattern,
155            unwrap_pattern: scan_input.unwrap_pattern,
156            compiled_trivial: scan_input.compiled_trivial,
157        };
158        violations.extend(analyze_test_function_body(&input));
159    }
160
161    violations
162}

164fn analyze_test_function_body(input: &TestBodyInput<'_>) -> Vec<HygieneViolation> {
165    let Some((body_lines, _)) = crate::scan::extract_balanced_block(input.lines, input.fn_line_idx)
166    else {
167        return Vec::new();
168    };
169
170    let mut violations = Vec::new();
171    let mut has_assertion = false;
172    let mut has_unwrap = false;
173    let mut has_code = false;
174
175    for (offset, body_line) in body_lines.iter().enumerate() {
176        let body_line_idx = input.fn_line_idx + offset;
177        let trimmed = body_line.trim();
178        if should_skip_body_line(trimmed) {
179            continue;
180        }
181
182        has_code = true;
183        if input.real_assert_pattern.is_match(trimmed) {
184            has_assertion = true;
185        }
186        if input.unwrap_pattern.is_match(trimmed) {
187            has_unwrap = true;
188        }
189
190        for (regex, desc) in input.compiled_trivial {
191            if regex.is_match(trimmed) {
192                violations.push(HygieneViolation::TrivialAssertion {
193                    file: input.path.to_path_buf(),
194                    line: body_line_idx + 1,
195                    function_name: input.fn_name.to_owned(),
196                    assertion: (*desc).to_owned(),
197                    severity: Severity::Warning,
198                });
199            }
200        }
201    }
202
203    if !has_code {
204        violations.push(HygieneViolation::CommentOnlyTest {
205            file: input.path.to_path_buf(),
206            line: input.fn_line_idx + 1,
207            function_name: input.fn_name.to_owned(),
208            severity: Severity::Warning,
209        });
210    } else if !has_assertion && has_unwrap {
211        violations.push(HygieneViolation::UnwrapOnlyAssertion {
212            file: input.path.to_path_buf(),
213            line: input.fn_line_idx + 1,
214            function_name: input.fn_name.to_owned(),
215            severity: Severity::Warning,
216        });
217    }
218
219    violations
220}
221
222fn should_skip_body_line(trimmed: &str) -> bool {
223    trimmed.is_empty()
224        || trimmed.starts_with(COMMENT_PREFIX)
225        || trimmed.starts_with(FN_PREFIX)
226        || matches!(trimmed, "{" | "}")
227}

229/// Checks for forbidden patterns in test files.
230/// Checks for forbidden patterns in test files.
231fn check_forbidden_patterns(
232    file: &std::path::Path,
233    lines: &[&str],
234    patterns: &QualityPatterns,
235    violations: &mut Vec<HygieneViolation>,
236) {
237    for (idx, line) in lines.iter().enumerate() {
238        let line_no = idx + 1;
239
240        for mat in patterns.mock_type.find_iter(line) {
241            violations.push(HygieneViolation::MockTypeUsage {
242                file: file.to_path_buf(),
243                line: line_no,
244                token: mat.as_str().to_owned(),
245                severity: Severity::Error,
246            });
247        }
248
249        if patterns.skip_message.is_match(line) {
250            violations.push(HygieneViolation::SkipBranchUsage {
251                file: file.to_path_buf(),
252                line: line_no,
253                severity: Severity::Error,
254            });
255        }
256
257        if patterns.todo.is_match(line) {
258            violations.push(HygieneViolation::StubMacroUsage {
259                file: file.to_path_buf(),
260                line: line_no,
261                macro_name: "todo".to_owned(),
262                severity: Severity::Error,
263            });
264        }
265
266        if patterns.unimplemented.is_match(line) {
267            violations.push(HygieneViolation::StubMacroUsage {
268                file: file.to_path_buf(),
269                line: line_no,
270                macro_name: "unimplemented".to_owned(),
271                severity: Severity::Error,
272            });
273        }
274    }
275
276    for idx in 0..lines.len().saturating_sub(2) {
277        let current = lines[idx].trim();
278        let next = lines[idx + 1].trim();
279        let next2 = lines[idx + 2].trim();
280        if current.starts_with("let Ok(") && next.starts_with("else") && next2.contains("return;") {
281            violations.push(HygieneViolation::SkipBranchUsage {
282                file: file.to_path_buf(),
283                line: idx + 1,
284                severity: Severity::Error,
285            });
286        }
287    }
