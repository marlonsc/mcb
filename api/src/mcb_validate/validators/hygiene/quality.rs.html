21struct QualityScanInput<'a> {
22    test_attr_pattern: &'a Regex,
23    fn_pattern: &'a Regex,
24    real_assert_pattern: &'a Regex,
25    unwrap_pattern: &'a Regex,
26    compiled_trivial: &'a [(Regex, &'a str)],
27}

29/// Owns the compiled regexes used to scan test files so they outlive the
30/// borrowed [`QualityScanInput`] view.
31struct QualityScanRegexes {
32    test_attr_pattern: Regex,
33    fn_pattern: Regex,
34    real_assert_pattern: Regex,
35    unwrap_pattern: Regex,
36    compiled_trivial: Vec<(Regex, &'static str)>,
37}
38
39impl QualityScanRegexes {
40    /// Trivial assertion patterns (regex, human-readable label).
41    const TRIVIAL_PATTERNS: [(&'static str, &'static str); 6] = [
42        (r"assert!\s*\(\s*true\s*\)", "assert!(true)"),
43        (r"assert!\s*\(\s*!false\s*\)", "assert!(!false)"),
44        (
45            r"assert_eq!\s*\(\s*true\s*,\s*true\s*\)",
46            "assert_eq!(true, true)",
47        ),
48        (r"assert_eq!\s*\(\s*1\s*,\s*1\s*\)", "assert_eq!(1, 1)"),
49        (r"assert_ne!\s*\(\s*1\s*,\s*2\s*\)", "assert_ne!(1, 2)"),
50        (
51            r"assert_ne!\s*\(\s*true\s*,\s*false\s*\)",
52            "assert_ne!(true, false)",
53        ),
54    ];
55
56    /// Compile every scan regex.
57    ///
58    /// # Errors
59    ///
60    /// Returns an error if any pattern fails to compile.
61    fn compile() -> Result<Self> {
62        Ok(Self {
63            test_attr_pattern: compile_regex(r"#\[(?:tokio::)?test\]")?,
64            fn_pattern: compile_regex(r"(?:async\s+)?fn\s+([a-z_][a-z0-9_]*)\s*\(")?,
65            // Match common assertion macros, allowing leading whitespace/punctuation
66            // to avoid false positives like "some_assert!".
67            real_assert_pattern: compile_regex(
68                r"(?:^|\s)(assert[a-z_]*!|assert_[a-z_]+\(|debug_assert[a-z_]*!|panic!)",
69            )?,
70            unwrap_pattern: compile_regex(r"\.unwrap\(|\.expect\(")?,
71            compiled_trivial: compile_regex_pairs(&Self::TRIVIAL_PATTERNS)?,
72        })
73    }
74
75    /// Borrow these regexes as a [`QualityScanInput`].
76    fn as_scan_input(&self) -> QualityScanInput<'_> {
77        QualityScanInput {
78            test_attr_pattern: &self.test_attr_pattern,
79            fn_pattern: &self.fn_pattern,
80            real_assert_pattern: &self.real_assert_pattern,
81            unwrap_pattern: &self.unwrap_pattern,
82            compiled_trivial: &self.compiled_trivial,
83        }
84    }
85}

91    fn_name: &'a str,

93    unwrap_pattern: &'a Regex,
94    compiled_trivial: &'a [(Regex, &'a str)],
95}

97/// Validates test quality by checking for trivial assertions, unwrap-only tests, and comment-only tests.
98///
99/// # Errors
100///
101/// Returns an error if regex compilation, directory enumeration, or file reading fails.
102pub fn validate_test_quality(config: &ValidationConfig) -> Result<Vec<HygieneViolation>> {
103    let mut violations = Vec::new();
104
105    let patterns = QualityPatterns {
106        mock_type: compile_regex(r"\bMock[A-Za-z0-9_]+\b")?,
107        skip_message: compile_regex("skipping:")?,
108        todo: compile_regex(r"\btodo!\(")?,
109        unimplemented: compile_regex(r"\bunimplemented!\(")?,
110    };
111
112    let scan_regexes = QualityScanRegexes::compile()?;
113    let scan_input = scan_regexes.as_scan_input();
114
115    for crate_dir in config.get_source_dirs()? {
116        let tests_dir = crate_dir.join("tests");
117        if !tests_dir.exists() {
118            continue;
119        }
120
121        for_each_file_under_root(config, &tests_dir, Some(LanguageId::Rust), |entry| {
122            let path = &entry.absolute_path;
123            if path.to_str().is_some_and(|s| s.contains("/fixtures/")) {
124                return Ok(());
125            }
126
127            let content = std::fs::read_to_string(path)?;
128            let lines: Vec<&str> = content.lines().collect();
129
130            violations.extend(process_quality_file(path, &lines, &patterns, &scan_input));
131            Ok(())
132        })?;
133    }
134
135    Ok(violations)
136}

138fn find_next_test_fn(
139    lines: &[&str],
140    start_idx: usize,
141    fn_pattern: &Regex,
142) -> Option<(usize, String)> {
143    lines
144        .iter()
145        .enumerate()
146        .skip(start_idx)
147        .find_map(|(line_idx, candidate)| {
148            let captures = fn_pattern.captures(candidate)?;
149            let fn_name = captures.get(1).map_or("", |m| m.as_str());
150            Some((line_idx, fn_name.to_owned()))
151        })
152}

154fn process_quality_file(
155    path: &std::path::Path,
156    lines: &[&str],
157    patterns: &QualityPatterns,
158    scan_input: &QualityScanInput<'_>,
159) -> Vec<HygieneViolation> {
160    let mut violations = Vec::new();
161    check_forbidden_patterns(path, lines, patterns, &mut violations);
162
163    for (line_idx, line) in lines.iter().enumerate() {
164        if line.trim().starts_with(MODULE_DOC_PREFIX)
165            || !scan_input.test_attr_pattern.is_match(line)
166        {
167            continue;
168        }
169
170        let Some((fn_line_idx, fn_name)) =
171            find_next_test_fn(lines, line_idx + 1, scan_input.fn_pattern)
172        else {
173            continue;
174        };
175
176        let input = TestBodyInput {
177            path,
178            lines,
179            fn_line_idx,
180            fn_name: &fn_name,
181            real_assert_pattern: scan_input.real_assert_pattern,
182            unwrap_pattern: scan_input.unwrap_pattern,
183            compiled_trivial: scan_input.compiled_trivial,
184        };
185        violations.extend(analyze_test_function_body(&input));
186    }
187
188    violations
189}

191/// Tracks which kinds of meaningful code a test body contains.
192#[derive(Default)]
193struct TestBodyFacts {
194    has_assertion: bool,
195    has_unwrap: bool,
196    has_code: bool,
197}
198
199fn analyze_test_function_body(input: &TestBodyInput<'_>) -> Vec<HygieneViolation> {
200    let Some((body_lines, _)) = crate::scan::extract_balanced_block(input.lines, input.fn_line_idx)
201    else {
202        return Vec::new();
203    };
204
205    let mut violations = Vec::new();
206    let mut facts = TestBodyFacts::default();
207
208    for (offset, body_line) in body_lines.iter().enumerate() {
209        let trimmed = body_line.trim();
210        if should_skip_body_line(trimmed) {
211            continue;
212        }
213        scan_test_body_line(
214            input,
215            trimmed,
216            input.fn_line_idx + offset,
217            &mut facts,
218            &mut violations,
219        );
220    }
221
222    if !facts.has_code {
223        violations.push(HygieneViolation::CommentOnlyTest {
224            file: input.path.to_path_buf(),
225            line: input.fn_line_idx + 1,
226            function_name: input.fn_name.to_owned(),
227            severity: Severity::Warning,
228        });
229    } else if !facts.has_assertion && facts.has_unwrap {
230        violations.push(HygieneViolation::UnwrapOnlyAssertion {
231            file: input.path.to_path_buf(),
232            line: input.fn_line_idx + 1,
233            function_name: input.fn_name.to_owned(),
234            severity: Severity::Warning,
235        });
236    }
237
238    violations
239}

241/// Inspect one meaningful body line, updating `facts` and pushing a
242/// `TrivialAssertion` violation for each trivial-assert pattern matched.
243fn scan_test_body_line(
244    input: &TestBodyInput<'_>,
245    trimmed: &str,
246    body_line_idx: usize,
247    facts: &mut TestBodyFacts,
248    violations: &mut Vec<HygieneViolation>,
249) {
250    facts.has_code = true;
251    if input.real_assert_pattern.is_match(trimmed) {
252        facts.has_assertion = true;
253    }
254    if input.unwrap_pattern.is_match(trimmed) {
255        facts.has_unwrap = true;
256    }
257
258    for (regex, desc) in input.compiled_trivial {
259        if regex.is_match(trimmed) {
260            violations.push(HygieneViolation::TrivialAssertion {
261                file: input.path.to_path_buf(),
262                line: body_line_idx + 1,
263                function_name: input.fn_name.to_owned(),
264                assertion: (*desc).to_owned(),
265                severity: Severity::Warning,
266            });
267        }
268    }
269}
270
271fn should_skip_body_line(trimmed: &str) -> bool {
272    trimmed.is_empty()
273        || trimmed.starts_with(COMMENT_PREFIX)
274        || trimmed.starts_with(FN_PREFIX)
275        || matches!(trimmed, "{" | "}")
276}

278/// Checks for forbidden patterns in test files (mock types, skip messages, stub
279/// macros, and `let Ok(..) else { return; }` skip branches).
280fn check_forbidden_patterns(
281    file: &std::path::Path,
282    lines: &[&str],
283    patterns: &QualityPatterns,
284    violations: &mut Vec<HygieneViolation>,
285) {
286    for (idx, line) in lines.iter().enumerate() {
287        check_forbidden_line(file, idx + 1, line, patterns, violations);
288    }
289    check_skip_branches(file, lines, violations);
290}

292/// Push violations for forbidden tokens (mock types, skip messages, stub macros)
293/// found on a single line.
294fn check_forbidden_line(
295    file: &std::path::Path,
296    line_no: usize,
297    line: &str,
298    patterns: &QualityPatterns,
299    violations: &mut Vec<HygieneViolation>,
300) {
301    for mat in patterns.mock_type.find_iter(line) {
302        violations.push(HygieneViolation::MockTypeUsage {
303            file: file.to_path_buf(),
304            line: line_no,
305            token: mat.as_str().to_owned(),
306            severity: Severity::Error,
307        });
308    }
309
310    if patterns.skip_message.is_match(line) {
311        violations.push(HygieneViolation::SkipBranchUsage {
312            file: file.to_path_buf(),
313            line: line_no,
314            severity: Severity::Error,
315        });
316    }
317
318    for (pattern, macro_name) in [
319        (&patterns.todo, "todo"),
320        (&patterns.unimplemented, "unimplemented"),
321    ] {
322        if pattern.is_match(line) {
323            violations.push(HygieneViolation::StubMacroUsage {
324                file: file.to_path_buf(),
325                line: line_no,
326                macro_name: macro_name.to_owned(),
327                severity: Severity::Error,
328            });
329        }
330    }
331}

333/// Push a `SkipBranchUsage` violation for each `let Ok(..) else { return; }`
334/// skip branch spanning three consecutive lines.
335fn check_skip_branches(
336    file: &std::path::Path,
337    lines: &[&str],
338    violations: &mut Vec<HygieneViolation>,
339) {
340    for idx in 0..lines.len().saturating_sub(2) {
341        let current = lines[idx].trim();
342        let next = lines[idx + 1].trim();
343        let next2 = lines[idx + 2].trim();
344        if current.starts_with("let Ok(") && next.starts_with("else") && next2.contains("return;") {
345            violations.push(HygieneViolation::SkipBranchUsage {
346                file: file.to_path_buf(),
347                line: idx + 1,
348                severity: Severity::Error,
349            });
350        }
351    }
