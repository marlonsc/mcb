10/// Verifies that no inline test declarations exist in src/ directories.
11///
12/// # Errors
13///
14/// Returns an error if pattern loading, directory enumeration, or file reading fails.
15pub fn validate_no_inline_tests(config: &ValidationConfig) -> Result<Vec<HygieneViolation>> {
16    let mut violations = Vec::new();
17    let cfg_test_pattern = required_pattern("TEST001.cfg_test")?;
18    let mod_tests_pattern = required_pattern("TEST001.mod_tests")?;
19    let [test_attr_pattern, tokio_test_attr_pattern] =
20        required_patterns(["TEST001.test_attr", "TEST001.tokio_test_attr"])?
21            .try_into()
22            .map_err(|_| crate::ValidationError::Config("invalid test pattern set".to_owned()))?;
23
24    for_each_rust_file(config, |path, lines| {
25        if path.to_str().is_some_and(|s| s.contains("/fixtures/")) {
26            return Ok(());
27        }
28
29        let mut push_violation = |line: usize| {
30            violations.push(HygieneViolation::InlineTestModule {
31                file: path.clone(),
32                line,
33                severity: Severity::Warning,
34            });
35        };
36
37        let mut last_cfg_test_line: Option<usize> = None;
38        let mut has_inline_module_marker = false;
39
40        for (line_num, line) in lines.iter().enumerate() {
41            let has_recent_cfg =
42                last_cfg_test_line.is_some_and(|cfg_line| line_num <= cfg_line + 5);
43            let is_cfg_test = cfg_test_pattern.is_match(line);
44            let is_orphan_test_mod = mod_tests_pattern.is_match(line) && !has_recent_cfg;
45
46            if is_cfg_test || is_orphan_test_mod {
47                if is_cfg_test {
48                    last_cfg_test_line = Some(line_num);
49                }
50                has_inline_module_marker = true;
51                push_violation(line_num + 1);
52            }
53        }
54
55        if !has_inline_module_marker
56            && let Some((line_num, _)) = lines.iter().enumerate().find(|(_, line)| {
57                test_attr_pattern.is_match(line) || tokio_test_attr_pattern.is_match(line)
58            })
59        {
60            push_violation(line_num + 1);
61        }
62
63        Ok(())
64    })?;
65
66    Ok(violations)
67}
