11/// Naming suggestion for an `integration/` test file lacking a purpose marker.
12fn integration_naming(file_name: &str) -> Option<(String, Severity)> {
13    let ok = ["integration", "workflow"]
14        .iter()
15        .any(|kw| file_name.contains(kw) || file_name.ends_with(&format!("_{kw}")));
16    (!ok).then(|| {
17        (
18            format!(
19                "{file_name}_integration.rs or {file_name}_workflow.rs (integration tests should indicate their purpose)"
20            ),
21            Severity::Info,
22        )
23    })
24}

26/// Naming suggestion for an `e2e/` test file lacking an end-to-end marker.
27fn e2e_naming(file_name: &str) -> Option<(String, Severity)> {
28    let ok = ["e2e", "end_to_end"]
29        .iter()
30        .any(|kw| file_name.contains(kw))
31        || file_name.starts_with("test_");
32    (!ok).then(|| {
33        (
34            format!(
35                "{file_name}_e2e.rs or test_{file_name}.rs (e2e tests should indicate they're end-to-end)"
36            ),
37            Severity::Info,
38        )
39    })
40}

42fn expected_naming_for_parent(
43    parent_dir: &str,
44    file_name: &str,
45    file_full: &str,
46) -> Option<(String, Severity)> {
47    match parent_dir {
48        "unit" => (!file_name.ends_with("_tests")).then(|| {
49            (
50                format!("{file_name}_tests.rs (unit tests must end with _tests)"),
51                Severity::Warning,
52            )
53        }),
54        "integration" => integration_naming(file_name),
55        "e2e" => e2e_naming(file_name),
56        "tests" => (!matches!(
57            file_full,
58            "lib.rs" | "mod.rs" | "utils.rs" | "unit.rs" | "integration.rs" | "e2e.rs"
59        ))
60        .then(|| {
61            (
62                "Move to a subdirectory (e.g., tests/unit/)".to_owned(),
63                Severity::Warning,
64            )
65        }),
66        _other => None,
67    }
68}

70/// Checks test file naming conventions and directory structure compliance.
71///
72/// # Errors
73///
74/// Returns an error if directory enumeration or file scanning fails.
75pub fn validate_test_naming(config: &ValidationConfig) -> Result<Vec<HygieneViolation>> {
76    let mut violations = Vec::new();
77
78    for crate_dir in config.get_source_dirs()? {
79        let tests_dir = crate_dir.join("tests");
80        if !tests_dir.exists() {
81            continue;
82        }
83
84        for_each_file_under_root(config, &tests_dir, Some(LanguageId::Rust), |entry| {
85            if let Some(violation) = check_test_file_naming(&entry.absolute_path) {
86                violations.push(violation);
87            }
88            Ok(())
89        })?;
90    }
91
92    Ok(violations)
93}

95/// Returns a `BadTestFileName` violation when `path` (a test file) does not
96/// follow the naming convention for its parent directory, or `None` for files
97/// that are exempt (entry points, helpers, fixtures, mocks).
98fn check_test_file_naming(path: &std::path::Path) -> Option<HygieneViolation> {
99    let file_name = path.file_stem().and_then(|s| s.to_str()).unwrap_or("");
100
101    if ["lib", "mod"].contains(&file_name) {
102        return None;
103    }
104
105    let path_str = path.to_str()?;
106    if path_str.contains("utils")
107        || ["mock", "fixture", "helper"]
108            .iter()
109            .any(|kw| file_name.contains(kw))
110    {
111        return None;
112    }
113
114    let parent_dir = path
115        .parent()
116        .and_then(|p| p.file_name())
117        .and_then(|n| n.to_str())
118        .unwrap_or("");
119    let file_full = path.file_name().and_then(|n| n.to_str()).unwrap_or("");
120
121    expected_naming_for_parent(parent_dir, file_name, file_full).map(|(suggestion, severity)| {
122        HygieneViolation::BadTestFileName {
123            file: path.to_path_buf(),
124            suggestion,
125            severity,
126        }
127    })
