11const SMOKE_TEST_PATTERNS: [&str; 5] = [
12    "_trait_object",
13    "_exists",
14    "_creation",
15    "_compiles",
16    "_factory",
17];

19struct NamingScanInput<'a> {
20    test_attr_pattern: &'a regex::Regex,
21    tokio_test_pattern: &'a regex::Regex,
22    fn_pattern: &'a regex::Regex,
23    assert_pattern: &'a regex::Regex,
24}
25
26/// Verifies that test functions follow the `test_*` naming pattern.
27///
28/// # Errors
29///
30/// Returns an error if source directory enumeration or file reading fails.
31pub fn validate_test_function_naming(config: &ValidationConfig) -> Result<Vec<HygieneViolation>> {
32    let mut violations = Vec::new();
33    let test_attr_pattern = compile_regex(r"#\[test\]")?;
34    let tokio_test_pattern = compile_regex(r"#\[tokio::test\]")?;
35    let fn_pattern = compile_regex(r"(?:async\s+)?fn\s+([a-z_][a-z0-9_]*)\s*\(")?;
36    let assert_pattern = compile_regex(
37        r"assert!|assert_eq!|assert_ne!|panic!|should_panic|\.unwrap\(|\.expect\(|Box<dyn\s|type_name::",
38    )?;
39    let scan_input = NamingScanInput {
40        test_attr_pattern: &test_attr_pattern,
41        tokio_test_pattern: &tokio_test_pattern,
42        fn_pattern: &fn_pattern,
43        assert_pattern: &assert_pattern,
44    };
45
46    for crate_dir in config.get_source_dirs()? {
47        let tests_dir = crate_dir.join("tests");
48        if !tests_dir.exists() {
49            continue;
50        }
51
52        for_each_file_under_root(config, &tests_dir, Some(LanguageId::Rust), |entry| {
53            let path = &entry.absolute_path;
54            let content = std::fs::read_to_string(path)?;
55            let lines: Vec<&str> = content.lines().collect();
56
57            violations.extend(collect_naming_violations_for_file(
58                path,
59                &lines,
60                &scan_input,
61            ));
62            Ok(())
63        })?;
64    }
65
66    Ok(violations)
67}

69fn is_test_attr_line(
70    line: &str,
71    test_attr_pattern: &regex::Regex,
72    tokio_test_pattern: &regex::Regex,
73) -> bool {
74    test_attr_pattern.is_match(line) || tokio_test_pattern.is_match(line)
75}
76
77fn find_next_test_function(
78    lines: &[&str],
79    start_idx: usize,
80    fn_pattern: &regex::Regex,
81) -> Option<(usize, String)> {
82    lines
83        .iter()
84        .enumerate()
85        .skip(start_idx)
86        .find_map(|(line_idx, potential_fn)| {
87            let captures = fn_pattern.captures(potential_fn)?;
88            let fn_name = captures.get(1).map_or("", |m| m.as_str());
89            Some((line_idx, fn_name.to_owned()))
90        })
91}

93fn append_naming_violation_if_needed(
94    violations: &mut Vec<HygieneViolation>,
95    path: &std::path::Path,
96    fn_line_idx: usize,
97    fn_name: &str,
98) {
99    if fn_name.starts_with("test_") {
100        return;
101    }
102
103    violations.push(HygieneViolation::BadTestFunctionName {
104        file: path.to_path_buf(),
105        line: fn_line_idx + 1,
106        function_name: fn_name.to_owned(),
107        suggestion: format!("test_{fn_name}"),
108        severity: Severity::Warning,
109    });
110}
111
112fn has_assertion_in_test_body(
113    lines: &[&str],
114    fn_line_idx: usize,
115    assert_pattern: &regex::Regex,
116) -> bool {
117    let Some((body_lines, _)) = crate::scan::extract_balanced_block(lines, fn_line_idx) else {
118        return false;
119    };
120
121    body_lines.iter().any(|line| assert_pattern.is_match(line))
122}
123
124fn is_smoke_test(fn_name: &str) -> bool {
125    SMOKE_TEST_PATTERNS
126        .iter()
127        .any(|pattern| fn_name.ends_with(pattern))
128}
129
130fn collect_naming_violations_for_file(
131    path: &std::path::Path,
132    lines: &[&str],
133    scan_input: &NamingScanInput<'_>,
134) -> Vec<HygieneViolation> {
135    let mut violations = Vec::new();
136
137    for (line_idx, line) in lines.iter().enumerate() {
138        if !is_test_attr_line(
139            line,
140            scan_input.test_attr_pattern,
141            scan_input.tokio_test_pattern,
142        ) {
143            continue;
144        }
145
146        let Some((fn_line_idx, fn_name)) =
147            find_next_test_function(lines, line_idx + 1, scan_input.fn_pattern)
148        else {
149            continue;
150        };
151
152        append_naming_violation_if_needed(&mut violations, path, fn_line_idx, &fn_name);
153
154        let has_assertion =
155            has_assertion_in_test_body(lines, fn_line_idx, scan_input.assert_pattern);
156        if !has_assertion && !is_smoke_test(&fn_name) {
157            violations.push(HygieneViolation::TestWithoutAssertion {
158                file: path.to_path_buf(),
159                line: fn_line_idx + 1,
160                function_name: fn_name,
161                severity: Severity::Warning,
162            });
163        }
164    }
165
166    violations
167}
