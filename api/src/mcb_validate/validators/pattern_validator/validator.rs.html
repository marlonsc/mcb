15/// Validates code patterns across the workspace.
16///
17/// checks:
18/// - Async trait usage
19/// - Dependency Injection patterns (Arc<dyn T>)
20/// - Result type usage
21pub struct PatternValidator {
22    config: ValidationConfig,
23    rules: PatternRulesConfig,
24}
25
26crate::impl_rules_validator_new!(PatternValidator, patterns);
27
28impl PatternValidator {
29    /// Creates a new validator with the given configuration.
30    #[must_use]
31    pub fn with_config(config: ValidationConfig, rules: &PatternRulesConfig) -> Self {
32        Self {
33            config,
34            rules: rules.clone(),
35        }
36    }
37
38    /// Runs all pattern validations.
39    ///
40    /// # Errors
41    ///
42    /// Returns an error if file scanning or regex compilation fails.
43    pub fn validate_all(&self) -> Result<Vec<PatternViolation>> {
44        if !self.rules.enabled {
45            return Ok(Vec::new());
46        }
47
48        let mut violations = Vec::new();
49
50        // Compile regex for DI check once
51        let arc_pattern = compile_regex("Arc<([A-Z][a-zA-Z0-9_]*)>")?;
52
53        for_each_crate_file(
54            &self.config,
55            Some(LanguageId::Rust),
56            |entry, _src_dir, _crate_name| {
57                let path = &entry.absolute_path;
58                let content = std::fs::read_to_string(path)?;
59
60                // Async Trait Check
61                violations.extend(check_async_traits(path, &content)?);
62
63                // DI Usage Check
64                violations.extend(check_arc_usage(
65                    path,
66                    &content,
67                    &arc_pattern,
68                    &self.rules.allowed_concrete_types,
69                    &self.rules.provider_trait_suffixes,
70                ));
71
72                // Result Type Check
73                violations.extend(check_result_types(path, &content)?);
74
75                Ok(())
76            },
77        )?;
78
79        Ok(violations)
80    }
81}
