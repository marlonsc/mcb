4use std::path::Path;
5
6use super::violation::PatternViolation;
7use crate::constants::common::ATTR_SEARCH_LINES;
8use crate::pattern_registry::compile_regex;
9use crate::traits::violation::Severity;
10
11/// Checks for async trait usage in a single file.
12pub fn check_async_traits(path: &Path, content: &str) -> crate::Result<Vec<PatternViolation>> {
13    let mut violations = Vec::new();
14    let lines: Vec<&str> = content.lines().collect();
15
16    let trait_pattern = compile_regex(r"pub\s+trait\s+([A-Z][a-zA-Z0-9_]*)")?;
17    let async_fn_pattern = compile_regex(r"async\s+fn\s+")?;
18    let send_sync_pattern = compile_regex(r":\s*.*Send\s*\+\s*Sync")?;
19    let async_trait_attr = compile_regex(r"#\[(async_trait::)?async_trait\]")?;
20    let allow_async_fn_trait = compile_regex(r"#\[allow\(async_fn_in_trait\)\]")?;
21
22    let has_async_methods = |trait_line: usize| {
23        crate::scan::extract_balanced_block(&lines, trait_line).is_some_and(|(body_lines, _)| {
24            body_lines.into_iter().any(|l| async_fn_pattern.is_match(l))
25        })
26    };
27
28    let has_attr_nearby = |line_num: usize, pattern: &regex::Regex| {
29        line_num > 0
30            && lines[..line_num]
31                .iter()
32                .rev()
33                .take(ATTR_SEARCH_LINES)
34                .any(|l| pattern.is_match(l))
35    };
36
37    for (line_num, line) in lines.iter().enumerate() {
38        if let Some(cap) = trait_pattern.captures(line) {
39            let trait_name = cap.get(1).map_or("", |m| m.as_str());
40            if !has_async_methods(line_num) {
41                continue;
42            }
43
44            let uses_native_async = has_attr_nearby(line_num, &allow_async_fn_trait);
45            let has_async_trait_attr =
46                uses_native_async || has_attr_nearby(line_num, &async_trait_attr);
47
48            if !has_async_trait_attr {
49                violations.push(PatternViolation::MissingAsyncTrait {
50                    file: path.to_path_buf(),
51                    line: line_num + 1,
52                    trait_name: trait_name.to_owned(),
53                    severity: Severity::Error,
54                });
55            }
56
57            if !send_sync_pattern.is_match(line) && !uses_native_async {
58                violations.push(PatternViolation::MissingSendSync {
59                    file: path.to_path_buf(),
60                    line: line_num + 1,
61                    trait_name: trait_name.to_owned(),
62                    missing_bound: "Send + Sync".to_owned(),
63                    severity: Severity::Warning,
64                });
65            }
66        }
67    }
68    Ok(violations)
69}
