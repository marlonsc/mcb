4use std::path::Path;
5
6use super::violation::PatternViolation;
7use crate::constants::common::{COMMENT_PREFIX, ERROR_FILE_PREFIX, ERROR_MODULE_FILE, USE_PREFIX};
8use crate::pattern_registry::compile_regex;
9use crate::traits::violation::Severity;
10
11/// Checks for result type usage violations in a single file.
12pub fn check_result_types(path: &Path, content: &str) -> crate::Result<Vec<PatternViolation>> {
13    let mut violations = Vec::new();
14
15    // Pattern to find std::result::Result usage
16    let std_result_pattern = compile_regex("std::result::Result<")?;
17
18    // Pattern to find Result<T, E> with explicit error type (not crate::Result)
19    let explicit_result_pattern = compile_regex(r"Result<[^,]+,\s*([A-Za-z][A-Za-z0-9_:]+)>")?;
20
21    // Skip error-related files (they define/extend error types)
22    let file_name = path.file_name().and_then(|n| n.to_str());
23    let parent_name = path
24        .parent()
25        .and_then(|p| p.file_name())
26        .and_then(|n| n.to_str());
27    if file_name.is_some_and(|n| {
28        n == ERROR_MODULE_FILE || n == "error_ext.rs" || n.starts_with(ERROR_FILE_PREFIX)
29    }) || parent_name.is_some_and(|p| p == ERROR_FILE_PREFIX)
30    {
31        return Ok(violations);
32    }
33
34    for (line_num, line) in content.lines().enumerate() {
35        let trimmed = line.trim();
36
37        // Skip comments and use statements
38        if trimmed.starts_with(COMMENT_PREFIX) || trimmed.starts_with(USE_PREFIX) {
39            continue;
40        }
41
42        // Check for std::result::Result
43        if std_result_pattern.is_match(line) {
44            violations.push(PatternViolation::RawResultType {
45                file: path.to_path_buf(),
46                line: line_num + 1,
47                context: trimmed.to_owned(),
48                suggestion: "crate::Result<T>".to_owned(),
49                severity: Severity::Warning,
50            });
51        }
52
53        // Check for Result<T, SomeError> with explicit error type
54        if let Some(cap) = explicit_result_pattern.captures(line) {
55            let error_type = cap.get(1).map_or("", |m| m.as_str());
56
57            // Allow certain standard error types
58            let allowed_errors = [
59                "Error",
60                "crate::Error",
61                "crate::error::Error",
62                "ValidationError",
63                "std::io::Error",
64                "anyhow::Error",
65            ];
66
67            if !allowed_errors.contains(&error_type)
68                && !error_type.starts_with("crate::")
69                && !error_type.starts_with("self::")
70            {
71                // This is informational - sometimes explicit error types are needed
72                // We won't flag this as a violation for now
73            }
74        }
75    }
76    Ok(violations)
77}
