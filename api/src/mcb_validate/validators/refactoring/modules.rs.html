14/// Check for mod declarations that reference non-existent files
15pub fn validate_mod_declarations(
16    validator: &RefactoringValidator,
17) -> Result<Vec<RefactoringViolation>> {
18    let mut violations = Vec::new();
19    let mod_pattern = compile_regex(r"(?:pub\s+)?mod\s+([a-z_][a-z0-9_]*)(?:\s*;)")?;
20
21    for_each_scan_file(
22        &validator.config,
23        Some(LanguageId::Rust),
24        false,
25        |entry, candidate_src_dir| {
26            if validator.should_skip_crate(candidate_src_dir) {
27                return Ok(());
28            }
29
30            let path = &entry.absolute_path;
31            let content = std::fs::read_to_string(path)?;
32            collect_deleted_module_refs(path, &content, &mod_pattern, &mut violations);
33            Ok(())
34        },
35    )?;
36
37    Ok(violations)
38}

40/// Whether a file backing `mod <mod_name>;` declared in `path` exists, checking
41/// both sibling and nested (`<file_stem>/`) layouts.
42fn module_decl_exists(path: &Path, mod_name: &str) -> bool {
43    let parent_dir = path.parent().unwrap_or(Path::new("."));
44    let file_stem = path.file_stem().and_then(|s| s.to_str());
45
46    let direct = [
47        parent_dir.join(format!("{mod_name}.rs")),
48        parent_dir.join(mod_name).join("mod.rs"),
49    ];
50    let nested = file_stem.map(|stem| {
51        [
52            parent_dir.join(stem).join(format!("{mod_name}.rs")),
53            parent_dir.join(stem).join(mod_name).join("mod.rs"),
54        ]
55    });
56
57    direct.into_iter().any(|p| p.exists())
58        || nested.is_some_and(|paths| paths.into_iter().any(|p| p.exists()))
59}

61/// Push a `DeletedModuleReference` violation for each `mod x;` in `content`
62/// whose backing file no longer exists.
63fn collect_deleted_module_refs(
64    path: &Path,
65    content: &str,
66    mod_pattern: &regex::Regex,
67    violations: &mut Vec<RefactoringViolation>,
68) {
69    for (line_num, line) in content.lines().enumerate() {
70        let Some(cap) = mod_pattern.captures(line) else {
71            continue;
72        };
73        let mod_name = cap.get(1).map_or("", |m| m.as_str());
74        if !module_decl_exists(path, mod_name) {
75            violations.push(RefactoringViolation::DeletedModuleReference {
76                referencing_file: path.to_path_buf(),
77                line: line_num + 1,
78                deleted_module: mod_name.to_owned(),
79                severity: Severity::Warning,
80            });
81        }
82    }
