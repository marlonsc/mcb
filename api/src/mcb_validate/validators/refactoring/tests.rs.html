14fn insert_test_keys(test_files: &mut HashSet<String>, stem: &str) {
15    test_files.insert(stem.to_owned());
16    test_files.insert(format!("{stem}_test"));
17    test_files.insert(format!("{stem}_tests"));
18    for suffix in ["_test", "_tests"] {
19        if let Some(base) = stem.strip_suffix(suffix) {
20            test_files.insert(base.to_owned());
21        }
22    }
23}
24
25fn has_test_key(test_files: &HashSet<String>, stem: &str) -> bool {
26    [stem, &format!("{stem}_test"), &format!("{stem}_tests")]
27        .iter()
28        .any(|name| test_files.contains(*name))
29}

31fn has_test_coverage(
32    relative: &Path,
33    test_files: &HashSet<String>,
34    test_dirs: &HashSet<String>,
35) -> bool {
36    let file_name = relative.file_stem().and_then(|s| s.to_str()).unwrap_or("");
37    if has_test_key(test_files, file_name) {
38        return true;
39    }
40
41    // Tests are organized by module/feature under tests/ (e.g. tests/unit/<module>/),
42    // not 1:1 with source files. A file counts as covered when any of its module
43    // path segments has a matching test directory or file.
44    relative
45        .parent()
46        .into_iter()
47        .flat_map(Path::components)
48        .filter_map(|c| c.as_os_str().to_str())
49        .any(|segment| test_dirs.contains(segment) || has_test_key(test_files, segment))
50}

52fn collect_test_index(
53    config: &crate::ValidationConfig,
54    tests_dir: &Path,
55) -> Result<(HashSet<String>, HashSet<String>)> {
56    let mut test_files: HashSet<String> = HashSet::new();
57    let mut test_dirs: HashSet<String> = HashSet::new();
58
59    for_each_file_under_root(config, tests_dir, Some(LanguageId::Rust), |entry| {
60        let path = &entry.absolute_path;
61        if let Some(stem) = path.file_stem().and_then(|s| s.to_str()) {
62            insert_test_keys(&mut test_files, stem);
63        }
64
65        for dir in path
66            .parent()
67            .into_iter()
68            .flat_map(std::path::Path::ancestors)
69            .take_while(|dir| dir.starts_with(tests_dir))
70        {
71            if let Some(name) = dir.file_name().and_then(|n| n.to_str()) {
72                test_dirs.insert(name.to_owned());
73            }
74        }
75
76        Ok(())
77    })?;
78
79    Ok((test_files, test_dirs))
80}

82/// Check for source files without corresponding test files
83pub fn validate_missing_test_files(
84    validator: &RefactoringValidator,
85) -> Result<Vec<RefactoringViolation>> {
86    let mut violations = Vec::new();
87
88    for crate_dir in validator.get_crate_dirs()? {
89        let src_dir = crate_dir.join("src");
90        let tests_dir = crate_dir.join("tests");
91
92        if !src_dir.exists() || !tests_dir.exists() || validator.should_skip_crate(&crate_dir) {
93            continue;
94        }
95
96        let (test_files, test_dirs) = collect_test_index(&validator.config, &tests_dir)?;
97
98        for_each_file_under_root(
99            &validator.config,
100            &src_dir,
101            Some(LanguageId::Rust),
102            |entry| {
103                if let Some(violation) = missing_test_file_violation(
104                    validator,
105                    &entry.absolute_path,
106                    &src_dir,
107                    &tests_dir,
108                    (&test_files, &test_dirs),
109                )? {
110                    violations.push(violation);
111                }
112                Ok(())
113            },
114        )?;
115    }
116
117    Ok(violations)
118}

120/// Returns a `MissingTestFile` violation when `path` is a non-skipped,
121/// non-inline-tested source file lacking corresponding test coverage.
122///
123/// # Errors
124///
125/// Returns an error if the file content cannot be read.
126fn missing_test_file_violation(
127    validator: &RefactoringValidator,
128    path: &Path,
129    src_dir: &Path,
130    tests_dir: &Path,
131    test_index: (&HashSet<String>, &HashSet<String>),
132) -> Result<Option<RefactoringViolation>> {
133    let (test_files, test_dirs) = test_index;
134    let file_name = path.file_stem().and_then(|s| s.to_str()).unwrap_or("");
135    // skip_files is configured with full names (e.g. "lib.rs"), so the skip check
136    // must compare the file name including its extension.
137    let full_name = path.file_name().and_then(|s| s.to_str()).unwrap_or("");
138    let relative = path.strip_prefix(src_dir).unwrap_or(path);
139    let Some(path_str) = relative.to_str() else {
140        return Ok(None);
141    };
142
143    if validator.skip_files.contains(full_name)
144        || validator
145            .skip_dir_patterns
146            .iter()
147            .any(|pattern| path_str.contains(pattern))
148    {
149        return Ok(None);
150    }
151
152    let content = std::fs::read_to_string(path)?;
153    if content.contains(CFG_TEST_MARKER) || has_test_coverage(relative, test_files, test_dirs) {
154        return Ok(None);
155    }
156
157    Ok(Some(RefactoringViolation::MissingTestFile {
158        source_file: path.to_path_buf(),
159        expected_test: tests_dir.join(format!("{file_name}_test.rs")),
160        severity: Severity::Warning,
161    }))
