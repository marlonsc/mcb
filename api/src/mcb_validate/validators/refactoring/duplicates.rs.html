18/// Check for same type defined in multiple locations
19pub fn validate_duplicate_definitions(
20    validator: &RefactoringValidator,
21) -> Result<Vec<RefactoringViolation>> {
22    let mut violations = Vec::new();
23    let definition_pattern = compile_regex(TYPE_DEFINITION_REGEX)?;
24
25    // Map: type_name -> Vec<file_path>
26    let mut definitions: HashMap<String, Vec<PathBuf>> = HashMap::new();
27    let crates_from_locations = |locations: &[PathBuf]| -> HashSet<String> {
28        locations
29            .iter()
30            .filter_map(|p| {
31                p.to_str()?
32                    .split(CRATE_PATH_DELIMITER)
33                    .nth(1)
34                    .and_then(|s| s.split('/').next())
35                    .map(str::to_owned)
36            })
37            .collect()
38    };
39    let duplicate_dirs_count = |locations: &[PathBuf]| -> usize {
40        locations
41            .iter()
42            .filter_map(|p| p.parent()?.to_str().map(str::to_owned))
43            .collect::<HashSet<_>>()
44            .len()
45    };
46    let duplicate_suggestion = |type_name: &str, crates: &HashSet<String>, severity: Severity| {
47        match severity {
48            Severity::Info => format!(
49                "Type '{type_name}' exists in {crates:?}. This is a known migration pattern - consolidate when migration completes."
50            ),
51            Severity::Warning => format!(
52                "Type '{type_name}' is defined in {crates:?}. Consider consolidating to one location."
53            ),
54            Severity::Error => format!(
55                "Type '{type_name}' is unexpectedly defined in multiple crates: {crates:?}. This requires immediate consolidation."
56            ),
57        }
58    };
59
60    for_each_scan_file(
61        &validator.config,
62        Some(LanguageId::Rust),
63        false,
64        |entry, candidate_src_dir| {
65            if validator.should_skip_crate(candidate_src_dir) {
66                return Ok(());
67            }
68
69            let path = &entry.absolute_path;
70            let Some(path_str) = path.to_str() else {
71                return Ok(());
72            };
73
74            // Skip test files and archived directories
75            if REFACTORING_SKIP_PATTERNS
76                .iter()
77                .any(|p| path_str.contains(p))
78            {
79                return Ok(());
80            }
81
82            let content = std::fs::read_to_string(path)?;
83
84            for cap in definition_pattern.captures_iter(&content) {
85                let type_name = cap.get(1).map_or("", |m| m.as_str());
86
87                // Skip generic names that are expected to appear in multiple places
88                if validator.generic_type_names.contains(type_name) {
89                    continue;
90                }
91
92                definitions
93                    .entry(type_name.to_owned())
94                    .or_default()
95                    .push(path.clone());
96            }
97
98            Ok(())
99        },
100    )?;
101
102    for (type_name, locations) in definitions {
103        if locations.len() <= 1 {
104            continue;
105        }
106
107        let crates = crates_from_locations(&locations);
108
109        if crates.len() > 1 {
110            let severity = categorize_duplicate_severity(validator, &type_name, &crates);
111            violations.push(RefactoringViolation::DuplicateDefinition {
112                type_name: type_name.clone(),
113                locations: locations.clone(),
114                suggestion: duplicate_suggestion(&type_name, &crates, severity),
115                severity,
116            });
117            continue;
118        }
119
120        if duplicate_dirs_count(&locations) >= 2 {
121            violations.push(RefactoringViolation::DuplicateDefinition {
122                type_name: type_name.clone(),
123                locations: locations.clone(),
124                suggestion: format!(
125                    "Type '{}' is defined {} times in different directories within the same crate. Consolidate to single location.",
126                    type_name,
127                    locations.len()
128                ),
129                severity: Severity::Warning,
130            });
131        }
132    }
133
134    Ok(violations)
135}

137/// Categorize duplicate severity based on known patterns
138fn categorize_duplicate_severity(
139    validator: &RefactoringValidator,
140    type_name: &str,
141    crates: &HashSet<String>,
142) -> Severity {
143    // Check if this is an intentionally duplicated utility type
144    if validator.utility_types.contains(type_name) {
145        return Severity::Info;
146    }
147
148    // Check if the crates match a known migration pattern
149    let crate_vec: Vec<&String> = crates.iter().collect();
150    if crate_vec.len() == 2 {
151        for (crate_a, crate_b) in &validator.known_migration_pairs {
152            if (crate_vec[0].as_str() == *crate_a && crate_vec[1].as_str() == *crate_b)
153                || (crate_vec[0].as_str() == *crate_b && crate_vec[1].as_str() == *crate_a)
154            {
155                // This is a known migration pair - downgrade to Info
156                return Severity::Info;
157            }
158        }
159    }
160
161    // Check for patterns that suggest migration in progress
162    // Types ending with Provider, Processor, etc. between known pairs
163    if MIGRATION_TYPE_SUFFIXES
164        .iter()
165        .any(|p| type_name.ends_with(p))
166    {
167        // Check if any known migration pair is involved
168        for (crate_a, crate_b) in &validator.known_migration_pairs {
169            if crates.contains(crate_a) || crates.contains(crate_b) {
170                return Severity::Warning; // Migration-related, but should be tracked
171            }
172        }
173    }
174
175    Severity::Warning
176}
