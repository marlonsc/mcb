16const DECLARATION_PATTERN: &str = r"\bpub\s+(?:trait|struct)\s+([A-Z][A-Za-z0-9_]*)\b";
17const LEGACY_IMPORT_PATTERN: &str = r"\b(?:pub\s+)?use\s+mcb_domain::repositories::";
18const FORBIDDEN_SCHEMA_SYMBOL: &str = r"\b(ProjectSchema|MemorySchema|MemorySchemaDdlGenerator)\b";
19const FORBIDDEN_SCHEMA_MACRO_PATH: &str = r"\$crate::schema::memory::";
20const FORBIDDEN_SCHEMA_IMPORT_PATTERN: &str =
21    r"\b(?:pub\s+)?use\s+[^;]*\b(ProjectSchema|MemorySchema)\b";
22const FORBIDDEN_ROOT_SCHEMA_PATH_PATTERN: &str = r"\bmcb_domain::(Schema|SchemaDdlGenerator|ColumnDef|ColumnType|TableDef|IndexDef|FtsDef|ForeignKeyDef|UniqueConstraintDef|COL_OBSERVATION_TYPE)\b";
23const FORBIDDEN_ROOT_SCHEMA_IMPORT_PATTERN: &str = r"\b(?:pub\s+)?use\s+mcb_domain::\{[^}]*\b(Schema|SchemaDdlGenerator|ColumnDef|ColumnType|TableDef|IndexDef|FtsDef|ForeignKeyDef|UniqueConstraintDef|COL_OBSERVATION_TYPE)\b[^}]*\}";
24const FORBIDDEN_RAW_ID_FIELD_PATTERN: &str = r"\bpub\s+([a-z_][a-z0-9_]*)\s*:\s*(String|Uuid)\b";

26fn push_line_match(
27    violations: &mut Vec<SsotViolation>,
28    pattern: &regex::Regex,
29    line: &str,
30    make: impl Fn(String) -> SsotViolation,
31) {
32    if pattern.is_match(line) {
33        violations.push(make(line.trim().to_owned()));
34    }
35}
36
37fn push_capture_group(
38    violations: &mut Vec<SsotViolation>,
39    pattern: &regex::Regex,
40    line: &str,
41    group: usize,
42    make: impl Fn(&str) -> SsotViolation,
43) {
44    for cap in pattern.captures_iter(line) {
45        if let Some(m) = cap.get(group) {
46            violations.push(make(m.as_str()));
47        }
48    }
49}

51fn push_duplicate_declarations(
52    violations: &mut Vec<SsotViolation>,
53    declaration_locations: BTreeMap<String, Vec<(PathBuf, usize)>>,
54) {
55    for (declaration_name, locations) in declaration_locations {
56        let unique_paths = locations
57            .iter()
58            .map(|(path, _line)| path.clone())
59            .collect::<HashSet<_>>();
60
61        if unique_paths.len() < 2 {
62            continue;
63        }
64
65        let mut sorted_locations = locations;
66        sorted_locations.sort_by(|a, b| a.0.cmp(&b.0).then(a.1.cmp(&b.1)));
67
68        let Some((file, line)) = sorted_locations.first().cloned() else {
69            continue;
70        };
71
72        let locations_text = sorted_locations
73            .iter()
74            .map(|(path, location_line)| format!("{}:{location_line}", path.display()))
75            .collect::<Vec<_>>()
76            .join(", ");
77
78        violations.push(SsotViolation::DuplicatePortDeclaration {
79            file,
80            line,
81            declaration_name,
82            locations: locations_text,
83            severity: Severity::Error,
84        });
85    }
86}

88fn record_declarations(
89    declaration_locations: &mut BTreeMap<String, Vec<(PathBuf, usize)>>,
90    declaration_pattern: &regex::Regex,
91    line: &str,
92    path: &Path,
93    line_number: usize,
94) {
95    for cap in declaration_pattern.captures_iter(line) {
96        if let Some(name_match) = cap.get(1) {
97            declaration_locations
98                .entry(name_match.as_str().to_owned())
99                .or_default()
100                .push((path.to_path_buf(), line_number));
101        }
102    }
103}

105fn push_raw_id_field_violations(
106    violations: &mut Vec<SsotViolation>,
107    forbidden_raw_id_field_pattern: &regex::Regex,
108    path: &Path,
109    line: &str,
110    line_number: usize,
111) {
112    for cap in forbidden_raw_id_field_pattern.captures_iter(line) {
113        let Some(field_name_match) = cap.get(1) else {
114            continue;
115        };
116        let Some(field_type_match) = cap.get(2) else {
117            continue;
118        };
119
120        if !field_name_match.as_str().ends_with("id") {
121            continue;
122        }
123
124        violations.push(SsotViolation::ForbiddenRawIdFieldType {
125            file: path.to_path_buf(),
126            line: line_number,
127            field_name: field_name_match.as_str().to_owned(),
128            field_type: field_type_match.as_str().to_owned(),
129            severity: Severity::Error,
130        });
131    }
132}

134/// Validator for single-source-of-truth invariants.
135pub struct SsotValidator {
136    config: ValidationConfig,
137}
138
139crate::impl_simple_validator_new!(SsotValidator);
140
141impl SsotValidator {
142    /// Scan the configured workspace and return SSOT violations.
143    ///
144    /// # Errors
145    ///
146    /// Returns an error if file scanning or regex compilation fails.
147    pub fn validate_all(&self) -> Result<Vec<SsotViolation>> {
148        let mut files = Vec::new();
149
150        for_each_scan_file(
151            &self.config,
152            Some(LanguageId::Rust),
153            false,
154            |entry, _candidate_src_dir| {
155                let content = std::fs::read_to_string(&entry.absolute_path)?;
156                files.push((entry.absolute_path.clone(), content));
157                Ok(())
158            },
159        )?;
160
161        Self::analyze_files(files)
162    }
163
164    /// Validate an in-memory synthetic file map (used by unit tests).
165    ///
166    /// # Errors
167    ///
168    /// Returns an error if regex compilation fails.
169    pub fn validate_synthetic_files(files: &HashMap<String, String>) -> Result<Vec<SsotViolation>> {
170        let synthetic_files = files
171            .iter()
172            .map(|(path, content)| (PathBuf::from(path), content.clone()))
173            .collect::<Vec<_>>();
174        Self::analyze_files(synthetic_files)
175    }
176
177    fn analyze_files(mut files: Vec<(PathBuf, String)>) -> Result<Vec<SsotViolation>> {
178        let declaration_pattern = compile_regex(DECLARATION_PATTERN)?;
179        let legacy_import_pattern = compile_regex(LEGACY_IMPORT_PATTERN)?;
180        let forbidden_schema_symbol_pattern = compile_regex(FORBIDDEN_SCHEMA_SYMBOL)?;
181        let forbidden_schema_macro_path_pattern = compile_regex(FORBIDDEN_SCHEMA_MACRO_PATH)?;
182        let forbidden_schema_import_pattern = compile_regex(FORBIDDEN_SCHEMA_IMPORT_PATTERN)?;
183        let forbidden_root_schema_path_pattern = compile_regex(FORBIDDEN_ROOT_SCHEMA_PATH_PATTERN)?;
184        let forbidden_root_schema_import_pattern =
185            compile_regex(FORBIDDEN_ROOT_SCHEMA_IMPORT_PATTERN)?;
186        let forbidden_raw_id_field_pattern = compile_regex(FORBIDDEN_RAW_ID_FIELD_PATTERN)?;
187
188        files.sort_by(|a, b| a.0.cmp(&b.0));
189
190        let mut declaration_locations: BTreeMap<String, Vec<(PathBuf, usize)>> = BTreeMap::new();
191        let mut violations = Vec::new();
192
193        for (path, content) in &files {
194            for (line_index, line) in content.lines().enumerate() {
195                let line_number = line_index + 1;
196                record_declarations(
197                    &mut declaration_locations,
198                    &declaration_pattern,
199                    line,
200                    path,
201                    line_number,
202                );
203
204                push_line_match(
205                    &mut violations,
206                    &legacy_import_pattern,
207                    line,
208                    |import_path| SsotViolation::ForbiddenLegacyImport {
209                        file: path.clone(),
210                        line: line_number,
211                        import_path,
212                        severity: Severity::Error,
213                    },
214                );
215
216                push_capture_group(
217                    &mut violations,
218                    &forbidden_schema_symbol_pattern,
219                    line,
220                    1,
221                    |symbol_name| SsotViolation::ForbiddenLegacySchemaSymbol {
222                        file: path.clone(),
223                        line: line_number,
224                        symbol_name: symbol_name.to_owned(),
225                        severity: Severity::Error,
226                    },
227                );
228
229                push_line_match(
230                    &mut violations,
231                    &forbidden_schema_macro_path_pattern,
232                    line,
233                    |macro_path| SsotViolation::ForbiddenSchemaMemoryMacroPath {
234                        file: path.clone(),
235                        line: line_number,
236                        macro_path,
237                        severity: Severity::Error,
238                    },
239                );
240
241                push_line_match(
242                    &mut violations,
243                    &forbidden_schema_import_pattern,
244                    line,
245                    |import_path| SsotViolation::ForbiddenLegacySchemaImport {
246                        file: path.clone(),
247                        line: line_number,
248                        import_path,
249                        severity: Severity::Error,
250                    },
251                );
252
253                push_line_match(
254                    &mut violations,
255                    &forbidden_root_schema_import_pattern,
256                    line,
257                    |path_text| SsotViolation::ForbiddenRootSchemaPath {
258                        file: path.clone(),
259                        line: line_number,
260                        path: path_text,
261                        severity: Severity::Error,
262                    },
263                );
264
265                push_capture_group(
266                    &mut violations,
267                    &forbidden_root_schema_path_pattern,
268                    line,
269                    0,
270                    |path_text| SsotViolation::ForbiddenRootSchemaPath {
271                        file: path.clone(),
272                        line: line_number,
273                        path: path_text.to_owned(),
274                        severity: Severity::Error,
275                    },
276                );
277
278                if is_domain_model_file(path) {
279                    push_raw_id_field_violations(
280                        &mut violations,
281                        &forbidden_raw_id_field_pattern,
282                        path,
283                        line,
284                        line_number,
285                    );
286                }
287            }
288        }
289
290        push_duplicate_declarations(&mut violations, declaration_locations);
291
292        Ok(violations)
293    }
294}

296fn is_domain_model_file(path: &std::path::Path) -> bool {
297    let path = path.to_string_lossy().replace('\\', "/");
298    path.contains("mcb-domain/src/entities/") || path.contains("mcb-domain/src/value_objects/")
299}

301impl Validator for SsotValidator {
302    fn name(&self) -> &'static str {
303        "ssot"
304    }
305
306    fn description(&self) -> &'static str {
307        "Detects duplicate declarations and forbidden legacy schema/repository references"
308    }
309
310    fn enabled_by_default(&self) -> bool {
311        false
312    }
313
314    fn validate(&self, _config: &ValidationConfig) -> Result<Vec<Box<dyn Violation>>> {
315        let violations = self.validate_all()?;
316        Ok(violations
317            .into_iter()
318            .map(|v| Box::new(v) as Box<dyn Violation>)
319            .collect())
320    }
