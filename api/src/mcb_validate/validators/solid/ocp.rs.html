17/// OCP: Check for excessive match statements
18///
19/// # Errors
20/// Returns an error if pattern compilation fails.
21pub fn validate_ocp(
22    config: &ValidationConfig,
23    max_match_arms: usize,
24) -> Result<Vec<SolidViolation>> {
25    let mut violations = Vec::new();
26    let match_pattern = required_pattern("SOLID003.match_keyword")?;
27
28    for_each_rust_file(config, |path, lines| {
29        for (line_num, line) in lines.iter().enumerate() {
30            if line.trim().starts_with("//") || !match_pattern.is_match(line) {
31                continue;
32            }
33            if let Some(violation) =
34                excessive_match_violation(&path, &lines, line_num, max_match_arms)?
35            {
36                violations.push(violation);
37            }
38        }
39        Ok(())
40    })?;
41
42    Ok(violations)
43}

45/// Returns an `ExcessiveMatchArms` violation when the match at `line_num`
46/// exceeds `max_match_arms`, else `None`.
47///
48/// # Errors
49///
50/// Returns an error if arm counting fails.
51fn excessive_match_violation(
52    path: &Path,
53    lines: &[&str],
54    line_num: usize,
55    max_match_arms: usize,
56) -> Result<Option<SolidViolation>> {
57    let arm_count = count_match_arms(lines, line_num)?;
58    Ok(
59        (arm_count > max_match_arms).then(|| SolidViolation::ExcessiveMatchArms {
60            file: path.to_path_buf(),
61            line: line_num + 1,
62            arm_count,
63            max_recommended: max_match_arms,
64            suggestion: "Consider using visitor pattern, enum dispatch, or trait objects"
65                .to_owned(),
66            severity: Severity::Info,
67        }),
68    )
69}

71/// OCP: Check for string-based type dispatch
72///
73/// # Errors
74/// Returns an error if pattern compilation fails.
75pub fn validate_string_dispatch(config: &ValidationConfig) -> Result<Vec<SolidViolation>> {
76    let mut violations = Vec::new();
77    let string_match_pattern = required_pattern("SOLID003.string_match")?;
78    let string_arm_pattern = required_pattern("SOLID003.string_arm")?;
79
80    for_each_rust_file(config, |path, lines| {
81        for (line_num, line) in lines.iter().enumerate() {
82            let trimmed = line.trim();
83            if trimmed.starts_with("//") || !string_match_pattern.is_match(line) {
84                continue;
85            }
86            if let Some(violation) =
87                string_dispatch_violation(&path, &lines, line_num, trimmed, string_arm_pattern)
88            {
89                violations.push(violation);
90            }
91        }
92        Ok(())
93    })?;
94
95    Ok(violations)
96}
97
98/// Returns a `StringBasedDispatch` violation when the match at `line_num` has
99/// enough string arms to suggest string-based dispatch, else `None`.
100fn string_dispatch_violation(
101    path: &Path,
102    lines: &[&str],
103    line_num: usize,
104    trimmed: &str,
105    string_arm_pattern: &Regex,
106) -> Option<SolidViolation> {
107    let string_arm_count = count_matches_in_block(lines, line_num, string_arm_pattern);
108    (string_arm_count >= MIN_STRING_MATCH_ARMS_FOR_DISPATCH).then(|| {
109        SolidViolation::StringBasedDispatch {
110            file: path.to_path_buf(),
111            line: line_num + 1,
112            match_expression: trimmed.chars().take(SHORT_PREVIEW_LENGTH).collect(),
113            suggestion: "Consider using enum types with FromStr or a registry pattern".to_owned(),
114            severity: Severity::Info,
115        }
116    })
117}
