23    pub enum DocumentationViolation {
24        /// Missing module-level documentation
25        #[violation(
26            id = "DOC001",
27            severity = Warning,
28            message = "Missing module doc: {file}",
29            suggestion = "Add //! module-level documentation at the top of the file"
30        )]
31        MissingModuleDoc {
32            file: PathBuf,
33            severity: Severity,
34        },
35        /// Missing documentation on public item
36        #[violation(
37            id = "DOC002",
38            severity = Warning,
39            message = "Missing {item_kind} doc: {file}:{line} - {item_name}",
40            suggestion = "Add /// documentation for {item_kind} {item_name}"
41        )]
42        MissingPubItemDoc {
43            file: PathBuf,
44            line: usize,
45            item_name: String,
46            item_kind: String,
47            severity: Severity,
48        },
49        /// Missing example code in documentation
50        #[violation(
51            id = "DOC003",
52            severity = Info,
53            message = "Missing example: {file}:{line} - {item_name}",
54            suggestion = "Add # Example section to {item_name} documentation"
55        )]
56        MissingExampleCode {
57            file: PathBuf,
58            line: usize,
59            item_name: String,
60            item_kind: String,
61            severity: Severity,
62        },
63    }
64}

75    /// Verify module-level documentation exists
76    ///
77    /// # Errors
78    ///
79    /// Returns an error if regex compilation or file scanning fails.
80    pub fn validate_module_docs(config: &ValidationConfig) -> Result<Vec<DocumentationViolation>> {
81        let mut violations = Vec::new();
82        let module_doc_pattern = compile_regex(MODULE_DOC_REGEX)?;
83
84        for_each_crate_file(
85            config,
86            Some(LanguageId::Rust),
87            |entry, _src_dir, _crate_name| {
88                let path = &entry.absolute_path;
89                let content = std::fs::read_to_string(path)?;
90                let file_name = path.file_name().and_then(|s| s.to_str()).unwrap_or("");
91
92                // Only check module files that require documentation
93                if !MODULE_FILE_NAMES.contains(&file_name) {
94                    return Ok(());
95                }
96
97                // Check if first non-empty line is module doc
98                let has_module_doc = content
99                    .lines()
100                    .filter(|line| !line.trim().is_empty())
101                    .take(1)
102                    .any(|line| module_doc_pattern.is_match(line));
103
104                if !has_module_doc {
105                    violations.push(DocumentationViolation::MissingModuleDoc {
106                        file: path.clone(),
107                        severity: Severity::Warning,
108                    });
109                }
110
111                Ok(())
112            },
113        )?;
114
115        Ok(violations)
116    }

118    /// Verify all pub items have rustdoc
119    ///
120    /// # Errors
121    ///
122    /// Returns an error if regex compilation or file scanning fails.
123    pub fn validate_pub_item_docs(
124        config: &ValidationConfig,
125    ) -> Result<Vec<DocumentationViolation>> {
126        let mut violations = Vec::new();
127
128        // Patterns for public items
129        let pub_struct_pattern = compile_regex(PUB_STRUCT_REGEX)?;
130        let pub_enum_pattern = compile_regex(PUB_ENUM_REGEX)?;
131        let pub_trait_pattern = compile_regex(PUB_TRAIT_REGEX)?;
132        let pub_fn_pattern = compile_regex(PUB_FN_REGEX)?;
133        let example_pattern = compile_regex(EXAMPLE_SECTION_REGEX)?;
134        let doc_comment_re = compile_regex(DOC_COMMENT_REGEX)?;
135        let doc_comment_capture_re = compile_regex(DOC_COMMENT_CAPTURE_REGEX)?;
136        let attr_re = compile_regex(ATTR_REGEX)?;
137        let simple_pub_item_specs = [
138            SimplePubItemSpec {
139                pattern: &pub_struct_pattern,
140                item_kind: ITEM_KIND_STRUCT,
141            },
142            SimplePubItemSpec {
143                pattern: &pub_enum_pattern,
144                item_kind: ITEM_KIND_ENUM,
145            },
146        ];
147
148        for_each_crate_file(
149            config,
150            Some(LanguageId::Rust),
151            |entry, _src_dir, _crate_name| {
152                let path = &entry.absolute_path;
153                let content = std::fs::read_to_string(path)?;
154                let lines: Vec<&str> = content.lines().collect();
155                let path_str = path.to_str().unwrap_or_default();
156                let regex_ctx = DocRegexContext {
157                    doc_comment_re: &doc_comment_re,
158                    doc_comment_capture_re: &doc_comment_capture_re,
159                    attr_re: &attr_re,
160                    example_pattern: &example_pattern,
161                };
162
163                for (line_num, line) in lines.iter().enumerate() {
164                    let scan_ctx = ScanLineContext {
165                        path,
166                        lines: &lines,
167                        line_num,
168                        line,
169                    };
170                    Self::check_simple_public_item_docs(
171                        &mut violations,
172                        &scan_ctx,
173                        &simple_pub_item_specs,
174                        &regex_ctx,
175                    );
176
177                    if let Some(cap) = pub_trait_pattern.captures(scan_ctx.line) {
178                        let name = cap.get(1).map_or("", |m: regex::Match| m.as_str());
179                        let item_ctx = Self::build_item_ctx(
180                            scan_ctx.path,
181                            scan_ctx.lines,
182                            scan_ctx.line_num,
183                            name,
184                        );
185                        Self::check_public_trait_docs(
186                            &mut violations,
187                            &item_ctx,
188                            path_str,
189                            &regex_ctx,
190                        );
191                    }
192
193                    if let Some(cap) = pub_fn_pattern.captures(scan_ctx.line) {
194                        let name = cap.get(1).map_or("", |m: regex::Match| m.as_str());
195                        let item_ctx = Self::build_item_ctx(
196                            scan_ctx.path,
197                            scan_ctx.lines,
198                            scan_ctx.line_num,
199                            name,
200                        );
201                        Self::check_public_function_docs(
202                            &mut violations,
203                            &item_ctx,
204                            scan_ctx.line,
205                            &regex_ctx,
206                        );
207                    }
208                }
209
210                Ok(())
211            },
212        )?;
213
214        Ok(violations)
215    }

217    fn build_item_ctx<'a>(
218        path: &'a std::path::Path,
219        lines: &'a [&'a str],
220        line_num: usize,
221        item_name: &'a str,
222    ) -> DocItemContext<'a> {
223        DocItemContext {
224            path,
225            lines,
226            line_num,
227            item_name,
228        }
229    }

231    fn check_simple_public_item_docs(
232        violations: &mut Vec<DocumentationViolation>,
233        scan_ctx: &ScanLineContext<'_>,
234        specs: &[SimplePubItemSpec<'_>],
235        regex_ctx: &DocRegexContext<'_>,
236    ) {
237        for spec in specs {
238            let Some(cap) = spec.pattern.captures(scan_ctx.line) else {
239                continue;
240            };
241
242            let name = cap.get(1).map_or("", |m: regex::Match| m.as_str());
243            let item_ctx =
244                Self::build_item_ctx(scan_ctx.path, scan_ctx.lines, scan_ctx.line_num, name);
245            Self::push_missing_pub_item_doc_if_needed(
246                violations,
247                &item_ctx,
248                &MissingDocSpec {
249                    item_kind: spec.item_kind,
250                    severity: Severity::Warning,
251                },
252                regex_ctx,
253            );
254        }
255    }
256
257    fn push_missing_pub_item_doc_if_needed(
258        violations: &mut Vec<DocumentationViolation>,
259        item_ctx: &DocItemContext<'_>,
260        spec: &MissingDocSpec<'_>,
261        regex_ctx: &DocRegexContext<'_>,
262    ) {
263        if has_doc_comment(
264            item_ctx.lines,
265            item_ctx.line_num,
266            regex_ctx.doc_comment_re,
267            regex_ctx.attr_re,
268        ) {
269            return;
270        }
271
272        violations.push(DocumentationViolation::MissingPubItemDoc {
273            file: item_ctx.path.to_path_buf(),
274            line: item_ctx.line_num + 1,
275            item_name: item_ctx.item_name.to_owned(),
276            item_kind: spec.item_kind.to_owned(),
277            severity: spec.severity,
278        });
279    }
280
281    fn check_public_trait_docs(
282        violations: &mut Vec<DocumentationViolation>,
283        item_ctx: &DocItemContext<'_>,
284        path_str: &str,
285        regex_ctx: &DocRegexContext<'_>,
286    ) {
287        if !has_doc_comment(
288            item_ctx.lines,
289            item_ctx.line_num,
290            regex_ctx.doc_comment_re,
291            regex_ctx.attr_re,
292        ) {
293            Self::push_missing_pub_item_doc_if_needed(
294                violations,
295                item_ctx,
296                &MissingDocSpec {
297                    item_kind: ITEM_KIND_TRAIT,
298                    severity: Severity::Warning,
299                },
300                regex_ctx,
301            );
302            return;
303        }
304
305        let is_di_or_port_trait =
306            path_str.contains(DI_MODULES_PATH) || path_str.contains(PORTS_PATH);
307        if is_di_or_port_trait {
308            return;
309        }
310
311        let doc_section = get_doc_comment_section(
312            item_ctx.lines,
313            item_ctx.line_num,
314            regex_ctx.doc_comment_capture_re,
315            regex_ctx.attr_re,
316        );
317        if regex_ctx.example_pattern.is_match(&doc_section) {
318            return;
319        }
320
321        violations.push(DocumentationViolation::MissingExampleCode {
322            file: item_ctx.path.to_path_buf(),
323            line: item_ctx.line_num + 1,
324            item_name: item_ctx.item_name.to_owned(),
325            item_kind: ITEM_KIND_TRAIT.to_owned(),
326            severity: Severity::Info,
327        });
328    }
329
330    fn check_public_function_docs(
331        violations: &mut Vec<DocumentationViolation>,
332        item_ctx: &DocItemContext<'_>,
333        line: &str,
334        regex_ctx: &DocRegexContext<'_>,
335    ) {
336        if line.starts_with("    ") || line.starts_with('\t') {
337            return;
338        }
339
340        Self::push_missing_pub_item_doc_if_needed(
341            violations,
342            item_ctx,
343            &MissingDocSpec {
344                item_kind: ITEM_KIND_FUNCTION,
345                severity: Severity::Info,
346            },
347            regex_ctx,
348        );
349    }
