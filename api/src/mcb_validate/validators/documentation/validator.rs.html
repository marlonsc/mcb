21/// Compiled regexes for scanning public items that require documentation.
22struct PubDocPatterns {
23    pub_struct: Regex,
24    pub_enum: Regex,
25    pub_trait: Regex,
26    pub_fn: Regex,
27    example: Regex,
28    doc_comment: Regex,
29    doc_comment_capture: Regex,
30    attr: Regex,
31    /// Items declared inside a macro invocation (e.g. `define_violations! { pub enum ... }`)
32    /// are documented by the macro expansion, not the source text, so the text
33    /// scanner must skip them.
34    macro_invocation: Regex,
35}
36
37impl PubDocPatterns {
38    /// Compile every pattern used by the public-item documentation scan.
39    fn compile() -> Result<Self> {
40        Ok(Self {
41            pub_struct: compile_regex(PUB_STRUCT_REGEX)?,
42            pub_enum: compile_regex(PUB_ENUM_REGEX)?,
43            pub_trait: compile_regex(PUB_TRAIT_REGEX)?,
44            pub_fn: compile_regex(PUB_FN_REGEX)?,
45            example: compile_regex(EXAMPLE_SECTION_REGEX)?,
46            doc_comment: compile_regex(DOC_COMMENT_REGEX)?,
47            doc_comment_capture: compile_regex(DOC_COMMENT_CAPTURE_REGEX)?,
48            attr: compile_regex(ATTR_REGEX)?,
49            macro_invocation: compile_regex(r"^\s*[a-z_][a-z0-9_]*!\s*\{")?,
50        })
51    }
52
53    /// Borrowed regex bundle for doc-comment detection helpers.
54    fn regex_context(&self) -> DocRegexContext<'_> {
55        DocRegexContext {
56            doc_comment_re: &self.doc_comment,
57            doc_comment_capture_re: &self.doc_comment_capture,
58            attr_re: &self.attr,
59            example_pattern: &self.example,
60        }
61    }
62
63    /// Specs for simple public items (struct, enum) checked uniformly.
64    fn simple_item_specs(&self) -> [SimplePubItemSpec<'_>; 2] {
65        [
66            SimplePubItemSpec {
67                pattern: &self.pub_struct,
68                item_kind: ITEM_KIND_STRUCT,
69            },
70            SimplePubItemSpec {
71                pattern: &self.pub_enum,
72                item_kind: ITEM_KIND_ENUM,
73            },
74        ]
75    }
76}

81    pub enum DocumentationViolation {
82        /// Missing module-level documentation
83        #[violation(
84            id = "DOC001",
85            severity = Warning,
86            message = "Missing module doc: {file}",
87            suggestion = "Add //! module-level documentation at the top of the file"
88        )]
89        MissingModuleDoc {
90            file: PathBuf,
91            severity: Severity,
92        },
93        /// Missing documentation on public item
94        #[violation(
95            id = "DOC002",
96            severity = Warning,
97            message = "Missing {item_kind} doc: {file}:{line} - {item_name}",
98            suggestion = "Add /// documentation for {item_kind} {item_name}"
99        )]
100        MissingPubItemDoc {
101            file: PathBuf,
102            line: usize,
103            item_name: String,
104            item_kind: String,
105            severity: Severity,
106        },
107        /// Missing example code in documentation
108        #[violation(
109            id = "DOC003",
110            severity = Info,
111            message = "Missing example: {file}:{line} - {item_name}",
112            suggestion = "Add # Example section to {item_name} documentation"
113        )]
114        MissingExampleCode {
115            file: PathBuf,
116            line: usize,
117            item_name: String,
118            item_kind: String,
119            severity: Severity,
120        },
121    }
122}

132impl DocumentationValidator {
133    /// Verify module-level documentation exists
134    ///
135    /// # Errors
136    ///
137    /// Returns an error if regex compilation or file scanning fails.
138    pub fn validate_module_docs(config: &ValidationConfig) -> Result<Vec<DocumentationViolation>> {
139        let mut violations = Vec::new();
140        let module_doc_pattern = compile_regex(MODULE_DOC_REGEX)?;
141
142        for_each_crate_file(
143            config,
144            Some(LanguageId::Rust),
145            |entry, _src_dir, _crate_name| {
146                let path = &entry.absolute_path;
147                let content = std::fs::read_to_string(path)?;
148                let file_name = path.file_name().and_then(|s| s.to_str()).unwrap_or("");
149
150                // Only check module files that require documentation
151                if !MODULE_FILE_NAMES.contains(&file_name) {
152                    return Ok(());
153                }
154
155                // Check if first non-empty line is module doc
156                let has_module_doc = content
157                    .lines()
158                    .filter(|line| !line.trim().is_empty())
159                    .take(1)
160                    .any(|line| module_doc_pattern.is_match(line));
161
162                if !has_module_doc {
163                    violations.push(DocumentationViolation::MissingModuleDoc {
164                        file: path.clone(),
165                        severity: Severity::Warning,
166                    });
167                }
168
169                Ok(())
170            },
171        )?;
172
173        Ok(violations)
174    }

176    /// Verify all pub items have rustdoc
177    ///
178    /// # Errors
179    ///
180    /// Returns an error if regex compilation or file scanning fails.
181    pub fn validate_pub_item_docs(
182        config: &ValidationConfig,
183    ) -> Result<Vec<DocumentationViolation>> {
184        let mut violations = Vec::new();
185        let patterns = PubDocPatterns::compile()?;
186
187        for_each_crate_file(
188            config,
189            Some(LanguageId::Rust),
190            |entry, _src_dir, _crate_name| {
191                let content = std::fs::read_to_string(&entry.absolute_path)?;
192                let lines: Vec<&str> = content.lines().collect();
193                Self::scan_file_for_pub_docs(
194                    &entry.absolute_path,
195                    &lines,
196                    &patterns,
197                    &mut violations,
198                );
199                Ok(())
200            },
201        )?;
202
203        Ok(violations)
204    }

206    /// Scan a single file's lines for undocumented public items.
207    fn scan_file_for_pub_docs(
208        path: &Path,
209        lines: &[&str],
210        patterns: &PubDocPatterns,
211        violations: &mut Vec<DocumentationViolation>,
212    ) {
213        // INTENTIONAL: Path to string; non-UTF8 paths yield empty string (best-effort)
214        let path_str = path.to_str().unwrap_or_default();
215        let regex_ctx = patterns.regex_context();
216        let simple_pub_item_specs = patterns.simple_item_specs();
217
218        let mut macro_depth: i32 = 0;
219        for (line_num, line) in lines.iter().enumerate() {
220            if Self::advance_macro_depth(&mut macro_depth, line, &patterns.macro_invocation) {
221                continue;
222            }
223
224            let scan_ctx = ScanLineContext {
225                path,
226                lines,
227                line_num,
228                line,
229            };
230            Self::check_simple_public_item_docs(
231                violations,
232                &scan_ctx,
233                &simple_pub_item_specs,
234                &regex_ctx,
235            );
236            Self::check_trait_and_fn_docs(
237                violations,
238                &scan_ctx,
239                path_str,
240                &regex_ctx,
241                (&patterns.pub_trait, &patterns.pub_fn),
242            );
243        }
244    }
245
246    /// Update `macro_depth` for `line` and return `true` when the line is inside
247    /// (or opening) a macro-invocation block whose pub items must be skipped.
248    fn advance_macro_depth(
249        macro_depth: &mut i32,
250        line: &str,
251        macro_invocation_re: &regex::Regex,
252    ) -> bool {
253        let entering_macro = *macro_depth == 0
254            && macro_invocation_re.is_match(line)
255            && !line.contains("macro_rules!");
256        if *macro_depth > 0 || entering_macro {
257            *macro_depth += i32::try_from(line.matches('{').count()).unwrap_or(0);
258            *macro_depth -= i32::try_from(line.matches('}').count()).unwrap_or(0);
259            *macro_depth = (*macro_depth).max(0);
260            return true;
261        }
262        false
263    }

265    /// Check the current line for an undocumented public trait and/or function,
266    /// appending any violations found.
267    fn check_trait_and_fn_docs(
268        violations: &mut Vec<DocumentationViolation>,
269        scan_ctx: &ScanLineContext<'_>,
270        path_str: &str,
271        regex_ctx: &DocRegexContext<'_>,
272        patterns: (&regex::Regex, &regex::Regex),
273    ) {
274        let (pub_trait_pattern, pub_fn_pattern) = patterns;
275        if let Some(cap) = pub_trait_pattern.captures(scan_ctx.line) {
276            let name = cap.get(1).map_or("", |m: regex::Match| m.as_str());
277            let item_ctx =
278                Self::build_item_ctx(scan_ctx.path, scan_ctx.lines, scan_ctx.line_num, name);
279            Self::check_public_trait_docs(violations, &item_ctx, path_str, regex_ctx);
280        }
281
282        if let Some(cap) = pub_fn_pattern.captures(scan_ctx.line) {
283            let name = cap.get(1).map_or("", |m: regex::Match| m.as_str());
284            let item_ctx =
285                Self::build_item_ctx(scan_ctx.path, scan_ctx.lines, scan_ctx.line_num, name);
286            Self::check_public_function_docs(violations, &item_ctx, scan_ctx.line, regex_ctx);
287        }
288    }

290    fn build_item_ctx<'a>(
291        path: &'a std::path::Path,
292        lines: &'a [&'a str],
293        line_num: usize,
294        item_name: &'a str,
295    ) -> DocItemContext<'a> {

297            path,
298            lines,
299            line_num,
300            item_name,
301        }

304    fn check_simple_public_item_docs(
305        violations: &mut Vec<DocumentationViolation>,
306        scan_ctx: &ScanLineContext<'_>,
307        specs: &[SimplePubItemSpec<'_>],
308        regex_ctx: &DocRegexContext<'_>,
309    ) {
310        for spec in specs {
311            let Some(cap) = spec.pattern.captures(scan_ctx.line) else {
312                continue;
313            };
314
315            let name = cap.get(1).map_or("", |m: regex::Match| m.as_str());
316            let item_ctx =
317                Self::build_item_ctx(scan_ctx.path, scan_ctx.lines, scan_ctx.line_num, name);
318            Self::push_missing_pub_item_doc_if_needed(
319                violations,
320                &item_ctx,
321                &MissingDocSpec {
322                    item_kind: spec.item_kind,
323                    severity: Severity::Warning,
324                },
325                regex_ctx,
326            );
327        }
328    }

330    fn push_missing_pub_item_doc_if_needed(
331        violations: &mut Vec<DocumentationViolation>,
332        item_ctx: &DocItemContext<'_>,
333        spec: &MissingDocSpec<'_>,
334        regex_ctx: &DocRegexContext<'_>,
335    ) {
336        if has_doc_comment(
337            item_ctx.lines,
338            item_ctx.line_num,
339            regex_ctx.doc_comment_re,
340            regex_ctx.attr_re,
341        ) {
342            return;
343        }
344
345        violations.push(DocumentationViolation::MissingPubItemDoc {
346            file: item_ctx.path.to_path_buf(),
347            line: item_ctx.line_num + 1,
348            item_name: item_ctx.item_name.to_owned(),
349            item_kind: spec.item_kind.to_owned(),
350            severity: spec.severity,
351        });
352    }
353
354    fn check_public_trait_docs(
355        violations: &mut Vec<DocumentationViolation>,
356        item_ctx: &DocItemContext<'_>,
357        path_str: &str,
358        regex_ctx: &DocRegexContext<'_>,
359    ) {
360        if !has_doc_comment(
361            item_ctx.lines,
362            item_ctx.line_num,
363            regex_ctx.doc_comment_re,
364            regex_ctx.attr_re,
365        ) {
366            Self::push_missing_pub_item_doc_if_needed(
367                violations,
368                item_ctx,
369                &MissingDocSpec {
370                    item_kind: ITEM_KIND_TRAIT,
371                    severity: Severity::Warning,
372                },
373                regex_ctx,
374            );
375            return;
376        }
377
378        let is_di_or_port_trait =
379            path_str.contains(DI_MODULES_PATH) || path_str.contains(PORTS_DIR);
380        if is_di_or_port_trait {
381            return;
382        }
383
384        let doc_section = get_doc_comment_section(
385            item_ctx.lines,
386            item_ctx.line_num,
387            regex_ctx.doc_comment_capture_re,
388            regex_ctx.attr_re,
389        );
390        if regex_ctx.example_pattern.is_match(&doc_section) {
391            return;
392        }
393
394        violations.push(DocumentationViolation::MissingExampleCode {
395            file: item_ctx.path.to_path_buf(),
396            line: item_ctx.line_num + 1,
397            item_name: item_ctx.item_name.to_owned(),
398            item_kind: ITEM_KIND_TRAIT.to_owned(),
399            severity: Severity::Info,
400        });
401    }
402
403    fn check_public_function_docs(
404        violations: &mut Vec<DocumentationViolation>,
405        item_ctx: &DocItemContext<'_>,
406        line: &str,
407        regex_ctx: &DocRegexContext<'_>,
408    ) {
409        if line.starts_with("    ") || line.starts_with('\t') {
410            return;
411        }
412
413        Self::push_missing_pub_item_doc_if_needed(
414            violations,
415            item_ctx,
416            &MissingDocSpec {
417                item_kind: ITEM_KIND_FUNCTION,
418                severity: Severity::Info,
419            },
420            regex_ctx,
421        );
422    }
