12/// Validate no forbidden use statements in source code
13pub fn validate_use_statements(
14    validator: &DependencyValidator,
15) -> Result<Vec<DependencyViolation>> {
16    let mut violations = Vec::new();
17    let use_pattern = compile_regex(r"use\s+(mcb_[a-z_]+)")?;
18
19    for (crate_name, allowed) in &validator.allowed_deps {
20        let crate_src = validator
21            .config
22            .workspace_root
23            .join("crates")
24            .join(crate_name)
25            .join("src");
26
27        if !crate_src.exists() {
28            continue;
29        }
30
31        for_each_file_under_root(
32            &validator.config,
33            &crate_src,
34            Some(LanguageId::Rust),
35            |entry| {
36                let path = &entry.absolute_path;
37                let content = std::fs::read_to_string(path)?;
38
39                crate::validators::for_each_non_test_non_comment_line(
40                    &content,
41                    |line_num, line, _trimmed| {
42                        if line.trim().starts_with("/*") || line.contains('"') {
43                            return;
44                        }
45                        violations.extend(use_pattern.captures_iter(line).filter_map(|cap| {
46                            let used_crate_kebab =
47                                cap.get(1).map_or("", |m| m.as_str()).replace('_', "-");
48                            (used_crate_kebab != *crate_name
49                                && !allowed.contains(&used_crate_kebab))
50                            .then(|| DependencyViolation::ForbiddenUseStatement {
51                                crate_name: crate_name.clone(),
52                                forbidden_dep: used_crate_kebab,
53                                file: path.clone(),
54                                line: line_num + 1,
55                                context: line.trim().to_owned(),
56                                severity: Severity::Error,
57                            })
58                        }));
59                    },
60                );
61
62                Ok(())
63            },
64        )?;
65    }
66
67    Ok(violations)
68}
