12/// Detect circular dependencies using topological sort
13pub fn detect_circular_dependencies(
14    validator: &DependencyValidator,
15) -> Result<Vec<DependencyViolation>> {
16    let mut violations = Vec::new();
17    let mut graph: HashMap<String, HashSet<String>> = HashMap::new();
18
19    // Build dependency graph from Cargo.toml files
20    for crate_name in validator.allowed_deps.keys() {
21        let cargo_toml = validator
22            .config
23            .workspace_root
24            .join("crates")
25            .join(crate_name)
26            .join(CARGO_TOML_FILENAME);
27
28        if !cargo_toml.exists() {
29            continue;
30        }
31
32        let content = std::fs::read_to_string(&cargo_toml)?;
33        let parsed: toml::Value = toml::from_str(&content)?;
34
35        let mut deps = HashSet::new();
36        if let Some(dependencies) = parsed.get("dependencies").and_then(|d| d.as_table()) {
37            for dep_name in dependencies.keys() {
38                if dep_name.starts_with(MCB_DEPENDENCY_PREFIX) {
39                    deps.insert(dep_name.replace('_', "-"));
40                }
41            }
42        }
43        graph.insert(crate_name.clone(), deps);
44    }
45
46    // Detect cycles using DFS
47    for start in graph.keys() {
48        let mut visited = HashSet::new();
49        let mut path = Vec::new();
50        if let Some(cycle) = find_cycle_impl(&graph, start, &mut visited, &mut path) {
51            violations.push(DependencyViolation::CircularDependency {
52                cycle: DependencyCycle(cycle),
53                severity: Severity::Error,
54            });
55        }
56    }
57
58    Ok(violations)
59}

61/// Detects cycles in the dependency graph using depth-first search.
62///
63/// Recursively traverses the dependency graph to find circular dependencies.
64/// Returns the cycle path if found, or None if no cycle exists from this node.
65///
66/// # Arguments
67/// * `graph` - The dependency graph mapping crate names to their dependencies
68/// * `node` - The current node being visited
69/// * `visited` - Set of nodes already fully explored
70/// * `path` - Current path being explored (used to detect cycles)
71fn find_cycle_impl(
72    graph: &HashMap<String, HashSet<String>>,
73    node: &str,
74    visited: &mut HashSet<String>,
75    path: &mut Vec<String>,
76) -> Option<Vec<String>> {
77    if path.contains(&node.to_owned()) {
78        let cycle_start = path.iter().position(|n| n == node)?;
79        let mut cycle: Vec<String> = path[cycle_start..].to_vec();
80        cycle.push(node.to_owned());
81        return Some(cycle);
82    }
83
84    if visited.contains(node) {
85        return None;
86    }
87
88    visited.insert(node.to_owned());
89    path.push(node.to_owned());
90
91    if let Some(deps) = graph.get(node) {
92        for dep in deps {
93            if let Some(cycle) = find_cycle_impl(graph, dep, visited, path) {
94                return Some(cycle);
95            }
96        }
97    }
98
99    path.pop();
100    None
101}
