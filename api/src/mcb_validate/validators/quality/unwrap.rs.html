14fn has_ignore_hint(line: &str, violation_type: &str) -> bool {
15    line.contains(&format!("mcb-validate-ignore: {violation_type}"))
16}
17
18fn should_skip_source_file(path: &std::path::Path) -> bool {
19    let Some(path_str) = path.to_str() else {
20        return true;
21    };
22    path.extension().is_none_or(|ext| ext != "rs")
23        || TEST_PATH_PATTERNS.iter().any(|p| path_str.contains(p))
24}
25
26fn has_ignore_nearby(lines: &[&str], detection_line: usize) -> bool {
27    if detection_line == 0 || detection_line > lines.len() {
28        return false;
29    }
30
31    let start = detection_line.saturating_sub(COMMENT_SEARCH_RADIUS + 1);
32    let end = (detection_line + 2).min(lines.len() - 1);
33    (start..=end).any(|idx| {
34        let line = lines[idx];
35        IGNORE_HINT_KEYWORDS
36            .iter()
37            .any(|k| has_ignore_hint(line, k))
38    })
39}
40
41fn is_lock_poisoning_expect(method: &str, line_content: &str) -> bool {
42    method == "expect"
43        && LOCK_POISONING_STRINGS
44            .iter()
45            .any(|s| line_content.contains(s))
46}
47
48fn should_skip_detection(
49    detection: &crate::ast::unwrap_detector::UnwrapDetection,
50    lines: &[&str],
51) -> bool {
52    if detection.in_test || detection.line == 0 || detection.line > lines.len() {
53        return true;
54    }
55
56    let line_content = lines[detection.line - 1];
57    SAFETY_COMMENT_MARKERS
58        .iter()
59        .any(|m| line_content.contains(m))
60        || has_ignore_nearby(lines, detection.line - 1)
61        || is_lock_poisoning_expect(&detection.method, line_content)
62}

64fn push_violation(
65    violations: &mut Vec<QualityViolation>,
66    file: &std::path::Path,
67    detection: crate::ast::unwrap_detector::UnwrapDetection,
68) {
69    match detection.method.as_str() {
70        "unwrap" => {
71            violations.push(QualityViolation::UnwrapInProduction {
72                file: file.to_path_buf(),
73                line: detection.line,
74                context: detection.context,
75                severity: Severity::Warning,
76            });
77        }
78        "expect" => {
79            violations.push(QualityViolation::ExpectInProduction {
80                file: file.to_path_buf(),
81                line: detection.line,
82                context: detection.context,
83                severity: Severity::Warning,
84            });
85        }
86        other => {
87            mcb_domain::debug!(
88                "unwrap",
89                "unhandled detection method type",
90                &format!(
91                    "method = {:?}, file = {}, line = {}",
92                    other,
93                    file.display(),
94                    detection.line
95                )
96            );
97        }
98    }
99}

101/// Scans production code for usage of `unwrap()` and `expect()` methods.
102///
103/// Uses AST-based detection to accurately identify method calls while ignoring
104/// test files and allowed patterns.
105pub fn validate(validator: &QualityValidator) -> Result<Vec<QualityViolation>> {
106    let mut violations = Vec::new();
107    let mut detector = UnwrapDetector::new()?;
108
109    for_each_scan_file(
110        &validator.config,
111        Some(LanguageId::Rust),
112        false,
113        |entry, _src_dir| {
114            if should_skip_source_file(&entry.absolute_path) {
115                return Ok(());
116            }
117            let content = std::fs::read_to_string(&entry.absolute_path)?;
118            let lines: Vec<&str> = content.lines().collect();
119            let detections = detector.detect_in_file(&entry.absolute_path)?;
120
121            for detection in detections {
122                if should_skip_detection(&detection, &lines) {
123                    continue;
124                }
125                push_violation(&mut violations, &entry.absolute_path, detection);
126            }
127
128            Ok(())
129        },
130    )?;
131
132    Ok(violations)
133}
