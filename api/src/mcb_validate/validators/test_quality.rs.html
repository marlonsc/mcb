1//! Test Quality Validation
2//!
3//! Validates test code quality:
4//! - Detects `#[ignore]` attributes without proper justification (attribute without documentation)
5//! - Detects `todo!()` macros in test fixtures outside intentional stubs
6//! - Detects missing test implementations
7//! - Ensures tests have proper documentation
8

24define_violations! {
25    dynamic_severity,
26    ViolationCategory::Testing,
27    pub enum TestQualityViolation {
28        /// Test with `#[ignore]` attribute missing justification
29        #[violation(
30            id = "TST001",
31            severity = Warning,
32            message = "{file}:{line} - Test '{test_name}' has #[ignore] without justification comment",
33            suggestion = "Add a comment explaining why the test is ignored (e.g., // Requires external tool: ruff)"
34        )]
35        IgnoreWithoutJustification {
36            file: PathBuf,
37            line: usize,
38            test_name: String,
39            severity: Severity,
40        },
41        /// todo!() macro in test fixture without proper stub marker
42        #[violation(
43            id = "TST002",
44            severity = Warning,
45            message = "{file}:{line} - Function '{function_name}' in test fixture contains todo!() - implement or mark as intentional stub",
46            suggestion = "Implement the test fixture function or add comment: // Intentional stub for X"
47        )]
48        TodoInTestFixture {
49            file: PathBuf,
50            line: usize,
51            function_name: String,
52            severity: Severity,
53        },
54        /// Test function with empty body
55        #[violation(
56            id = "TST003",
57            severity = Warning,
58            message = "{file}:{line} - Test '{test_name}' has empty body - implement or remove",
59            suggestion = "Implement the test logic or remove the test function"
60        )]
61        EmptyTestBody {
62            file: PathBuf,
63            line: usize,
64            test_name: String,
65            severity: Severity,
66        },
67        /// Test missing documentation comment
68        #[violation(
69            id = "TST004",
70            severity = Warning,
71            message = "{file}:{line} - Test '{test_name}' missing documentation comment explaining what it tests",
72            suggestion = "Add documentation comment: /// Tests that [scenario] [expected behavior]"
73        )]
74        TestMissingDocumentation {
75            file: PathBuf,
76            line: usize,
77            test_name: String,
78            severity: Severity,
79        },
80        /// Test with only assert!(true) or similar stub
81        #[violation(
82            id = "TST005",
83            severity = Warning,
84            message = "{file}:{line} - Test '{test_name}' contains stub assertion (assert!(true)) - implement real test",
85            suggestion = "Replace assert!(true) with actual test logic and assertions"
86        )]
87        StubTestAssertion {
88            file: PathBuf,
89            line: usize,
90            test_name: String,
91            severity: Severity,
92        },
93    }
94}

96/// Test quality validator
97pub struct TestQualityValidator {
98    /// Configuration for validation scans
99    config: ValidationConfig,
100    rules: TestQualityRulesConfig,
101}
102
103impl crate::traits::validator::Validator for TestQualityValidator {
104    fn name(&self) -> &'static str {
105        "test-quality"
106    }
107
108    fn description(&self) -> &'static str {
109        "Validates test quality rules (ignore usage, stubs, and test documentation)"
110    }
111
112    fn validate(
113        &self,
114        _config: &crate::ValidationConfig,
115    ) -> crate::Result<Vec<Box<dyn crate::traits::violation::Violation>>> {
116        let violations = self.validate()?;
117        Ok(violations
118            .into_iter()
119            .map(|v| Box::new(v) as Box<dyn crate::traits::violation::Violation>)
120            .collect())
121    }
122}

126impl TestQualityValidator {
127    /// Create a validator with a custom configuration
128    #[must_use]
129    pub fn with_config(config: ValidationConfig, rules: &TestQualityRulesConfig) -> Self {
130        Self {
131            config,
132            rules: rules.clone(),
133        }
134    }
135
136    /// Validate test quality across all test files
137    ///
138    /// # Errors
139    /// Returns an error if pattern compilation fails or file reading fails.
140    pub fn validate(&self) -> Result<Vec<TestQualityViolation>> {
141        if !self.rules.enabled {
142            return Ok(Vec::new());
143        }
144        let mut violations = Vec::new();
145
146        // Regex patterns
147        let test_pattern = compile_regex(r"#\[test\]|#\[tokio::test\]")?;
148        let fn_pattern = compile_regex(r"fn\s+(\w+)")?;
149        let empty_body_pattern = compile_regex(r"\{\s*\}")?;
150        let stub_assert_pattern = compile_regex(r"assert!\(true\)|assert_eq!\(true,\s*true\)")?;
151
152        for_each_scan_file(
153            &self.config,
154            Some(LanguageId::Rust),
155            false,
156            |entry, _src_dir| {
157                if !entry
158                    .absolute_path
159                    .to_str()
160                    .is_some_and(|s| s.contains(TEST_DIR_FRAGMENT) || s.contains("/test_"))
161                {
162                    return Ok(());
163                }
164
165                let content = std::fs::read_to_string(&entry.absolute_path)?;
166                let lines: Vec<&str> = content.lines().collect();
167
168                Self::check_ignored_tests(
169                    &entry.absolute_path,
170                    &lines,
171                    &fn_pattern,
172                    &mut violations,
173                );
174                self.check_todo_in_fixtures(
175                    &entry.absolute_path,
176                    &lines,
177                    &fn_pattern,
178                    &mut violations,
179                );
180                Self::check_empty_test_bodies(
181                    &entry.absolute_path,
182                    &lines,
183                    &test_pattern,
184                    &fn_pattern,
185                    &empty_body_pattern,
186                    &mut violations,
187                );
188                Self::check_stub_assertions(
189                    &entry.absolute_path,
190                    &lines,
191                    &test_pattern,
192                    &stub_assert_pattern,
193                    &fn_pattern,
194                    &mut violations,
195                );
196                Ok(())
197            },
198        )?;
199
200        Ok(violations)
201    }
202
203    fn check_ignored_tests(
204        file: &Path,
205        lines: &[&str],
206        fn_pattern: &Regex,
207        violations: &mut Vec<TestQualityViolation>,
208    ) {
209        for (i, line) in lines.iter().enumerate() {
210            if line.contains("#[ignore]") {
211                // Check if there's a justification comment above
212                let has_justification = i > 0 && {
213                    let prev_line = lines[i - 1];
214                    prev_line.contains("Requires")
215                        || prev_line.contains("requires")
216                        || prev_line.contains(crate::constants::labels::PENDING_LABEL_TODO)
217                        || prev_line.contains("WIP")
218                };
219
220                if !has_justification {
221                    // Find the test function name
222                    if let Some(test_name) = Self::find_test_name(lines, i, fn_pattern) {
223                        violations.push(TestQualityViolation::IgnoreWithoutJustification {
224                            file: file.to_path_buf(),
225                            line: i + 1,
226                            test_name,
227                            severity: Severity::Warning,
228                        });
229                    }
230                }
231            }
232        }
233    }
234
235    fn check_todo_in_fixtures(
236        &self,
237        file: &Path,
238        lines: &[&str],
239        fn_pattern: &Regex,
240        violations: &mut Vec<TestQualityViolation>,
241    ) {
242        if self.should_skip_path(file) {
243            return;
244        }
245
246        for (i, line) in lines.iter().enumerate() {
247            if line.contains("todo!(") {
248                // Check if it's NOT marked as intentional stub
249                let has_stub_marker = i > 0 && {
250                    let prev_line = lines[i - 1];
251                    prev_line.contains("Intentional stub") || prev_line.contains("Test stub")
252                };
253
254                if !has_stub_marker {
255                    // Find the function name
256                    if let Some(function_name) = Self::find_function_name(lines, i, fn_pattern) {
257                        violations.push(TestQualityViolation::TodoInTestFixture {
258                            file: file.to_path_buf(),
259                            line: i + 1,
260                            function_name,
261                            severity: Severity::Error,
262                        });
263                    }
264                }
265            }
266        }
267    }

269    fn check_empty_test_bodies(
270        file: &Path,
271        lines: &[&str],
272        test_pattern: &Regex,
273        fn_pattern: &Regex,
274        empty_body_pattern: &Regex,
275        violations: &mut Vec<TestQualityViolation>,
276    ) {
277        for (i, line) in lines.iter().enumerate() {
278            if test_pattern.is_match(line) {
279                // Find the function declaration
280                if let Some(fn_line_idx) = (i..i + FORWARD_SEARCH_LINES)
281                    .find(|&idx| idx < lines.len() && fn_pattern.is_match(lines[idx]))
282                {
283                    // Check if the function body is empty (just {})
284                    if let Some(body_start) = (fn_line_idx..fn_line_idx + 3)
285                        .find(|&idx| idx < lines.len() && lines[idx].contains('{'))
286                        && (empty_body_pattern.is_match(lines[body_start])
287                            || (body_start + 1 < lines.len()
288                                && lines[body_start + 1].trim() == "}"))
289                        && let Some(test_name) = fn_pattern
290                            .captures(lines[fn_line_idx])
291                            .and_then(|c| c.get(1))
292                    {
293                        violations.push(TestQualityViolation::EmptyTestBody {
294                            file: file.to_path_buf(),
295                            line: fn_line_idx + 1,
296                            test_name: test_name.as_str().to_owned(),
297                            severity: Severity::Error,
298                        });
299                    }
300                }
301            }
302        }
303    }
304
305    fn check_stub_assertions(
306        file: &Path,
307        lines: &[&str],
308        test_pattern: &Regex,
309        stub_assert_pattern: &Regex,
310        fn_pattern: &Regex,
311        violations: &mut Vec<TestQualityViolation>,
312    ) {
313        for (i, line) in lines.iter().enumerate() {
314            if test_pattern.is_match(line) {
315                for offset in 0..MAX_BLOCK_SEARCH_OFFSET {
316                    if i + offset >= lines.len() {
317                        break;
318                    }
319                    if stub_assert_pattern.is_match(lines[i + offset]) {
320                        if let Some(test_name) = Self::find_test_name(lines, i, fn_pattern) {
321                            violations.push(TestQualityViolation::StubTestAssertion {
322                                file: file.to_path_buf(),
323                                line: i + offset + 1,
324                                test_name,
325                                severity: Severity::Warning,
326                            });
327                        }
328                        break;
329                    }
330                }
331            }
332        }
333    }
334
335    fn find_test_name(lines: &[&str], start_idx: usize, fn_pattern: &Regex) -> Option<String> {
336        let end = std::cmp::min(start_idx + FORWARD_SEARCH_LINES, lines.len());
337        for line in lines.iter().take(end).skip(start_idx) {
338            if let Some(captures) = fn_pattern.captures(line)
339                && let Some(name) = captures.get(1)
340            {
341                return Some(name.as_str().to_owned());
342            }
343        }
344        None
345    }
346
347    fn find_function_name(lines: &[&str], start_idx: usize, fn_pattern: &Regex) -> Option<String> {
348        // Look backwards for function name
349        for i in (0..=start_idx).rev().take(FUNCTION_NAME_SEARCH_LINES) {
350            if let Some(captures) = fn_pattern.captures(lines[i])
351                && let Some(name) = captures.get(1)
352            {
353                return Some(name.as_str().to_owned());
354            }
355        }
356        None
357    }
358
359    /// Check if a path should be skipped based on configuration
360    fn should_skip_path(&self, path: &Path) -> bool {
361        let Some(path_str) = path.to_str() else {
362            return false;
363        };
364        self.rules
365            .excluded_paths
366            .iter()
367            .any(|excluded| path_str.contains(excluded))
368    }
