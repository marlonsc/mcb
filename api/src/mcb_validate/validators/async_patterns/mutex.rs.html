4use super::constants::WRONG_MUTEX_PATTERNS;
5use crate::constants::common::TEST_PATH_PATTERNS;
6use crate::filters::LanguageId;
7use crate::pattern_registry::{compile_regex_triples, required_pattern};
8use crate::scan::for_each_scan_file;
9use crate::{Result, Severity, ValidationConfig};

11use super::violation::AsyncViolation;

13/// Detect `std::sync::Mutex` usage in files with async code
14pub fn validate_mutex_types(config: &ValidationConfig) -> Result<Vec<AsyncViolation>> {
15    let mut violations = Vec::new();
16
17    let async_indicator = required_pattern("ASYNC001.async_indicator")?;
18    let compiled_mutex = compile_regex_triples(WRONG_MUTEX_PATTERNS)?;
19
20    for_each_scan_file(config, Some(LanguageId::Rust), false, |entry, _src_dir| {
21        let path = &entry.absolute_path;
22        if path
23            .to_str()
24            .is_some_and(|s| TEST_PATH_PATTERNS.iter().any(|p| s.contains(p)))
25        {
26            return Ok(());
27        }
28
29        let content = std::fs::read_to_string(path)?;
30
31        // Only check files that have async code
32        if !async_indicator.is_match(&content) {
33            return Ok(());
34        }
35
36        crate::validators::for_each_non_test_non_comment_line(
37            &content,
38            |line_num, line, _trimmed| {
39                // Check for std mutex patterns
40                for (pattern, desc, sugg) in &compiled_mutex {
41                    if pattern.is_match(line) {
42                        violations.push(AsyncViolation::WrongMutexType {
43                            file: path.clone(),
44                            line: line_num + 1,
45                            mutex_type: desc.to_string(),
46                            suggestion: sugg.to_string(),
47                            severity: Severity::Warning,
48                        });
49                    }
50                }
51            },
52        );
53
54        Ok(())
55    })?;
56
57    Ok(violations)
58}
