20/// Source used to build the file inventory.
21#[derive(Debug, Clone, Copy, PartialEq, Eq)]
22pub enum FileInventorySource {
23    /// Inventory built using `git ls-files`
24    Git,
25    /// Inventory built by walking the directory recursively
26    WalkDir,
27}
28
29impl FileInventorySource {
30    /// Return the string representation of the source
31    #[must_use]
32    pub const fn as_str(self) -> &'static str {
33        match self {
34            Self::Git => "git",
35            Self::WalkDir => "walkdir",
36        }
37    }
38}

40/// A single file entry in the validation inventory.
41#[derive(Debug, Clone)]
42pub struct InventoryEntry {
43    /// Absolute path to the file on disk
44    pub absolute_path: PathBuf,
45    /// Path relative to the workspace root
46    pub relative_path: PathBuf,
47    /// Language detected during inventory building (single-pass).
48    pub detected_language: Option<LanguageId>,
49}

51/// Context for a single validation run.
52///
53/// Contains the workspace layout, file inventory, and shared caches used during validation.
54#[derive(Debug)]
55pub struct ValidationRunContext {
56    workspace_root: PathBuf,
57    trace_id: String,
58    file_inventory: Arc<Vec<InventoryEntry>>,
59    file_inventory_source: FileInventorySource,
60    content_cache: Mutex<HashMap<PathBuf, Arc<str>>>,
61    rca_cache: Mutex<HashMap<PathBuf, Option<FuncSpace>>>,
62}
63
64thread_local! {
65    static ACTIVE_RUN_CONTEXT: RefCell<Option<Arc<ValidationRunContext>>> = const { RefCell::new(None) };
66}

68impl ValidationRunContext {
69    /// Create a new validation context.
70    ///
71    /// # Errors
72    /// Returns an error if file inventory enumeration fails.
73    pub fn build(config: &ValidationConfig) -> Result<Self> {
74        let file_config = FileConfig::load(&config.workspace_root);
75        let mut ignore_patterns = file_config.general.exclude_patterns.clone();
76        ignore_patterns.extend(config.exclude_patterns.iter().cloned());
77
78        let (entries, source) = enumerate_inventory(&config.workspace_root, &ignore_patterns)?;
79
80        Ok(Self {
81            workspace_root: config.workspace_root.clone(),
82            trace_id: build_trace_id(),
83            file_inventory: Arc::new(entries),
84            file_inventory_source: source,
85            content_cache: Mutex::new(HashMap::new()),
86            rca_cache: Mutex::new(HashMap::new()),
87        })
88    }

152    /// Read file content, using cache if available.
153    ///
154    /// # Errors
155    /// Returns an error if file reading fails.
156    pub fn read_cached(&self, path: &Path) -> std::io::Result<Arc<str>> {
157        let normalized = normalize_path(path)?;
158        if let Ok(cache) = self.content_cache.lock()
159            && let Some(content) = cache.get(&normalized)
160        {
161            return Ok(Arc::clone(content));
162        }
163
164        let content = std::fs::read_to_string(&normalized)?;
165        let value: Arc<str> = Arc::from(content);
166
167        if let Ok(mut cache) = self.content_cache.lock() {
168            cache.insert(normalized, Arc::clone(&value));
169        }
170
171        Ok(value)
172    }
173
174    /// Parse file with RCA, using cache if available.
175    /// Returns cached `FuncSpace` clone on cache hit, otherwise parses and caches.
176    #[must_use]
177    pub fn parse_rca_cached(&self, path: &Path, content: &str) -> Option<FuncSpace> {
178        let normalized = std::fs::canonicalize(path).ok()?;
179
180        // Check cache
181        if let Ok(cache) = self.rca_cache.lock()
182            && let Some(result) = cache.get(&normalized)
183        {
184            return result.clone();
185        }
186
187        // Parse (uncached)
188        let result = crate::ast::rca_helpers::parse_file_spaces_raw(path, content);
189
190        // Store in cache
191        if let Ok(mut cache) = self.rca_cache.lock() {
192            cache.insert(normalized, result.clone());
193        }
194
195        result
196    }

198    /// Execute a closure with the given context set as active for the current thread
199    pub fn with_active<T>(context: &Arc<Self>, f: impl FnOnce() -> T) -> T {
200        ACTIVE_RUN_CONTEXT.with(|slot| {
201            let previous = slot.replace(Some(Arc::clone(context)));
202            let output = f();
203            slot.replace(previous);
204            output
205        })
206    }
207
208    /// Get the active validation context for the current thread
209    #[must_use]
210    pub fn active() -> Option<Arc<Self>> {
211        ACTIVE_RUN_CONTEXT.with(|slot| slot.borrow().as_ref().map(Arc::clone))
212    }
213
214    /// Get the active context or build a new one.
215    ///
216    /// Only reuses the thread-local active context when its `workspace_root`
217    /// matches `config`'s; otherwise builds a fresh one. Without this guard, a
218    /// prior validation on the same thread (e.g. another test with a different
219    /// temp-dir root) would leak its file inventory, yielding zero matching
220    /// files for the current root — a cross-test flake.
221    ///
222    /// # Errors
223    /// Returns an error if the context needs to be built and it fails.
224    pub fn active_or_build(config: &ValidationConfig) -> Result<Arc<Self>> {
225        if let Some(active) = Self::active()
226            && active.workspace_root == config.workspace_root
227        {
228            return Ok(active);
229        }
230
231        Ok(Arc::new(Self::build(config)?))
232    }
233}

235fn enumerate_inventory(
236    workspace_root: &Path,
237    ignore_patterns: &[String],
238) -> Result<(Vec<InventoryEntry>, FileInventorySource)> {
239    let detector = LanguageDetector::new();
240
241    if workspace_root.join(".git").exists()
242        && let Ok(Some(entries)) = enumerate_with_git(workspace_root, ignore_patterns, &detector)
243    {
244        return Ok((entries, FileInventorySource::Git));
245    }
246
247    let entries = enumerate_with_walkdir(workspace_root, ignore_patterns, &detector)?;
248    Ok((entries, FileInventorySource::WalkDir))
249}

251fn enumerate_with_git(
252    workspace_root: &Path,
253    ignore_patterns: &[String],
254    detector: &LanguageDetector,
255) -> std::io::Result<Option<Vec<InventoryEntry>>> {
256    let output = Command::new("git")
257        .arg("-C")
258        .arg(workspace_root)
259        .arg("ls-files")
260        .arg("-co")
261        .arg("--exclude-standard")
262        .output()?;
263
264    if !output.status.success() {
265        return Ok(None);
266    }
267
268    let stdout = String::from_utf8_lossy(&output.stdout);
269    let mut seen = HashSet::new();
270    let mut entries = Vec::new();
271
272    for line in stdout.lines() {
273        if line.is_empty() {
274            continue;
275        }
276
277        let relative = PathBuf::from(line);
278        let absolute = workspace_root.join(&relative);
279
280        if !absolute.is_file() {
281            continue;
282        }
283
284        if should_ignore(line, ignore_patterns) {
285            continue;
286        }
287
288        if seen.insert(relative.clone()) {
289            let abs = normalize_path(&absolute)?;
290            let lang = detector.detect(&abs, None);
291            entries.push(InventoryEntry {
292                absolute_path: abs,
293                relative_path: relative,
294                detected_language: lang,
295            });
296        }
297    }
298
299    Ok(Some(entries))
300}

302fn enumerate_with_walkdir(
303    workspace_root: &Path,
304    ignore_patterns: &[String],
305    detector: &LanguageDetector,
306) -> std::io::Result<Vec<InventoryEntry>> {
307    let mut seen = HashSet::new();
308    let mut entries = Vec::new();
309
310    // Canonicalize workspace_root so strip_prefix works consistently with
311    // canonicalized file paths. On macOS /tmp → /private/tmp symlink and on
312    // Windows the \\?\ prefix would otherwise cause strip_prefix to fail.
313    let canonical_root = normalize_path(workspace_root)?;
314
315    for entry in WalkDir::new(&canonical_root)
316        .follow_links(false)
317        .into_iter()
318        .filter_map(std::result::Result::ok)
319    {
320        if let Some(inventory_entry) = walk_entry_to_inventory(
321            entry.path(),
322            &canonical_root,
323            ignore_patterns,
324            detector,
325            &mut seen,
326        )? {
327            entries.push(inventory_entry);
328        }
329    }
330
331    Ok(entries)
332}

334/// Convert one walked path into an [`InventoryEntry`], skipping ignored/duplicate/non-file paths.
335fn walk_entry_to_inventory(
336    path: &Path,
337    canonical_root: &Path,
338    ignore_patterns: &[String],
339    detector: &LanguageDetector,
340    seen: &mut HashSet<std::path::PathBuf>,
341) -> std::io::Result<Option<InventoryEntry>> {
342    if !path.is_file() {
343        return Ok(None);
344    }
345
346    let absolute = normalize_path(path)?;
347    let Ok(relative) = absolute.strip_prefix(canonical_root) else {
348        return Ok(None);
349    };
350    let relative = relative.to_path_buf();
351
352    let Some(relative_str) = relative.to_str() else {
353        return Ok(None);
354    };
355    if should_ignore(relative_str, ignore_patterns)
356        || relative_str.contains("/.git/")
357        || relative_str.starts_with(".git/")
358    {
359        return Ok(None);
360    }
361
362    if !seen.insert(relative.clone()) {
363        return Ok(None);
364    }
365
366    let lang = detector.detect(&absolute, None);
367    Ok(Some(InventoryEntry {
368        absolute_path: absolute,
369        relative_path: relative,
370        detected_language: lang,
371    }))
372}
373
374fn should_ignore(path: &str, ignore_patterns: &[String]) -> bool {
375    ignore_patterns.iter().any(|pattern| path.contains(pattern))
376}

378fn build_trace_id() -> String {
