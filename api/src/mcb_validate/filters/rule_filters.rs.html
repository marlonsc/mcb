17/// File/directory pattern filter for allow/deny/skip applicability.
18#[derive(Debug, Clone, Default, Serialize, Deserialize)]
19pub struct ApplicabilityFilter {
20    /// Glob patterns matching file names or paths.
21    #[serde(alias = "files")]
22    pub file_patterns: Option<Vec<String>>,
23    /// Glob patterns matching directory names or paths.
24    pub directory_patterns: Option<Vec<String>>,
25}
26
27impl ApplicabilityFilter {
28    /// Returns `true` when neither file nor directory patterns are defined.
29    #[must_use]
30    pub fn is_empty(&self) -> bool {
31        self.file_patterns
32            .as_ref()
33            .is_none_or(std::vec::Vec::is_empty)
34            && self
35                .directory_patterns
36                .as_ref()
37                .is_none_or(std::vec::Vec::is_empty)
38    }
39}

41/// Filter configuration for a rule.
42///
43/// Precedence: `skip` > `deny` > `allow`.
44#[derive(Debug, Clone, Serialize, Deserialize)]
45pub struct RuleFilters {
46    /// Language identifiers the rule applies to (e.g. `["rust", "python"]`).
47    pub languages: Option<Vec<String>>,
48    /// Crate/package names whose presence activates this rule.
49    pub dependencies: Option<Vec<String>>,
50    /// Glob patterns for files the rule should match.
51    pub file_patterns: Option<Vec<String>>,
52    /// Paths explicitly allowed (overridden by `deny` and `skip`).
53    pub allow: Option<ApplicabilityFilter>,
54    /// Paths explicitly denied (overridden by `skip`).
55    pub deny: Option<ApplicabilityFilter>,
56    /// Paths unconditionally excluded from this rule.
57    pub skip: Option<ApplicabilityFilter>,
58}
59
60impl RuleFilters {
61    /// Returns `true` when no filter criteria are configured.
62    pub fn is_empty(&self) -> bool {
63        let no_file_filters =
64            self.languages.is_none() && self.dependencies.is_none() && self.file_patterns.is_none();
65        let no_allow = self
66            .allow
67            .as_ref()
68            .is_none_or(ApplicabilityFilter::is_empty);
69        let no_deny = self.deny.as_ref().is_none_or(ApplicabilityFilter::is_empty);
70        let no_skip = self.skip.as_ref().is_none_or(ApplicabilityFilter::is_empty);
71        no_file_filters && no_allow && no_deny && no_skip
72    }
73}

75/// Executor for rule filters
76pub struct RuleFilterExecutor {
77    workspace_root: std::path::PathBuf,
78    language_detector: LanguageDetector,
79    dependency_parser: CargoDependencyParser,
80    file_matcher: FilePatternMatcher,
81}
82
83impl RuleFilterExecutor {
84    /// Create a new filter executor
85    #[must_use]
86    pub fn new(workspace_root: std::path::PathBuf) -> Self {
87        Self {
88            workspace_root: workspace_root.clone(),
89            language_detector: LanguageDetector::new(),
90            dependency_parser: CargoDependencyParser::new(workspace_root),
91            file_matcher: FilePatternMatcher::default(),
92        }
93    }
94
95    /// Check if a rule should execute on a given file
96    ///
97    /// # Arguments
98    /// * `filters` - Filter configuration for the rule
99    /// * `file_path` - Path to the file being checked
100    /// * `file_content` - Optional content of the file (for language detection)
101    /// * `workspace_deps` - Workspace dependency information
102    ///
103    /// # Returns
104    /// true if the rule should execute on this file
105    /// Check if a rule should execute on a given file
106    ///
107    /// # Arguments
108    /// * `filters` - Filter configuration for the rule
109    /// * `file_path` - Path to the file being checked
110    /// * `file_content` - Optional content of the file (for language detection)
111    /// * `workspace_deps` - Workspace dependency information
112    ///
113    /// # Returns
114    /// true if the rule should execute on this file
115    ///
116    /// # Errors
117    ///
118    /// Returns an error if filter evaluation encounters a configuration issue.
119    pub fn should_execute_rule(
120        &self,
121        filters: &RuleFilters,
122        file_path: &Path,
123        file_content: Option<&str>,
124        workspace_deps: &WorkspaceDependencies,
125    ) -> crate::Result<bool> {
126        if filters.is_empty() {
127            return Ok(true);
128        }
129
130        let language_ok = filters.languages.as_ref().is_none_or(|languages| {
131            self.language_detector
132                .matches_languages(file_path, file_content, languages)
133        });
134
135        let dependencies_ok = filters.dependencies.as_ref().is_none_or(|required_deps| {
136            Self::check_dependencies(required_deps, file_path, workspace_deps)
137        });
138
139        let patterns_ok = filters
140            .file_patterns
141            .as_ref()
142            .is_none_or(|patterns| self.matches_patterns_with_fallback(file_path, patterns));
143
144        let skip_match = filters
145            .skip
146            .as_ref()
147            .is_some_and(|skip| self.matches_filter(file_path, skip));
148
149        let deny_match = filters
150            .deny
151            .as_ref()
152            .is_some_and(|deny| self.matches_filter(file_path, deny));
153        let allow_match = filters
154            .allow
155            .as_ref()
156            .is_some_and(|allow| self.matches_filter(file_path, allow));
157
158        let access_ok = !skip_match && (!deny_match || allow_match);
159
160        Ok(language_ok && dependencies_ok && patterns_ok && access_ok)
161    }
162
163    fn relative_path<'a>(&'a self, file_path: &'a Path) -> &'a Path {
164        file_path
165            .strip_prefix(&self.workspace_root)
166            .unwrap_or(file_path)
167    }
168
169    fn matches_patterns_with_fallback(&self, file_path: &Path, patterns: &[String]) -> bool {
170        let rel_path = self.relative_path(file_path);
171        self.file_matcher.matches_any(rel_path, patterns)
172            || (rel_path != file_path && self.file_matcher.matches_any(file_path, patterns))
173    }
174
175    fn matches_filter(&self, file_path: &Path, filter: &ApplicabilityFilter) -> bool {
176        let file_match = filter
177            .file_patterns
178            .as_ref()
179            .is_some_and(|patterns| self.matches_patterns_with_fallback(file_path, patterns));
180        let dir_match = filter.directory_patterns.as_ref().is_some_and(|patterns| {
181            self.file_matcher
182                .matches_any(self.relative_path(file_path), patterns)
183        });
184        file_match || dir_match
185    }
186
187    /// Check if required dependencies are present in the file's crate
188    fn check_dependencies(
189        required_deps: &[String],
190        file_path: &Path,
191        workspace_deps: &WorkspaceDependencies,
192    ) -> bool {
193        if let Some(crate_deps) = workspace_deps.find_crate_deps(file_path) {
194            required_deps
195                .iter()
196                .all(|dep| crate_deps.has_dependency(dep))
197        } else {
198            // If we can't determine the crate, assume dependencies are not present
199            false
200        }
201    }
202
203    /// Parse workspace dependencies (can be cached)
204    ///
205    /// # Errors
206    ///
207    /// Returns an error if Cargo.toml parsing fails.
208    pub fn parse_workspace_dependencies(&self) -> crate::Result<WorkspaceDependencies> {
209        self.dependency_parser.parse_workspace_deps()
210    }
211
212    /// Create a file matcher for specific patterns
213    ///
214    /// # Errors
215    ///
216    /// Returns an error if any glob pattern is invalid.
217    pub fn create_file_matcher(&self, patterns: &[String]) -> crate::Result<FilePatternMatcher> {
218        FilePatternMatcher::from_mixed_patterns(patterns)
219            .map_err(|e| crate::ValidationError::Config(format!("Invalid file pattern: {e}")))
220    }

222    /// Get the language detector for direct use
223    #[must_use]
224    pub fn language_detector(&self) -> &LanguageDetector {
225        &self.language_detector
226    }
227
228    /// Get the dependency parser for direct use
229    #[must_use]
230    pub fn dependency_parser(&self) -> &CargoDependencyParser {
231        &self.dependency_parser
232    }
