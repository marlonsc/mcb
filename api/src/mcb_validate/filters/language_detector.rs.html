7/// Map a `LanguageId` to its `rust-code-analysis` (RCA) equivalent.
8#[must_use]
9pub fn language_to_rca(lang: LanguageId) -> LANG {
10    match lang {
11        LanguageId::Rust => LANG::Rust,
12        LanguageId::Python => LANG::Python,
13        LanguageId::JavaScript => LANG::Mozjs,
14        LanguageId::TypeScript => LANG::Typescript,
15        LanguageId::Tsx => LANG::Tsx,
16        LanguageId::Java => LANG::Java,
17        LanguageId::C | LanguageId::Cpp => LANG::Cpp,
18        LanguageId::Kotlin => LANG::Kotlin,
19        LanguageId::Go
20        | LanguageId::Ruby
21        | LanguageId::Shell
22        | LanguageId::Yaml
23        | LanguageId::Toml
24        | LanguageId::Json
25        | LanguageId::Markdown
26        | LanguageId::Html
27        | LanguageId::Css
28        | LanguageId::Sql
29        | LanguageId::Dockerfile
30        | LanguageId::Makefile
31        | LanguageId::Php
32        | LanguageId::Swift
33        | LanguageId::Protobuf => LANG::Preproc,
34    }
35}

37/// Map a `rust-code-analysis` (RCA) language to our `LanguageId`.
38#[must_use]
39pub fn language_from_rca(lang: LANG) -> Option<LanguageId> {
40    match lang {
41        LANG::Rust => Some(LanguageId::Rust),
42        LANG::Python => Some(LanguageId::Python),
43        LANG::Mozjs | LANG::Javascript => Some(LanguageId::JavaScript),
44        LANG::Typescript => Some(LanguageId::TypeScript),
45        LANG::Tsx => Some(LanguageId::Tsx),
46        LANG::Java => Some(LanguageId::Java),
47        LANG::Cpp => Some(LanguageId::Cpp),
48        LANG::Kotlin => Some(LanguageId::Kotlin),
49        LANG::Ccomment | LANG::Preproc => None,
50    }
51}

53/// Component for detecting the programming language of a file.
54pub struct LanguageDetector {

58impl Default for LanguageDetector {

64impl LanguageDetector {
65    /// Create a new language detector instance.
66    #[must_use]
67    pub fn new() -> Self {
68        Self { _private: () }
69    }

71    /// Detect the language of a file at the given path, optionally using its content.
72    pub fn detect(&self, path: &Path, content: Option<&str>) -> Option<LanguageId> {
73        if let Some(ext) = path.extension().and_then(|ext| ext.to_str())
74            && let Some(lang) = LanguageId::from_extension(ext)
75        {
76            return Some(lang);
77        }
78
79        if let Some(filename) = path.file_name().and_then(|name| name.to_str())
80            && let Some(lang) = LanguageId::from_filename(filename)
81        {
82            return Some(lang);
83        }
84
85        let source = content.map_or_else(
86            || std::fs::read(path).unwrap_or_default(),
87            |c| c.as_bytes().to_vec(),
88        );
89        let (rca_lang, _) = guess_language(&source, path);
90        if let Some(lang) = rca_lang.and_then(language_from_rca) {
91            return Some(lang);
92        }
93
94        let first_line = content.and_then(|body| body.lines().next()).or_else(|| {
95            std::str::from_utf8(&source)
96                .ok()
97                .and_then(|body| body.lines().next())
98        });
99
100        first_line.and_then(LanguageId::from_shebang)
101    }
102
103    /// Detect language and return its name.
104    #[must_use]
105    pub fn detect_name(&self, path: &Path, content: Option<&str>) -> Option<String> {
106        self.detect(path, content).map(|id| id.name().to_owned())
107    }
108
109    /// Detect language and return the specific RCA enum if supported.
110    #[must_use]
111    pub fn detect_rca_lang(&self, path: &Path, content: Option<&str>) -> Option<LANG> {
112        if let Some(ext) = path.extension().and_then(|e| e.to_str())
113            && let Some(lang) = LanguageId::from_extension(ext)
114        {
115            return Some(language_to_rca(lang));
116        }
117
118        let source = content.map_or_else(
119            || std::fs::read(path).unwrap_or_default(),
120            |c| c.as_bytes().to_vec(),
121        );
122        let (rca_lang, _) = guess_language(&source, path);
123        rca_lang.and_then(|lang| language_from_rca(lang).map(|_| lang))
124    }

127    #[must_use]

152
153    /// Check if the detected language of a file is in the allowed list.
154    #[must_use]
155    pub fn matches_languages(
156        &self,
157        path: &Path,
158        content: Option<&str>,
159        allowed_languages: &[String],
160    ) -> bool {
161        self.detect_name(path, content)
162            .is_some_and(|language| allowed_languages.contains(&language))
163    }
