4//! Language Detection for Rule Filtering
5//!
6//! Detects programming language from file extensions and content patterns.
7//! Uses `rust-code-analysis` directly for language detection.
8//!
9//! Previously delegated to the `mcb-language-support` crate; all logic is now
10//! inlined here to remove that dependency.

17/// Supported programming languages.
18///
19/// Maps to `rust_code_analysis::LANG` while providing a stable,
20/// serializable representation. Only languages that RCA actually
21/// supports are included.
22#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, serde::Serialize, serde::Deserialize)]
23#[serde(rename_all = "lowercase")]
24pub enum LanguageId {
25    /// Rust
26    Rust,
27    /// Python
28    Python,
29    /// JavaScript (including JSX)
30    JavaScript,
31    /// TypeScript (including TSX)
32    TypeScript,
33    /// Java
34    Java,
35    /// C / C++ (RCA treats them identically)
36    Cpp,
37    /// Kotlin
38    Kotlin,
39    /// Go
40    Go,
41    /// Ruby
42    Ruby,
43    /// Shell (bash, zsh, ksh)
44    Shell,
45    /// YAML
46    Yaml,
47    /// TOML
48    Toml,
49    /// JSON / JSONC
50    Json,
51    /// Markdown
52    Markdown,
53    /// HTML / XHTML
54    Html,
55    /// CSS / SCSS / SASS / Less
56    Css,
57    /// SQL
58    Sql,
59    /// Dockerfile
60    Dockerfile,
61    /// Makefile
62    Makefile,
63    /// Protocol Buffers
64    Protobuf,
65}

119    /// Get the canonical name of the language.
120    #[must_use]

124            Self::Python => "python",
125            Self::JavaScript => "javascript",

146    /// Convert to `rust_code_analysis::LANG`.
147    #[must_use]
148    pub fn to_rca_lang(&self) -> LANG {
149        match self {
150            Self::Rust => LANG::Rust,
151            Self::Python => LANG::Python,
152            Self::JavaScript => LANG::Mozjs,
153            Self::TypeScript => LANG::Typescript,
154            Self::Java => LANG::Java,
155            Self::Cpp => LANG::Cpp,
156            Self::Kotlin => LANG::Kotlin,
157            Self::Go
158            | Self::Ruby
159            | Self::Shell
160            | Self::Yaml
161            | Self::Toml
162            | Self::Json
163            | Self::Markdown
164            | Self::Html
165            | Self::Css
166            | Self::Sql
167            | Self::Dockerfile
168            | Self::Makefile
169            | Self::Protobuf => LANG::Preproc,
170        }
171    }

173    /// Create from `rust_code_analysis::LANG`.
174    #[must_use]
175    pub fn from_rca_lang(lang: LANG) -> Option<Self> {
176        match lang {
177            LANG::Rust => Some(Self::Rust),
178            LANG::Python => Some(Self::Python),
179            LANG::Mozjs | LANG::Javascript => Some(Self::JavaScript),
180            LANG::Typescript | LANG::Tsx => Some(Self::TypeScript),
181            LANG::Java => Some(Self::Java),
182            LANG::Cpp => Some(Self::Cpp),
183            LANG::Kotlin => Some(Self::Kotlin),
184            LANG::Ccomment | LANG::Preproc => None,
185        }
186    }

215    /// Try to create from a string name (case-insensitive).
216    #[must_use]
217    pub fn from_name(name: &str) -> Option<Self> {
218        let lower = name.to_ascii_lowercase();
219        Self::NAME_EQUIVALENTS
220            .iter()
221            .find_map(|(name, lang)| (*name == lower).then_some(*lang))
222    }

224    /// Resolve a `LanguageId` from a file extension (case-insensitive, leading dot stripped).
225    #[must_use]
226    pub fn from_extension(ext: &str) -> Option<Self> {
227        let normalized = ext.trim().trim_start_matches('.').to_ascii_lowercase();
228        match normalized.as_str() {
229            "rs" => Some(Self::Rust),
230            "py" | "pyi" | "pyw" => Some(Self::Python),
231            "js" | "mjs" | "cjs" | "jsx" => Some(Self::JavaScript),
232            "ts" | "mts" | "cts" | "tsx" => Some(Self::TypeScript),
233            "java" => Some(Self::Java),
234            "c" | "h" | "cpp" | "cc" | "cxx" | "hpp" | "hxx" | "mm" | "m" => Some(Self::Cpp),
235            "kt" | "kts" => Some(Self::Kotlin),
236            "go" => Some(Self::Go),
237            "rb" | "rake" | "gemspec" => Some(Self::Ruby),
238            "sh" | "bash" | "zsh" | "ksh" => Some(Self::Shell),
239            "yaml" | "yml" => Some(Self::Yaml),
240            "toml" => Some(Self::Toml),
241            "json" | "jsonc" => Some(Self::Json),
242            "md" | "markdown" | "mdown" | "mkd" => Some(Self::Markdown),
243            "html" | "htm" | "xhtml" => Some(Self::Html),
244            "css" | "scss" | "sass" | "less" => Some(Self::Css),
245            "sql" => Some(Self::Sql),
246            "proto" => Some(Self::Protobuf),
247            "mk" => Some(Self::Makefile),
248            _ => None,
249        }
250    }
251
252    /// Resolve a `LanguageId` from a well-known filename (e.g. `Dockerfile`, `Makefile`).
253    #[must_use]
254    pub fn from_filename(filename: &str) -> Option<Self> {
255        let lower = filename.to_ascii_lowercase();
256        match lower.as_str() {
257            "dockerfile" => Some(Self::Dockerfile),
258            "makefile" | "gnumakefile" => Some(Self::Makefile),
259            "cmakelists.txt" => Some(Self::Cpp),
260            _ => None,
261        }
262    }
263
264    /// Resolve a `LanguageId` from a shebang line (e.g. `#!/usr/bin/env python3`).
265    #[must_use]
266    pub fn from_shebang(first_line: &str) -> Option<Self> {
267        let line = first_line.trim().to_ascii_lowercase();
268        line.starts_with("#!")
269            .then(|| {
270                if line.contains("python") {
271                    Self::Python
272                } else if line.contains("ts-node") {
273                    Self::TypeScript
274                } else if line.contains("node") || line.contains("deno") {
275                    Self::JavaScript
276                } else if ["bash", "sh", "zsh", "ksh"]
277                    .iter()
278                    .any(|shell| line.contains(shell))
279                {
280                    Self::Shell
281                } else {
282                    Self::Ruby
283                }
284            })
285            .filter(|lang| *lang != Self::Ruby || line.contains("ruby"))
286    }
287}

289impl FromStr for LanguageId {
290    type Err = String;
291
292    fn from_str(s: &str) -> std::result::Result<Self, Self::Err> {
293        Self::from_name(s).ok_or_else(|| format!("Unknown language: {s}"))
294    }

303/// Detects programming language from file paths using `rust_code_analysis::guess_language`.
304pub struct LanguageDetector {
305    _private: (),
306}
307
308impl Default for LanguageDetector {
309    fn default() -> Self {
310        Self::new()
311    }
312}
313
314impl LanguageDetector {
315    /// Create a new language detector.
316    #[must_use]
317    pub fn new() -> Self {
318        Self { _private: () }
319    }
320
321    /// Detect language from file path, returning a [`LanguageId`].
322    pub fn detect(&self, path: &Path, content: Option<&str>) -> Option<LanguageId> {
323        let source = content.map_or_else(
324            || std::fs::read(path).unwrap_or_default(),
325            |c| c.as_bytes().to_vec(),
326        );
327        let (rca_lang, _) = guess_language(&source, path);
328        if let Some(lang) = rca_lang.and_then(LanguageId::from_rca_lang) {
329            return Some(lang);
330        }
331
332        if let Some(ext) = path.extension().and_then(|ext| ext.to_str())
333            && let Some(lang) = LanguageId::from_extension(ext)
334        {
335            return Some(lang);
336        }
337
338        if let Some(filename) = path.file_name().and_then(|name| name.to_str())
339            && let Some(lang) = LanguageId::from_filename(filename)
340        {
341            return Some(lang);
342        }
343
344        let first_line = content.and_then(|body| body.lines().next()).or_else(|| {
345            std::str::from_utf8(&source)
346                .ok()
347                .and_then(|body| body.lines().next())
348        });
349
350        first_line.and_then(LanguageId::from_shebang)
351    }
352
353    /// Detect language and return as string name.
354    #[must_use]
355    pub fn detect_name(&self, path: &Path, content: Option<&str>) -> Option<String> {
356        self.detect(path, content).map(|id| id.name().to_owned())
357    }
358
359    /// Detect language and return `LANG` enum for direct RCA usage.
360    #[must_use]
361    pub fn detect_rca_lang(&self, path: &Path, content: Option<&str>) -> Option<LANG> {
362        let source = content.map_or_else(
363            || std::fs::read(path).unwrap_or_default(),
364            |c| c.as_bytes().to_vec(),
365        );
366        let (rca_lang, _) = guess_language(&source, path);
367        rca_lang.and_then(|lang| LanguageId::from_rca_lang(lang).map(|_| lang))
368    }
369
370    /// All supported language names.
371    #[must_use]
372    pub fn supported_language_names(&self) -> Vec<String> {
373        vec![
374            "rust".to_owned(),
375            "python".to_owned(),
376            "javascript".to_owned(),
377            "typescript".to_owned(),
378            "go".to_owned(),
379            "java".to_owned(),
380            "cpp".to_owned(),
381            "kotlin".to_owned(),
382            "ruby".to_owned(),
383            "shell".to_owned(),
384            "yaml".to_owned(),
385            "toml".to_owned(),
386            "json".to_owned(),
387            "markdown".to_owned(),
388            "html".to_owned(),
389            "css".to_owned(),
390            "sql".to_owned(),
391            "dockerfile".to_owned(),
392            "makefile".to_owned(),
393            "protobuf".to_owned(),
394        ]
395    }
396
397    /// Check if a file matches any of the specified language names.
398    #[must_use]
399    pub fn matches_languages(
400        &self,
401        path: &Path,
402        content: Option<&str>,
403        allowed_languages: &[String],
404    ) -> bool {
405        self.detect_name(path, content)
406            .is_some_and(|language| allowed_languages.contains(&language))
407    }
