29/// Loaded and validated YAML rule
30#[derive(Debug, Clone)]
31pub struct ValidatedRule {
32    /// Unique identifier for the rule.
33    pub id: String,
34    /// Human-readable name of the rule.
35    pub name: String,
36    /// Category of the rule (e.g., quality, security).
37    pub category: String,
38    /// Severity level (error, warning, info).
39    pub severity: String,
40    /// Whether the rule is active.
41    pub enabled: bool,
42    /// Detailed description of what the rule checks.
43    pub description: String,
44    /// Explanation of why this rule exists.
45    pub rationale: String,
46    /// The engine used to execute this rule.
47    pub engine: String,
48    /// Engine-specific configuration.
49    pub config: serde_json::Value,
50    /// Raw rule definition.
51    pub rule_definition: serde_json::Value,
52    /// List of available automated fixes.
53    pub fixes: Vec<RuleFix>,
54    /// Linter codes to execute (e.g., `["F401"]` for Ruff, `["clippy::unwrap_used"]` for Clippy)
55    pub lint_select: Vec<String>,
56    /// Custom message for violations
57    pub message: Option<String>,
58    /// AST selectors for multi-language pattern matching (Phase 2)
59    pub selectors: Vec<AstSelector>,
60    /// Tree-sitter query string for complex AST matching (Phase 2)
61    pub ast_query: Option<String>,
62    /// Metrics configuration for schema v3 rules (Phase 4)
63    pub metrics: Option<MetricsConfig>,
64    /// Optional filters to restrict rule applicability by language, dependency, or file pattern.
65    pub filters: Option<RuleFilters>,
66}

187    /// Set embedded rules for the loader.
188    pub fn set_embedded_rules(&mut self, rules: Vec<(&str, &str)>) {
189        self.embedded_rules = Some(
190            rules
191                .into_iter()
192                .map(|(path, content)| (path.to_owned(), content.to_owned()))
193                .collect(),
194        );

197    /// Load all rules from embedded entries without filesystem access.
198    ///
199    /// # Errors
200    ///
201    /// Returns an error if template loading or rule parsing fails.
202    pub fn load_embedded_rules(&mut self) -> Result<Vec<ValidatedRule>> {
203        let mut rules = Vec::new();
204
205        if let Some(embedded_rules) = &self.embedded_rules {
206            self.template_engine
207                .load_templates_from_embedded(embedded_rules)?;
208
209            for (path, content) in embedded_rules {
210                if path.ends_with(".yml") && !path.contains("/templates/") {
211                    let loaded_rules = self.load_rule_from_str(Path::new(path), content)?;
212                    rules.extend(loaded_rules);
213                }
214            }
215        }
216
217        Ok(rules)
218    }

220    /// Synchronous variant of [`Self::load_all_rules`].
221    ///
222    /// # Errors
223    ///
224    /// Returns an error if template loading, file reading, or rule parsing fails.
225    pub fn load_all_rules_sync(&mut self) -> Result<Vec<ValidatedRule>> {
226        let mut rules = Vec::new();
227
228        if self.embedded_rules.is_some() {
229            rules.extend(self.load_embedded_rules()?);
230        }
231
232        if self.rules_dir.exists() {
233            self.template_engine.load_templates_sync(&self.rules_dir)?;
234
235            for path in collect_yaml_files(&self.rules_dir)? {
236                if Self::is_rule_file(&path) {
237                    let content =
238                        std::fs::read_to_string(&path).map_err(crate::ValidationError::Io)?;
239                    let loaded_rules = self.load_rule_from_str(&path, &content)?;
240                    rules.extend(loaded_rules);
241                }
242            }
243        }
244
245        Ok(rules)
246    }

248    /// Load all rules from the rules directory
249    ///
250    /// # Errors
251    ///
252    /// Returns an error if template loading, file reading, or rule parsing fails.
253    pub async fn load_all_rules(&mut self) -> Result<Vec<ValidatedRule>> {
254        if let Some(embedded_rules) = &self.embedded_rules
255            && !embedded_rules.is_empty()
256        {
257            return self.load_embedded_rules();
258        }
259
260        let mut rules = Vec::new();
261
262        // Load templates first
263        self.template_engine.load_templates(&self.rules_dir).await?;
264
265        // Load rule files
266        for path in collect_yaml_files(&self.rules_dir)? {
267            if Self::is_rule_file(&path) {
268                let loaded_rules = self.load_rule_file(&path).await?;
269                rules.extend(loaded_rules);
270            }
271        }
272
273        Ok(rules)
274    }

276    /// Load rules from a specific file
277    ///
278    /// # Errors
279    ///
280    /// Returns an error if the file cannot be read or YAML parsing fails.
281    pub async fn load_rule_file(&self, path: &Path) -> Result<Vec<ValidatedRule>> {
282        let content = tokio::fs::read_to_string(path)
283            .await
284            .map_err(crate::ValidationError::Io)?;
285
286        self.load_rule_from_str(path, &content)
287    }

289    fn load_rule_from_str(&self, path: &Path, content: &str) -> Result<Vec<ValidatedRule>> {
290        let yaml_value: serde_yaml::Value =
291            serde_yaml::from_str(content).map_err(|e| crate::ValidationError::Parse {
292                file: path.to_path_buf(),
293                message: format!("YAML parse error: {e}"),
294            })?;
295
296        // Check if this is a template
297        if yaml_value
298            .get(YAML_FIELD_BASE)
299            .and_then(serde_yaml::Value::as_bool)
300            .unwrap_or(false)
301        {
302            // This is a template, skip it
303            return Ok(vec![]);
304        }
305
306        // Apply template if specified
307        let processed_yaml = if let Some(template_name) =
308            yaml_value.get(YAML_FIELD_TEMPLATE).and_then(|v| v.as_str())
309        {
310            // Merge variables into rule definition so template specific logic can access them
311            let template_args = if let Some(vars) = &self.variables {
312                let mut args = vars.clone();
313                // Simple shallow merge of rule over variables (for template arguments)
314                if let serde_yaml::Value::Mapping(args_map) = &mut args
315                    && let serde_yaml::Value::Mapping(rule_map) = &yaml_value
316                {
317                    for (k, v) in rule_map {
318                        args_map.insert(k.clone(), v.clone());
319                    }
320                }
321                args
322            } else {
323                yaml_value.clone()
324            };
325
326            self.template_engine
327                .apply_template(template_name, &template_args)?
328        } else {
329            yaml_value
330        };
331
332        // Handle extends
333        let processed_yaml = if let Some(extends_name) = processed_yaml
334            .get(YAML_FIELD_EXTENDS)
335            .and_then(|v| v.as_str())
336        {
337            self.template_engine
338                .extend_rule(extends_name, &processed_yaml)?
339        } else {
340            processed_yaml
341        };
342
343        // Substitute globals if provided
344        let mut final_yaml = processed_yaml;
345        if let Some(vars) = &self.variables {
346            self.template_engine
347                .substitute_variables(&mut final_yaml, vars)?;
348        }
349
350        // Convert to JSON for validating
351        let json_value: serde_json::Value =
352            serde_json::to_value(final_yaml).map_err(|e| crate::ValidationError::Parse {
353                file: path.to_path_buf(),
354                message: format!("YAML to JSON conversion error: {e}"),
355            })?;
356
357        // Validate against schema
358        self.validator.validate_rule(&json_value)?;
359
360        // Convert to validated rule
361        let validated_rule = Self::yaml_to_validated_rule(&json_value)?;
362
363        Ok(vec![validated_rule])
364    }

366    /// Check if a file is a rule file
367    fn is_rule_file(path: &Path) -> bool {
368        path.extension().and_then(|ext| ext.to_str()) == Some("yml")
369            && !path.to_str().is_some_and(|s| s.contains("/templates/"))
370    }

372    /// Convert YAML/JSON value to `ValidatedRule`
373    fn yaml_to_validated_rule(value: &serde_json::Value) -> Result<ValidatedRule> {
374        let obj = value
375            .as_object()
376            .ok_or_else(|| crate::ValidationError::Config("Rule must be an object".to_owned()))?;
377
378        let id = obj
379            .get(YAML_FIELD_ID)
380            .and_then(|v| v.as_str())
381            .ok_or_else(|| {
382                crate::ValidationError::Config("Rule must have an 'id' field".to_owned())
383            })?
384            .to_owned();
385
386        let name = obj
387            .get(YAML_FIELD_NAME)
388            .and_then(|v| v.as_str())
389            .unwrap_or(DEFAULT_RULE_NAME)
390            .to_owned();
391
392        let category = obj
393            .get(YAML_FIELD_CATEGORY)
394            .and_then(|v| v.as_str())
395            .unwrap_or(DEFAULT_RULE_CATEGORY)
396            .to_owned();
397
398        let severity = obj
399            .get(YAML_FIELD_SEVERITY)
400            .and_then(|v| v.as_str())
401            .unwrap_or(DEFAULT_RULE_SEVERITY)
402            .to_owned();
403
404        let enabled = obj
405            .get(YAML_FIELD_ENABLED)
406            .and_then(serde_json::Value::as_bool)
407            .unwrap_or(true);
408
409        let description = obj
410            .get(YAML_FIELD_DESCRIPTION)
411            .and_then(|v| v.as_str())
412            .unwrap_or(DEFAULT_RULE_DESCRIPTION)
413            .to_owned();
414
415        let rationale = obj
416            .get(YAML_FIELD_RATIONALE)
417            .and_then(|v| v.as_str())
418            .unwrap_or(DEFAULT_RULE_RATIONALE)
419            .to_owned();
420
421        let engine = obj
422            .get(YAML_FIELD_ENGINE)
423            .and_then(|v| v.as_str())
424            .unwrap_or(DEFAULT_RULE_ENGINE)
425            .to_owned();
426
427        let config = obj
428            .get(YAML_FIELD_CONFIG)
429            .cloned()
430            .unwrap_or(serde_json::Value::Object(serde_json::Map::new()));
431
432        let rule_definition = obj
433            .get(YAML_FIELD_RULE)
434            .cloned()
435            .unwrap_or(serde_json::Value::Object(serde_json::Map::new()));
436
437        let fixes = obj
438            .get(YAML_FIELD_FIXES)
439            .and_then(|v| v.as_array())
440            .map(|arr| {
441                arr.iter()
442                    .filter_map(|fix| {
443                        if let Some(fix_obj) = fix.as_object() {
444                            Some(RuleFix {
445                                fix_type: fix_obj.get(YAML_FIELD_FIX_TYPE)?.as_str()?.to_owned(),
446                                pattern: fix_obj
447                                    .get(YAML_FIELD_PATTERN)
448                                    .and_then(|v| v.as_str())
449                                    .map(str::to_owned),
450                                message: fix_obj.get(YAML_FIELD_MESSAGE)?.as_str()?.to_owned(),
451                            })
452                        } else {
453                            None
454                        }
455                    })
456                    .collect()
457            })
458            .unwrap_or_default();
459
460        // Extract lint_select codes (for Ruff/Clippy integration)
461        let lint_select = obj
462            .get(YAML_FIELD_LINT_SELECT)
463            .and_then(|v| v.as_array())
464            .map(|arr| {
465                arr.iter()
466                    .filter_map(|code| code.as_str().map(str::to_owned))
467                    .collect()
468            })
469            .unwrap_or_default();
470
471        // Extract custom message
472        let message = obj
473            .get(YAML_FIELD_MESSAGE)
474            .and_then(|v| v.as_str())
475            .map(str::to_owned);
476
477        // Extract AST selectors (Phase 2)
478        let selectors = obj
479            .get(YAML_FIELD_SELECTORS)
480            .and_then(|v| v.as_array())
481            .map(|arr| {
482                arr.iter()
483                    .filter_map(|sel| {
484                        if let Some(sel_obj) = sel.as_object() {
485                            Some(AstSelector {
486                                language: sel_obj.get(YAML_FIELD_LANGUAGE)?.as_str()?.to_owned(),
487                                node_type: sel_obj.get(YAML_FIELD_NODE_TYPE)?.as_str()?.to_owned(),
488                                pattern: sel_obj
489                                    .get(YAML_FIELD_PATTERN)
490                                    .and_then(|v| v.as_str())
491                                    .map(str::to_owned),
492                            })
493                        } else {
494                            None
495                        }
496                    })
497                    .collect()
498            })
499            .unwrap_or_default();
500
501        // Extract ast_query (Phase 2)
502        let ast_query = obj
503            .get(YAML_FIELD_AST_QUERY)
504            .and_then(|v| v.as_str())
505            .map(str::to_owned);
506
507        // Extract metrics configuration (Phase 4 - rule/v3)
508        let metrics = obj
509            .get(YAML_FIELD_METRICS)
510            .and_then(|v| serde_json::from_value::<MetricsConfig>(v.clone()).ok());
511
512        let filters = obj
513            .get(YAML_FIELD_FILTERS)
514            .and_then(|v| serde_json::from_value::<RuleFilters>(v.clone()).ok());
515
516        Ok(ValidatedRule {
517            id,
518            name,
519            category,
520            severity,
521            enabled,
522            description,
523            rationale,
524            engine,
525            config,
526            rule_definition,
527            fixes,
528            lint_select,
529            message,
530            selectors,
531            ast_query,
532            metrics,
533            filters,
534        })
535    }

537    /// Get rule file path for a rule ID
538    #[must_use]
539    pub fn get_rule_path(&self, rule_id: &str) -> Option<PathBuf> {
540        // This would need a more sophisticated mapping
541        // For now, just search in the rules directory
542        for path in collect_yaml_files(&self.rules_dir).ok()? {
543            if Self::is_rule_file(&path)
544                && let Ok(content) = std::fs::read_to_string(&path)
545                && content.contains(&format!("id: {rule_id}"))
546            {
547                return Some(path.clone());
548            }
549        }
550        None
551    }
