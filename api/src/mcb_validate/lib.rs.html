1//! Architecture Validation for MCP Context Browser
2//!
3//! **Documentation**: [`docs/modules/validate.md`](../../../docs/modules/validate.md) |
4//! **Strategy**: [`ADR-020`](../../../docs/adr/020-testing-strategy-integration.md)
5//!
6//! This crate provides comprehensive validation of workspace crates against:
7//! - Clean Architecture principles (dependency direction)
8//! - Code quality standards (no unwrap/expect in production)
9//! - Professional patterns (DI, async traits, error types)
10//! - Test hygiene (no inline tests)
11//! - Documentation completeness
12//! - Naming conventions
13//! - SOLID principles (SRP, OCP, LSP, ISP, DIP)
14//! - Code organization (constants centralization, file placement)
15//!
16//! # Multi-Directory Validation
17//!
18//! Validation can scan multiple source directories (e.g., workspace crates + extra src/ trees):
19//!
20//! ```
21//! use mcb_validate::{GenericReporter, ValidationConfig, ValidatorRegistry};
22//!
23//! let tmp = tempfile::tempdir().unwrap();
24//! let config = ValidationConfig::new(tmp.path())
25//!     .with_exclude_pattern("target/");
26//!
27//! let registry = ValidatorRegistry::standard_for(&config.workspace_root);
28//! let violations = registry.validate_all(&config).unwrap();
29//! let report = GenericReporter::create_report(&violations, config.workspace_root.clone());
30//! assert!(report.summary.total_violations >= 0);
31//! ```

174/// Configuration for multi-directory validation
175///
176/// Allows scanning multiple source directories beyond the standard `crates/` directory.
177/// Useful for validating additional source trees alongside workspace crates.
178///
179/// # Example
180///
181/// ```
182/// use mcb_validate::ValidationConfig;
183///
184/// let config = ValidationConfig::new("/workspace")
185///     .with_additional_path("../src")
186///     .with_exclude_pattern("target/")
187///     .with_exclude_pattern("tests/fixtures/");
188/// ```
189#[derive(Debug, Clone)]
190pub struct ValidationConfig {
191    /// Root directory of the workspace (contains Cargo.toml with workspace manifest)
192    pub workspace_root: PathBuf,
193    /// Additional source paths to validate.
194    pub additional_src_paths: Vec<PathBuf>,
195    /// Patterns to exclude from validation (e.g., "target/", "tests/")
196    pub exclude_patterns: Vec<String>,
197}
198
199impl ValidationConfig {
200    /// Create a new validation config for the given workspace root.
201    ///
202    /// The workspace root is canonicalized so that all downstream path
203    /// comparisons (inventory `starts_with`, `strip_prefix`, etc.) work
204    /// correctly on platforms where temp-dir symlinks differ from the
205    /// canonical form (macOS `/tmp` â†’ `/private/tmp`, Windows `\\?\`).
206    pub fn new(workspace_root: impl Into<PathBuf>) -> Self {
207        let raw: PathBuf = workspace_root.into();
208        let canonical = std::fs::canonicalize(&raw).unwrap_or(raw);
209        Self {
210            workspace_root: canonical,
211            additional_src_paths: Vec::new(),
212            exclude_patterns: Vec::new(),
213        }
214    }
215
216    /// Add an additional source path to validate
217    ///
218    /// Paths can be absolute or relative to `workspace_root`.
219    #[must_use]
220    pub fn with_additional_path(mut self, path: impl Into<PathBuf>) -> Self {
221        self.additional_src_paths.push(path.into());
222        self
223    }
224
225    /// Add an exclude pattern (files/directories matching this will be skipped)
226    #[must_use]
227    pub fn with_exclude_pattern(mut self, pattern: impl Into<String>) -> Self {
228        self.exclude_patterns.push(pattern.into());
229        self
230    }
231
232    /// Check if a path should be excluded based on patterns
233    #[must_use]
234    pub fn should_exclude(&self, path: &Path) -> bool {
235        let Some(path_str) = path.to_str() else {
236            return false;
237        };
238        self.exclude_patterns
239            .iter()
240            .any(|pattern| path_str.contains(pattern))
241    }
242
243    /// Get all source directories to validate
244    ///
245    /// Returns crates/ subdirectories plus any additional paths.
246    ///
247    /// # Errors
248    ///
249    /// Returns an error if the crates directory cannot be read.
250    pub fn get_source_dirs(&self) -> Result<Vec<PathBuf>> {
251        let mut dirs = Vec::new();
252
253        // Load file configuration to get skip_crates
254        let file_config = FileConfig::load(&self.workspace_root);
255
256        // Original crates/ scanning
257        let crates_dir = self.workspace_root.join("crates");
258        if crates_dir.exists() {
259            for entry in std::fs::read_dir(&crates_dir)? {
260                let entry = entry?;
261                let path = entry.path();
262
263                // Skip crates specified in configuration (e.g., validate crate itself, facade crates)
264                if let Some(file_name) = path.file_name().and_then(|n| n.to_str())
265                    && file_config
266                        .general
267                        .skip_crates
268                        .iter()
269                        .any(|skip| skip == file_name)
270                {
271                    continue;
272                }
273
274                if path.is_dir() && !self.should_exclude(&path) {
275                    dirs.push(path);
276                }
277            }
278        }
279
280        // Additional paths from config
281        for path in &self.additional_src_paths {
282            let full_path = if path.is_absolute() {
283                path.clone()
284            } else {
285                self.workspace_root.join(path)
286            };
287            if full_path.exists() && !self.should_exclude(&full_path) {
288                dirs.push(full_path);
289            }
290        }
291
292        Ok(dirs)
293    }
294
295    /// Get actual source directories to scan for Rust files
296    ///
297    /// For crate directories (containing `src/` subdirectory), returns `<dir>/src/`.
298    ///
299    /// # Errors
300    ///
301    /// Returns an error if source directory enumeration fails.
302    pub fn get_scan_dirs(&self) -> Result<Vec<PathBuf>> {
303        let mut scan_dirs = Vec::new();
304
305        for dir in self.get_source_dirs()? {
306            let src_subdir = dir.join("src");
307            if src_subdir.exists() && src_subdir.is_dir() {
308                // Crate-style: has src/ subdirectory
309                scan_dirs.push(src_subdir);
310            }
311            // Standard crate without src/ directory yet - skip (implicit continue)
312        }
313
314        Ok(scan_dirs)
315    }
316}

318/// Validation error types
319#[derive(Error, Debug)]
320pub enum ValidationError {
321    /// I/O error
322    #[error("I/O error: {0}")]
323    Io(#[from] std::io::Error),
324
325    /// Parse error
326    #[error("Parse error in {file}: {message}")]
327    Parse {
328        /// Path to the file that failed to parse
329        file: PathBuf,
330        /// Error message
331        message: String,
332    },
333
334    /// TOML parse error
335    #[error("TOML parse error: {0}")]
336    Toml(#[from] toml::de::Error),
337
338    /// YAML parse error
339    #[error("YAML parse error: {0}")]
340    Yaml(#[from] serde_yaml::Error),
341
342    /// Configuration error
343    #[error("Configuration error: {0}")]
344    Config(String),
345
346    /// Invalid regex pattern
347    #[error("Invalid regex pattern: {0}")]
348    InvalidRegex(#[from] regex::Error),
349
350    /// Pattern not found
351    #[error("Pattern not found: {0}")]
352    PatternNotFound(String),
353
354    /// Validation run context must be active
355    #[error("Validation run context must be active")]
356    ContextNotActive,
357
358    /// Unknown validator name requested
359    #[error("Unknown validator(s): {names}. Available: {available}")]
360    UnknownValidator {
361        /// Validator names that were not found
362        names: String,
363        /// Available validator names
364        available: String,
365    },
366
367    /// Validator execution failed
368    #[error("Validator '{name}' failed: {message}")]
369    ValidatorFailed {
370        /// Validator name
371        name: String,
372        /// Failure description
373        message: String,
374        /// Original error
375        #[source]
376        source: Option<Box<dyn std::error::Error + Send + Sync>>,
377    },
378}

434/// Get the workspace root from the current directory
435#[must_use]
436pub fn find_workspace_root() -> Option<PathBuf> {
437    let current = std::env::current_dir().ok()?;
438    find_workspace_root_from(&current)
439}
440
441/// Find workspace root starting from a given path
442#[must_use]
443pub fn find_workspace_root_from(start: &Path) -> Option<PathBuf> {
444    let mut current = start.to_path_buf();
445    loop {
446        let cargo_toml = current.join(CARGO_TOML_FILENAME);
447        if cargo_toml.exists()
448            && let Ok(content) = std::fs::read_to_string(&cargo_toml)
449            && content.contains("[workspace]")
450        {
451            return Some(current);
452        }
453        if !current.pop() {
454            return None;
455        }
456    }
